28 /*
29 =================
30 SV_GetChallenge
31 
32 A "getchallenge" OOB command has been received
33 Returns a challenge number that can be used
34 in a subsequent connectResponse command.
35 We do this to prevent denial of service attacks that
36 flood the server with invalid connection IPs. With a
37 challenge, they must give a valid IP address.
38 
39 ioquake3: we added a possibility for clients to add a challenge
40 to their packets, to make it more difficult for malicious servers
41 to hi-jack client connections.
42 =================
43 */
44 void SV_GetChallenge(netadr_t from)
45 {
46  int i;
47  int oldest;
48  int oldestTime;
49  int oldestClientTime;
50  int clientChallenge;
51  challenge_t *challenge;
52  qboolean wasfound = qfalse;
53  char *gameName;
54  qboolean gameMismatch;
55 
56  // ignore if we are in single player
57  if ( Cvar_VariableValue( "g_gametype" ) == GT_SINGLE_PLAYER || Cvar_VariableValue("ui_singlePlayerActive")) {
58  return;
59  }
60 
61  gameName = Cmd_Argv(2);
62 
63 #ifdef LEGACY_PROTOCOL
64  // gamename is optional for legacy protocol
65  if (com_legacyprotocol->integer && !*gameName)
66  gameMismatch = qfalse;
67  else
68 #endif
69  gameMismatch = !*gameName || strcmp(gameName, com_gamename->string) != 0;
70 
71  // reject client if the gamename string sent by the client doesn't match ours
72  if (gameMismatch)
73  {
74  NET_OutOfBandPrint(NS_SERVER, from, "print\nGame mismatch: This is a %s server\n",
75  com_gamename->string);
76  return;
77  }
78 
79  oldest = 0;
80  oldestClientTime = oldestTime = 0x7fffffff;
81 
82  // see if we already have a challenge for this ip
83  challenge = &svs.challenges[0];
84  clientChallenge = atoi(Cmd_Argv(1));
85 
86  for(i = 0 ; i < MAX_CHALLENGES ; i++, challenge++)
87  {
88  if(!challenge->connected && NET_CompareAdr(from, challenge->adr))
89  {
90  wasfound = qtrue;
91 
92  if(challenge->time < oldestClientTime)
93  oldestClientTime = challenge->time;
94  }
95 
96  if(wasfound && i >= MAX_CHALLENGES_MULTI)
97  {
98  i = MAX_CHALLENGES;
99  break;
100  }
101 
102  if(challenge->time < oldestTime)
103  {
104  oldestTime = challenge->time;
105  oldest = i;
106  }
107  }
108 
109  if (i == MAX_CHALLENGES)
110  {
111  // this is the first time this client has asked for a challenge
112  challenge = &svs.challenges[oldest];
113  challenge->clientChallenge = clientChallenge;
114  challenge->adr = from;
115  challenge->connected = qfalse;
116  }
117 
118  // always generate a new challenge number, so the client cannot circumvent sv_maxping
119  challenge->challenge = ( (rand() << 16) ^ rand() ) ^ svs.time;
120  challenge->wasrefused = qfalse;
121  challenge->time = svs.time;
122 
123  challenge->pingTime = svs.time;
124  NET_OutOfBandPrint(NS_SERVER, challenge->adr, "challengeResponse %d %d %d",
125  challenge->challenge, clientChallenge, com_protocol->integer);
126 }

170 void SV_DirectConnect( netadr_t from ) {
171  char userinfo[MAX_INFO_STRING];
172  int i;
173  client_t *cl, *newcl;
174  client_t temp;
175  sharedEntity_t *ent;
176  int clientNum;
177  int version;
178  int qport;
179  int challenge;
180  char *password;
181  int startIndex;
182  intptr_t denied;
183  int count;
184  char *ip;
185 #ifdef LEGACY_PROTOCOL
186  qboolean compat = qfalse;
187 #endif
188 
189  Com_DPrintf ("SVC_DirectConnect ()\n");
190 
191  // Check whether this client is banned.
192  if(SV_IsBanned(&from, qfalse))
193  {
194  NET_OutOfBandPrint(NS_SERVER, from, "print\nYou are banned from this server.\n");
195  return;
196  }
197 
198  Q_strncpyz( userinfo, Cmd_Argv(1), sizeof(userinfo) );
199 
200  version = atoi(Info_ValueForKey(userinfo, "protocol"));
201 
202 #ifdef LEGACY_PROTOCOL
203  if(version > 0 && com_legacyprotocol->integer == version)
204  compat = qtrue;
205  else
206 #endif
207  {
208  if(version != com_protocol->integer)
209  {
210  NET_OutOfBandPrint(NS_SERVER, from, "print\nServer uses protocol version %i "
211  "(yours is %i).\n", com_protocol->integer, version);
212  Com_DPrintf(" rejected connect from version %i\n", version);
213  return;
214  }
215  }
216 
217  challenge = atoi( Info_ValueForKey( userinfo, "challenge" ) );
218  qport = atoi( Info_ValueForKey( userinfo, "qport" ) );
219 
220  // quick reject
221  for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
222  if ( cl->state == CS_FREE ) {
223  continue;
224  }
225  if ( NET_CompareBaseAdr( from, cl->netchan.remoteAddress )
226  && ( cl->netchan.qport == qport
227  || from.port == cl->netchan.remoteAddress.port ) ) {
228  if (( svs.time - cl->lastConnectTime)
229  < (sv_reconnectlimit->integer * 1000)) {
230  Com_DPrintf ("%s:reconnect rejected : too soon\n", NET_AdrToString (from));
231  return;
232  }
233  break;
234  }
235  }
236 
237  // don't let "ip" overflow userinfo string
238  if ( NET_IsLocalAddress (from) )
239  ip = "localhost";
240  else
241  ip = (char *)NET_AdrToString( from );
242  if( ( strlen( ip ) + strlen( userinfo ) + 4 ) >= MAX_INFO_STRING ) {
243  NET_OutOfBandPrint( NS_SERVER, from,
244  "print\nUserinfo string length exceeded. "
245  "Try removing setu cvars from your config.\n" );
246  return;
247  }
248  Info_SetValueForKey( userinfo, "ip", ip );
249 
250  // see if the challenge is valid (LAN clients don't need to challenge)
251  if (!NET_IsLocalAddress(from))
252  {
253  int ping;
254  challenge_t *challengeptr;
255 
256  for (i=0; i<MAX_CHALLENGES; i++)
257  {
258  if (NET_CompareAdr(from, svs.challenges[i].adr))
259  {
260  if(challenge == svs.challenges[i].challenge)
261  break;
262  }
263  }
264 
265  if (i == MAX_CHALLENGES)
266  {
267  NET_OutOfBandPrint( NS_SERVER, from, "print\nNo or bad challenge for your address.\n" );
268  return;
269  }
270 
271  challengeptr = &svs.challenges[i];
272 
273  if(challengeptr->wasrefused)
274  {
275  // Return silently, so that error messages written by the server keep being displayed.
276  return;
277  }
278 
279  ping = svs.time - challengeptr->pingTime;
280 
281  // never reject a LAN client based on ping
282  if ( !Sys_IsLANAddress( from ) ) {
283  if ( sv_minPing->value && ping < sv_minPing->value ) {
284  NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is for high pings only\n" );
285  Com_DPrintf ("Client %i rejected on a too low ping\n", i);
286  challengeptr->wasrefused = qtrue;
287  return;
288  }
289  if ( sv_maxPing->value && ping > sv_maxPing->value ) {
290  NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is for low pings only\n" );
291  Com_DPrintf ("Client %i rejected on a too high ping\n", i);
292  challengeptr->wasrefused = qtrue;
293  return;
294  }
295  }
296 
297  Com_Printf("Client %i connecting with %i challenge ping\n", i, ping);
298  challengeptr->connected = qtrue;
299  }
300 
301  newcl = &temp;
302  Com_Memset (newcl, 0, sizeof(client_t));
303 
304  // if there is already a slot for this ip, reuse it
305  for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
306  if ( cl->state == CS_FREE ) {
307  continue;
308  }
309  if ( NET_CompareBaseAdr( from, cl->netchan.remoteAddress )
310  && ( cl->netchan.qport == qport
311  || from.port == cl->netchan.remoteAddress.port ) ) {
312  Com_Printf ("%s:reconnect\n", NET_AdrToString (from));
313  newcl = cl;
314 
315  // this doesn't work because it nukes the players userinfo
316 
317 // // disconnect the client from the game first so any flags the
318 // // player might have are dropped
319 // VM_Call( gvm, GAME_CLIENT_DISCONNECT, newcl - svs.clients );
320  //
321  goto gotnewcl;
322  }
323  }
324 
325  // find a client slot
326  // if "sv_privateClients" is set > 0, then that number
327  // of client slots will be reserved for connections that
328  // have "password" set to the value of "sv_privatePassword"
329  // Info requests will report the maxclients as if the private
330  // slots didn't exist, to prevent people from trying to connect
331  // to a full server.
332  // This is to allow us to reserve a couple slots here on our
333  // servers so we can play without having to kick people.
334 
335  // check for privateClient password
336  password = Info_ValueForKey( userinfo, "password" );
337  if ( !strcmp( password, sv_privatePassword->string ) ) {
338  startIndex = 0;
339  } else {
340  // skip past the reserved slots
341  startIndex = sv_privateClients->integer;
342  }
343 
344  newcl = NULL;
345  for ( i = startIndex; i < sv_maxclients->integer ; i++ ) {
346  cl = &svs.clients[i];
347  if (cl->state == CS_FREE) {
348  newcl = cl;
349  break;
350  }
351  }
352 
353  if ( !newcl ) {
354  if ( NET_IsLocalAddress( from ) ) {
355  count = 0;
356  for ( i = startIndex; i < sv_maxclients->integer ; i++ ) {
357  cl = &svs.clients[i];
358  }
359  Com_Error( ERR_FATAL, "server is full on local connect" );
360  return;
361  }
362  else {
363  NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is full.\n" );
364  Com_DPrintf ("Rejected a connection.\n");
365  return;
366  }
367  }
368 
369  // we got a newcl, so reset the reliableSequence and reliableAcknowledge
370  cl->reliableAcknowledge = 0;
371  cl->reliableSequence = 0;
372 
373 gotnewcl:
374  // build a new connection
375  // accept the new client
376  // this is the only place a client_t is ever initialized
377  *newcl = temp;
378  clientNum = newcl - svs.clients;
379  ent = SV_GentityNum( clientNum );
380  newcl->gentity = ent;
381 
382  // save the challenge
383  newcl->challenge = challenge;
384 
385  // save the address
386 #ifdef LEGACY_PROTOCOL
387  newcl->compat = compat;
388  Netchan_Setup(NS_SERVER, &newcl->netchan, from, qport, challenge, compat);
389 #else
390  Netchan_Setup(NS_SERVER, &newcl->netchan, from, qport, challenge, qfalse);
391 #endif
392  // init the netchan queue
393  newcl->netchan_end_queue = &newcl->netchan_start_queue;
394 
395  // save the userinfo
396  Q_strncpyz( newcl->userinfo, userinfo, sizeof(newcl->userinfo) );
397 
398  // get the game a chance to reject this connection or modify the userinfo
399  denied = VM_Call( gvm, GAME_CLIENT_CONNECT, clientNum, qtrue, qfalse ); // firstTime = qtrue
400  if ( denied ) {
401  // we can't just use VM_ArgPtr, because that is only valid inside a VM_Call
402  char *str = VM_ExplicitArgPtr( gvm, denied );
403 
404  NET_OutOfBandPrint( NS_SERVER, from, "print\n%s\n", str );
405  Com_DPrintf ("Game rejected a connection: %s.\n", str);
406  return;
407  }
408 
409  SV_UserinfoChanged( newcl );
410 
411  // send the connect packet to the client
412  NET_OutOfBandPrint(NS_SERVER, from, "connectResponse %d", challenge);
413 
414  Com_DPrintf( "Going from CS_FREE to CS_CONNECTED for %s\n", newcl->name );
415 
416  newcl->state = CS_CONNECTED;
417  newcl->lastSnapshotTime = 0;
418  newcl->lastPacketTime = svs.time;
419  newcl->lastConnectTime = svs.time;
420 
421  // when we receive the first packet from the client, we will
422  // notice that it is from a different serverid and that the
423  // gamestate message was not just sent, forcing a retransmit
424  newcl->gamestateMessageNum = -1;
425 
426  // if this was the first client on the server, or the last client
427  // the server can hold, send a heartbeat to the master.
428  count = 0;
429  for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
430  if ( svs.clients[i].state >= CS_CONNECTED ) {
431  count++;
432  }
433  }
434  if ( count == 1 || count == sv_maxclients->integer ) {
435  SV_Heartbeat_f();
436  }
437 }

465 /*
466 =====================
467 SV_DropClient
468 
469 Called when the player is totally leaving the server, either willingly
470 or unwillingly. This is NOT called if the entire server is quiting
471 or crashing -- SV_FinalMessage() will handle that
472 =====================
473 */
474 void SV_DropClient( client_t *drop, const char *reason ) {
475  int i;
476  challenge_t *challenge;
477 
478  if ( drop->state == CS_ZOMBIE ) {
479  return; // already dropped
480  }
481 
482  // see if we already have a challenge for this ip
483  challenge = &svs.challenges[0];
484 
485  for (i = 0 ; i < MAX_CHALLENGES ; i++, challenge++)
486  {
487  if(NET_CompareAdr(drop->netchan.remoteAddress, challenge->adr))
488  {
489  Com_Memset(challenge, 0, sizeof(*challenge));
490  break;
491  }
492  }
493 
494  // Free all allocated data on the client structure
495  SV_FreeClient(drop);
496 
497  // tell everyone why they got dropped
498  SV_SendServerCommand( NULL, "print \"%s" S_COLOR_WHITE " %s\n\"", drop->name, reason );
499 
500  // call the prog function for removing a client
501  // this will remove the body, among other things
502  VM_Call( gvm, GAME_CLIENT_DISCONNECT, drop - svs.clients );
503 
504  // add the disconnect command
505  SV_SendServerCommand( drop, "disconnect \"%s\"", reason);
506 
507  // nuke user info
508  SV_SetUserinfo( drop - svs.clients, "" );
509 
510  Com_DPrintf( "Going to CS_ZOMBIE for %s\n", drop->name );
511  drop->state = CS_ZOMBIE; // become free in a few seconds
512 
513  // if this was the last client on the server, send a heartbeat
514  // to the master so it is known the server is empty
515  // send a heartbeat now so the master will get up to date info
516  // if there is already a slot for this ip, reuse it
517  for (i=0 ; i < sv_maxclients->integer ; i++ ) {
518  if ( svs.clients[i].state >= CS_CONNECTED ) {
519  break;
520  }
521  }
522  if ( i == sv_maxclients->integer ) {
523  SV_Heartbeat_f();
524  }
525 }

527 /*
528 ================
529 SV_SendClientGameState
530 
531 Sends the first message from the server to a connected client.
532 This will be sent on the initial connection and upon each new map load.
533 
534 It will be resent if the client acknowledges a later message but has
535 the wrong gamestate.
536 ================
537 */
538 static void SV_SendClientGameState( client_t *client ) {
539  int start;
540  entityState_t *base, nullstate;
541  msg_t msg;
542  byte msgBuffer[MAX_MSGLEN];
543 
544  Com_DPrintf ("SV_SendClientGameState() for %s\n", client->name);
545  Com_DPrintf( "Going from CS_CONNECTED to CS_PRIMED for %s\n", client->name );
546  client->state = CS_PRIMED;
547  client->pureAuthentic = 0;
548  client->gotCP = qfalse;
549 
550  // when we receive the first packet from the client, we will
551  // notice that it is from a different serverid and that the
552  // gamestate message was not just sent, forcing a retransmit
553  client->gamestateMessageNum = client->netchan.outgoingSequence;
554 
555  MSG_Init( &msg, msgBuffer, sizeof( msgBuffer ) );
556 
557  // NOTE, MRE: all server->client messages now acknowledge
558  // let the client know which reliable clientCommands we have received
559  MSG_WriteLong( &msg, client->lastClientCommand );
560 
561  // send any server commands waiting to be sent first.
562  // we have to do this cause we send the client->reliableSequence
563  // with a gamestate and it sets the clc.serverCommandSequence at
564  // the client side
565  SV_UpdateServerCommandsToClient( client, &msg );
566 
567  // send the gamestate
568  MSG_WriteByte( &msg, svc_gamestate );
569  MSG_WriteLong( &msg, client->reliableSequence );
570 
571  // write the configstrings
572  for ( start = 0 ; start < MAX_CONFIGSTRINGS ; start++ ) {
573  if (sv.configstrings[start][0]) {
574  MSG_WriteByte( &msg, svc_configstring );
575  MSG_WriteShort( &msg, start );
576  MSG_WriteBigString( &msg, sv.configstrings[start] );
577  }
578  }
579 
580  // write the baselines
581  Com_Memset( &nullstate, 0, sizeof( nullstate ) );
582  for ( start = 0 ; start < MAX_GENTITIES; start++ ) {
583  base = &sv.svEntities[start].baseline;
584  if ( !base->number ) {
585  continue;
586  }
587  MSG_WriteByte( &msg, svc_baseline );
588  MSG_WriteDeltaEntity( &msg, &nullstate, base, qtrue );
589  }
590 
591  MSG_WriteByte( &msg, svc_EOF );
592 
593  MSG_WriteLong( &msg, client - svs.clients);
594 
595  // write the checksum feed
596  MSG_WriteLong( &msg, sv.checksumFeed);
597 
598  // deliver this to the client
599  SV_SendMessageToClient( &msg, client );
600 }

748 SV_WriteDownloadToClient
749 
750 Check to see if the client wants a file, open it if needed and start pumping the client
751 Fill up msg with data, return number of download blocks added
752 ==================
753 */
754 int SV_WriteDownloadToClient(client_t *cl, msg_t *msg)
755 {
756  int curindex;
757  int unreferenced = 1;
758  char errorMessage[1024];
759  char pakbuf[MAX_QPATH], *pakptr;
760  int numRefPaks;
761 
762  if (!*cl->downloadName)
763  return 0; // Nothing being downloaded
764 
765  if(!cl->download)
766  {
767  qboolean idPack = qfalse;
768

770  Com_sprintf(pakbuf, sizeof(pakbuf), "%s", cl->downloadName);
771  pakptr = strrchr(pakbuf, '.');
772 
773  if(pakptr)
774  {
775  *pakptr = '\0';
776 
777  // Check for pk3 filename extension
778  if(!Q_stricmp(pakptr + 1, "pk3"))
779  {
780  const char *referencedPaks = FS_ReferencedPakNames();
781 
782  // Check whether the file appears in the list of referenced
783  // paks to prevent downloading of arbitrary files.
784  Cmd_TokenizeStringIgnoreQuotes(referencedPaks);
785  numRefPaks = Cmd_Argc();
786 
787  for(curindex = 0; curindex < numRefPaks; curindex++)
788  {
789  if(!FS_FilenameCompare(Cmd_Argv(curindex), pakbuf))
790  {
791  unreferenced = 0;
792 
793  // now that we know the file is referenced,
794  // check whether it's legal to download it.
795  idPack = idPack || FS_idPak(pakbuf, BASEGAME, NUM_ID_PAKS);
796 
797  break;
798  }
799  }
800  }
801  }
802 
803  cl->download = 0;
804 
805  // We open the file here
806  if ( !(sv_allowDownload->integer & DLF_ENABLE) ||
807  (sv_allowDownload->integer & DLF_NO_UDP) ||
808  idPack || unreferenced ||
809  ( cl->downloadSize = FS_SV_FOpenFileRead( cl->downloadName, &cl->download ) ) < 0 ) {
810  // cannot auto-download file
811  if(unreferenced)
812  {
813  Com_Printf("clientDownload: %d : \"%s\" is not referenced and cannot be downloaded.\n", (int) (cl - svs.clients), cl->downloadName);
814  Com_sprintf(errorMessage, sizeof(errorMessage), "File \"%s\" is not referenced and cannot be downloaded.", cl->downloadName);
815  }
816  else if (idPack) {
817  Com_Printf("clientDownload: %d : \"%s\" cannot download id pk3 files\n", (int) (cl - svs.clients), cl->downloadName);
818  Com_sprintf(errorMessage, sizeof(errorMessage), "Cannot autodownload id pk3 file \"%s\"", cl->downloadName);
819  }
820  else if ( !(sv_allowDownload->integer & DLF_ENABLE) ||
821  (sv_allowDownload->integer & DLF_NO_UDP) ) {
822 
823  Com_Printf("clientDownload: %d : \"%s\" download disabled", (int) (cl - svs.clients), cl->downloadName);
824  if (sv_pure->integer) {
825  Com_sprintf(errorMessage, sizeof(errorMessage), "Could not download \"%s\" because autodownloading is disabled on the server.\n\n"
826  "You will need to get this file elsewhere before you "
827  "can connect to this pure server.\n", cl->downloadName);
828  } else {
829  Com_sprintf(errorMessage, sizeof(errorMessage), "Could not download \"%s\" because autodownloading is disabled on the server.\n\n"
830  "The server you are connecting to is not a pure server, "
831  "set autodownload to No in your settings and you might be "
832  "able to join the game anyway.\n", cl->downloadName);
833  }
834  } else {
835  // NOTE TTimo this is NOT supposed to happen unless bug in our filesystem scheme?
836  // if the pk3 is referenced, it must have been found somewhere in the filesystem
837  Com_Printf("clientDownload: %d : \"%s\" file not found on server\n", (int) (cl - svs.clients), cl->downloadName);
838  Com_sprintf(errorMessage, sizeof(errorMessage), "File \"%s\" not found on server for autodownloading.\n", cl->downloadName);
839  }
840  MSG_WriteByte( msg, svc_download );
841  MSG_WriteShort( msg, 0 ); // client is expecting block zero
842  MSG_WriteLong( msg, -1 ); // illegal file size
843  MSG_WriteString( msg, errorMessage );
844 
845  *cl->downloadName = 0;
846 
847  if(cl->download)
848  FS_FCloseFile(cl->download);
849 
850  return 0;
851  }
852 
853  Com_Printf( "clientDownload: %d : beginning \"%s\"\n", (int) (cl - svs.clients), cl->downloadName );
854 
855  // Init
856  cl->downloadCurrentBlock = cl->downloadClientBlock = cl->downloadXmitBlock = 0;
857  cl->downloadCount = 0;
858  cl->downloadEOF = qfalse;
859  }
860 
861  // Perform any reads that we need to
862  while (cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW &&
863  cl->downloadSize != cl->downloadCount) {
864 
865  curindex = (cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW);
866 
867  if (!cl->downloadBlocks[curindex])
868  cl->downloadBlocks[curindex] = Z_Malloc(MAX_DOWNLOAD_BLKSIZE);
869 
870  cl->downloadBlockSize[curindex] = FS_Read( cl->downloadBlocks[curindex], MAX_DOWNLOAD_BLKSIZE, cl->download );
871 
872  if (cl->downloadBlockSize[curindex] < 0) {
873  // EOF right now
874  cl->downloadCount = cl->downloadSize;
875  break;
876  }
877 
878  cl->downloadCount += cl->downloadBlockSize[curindex];
879 
880  // Load in next block
881  cl->downloadCurrentBlock++;
882  }
883 
884  // Check to see if we have eof condition and add the EOF block
885  if (cl->downloadCount == cl->downloadSize &&
886  !cl->downloadEOF &&
887  cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW) {
888 
889  cl->downloadBlockSize[cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW] = 0;
890  cl->downloadCurrentBlock++;
891 
892  cl->downloadEOF = qtrue; // We have added the EOF block
893  }
894 
895  if (cl->downloadClientBlock == cl->downloadCurrentBlock)
896  return 0; // Nothing to transmit
897 
898  // Write out the next section of the file, if we have already reached our window,
899  // automatically start retransmitting
900  if (cl->downloadXmitBlock == cl->downloadCurrentBlock)
901  {
902  // We have transmitted the complete window, should we start resending?
903  if (svs.time - cl->downloadSendTime > 1000)
904  cl->downloadXmitBlock = cl->downloadClientBlock;
905  else
906  return 0;
907  }
908 
909  // Send current block
910  curindex = (cl->downloadXmitBlock % MAX_DOWNLOAD_WINDOW);
911 
912  MSG_WriteByte( msg, svc_download );
913  MSG_WriteShort( msg, cl->downloadXmitBlock );
914 
915  // block zero is special, contains file size
916  if ( cl->downloadXmitBlock == 0 )
917  MSG_WriteLong( msg, cl->downloadSize );
918 
919  MSG_WriteShort( msg, cl->downloadBlockSize[curindex] );
920 
921  // Write the block
922  if(cl->downloadBlockSize[curindex])
923  MSG_WriteData(msg, cl->downloadBlocks[curindex], cl->downloadBlockSize[curindex]);
924 
925  Com_DPrintf( "clientDownload: %d : writing block %d\n", (int) (cl - svs.clients), cl->downloadXmitBlock );
926 
927  // Move on to the next block
928  // It will get sent with next snap shot. The rate will keep us in line.
929  cl->downloadXmitBlock++;
930  cl->downloadSendTime = svs.time;
931 
932  return 1;
933 }

1021 
1022 If we are pure, disconnect the client if they do no meet the following conditions:
1023

1025 2. there are no any additional checksums that we do not have
1026

1028 
1029 =================
1030 */
1031 static void SV_VerifyPaks_f( client_t *cl ) {
1032  int nChkSum1, nChkSum2, nClientPaks, nServerPaks, i, j, nCurArg;
1033  int nClientChkSum[1024];
1034  int nServerChkSum[1024];
1035  const char *pPaks, *pArg;
1036  qboolean bGood = qtrue;
1037 
1038  // if we are pure, we "expect" the client to load certain things from
1039  // certain pk3 files, namely we want the client to have loaded the
1040  // ui and cgame that we think should be loaded based on the pure setting
1041  //
1042  if ( sv_pure->integer != 0 ) {
1043 
1044  nChkSum1 = nChkSum2 = 0;
1045  // we run the game, so determine which cgame and ui the client "should" be running
1046  bGood = (FS_FileIsInPAK("vm/cgame.qvm", &nChkSum1) == 1);
1047  if (bGood)
1048  bGood = (FS_FileIsInPAK("vm/ui.qvm", &nChkSum2) == 1);
1049 
1050  nClientPaks = Cmd_Argc();
1051 
1052  // start at arg 2 ( skip serverId cl_paks )
1053  nCurArg = 1;
1054 
1055  pArg = Cmd_Argv(nCurArg++);
1056  if(!pArg) {
1057  bGood = qfalse;
1058  }
1059  else
1060  {
1061  // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=475
1062  // we may get incoming cp sequences from a previous checksumFeed, which we need to ignore
1063  // since serverId is a frame count, it always goes up
1064  if (atoi(pArg) < sv.checksumFeedServerId)
1065  {
1066  Com_DPrintf("ignoring outdated cp command from client %s\n", cl->name);
1067  return;
1068  }
1069  }
1070 
1071  // we basically use this while loop to avoid using 'goto' :)
1072  while (bGood) {
1073 
1074  // must be at least 6: "cl_paks cgame ui @ firstref ... numChecksums"
1075  // numChecksums is encoded
1076  if (nClientPaks < 6) {
1077  bGood = qfalse;
1078  break;
1079  }
1080  // verify first to be the cgame checksum
1081  pArg = Cmd_Argv(nCurArg++);
1082  if (!pArg || *pArg == '@' || atoi(pArg) != nChkSum1 ) {
1083  bGood = qfalse;
1084  break;
1085  }
1086  // verify the second to be the ui checksum
1087  pArg = Cmd_Argv(nCurArg++);
1088  if (!pArg || *pArg == '@' || atoi(pArg) != nChkSum2 ) {
1089  bGood = qfalse;
1090  break;
1091  }
1092  // should be sitting at the delimeter now
1093  pArg = Cmd_Argv(nCurArg++);
1094  if (*pArg != '@') {
1095  bGood = qfalse;
1096  break;
1097  }
1098  // store checksums since tokenization is not re-entrant
1099  for (i = 0; nCurArg < nClientPaks; i++) {
1100  nClientChkSum[i] = atoi(Cmd_Argv(nCurArg++));
1101  }
1102 
1103  // store number to compare against (minus one cause the last is the number of checksums)
1104  nClientPaks = i - 1;
1105 
1106  // make sure none of the client check sums are the same
1107  // so the client can't send 5 the same checksums
1108  for (i = 0; i < nClientPaks; i++) {
1109  for (j = 0; j < nClientPaks; j++) {
1110  if (i == j)
1111  continue;
1112  if (nClientChkSum[i] == nClientChkSum[j]) {
1113  bGood = qfalse;
1114  break;
1115  }
1116  }
1117  if (bGood == qfalse)
1118  break;
1119  }
1120  if (bGood == qfalse)
1121  break;
1122 
1123  // get the pure checksums of the pk3 files loaded by the server
1124  pPaks = FS_LoadedPakPureChecksums();
1125  Cmd_TokenizeString( pPaks );
1126  nServerPaks = Cmd_Argc();
1127  if (nServerPaks > 1024)
1128  nServerPaks = 1024;
1129 
1130  for (i = 0; i < nServerPaks; i++) {
1131  nServerChkSum[i] = atoi(Cmd_Argv(i));
1132  }
1133 
1134  // check if the client has provided any pure checksums of pk3 files not loaded by the server
1135  for (i = 0; i < nClientPaks; i++) {
1136  for (j = 0; j < nServerPaks; j++) {
1137  if (nClientChkSum[i] == nServerChkSum[j]) {
1138  break;
1139  }
1140  }
1141  if (j >= nServerPaks) {
1142  bGood = qfalse;
1143  break;
1144  }
1145  }
1146  if ( bGood == qfalse ) {
1147  break;
1148  }
1149 
1150  // check if the number of checksums was correct
1151  nChkSum1 = sv.checksumFeed;
1152  for (i = 0; i < nClientPaks; i++) {
1153  nChkSum1 ^= nClientChkSum[i];
1154  }
1155  nChkSum1 ^= nClientPaks;
1156  if (nChkSum1 != nClientChkSum[nClientPaks]) {
1157  bGood = qfalse;
1158  break;
1159  }
1160 
1161  // break out
1162  break;
1163  }
1164 
1165  cl->gotCP = qtrue;
1166 
1167  if (bGood) {
1168  cl->pureAuthentic = 1;
1169  }
1170  else {
1171  cl->pureAuthentic = 0;
1172  cl->lastSnapshotTime = 0;
1173  cl->state = CS_ACTIVE;
1174  SV_SendClientSnapshot( cl );
1175  SV_DropClient( cl, "Unpure client detected. Invalid .PK3 files referenced!" );
1176  }
1177  }
1178 }
