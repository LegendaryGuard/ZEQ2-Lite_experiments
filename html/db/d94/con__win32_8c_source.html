48 /*
49 ==================
50 CON_CtrlHandler
51 
52 The Windows Console doesn't use signals for terminating the application
53 with Ctrl-C, logging off, window closing, etc. Instead it uses a special
54 handler routine. Fortunately, the values for Ctrl signals don't seem to
55 overlap with true signal codes that Windows provides, so calling
56 Sys_SigHandler() with those numbers should be safe for generating unique
57 shutdown messages.
58 ==================
59 */
60 static BOOL WINAPI CON_CtrlHandler( DWORD sig )
61 {
62  Sys_SigHandler( sig );
63  return TRUE;
64 }

135 CON_Show
136 ==================
137 */
138 static void CON_Show( void )
139 {
140  CONSOLE_SCREEN_BUFFER_INFO binfo;
141  COORD writeSize = { MAX_EDIT_LINE, 1 };
142  COORD writePos = { 0, 0 };
143  SMALL_RECT writeArea = { 0, 0, 0, 0 };
144  int i;
145  CHAR_INFO line[ MAX_EDIT_LINE ];
146 
147  GetConsoleScreenBufferInfo( qconsole_hout, &binfo );
148 
149  // if we're in the middle of printf, don't bother writing the buffer
150  if( binfo.dwCursorPosition.X != 0 )
151  return;
152 
153  writeArea.Left = 0;
154  writeArea.Top = binfo.dwCursorPosition.Y;
155  writeArea.Bottom = binfo.dwCursorPosition.Y;
156  writeArea.Right = MAX_EDIT_LINE;
157 
158  // build a space-padded CHAR_INFO array
159  for( i = 0; i < MAX_EDIT_LINE; i++ )
160  {
161  if( i < qconsole_linelen )
162  line[ i ].Char.AsciiChar = qconsole_line[ i ];
163  else
164  line[ i ].Char.AsciiChar = ' ';
165 
166  line[ i ].Attributes = qconsole_attrib;
167  }
168 
169  if( qconsole_linelen > binfo.srWindow.Right )
170  {
171  WriteConsoleOutput( qconsole_hout,
172  line + (qconsole_linelen - binfo.srWindow.Right ),
173  writeSize, writePos, &writeArea );
174  }
175  else
176  {
177  WriteConsoleOutput( qconsole_hout, line, writeSize,
178  writePos, &writeArea );
179  }
180 }

242 ==================
243 CON_Input
244 ==================
245 */
246 char *CON_Input( void )
247 {
248  INPUT_RECORD buff[ MAX_EDIT_LINE ];
249  DWORD count = 0, events = 0;
250  WORD key = 0;
251  int i;
252  int newlinepos = -1;
253 
254  if( !GetNumberOfConsoleInputEvents( qconsole_hin, &events ) )
255  return NULL;
256 
257  if( events < 1 )
258  return NULL;
259 
260  // if we have overflowed, start dropping oldest input events
261  if( events >= MAX_EDIT_LINE )
262  {
263  ReadConsoleInput( qconsole_hin, buff, 1, &events );
264  return NULL;
265  }
266 
267  if( !ReadConsoleInput( qconsole_hin, buff, events, &count ) )
268  return NULL;
269 
270  FlushConsoleInputBuffer( qconsole_hin );
271 
272  for( i = 0; i < count; i++ )
273  {
274  if( buff[ i ].EventType != KEY_EVENT )
275  continue;
276  if( !buff[ i ].Event.KeyEvent.bKeyDown )
277  continue;
278 
279  key = buff[ i ].Event.KeyEvent.wVirtualKeyCode;
280 
281  if( key == VK_RETURN )
282  {
283  newlinepos = i;
284  break;
285  }
286  else if( key == VK_UP )
287  {
288  CON_HistPrev();
289  break;
290  }
291  else if( key == VK_DOWN )
292  {
293  CON_HistNext();
294  break;
295  }
296  else if( key == VK_TAB )
297  {
298  field_t f;
299 
300  Q_strncpyz( f.buffer, qconsole_line,
301  sizeof( f.buffer ) );
302  Field_AutoComplete( &f );
303  Q_strncpyz( qconsole_line, f.buffer,
304  sizeof( qconsole_line ) );
305  qconsole_linelen = strlen( qconsole_line );
306  break;
307  }
308 
309  if( qconsole_linelen < sizeof( qconsole_line ) - 1 )
310  {
311  char c = buff[ i ].Event.KeyEvent.uChar.AsciiChar;
312 
313  if( key == VK_BACK )
314  {
315  int pos = ( qconsole_linelen > 0 ) ?
316  qconsole_linelen - 1 : 0;
317 
318  qconsole_line[ pos ] = '\0';
319  qconsole_linelen = pos;
320  }
321  else if( c )
322  {
323  qconsole_line[ qconsole_linelen++ ] = c;
324  qconsole_line[ qconsole_linelen ] = '\0';
325  }
326  }
327  }
328 
329  if( newlinepos < 0) {
330  CON_Show();
331  return NULL;
332  }
333 
334  if( !qconsole_linelen )
335  {
336  CON_Show();
337  Com_Printf( "\n" );
338  return NULL;
339  }
340 
341  CON_HistAdd();
342  Com_Printf( "%s\n", qconsole_line );
343 
344  qconsole_linelen = 0;
345  CON_Show();
346 
347  return qconsole_line;
348 }
