444 static void Controls_Update( void ) {
445  int i;
446  int j;
447  int y;
448  menucommon_s **controls;
449  menucommon_s *control;
450 
451 
452  controls = g_controls;
453 
454  // enable controls in active group (and count number of items for vertical centering)
455  for( j = 0; (control = controls[j]) ; j++ ) {
456  control->flags &= ~(QMF_GRAYED|QMF_HIDDEN|QMF_INACTIVE);
457  }
458 
459  // position controls
460  y = ( SCREEN_HEIGHT - j * SMALLCHAR_HEIGHT ) / 1.25;
461  for( j = 0; (control = controls[j]) ; j++, y += CONTROLSCHAR_HEIGHT ) {
462  control->x = 320;
463  control->y = y;
464  control->left = 320 - 19*SMALLCHAR_WIDTH;
465  control->right = 320 + 21*SMALLCHAR_WIDTH;
466  control->top = y;
467  control->bottom = y + CONTROLSCHAR_HEIGHT;
468  }
469 
470  if( s_controls.waitingforkey ) {
471  // disable everybody
472  for( i = 0; i < s_controls.menu.nitems; i++ ) {
473  ((menucommon_s*)(s_controls.menu.items[i]))->flags |= QMF_GRAYED;
474  }
475 
476  // enable action item
477  ((menucommon_s*)(s_controls.menu.items[s_controls.menu.cursor]))->flags &= ~QMF_GRAYED;
478 
479  // don't gray out player's name
480  s_controls.name.generic.flags &= ~QMF_GRAYED;
481 
482  return;
483  }
484 
485  // enable everybody
486  for( i = 0; i < s_controls.menu.nitems; i++ ) {
487  ((menucommon_s*)(s_controls.menu.items[i]))->flags &= ~QMF_GRAYED;
488  }
489 
490 }

595 static void Controls_GetKeyAssignment (char *command, int *twokeys)
596 {
597  int count;
598  int j;
599  char b[256];
600 
601  twokeys[0] = twokeys[1] = -1;
602  count = 0;
603 
604  for ( j = 0; j < 256; j++ )
605  {
606  trap_Key_GetBindingBuf( j, b, 256 );
607  if ( *b == 0 ) {
608  continue;
609  }
610  if ( !Q_stricmp( b, command ) ) {
611  twokeys[count] = j;
612  count++;
613  if (count == 2)
614  break;
615  }
616  }
617 }
