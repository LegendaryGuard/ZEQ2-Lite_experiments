28 /*
29 ===================
30 CG_PlaceString
31 
32 Also called by scoreboard drawing
33 ===================
34 */
35 const char *CG_PlaceString( int rank ) {
36  static char str[64];
37  char *s, *t;
38 
39  if ( rank & RANK_TIED_FLAG ) {
40  rank &= ~RANK_TIED_FLAG;
41  t = "Tied for ";
42  } else {
43  t = "";
44  }
45 
46  if ( rank == 1 ) {
47  s = S_COLOR_BLUE "1st" S_COLOR_WHITE; // draw in blue
48  } else if ( rank == 2 ) {
49  s = S_COLOR_RED "2nd" S_COLOR_WHITE; // draw in red
50  } else if ( rank == 3 ) {
51  s = S_COLOR_YELLOW "3rd" S_COLOR_WHITE; // draw in yellow
52  } else if ( rank == 11 ) {
53  s = "11th";
54  } else if ( rank == 12 ) {
55  s = "12th";
56  } else if ( rank == 13 ) {
57  s = "13th";
58  } else if ( rank % 10 == 1 ) {
59  s = va("%ist", rank);
60  } else if ( rank % 10 == 2 ) {
61  s = va("%ind", rank);
62  } else if ( rank % 10 == 3 ) {
63  s = va("%ird", rank);
64  } else {
65  s = va("%ith", rank);
66  }
67 
68  Com_sprintf( str, sizeof( str ), "%s%s", t, s );
69  return str;
70 }

73 =============
74 CG_Obituary
75 =============
76 */
77 static void CG_Obituary( entityState_t *ent ) {
78  int mod;
79  int target, attacker;
80  char *message;
81  char *message2;
82  const char *targetInfo;
83  const char *attackerInfo;
84  char targetName[32];
85  char attackerName[32];
86  clientInfo_t *ci;
87 
88  if ( !cg_displayObituary.integer ) {
89  return;
90  }
91 
92  target = ent->otherEntityNum;
93  attacker = ent->otherEntityNum2;
94  mod = ent->eventParm;
95 
96  if ( target < 0 || target >= MAX_CLIENTS ) {
97  CG_Error( "CG_Obituary: target out of range" );
98  }
99  ci = &cgs.clientinfo[target];
100 
101  if ( attacker < 0 || attacker >= MAX_CLIENTS ) {
102  attacker = ENTITYNUM_WORLD;
103  attackerInfo = NULL;
104  } else {
105  attackerInfo = CG_ConfigString( CS_PLAYERS + attacker );
106  }
107 
108  targetInfo = CG_ConfigString( CS_PLAYERS + target );
109  if ( !targetInfo ) {
110  return;
111  }
112  Q_strncpyz( targetName, Info_ValueForKey( targetInfo, "n" ), sizeof(targetName) - 2);
113  strcat( targetName, S_COLOR_WHITE );
114 
115  message2 = "";
116 
117  // check for single client messages
118 
119  message = NULL;
120 
121  if (message) {
122  CG_Printf( "%s %s.\n", targetName, message);
123  return;
124  }
125  // check for double client messages
126  if ( !attackerInfo ) {
127  attacker = ENTITYNUM_WORLD;
128  strcpy( attackerName, "noname" );
129  } else {
130  Q_strncpyz( attackerName, Info_ValueForKey( attackerInfo, "n" ), sizeof(attackerName) - 2);
131  strcat( attackerName, S_COLOR_WHITE );
132  // check for kill messages about the current clientNum
133  if ( target == cg.snap->ps.clientNum ) {
134  Q_strncpyz( cg.killerName, attackerName, sizeof( cg.killerName ) );
135  }
136  }
137 
138  // we don't know what it was
139  CG_Printf( "%s died.\n", targetName );
140 }

621 void CG_CheckEvents( centity_t *cent ) {
622  // check for event-only entities
623  if ( cent->currentState.eType > ET_EVENTS ) {
624  if ( cent->previousEvent ) {
625  return; // already fired
626  }
627  // if this is a player event set the entity number of the client entity number
628  if ( cent->currentState.eFlags & EF_PLAYER_EVENT ) {
629  cent->currentState.number = cent->currentState.otherEntityNum;
630  }
631 
632  cent->previousEvent = 1;
633 
634  cent->currentState.event = cent->currentState.eType - ET_EVENTS;
635  } else {
636  // check for events riding with another entity
637  if ( cent->currentState.event == cent->previousEvent ) {
638  return;
639  }
640  cent->previousEvent = cent->currentState.event;
641  if ( ( cent->currentState.event & ~EV_EVENT_BITS ) == 0 ) {
642  return;
643  }
644  }
645 
646  // calculate the position at exactly the frame time
647  BG_EvaluateTrajectory( &cent->currentState, &cent->currentState.pos, cg.snap->serverTime, cent->lerpOrigin );
648  CG_SetEntitySoundPosition( cent );
649 
650  CG_EntityEvent( cent, cent->lerpOrigin );
651 }
