252 static unsigned hashkey(const char *string, unsigned len) {
253  unsigned register hash, i;
254 
255  hash = 0;
256  for (i = 0; i < len && string[i] != '\0'; ++i) {
257  hash += string[i] * (119 + i);
258  }
259  hash = (hash ^ (hash >> 10) ^ (hash >> 20));
260  return hash;
261 }

397 static void compute_rexmodrmsib(u8* rex_r, u8* modrm_r, u8* sib_r, arg_t* arg1, arg_t* arg2)
398 {
399  u8 rex = 0;
400  u8 modrm = 0;
401  u8 sib = 0;
402 
403  if((arg1->type == T_REGISTER && arg2->type == T_REGISTER)
404  && ((arg1->v.reg & R_MSZ) != (arg2->v.reg & R_MSZ))
405  && !((arg1->v.reg & R_XMM) || (arg2->v.reg & R_XMM)))
406  crap("both registers must be of same width");
407 
408  if((arg1->type == T_REGISTER && arg1->v.reg & R_64)
409  || (arg2->type == T_REGISTER && arg2->v.reg & R_64))
410  {
411  rex |= REX_W;
412  }
413 
414  if(arg1->type == T_REGISTER)
415  {
416  if((arg1->v.reg & R_MGP) > 0x07)
417  rex |= REX_R;
418 
419  modrm |= (arg1->v.reg & 0x07) << 3;
420  }
421 
422  if(arg2->type == T_REGISTER)
423  {
424  if((arg2->v.reg & R_MGP) > 0x07)
425  rex |= REX_B;
426 
427  modrm |= (arg2->v.reg & 0x07);
428  }
429 
430  if(arg2->type == T_MEMORY)
431  {
432  if((arg2->v.mem.basetype == T_REGISTER && !(arg2->v.mem.base.reg & R_64))
433  || (arg2->v.mem.indextype == T_REGISTER && !(arg2->v.mem.index.reg & R_64)))
434  {
435  crap("only 64bit base/index registers are %x %x", arg2->v.mem.base.reg, arg2->v.mem.index.reg);
436  }
437 
438  if(arg2->v.mem.indextype == T_REGISTER)
439  {
440  modrm |= MODRM_RM_SIB;
441  if(!arg2->v.mem.disp)
442  {
443  modrm |= MODRM_MOD_00;
444  }
445  else if(iss8(arg2->v.mem.disp))
446  {
447  modrm |= MODRM_MOD_01;
448  }
449  else if(isu32(arg2->v.mem.disp))
450  {
451  modrm |= MODRM_MOD_10;
452  }
453  else
454  {
455  crap("invalid displacement");
456  }
457 
458  if((arg2->v.mem.index.reg & R_MGP) > 0x07)
459  rex |= REX_X;
460 
461  if((arg2->v.mem.base.reg & R_MGP) > 0x07)
462  rex |= REX_B;
463 
464  if(arg2->v.mem.basetype != T_REGISTER)
465  crap("base must be register");
466  switch(arg2->v.mem.scale)
467  {
468  case 1: break;
469  case 2: sib |= 1 << 6; break;
470  case 4: sib |= 2 << 6; break;
471  case 8: sib |= 3 << 6; break;
472  }
473  sib |= (arg2->v.mem.index.reg & 0x07) << 3;
474  sib |= (arg2->v.mem.base.reg & 0x07);
475  }
476  else if(arg2->v.mem.indextype == T_NONE)
477  {
478  if(!arg2->v.mem.disp)
479  {
480  modrm |= MODRM_MOD_00;
481  }
482  else if(iss8(arg2->v.mem.disp))
483  {
484  modrm |= MODRM_MOD_01;
485  }
486  else if(isu32(arg2->v.mem.disp))
487  {
488  modrm |= MODRM_MOD_10;
489  }
490  else
491  {
492  crap("invalid displacement");
493  }
494 
495  if(arg2->v.mem.basetype != T_REGISTER)
496  crap("todo: base != register");
497 
498  if((arg2->v.mem.base.reg & R_MGP) > 0x07)
499  rex |= REX_B;
500 
501  modrm |= arg2->v.mem.base.reg & 0x07;
502  }
503  else
504  {
505  crap("invalid indextype");
506  }
507  }
508  else
509  {
510  modrm |= MODRM_MOD_11;
511  }
512 
513  if(rex)
514  rex |= 0x40; // XXX
515 
516  *rex_r = rex;
517  *modrm_r = modrm;
518  *sib_r = sib;
519 }

620 static void emit_mov(const char* mnemonic, arg_t arg1, arg_t arg2, void* data)
621 {
622  u8 rex = 0;
623  u8 modrm = 0;
624  u8 sib = 0;
625 
626  if(arg1.type == T_IMMEDIATE && arg2.type == T_REGISTER)
627  {
628  u8 op = 0xb8;
629 
630  if(arg2.v.reg & R_8)
631  {
632  if(!iss8(arg1.v.imm))
633  crap("value too large for 8bit register");
634 
635  op = 0xb0;
636  }
637  else if(arg2.v.reg & R_16)
638  {
639  if(!isu16(arg1.v.imm))
640  crap("value too large for 16bit register");
641  emit1(0x66);
642  }
643  else if(!(arg2.v.reg & R_64))
644  {
645  if(!isu32(arg1.v.imm))
646  crap("value too large for 32bit register");
647  }
648 
649  compute_rexmodrmsib(&rex, &modrm, &sib, &arg1, &arg2);
650 
651  if(rex) emit1(rex);
652 
653  op |= (arg2.v.reg & 0x07);
654 
655  emit1(op);
656 
657  if(arg2.v.reg & R_8) emit1(arg1.v.imm);
658  else if(arg2.v.reg & R_16) emit2(arg1.v.imm);
659  else if(arg2.v.reg & R_64) emit8(arg1.v.imm);
660  else emit4(arg1.v.imm);
661  }
662  else if(arg1.type == T_IMMEDIATE && arg2.type == T_MEMORY)
663  {
664  if(!iss32(arg1.v.imm))
665  {
666  crap("only 32bit immediates supported");
667  }
668 
669  compute_rexmodrmsib(&rex, &modrm, &sib, &arg1, &arg2);
670  if(rex) emit1(rex);
671  emit1(0xc7); // mov reg/mem, imm
672  emit1(modrm);
673  if((modrm & 0x07) == MODRM_RM_SIB)
674  emit1(sib);
675 
676  emit4(arg1.v.imm);
677  }

1196  char label[20];

1200

1207  {

1209  ttype = nexttok(&s, NULL, &val);
1210  if(ttype == '-')
1211  {
1212  negative = -1;
1213  ttype = nexttok(&s, NULL, &val);
1214  }
1215  if(ttype != TOK_INT)
1216  crap("expected integer");
1217  arg.type = T_IMMEDIATE;
1218  arg.v.imm = negative * val;
1219  break;
1220  case '*' :
1221  if((ttype = nexttok(&s, NULL, NULL)) != '%')
1222  {

1224  goto tok_memory;

1238  break;
1239  case '-':
1240  negative = -1;
1241  if(nexttok(&s, NULL, &val) != TOK_INT)
1242  crap("expected integer");
1243  /* fall through */
1244  case TOK_INT:
