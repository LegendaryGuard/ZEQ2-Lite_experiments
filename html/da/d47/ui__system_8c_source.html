186 static int SystemSettings_FindBuiltinResolution( int mode )
187 {
188  int i;
189 
190  if( !resolutionsDetected )
191  return mode;
192 
193  if( mode < 0 )

195

200  }

229 /*
230 =================
231 SystemSettings_GetAspectRatios
232 =================
233 */
234 static void SystemSettings_GetAspectRatios( void )
235 {
236  int i, r;
237 
238  // build ratio list from resolutions
239  for( r = 0; resolutions[r]; r++ )
240  {
241  int w, h;
242  char *x;
243  char str[ sizeof(ratioBuf[0]) ];
244 
245  // calculate resolution's aspect ratio
246  x = strchr( resolutions[r], 'x' ) + 1;
247  Q_strncpyz( str, resolutions[r], x-resolutions[r] );
248  w = atoi( str );
249  h = atoi( x );
250  Com_sprintf( str, sizeof(str), "%.2f:1", (float)w / (float)h );
251 
252  // add ratio to list if it is new
253  // establish res/ratio relationship
254  for( i = 0; ratioBuf[i][0]; i++ )
255  {
256  if( !Q_stricmp( str, ratioBuf[i] ) )
257  break;
258  }
259  if( !ratioBuf[i][0] )
260  {
261  Q_strncpyz( ratioBuf[i], str, sizeof(ratioBuf[i]) );
262  ratioToRes[i] = r;
263  }
264  resToRatio[r] = i;
265  }
266 
267  // prepare itemlist pointer array
268  // rename common ratios ("1.33:1" -> "4:3")
269  for( r = 0; ratioBuf[r][0]; r++ )
270  {
271  for( i = 0; knownRatios[i][0]; i++ )
272  {
273  if( !Q_stricmp( ratioBuf[r], knownRatios[i][0] ) )
274  {
275  Q_strncpyz( ratioBuf[r], knownRatios[i][1], sizeof(ratioBuf[r]) );
276  break;
277  }
278  }
279  ratios[r] = ratioBuf[r];
280  }
281  ratios[r] = NULL;
282 }

606 static void SystemSettings_Event( void* ptr, int event ) {
607  InitialVideoOptions_s *ivo;
608 
609  if( event != QM_ACTIVATED ) {
610  return;
611  }
612 
613  switch( ((menucommon_s*)ptr)->id ) {
614  case ID_RATIO:
615  s_systemsettings.mode.curvalue =
616  ratioToRes[ s_systemsettings.ratio.curvalue ];
617  // fall through to back mode constraints
618 
619  case ID_MODE:
620  // clamp 3dfx video modes
621  if ( s_systemsettings.driver.curvalue == 1 )
622  {
623  if ( s_systemsettings.mode.curvalue < 2 )
624  s_systemsettings.mode.curvalue = 2;
625  else if ( s_systemsettings.mode.curvalue > 6 )
626  s_systemsettings.mode.curvalue = 6;
627  }
628  s_systemsettings.ratio.curvalue =
629  resToRatio[ s_systemsettings.mode.curvalue ];
630  break;
631 
632  case ID_LIST:
633  ivo = &s_ivo_templates[s_systemsettings.list.curvalue];
634 
635  s_systemsettings.mode.curvalue = SystemSettings_FindDetectedResolution(ivo->mode);
636  s_systemsettings.ratio.curvalue = resToRatio[ s_systemsettings.mode.curvalue ];
637  s_systemsettings.tq.curvalue = ivo->tq;
638  s_systemsettings.lighting.curvalue = ivo->lighting;
639  s_systemsettings.texturebits.curvalue = ivo->texturebits;
640  s_systemsettings.geometry.curvalue = ivo->geometry;
641  s_systemsettings.filter.curvalue = ivo->filter;
642  s_systemsettings.multisample.curvalue = ivo->multisample;
643  s_systemsettings.anisotropy.curvalue = ivo->anisotropy;
644  s_systemsettings.fs.curvalue = ivo->fullscreen;
645  s_systemsettings.vsync.curvalue = ivo->vsync;
646  Com_sprintf(s_systemsettings.framerate.field.buffer,sizeof(s_systemsettings.framerate.field.buffer),"%d",ivo->vsync);
647  break;
648 
649  case ID_MASTERVOLUME:
650  trap_Cvar_SetValue( "s_volume", s_systemsettings.mastervolume.curvalue / 10 );
651  break;
652 
653  case ID_MUSICVOLUME:
654  trap_Cvar_SetValue( "cg_music", s_systemsettings.musicvolume.curvalue / 10 );
655  trap_Cvar_SetValue( "s_musicvolume", s_systemsettings.musicvolume.curvalue / 10 );
656  break;
657 
658  case ID_QUALITY:
659  if( s_systemsettings.quality.curvalue ) {
660  trap_Cvar_SetValue( "s_khz", 22 );
661  trap_Cvar_SetValue( "s_compression", 0 );
662  }
663  else {
664  trap_Cvar_SetValue( "s_khz", 11 );
