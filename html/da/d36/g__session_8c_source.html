95 /*
96 ================
97 G_InitSessionData
98 
99 Called on a first-time connect
100 ================
101 */
102 void G_InitSessionData( gclient_t *client, char *userinfo ) {
103  clientSession_t *sess;
104  const char *value;
105 
106  sess = &client->sess;
107  // initial team determination
108  if ( g_gametype.integer >= GT_TEAM ) {
109  if ( g_teamAutoJoin.integer ) {
110  sess->sessionTeam = PickTeam( -1 );
111  BroadcastTeamChange( client, -1 );
112  } else {
113  // always spawn as spectator in team games
114  sess->sessionTeam = TEAM_SPECTATOR;
115  }
116  } else {
117  value = Info_ValueForKey( userinfo, "team" );
118  if ( value[0] == 's' ) {
119  // a willing spectator, not a waiting-in-line
120  sess->sessionTeam = TEAM_SPECTATOR;
121  } else {
122  switch ( g_gametype.integer ) {
123  default:
124  case GT_FFA:
125  case GT_STRUGGLE:
126  case GT_SINGLE_PLAYER:
127  if ( g_maxGameClients.integer > 0 &&
128  level.numNonSpectatorClients >= g_maxGameClients.integer ) {
129  sess->sessionTeam = TEAM_SPECTATOR;
130  } else {
131  sess->sessionTeam = TEAM_FREE;
132  }
133  break;
134  case GT_TOURNAMENT:
135  // if the game is full, go into a waiting mode
136  if ( level.numNonSpectatorClients >= 2 ) {
137  sess->sessionTeam = TEAM_SPECTATOR;
138  } else {
139  sess->sessionTeam = TEAM_FREE;
140  }
141  break;
142  }
143  }
144  }
145 
146  sess->spectatorState = SPECTATOR_FREE;
147  AddTournamentQueue(client);
148 
149  G_WriteClientSessionData( client );
150 }
