95 SelectNearestDeathmatchSpawnPoint
96 
97 Find the spot that we DON'T want to use

100 #define MAX_SPAWN_POINTS 128
101 gentity_t *SelectNearestDeathmatchSpawnPoint( vec3_t from ) {
102  gentity_t *spot;
103  vec3_t delta;
104  float dist, nearestDist;
105  gentity_t *nearestSpot;
106 
107  nearestDist = 999999;
108  nearestSpot = NULL;
109  spot = NULL;
110 
111  while ((spot = G_Find (spot, FOFS(classname), "info_player_deathmatch")) != NULL) {
112 
113  VectorSubtract( spot->s.origin, from, delta );

119  }

126 ================
127 SelectRandomDeathmatchSpawnPoint
128 
129 go to a random point that doesn't telefrag
130 ================
131 */
132 #define MAX_SPAWN_POINTS 128
133 gentity_t *SelectRandomDeathmatchSpawnPoint (void) {
134  gentity_t *spot;
135  int count;
136  int selection;
137  gentity_t *spots[MAX_SPAWN_POINTS];
138 
139  count = 0;
140  spot = NULL;
141 
142  while((spot = G_Find (spot, FOFS(classname), "info_player_deathmatch")) != NULL && count < MAX_SPAWN_POINTS)
143  {
144  if(SpotWouldTelefrag(spot))
145  continue;
146 
147  spots[count] = spot;
148  count++;
149  }
150

153  }
154 
155  selection = rand() % count;
156  return spots[ selection ];

159 /*
160 ===========
161 SelectRandomFurthestSpawnPoint
162 
163 Chooses a player start, deathmatch start, etc
164 ============
165 */
166 gentity_t *SelectRandomFurthestSpawnPoint ( vec3_t avoidPoint, vec3_t origin, vec3_t angles ) {
167  gentity_t *spot;
168  vec3_t delta;
169  float dist;
170  float list_dist[MAX_SPAWN_POINTS];
171  gentity_t *list_spot[MAX_SPAWN_POINTS];
172  int numSpots, rnd, i, j;
173 
174  numSpots = 0;
175  spot = NULL;
176 
177  while((spot = G_Find (spot, FOFS(classname), "info_player_deathmatch")) != NULL)
178  {
179  if(SpotWouldTelefrag(spot))
180  continue;
181 
182  VectorSubtract( spot->s.origin, avoidPoint, delta );
183  dist = VectorLength( delta );
184 
185  for (i = 0; i < numSpots; i++)
186  {
187  if(dist > list_dist[i])
188  {
189  if (numSpots >= MAX_SPAWN_POINTS)
190  numSpots = MAX_SPAWN_POINTS - 1;
191 
192  for(j = numSpots; j > i; j--)
193  {
194  list_dist[j] = list_dist[j-1];
195  list_spot[j] = list_spot[j-1];
196  }
197 
198  list_dist[i] = dist;
199  list_spot[i] = spot;
200 
201  numSpots++;
202  break;
203  }
204  }
205 
206  if(i >= numSpots && numSpots < MAX_SPAWN_POINTS)
207  {
208  list_dist[numSpots] = dist;
209  list_spot[numSpots] = spot;
210  numSpots++;
211  }
212  }
213 
214  if(!numSpots)
215  {
216  spot = G_Find(NULL, FOFS(classname), "info_player_deathmatch");
217 
218  if (!spot)
219  G_Error( "Couldn't find a spawn point" );
220 
221  VectorCopy (spot->s.origin, origin);
222  origin[2] += 9;
223  VectorCopy (spot->s.angles, angles);
224  return spot;
225  }
226 
227  // select a random spot from the spawn points furthest away
228  rnd = random() * (numSpots / 2);
229 
230  VectorCopy (list_spot[rnd]->s.origin, origin);
231  origin[2] += 9;
232  VectorCopy (list_spot[rnd]->s.angles, angles);
233 
234  return list_spot[rnd];
235 }

514 /*
515 ===========
516 ClientCheckName
517 ============
518 */
519 static void ClientCleanName(const char *in, char *out, int outSize)
520 {
521  int outpos = 0, colorlessLen = 0, spaces = 0;
522 
523  // discard leading spaces
524  for(; *in == ' '; in++);
525 
526  for(; *in && outpos < outSize - 1; in++)
527  {
528  out[outpos] = *in;
529 
530  if(*in == ' ')
531  {
532  // don't allow too many consecutive spaces
533  if(spaces > 2)
534  continue;
535 
536  spaces++;
537  }
538  else if(outpos > 0 && out[outpos - 1] == Q_COLOR_ESCAPE)
539  {
540  if(Q_IsColorString(&out[outpos - 1]))
541  {
542  colorlessLen--;
543 
544  if(ColorIndex(*in) == 0)
545  {
546  // Disallow color black in names to prevent players
547  // from getting advantage playing in front of black backgrounds
548  outpos--;
549  continue;
550  }
551  }
552  else
553  {
554  spaces = 0;
555  colorlessLen++;
556  }
557  }
558  else
559  {
560  spaces = 0;
561  colorlessLen++;
562  }
563 
564  outpos++;
565  }
566 
567  out[outpos] = '\0';
568 
569  // don't allow empty names
570  if( *out == '\0' || colorlessLen == 0)
571  Q_strncpyz(out, "UnnamedPlayer", outSize );
572 }

719 /*
720 ===========
721 ClientConnect
722 
723 Called when a player begins connecting to the server.
724 Called again for every map change or tournement restart.
725 
726 The session information will be valid after exit.
727 
728 Return NULL if the client should be allowed, otherwise return
729 a string with the reason for denial.
730 
731 Otherwise, the client will be sent the current gamestate
732 and will eventually get to ClientBegin.
733 
734 firstTime will be qtrue the very first time a client connects
735 to the server machine, but qfalse on map changes and tournement
736 restarts.
737 ============
738 */
739 char *ClientConnect( int clientNum, qboolean firstTime ) {
740  char *value;
741 // char *areabits;
742  gclient_t *client;
743  char userinfo[MAX_INFO_STRING];
744  gentity_t *ent;
745 
746  ent = &g_entities[ clientNum ];
747 
748  trap_GetUserinfo( clientNum, userinfo, sizeof( userinfo ) );
749 
750  // IP filtering
751  // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=500
752  // recommanding PB based IP / GUID banning, the builtin system is pretty limited
753  // check to see if they are on the banned IP list
754  value = Info_ValueForKey (userinfo, "ip");
755  if ( G_FilterPacket( value ) ) {
756  return "You are banned from this server.";
757  }
758 
759  // we don't check password for local client
