34 /*
35 ===================
36 CG_InitLocalEntities
37 
38 This is called at startup and for tournement restarts
39 ===================
40 */
41 void CG_InitLocalEntities( void ) {
42  int i;
43 
44  memset( cg_localEntities, 0, sizeof( cg_localEntities ) );
45  cg_activeLocalEntities.next = &cg_activeLocalEntities;
46  cg_activeLocalEntities.prev = &cg_activeLocalEntities;
47  cg_freeLocalEntities = cg_localEntities;
48  for ( i = 0 ; i < MAX_LOCAL_ENTITIES - 1 ; i++ ) {
49  cg_localEntities[i].next = &cg_localEntities[i+1];
50  }
51 }

54 /*
55 ==================
56 CG_FreeLocalEntity
57 ==================
58 */
59 void CG_FreeLocalEntity( localEntity_t *le ) {
60  if ( !le->prev ) {
61  CG_Error( "CG_FreeLocalEntity: not active" );
62  }
63 
64  // remove from the doubly linked active list
65  le->prev->next = le->next;
66  le->next->prev = le->prev;
67 
68  // the free list is only singly linked
69  le->next = cg_freeLocalEntities;
70  cg_freeLocalEntities = le;
71 }
72 
73 /*
74 ===================
75 CG_AllocLocalEntity
76 
77 Will allways succeed, even if it requires freeing an old active entity
78 ===================
79 */
80 localEntity_t *CG_AllocLocalEntity( void ) {
81  localEntity_t *le;
82 
83  if ( !cg_freeLocalEntities ) {
84  // no free entities, so free the one at the end of the chain
85  // remove the oldest active entity
86  CG_FreeLocalEntity( cg_activeLocalEntities.prev );
87  }
88 
89  le = cg_freeLocalEntities;
90  cg_freeLocalEntities = cg_freeLocalEntities->next;
91 
92  memset( le, 0, sizeof( *le ) );
93 
94  // link into the active list
95  le->next = cg_activeLocalEntities.next;
96  le->prev = &cg_activeLocalEntities;
97  cg_activeLocalEntities.next->prev = le;
98  cg_activeLocalEntities.next = le;
99  return le;
100 }

139 /*
140 ================
141 CG_ReflectVelocity
142 ================
143 */
144 void CG_ReflectVelocity( localEntity_t *le, trace_t *trace ) {
145  vec3_t velocity;
146  float dot;
147  int hitTime;
148 
149  // reflect the velocity on the trace plane
150  hitTime = cg.time - cg.frametime + cg.frametime * trace->fraction;
151  BG_EvaluateTrajectoryDelta( NULL, &le->pos, hitTime, velocity );
152  dot = DotProduct( velocity, trace->plane.normal );
153  VectorMA( velocity, -2*dot, trace->plane.normal, le->pos.trDelta );
154 
155  VectorScale( le->pos.trDelta, le->bounceFactor, le->pos.trDelta );
156 
157  VectorCopy( trace->endpos, le->pos.trBase );
158  le->pos.trTime = cg.time;
159 
160 
161  // check for stop, making sure that even on low FPS systems it doesn't bobble
162  if ( trace->allsolid ||
163  ( trace->plane.normal[2] > 0 &&
164  ( le->pos.trDelta[2] < 40 || le->pos.trDelta[2] < -cg.frametime * le->pos.trDelta[2] ) ) ) {
165  le->pos.trType = TR_STATIONARY;
166  } else {
167 
168  }
169 }

171 /*
172 ================
173 CG_AddFragment
174 ================
175 */
176 void CG_AddFragment( localEntity_t *le ) {
177  vec3_t newOrigin;
178  trace_t trace;
179 
180  if ( le->pos.trType == TR_STATIONARY ) {
181  // sink into the ground if near the removal time
182  int t;
183  float oldZ;
184 
185  t = le->endTime - cg.time;
186  if ( t < SINK_TIME ) {
187  // we must use an explicit lighting origin, otherwise the
188  // lighting would be lost as soon as the origin went
189  // into the ground
190  VectorCopy( le->refEntity.origin, le->refEntity.lightingOrigin );
191  le->refEntity.renderfx |= RF_LIGHTING_ORIGIN;
192  oldZ = le->refEntity.origin[2];
193  le->refEntity.origin[2] -= 16 * ( 1.0 - (float)t / SINK_TIME );
194  trap_R_AddRefEntityToScene( &le->refEntity );
195  le->refEntity.origin[2] = oldZ;
196  } else {
197  trap_R_AddRefEntityToScene( &le->refEntity );
198  }
199 
200  return;
201  }
202 
203  // calculate new position
204  BG_EvaluateTrajectory( NULL, &le->pos, cg.time, newOrigin );
205 
206  // trace a line from previous position to new position
207  CG_Trace( &trace, le->refEntity.origin, NULL, NULL, newOrigin, -1, CONTENTS_SOLID );
208  if ( trace.fraction == 1.0 ) {
209  // still in free fall
210  VectorCopy( newOrigin, le->refEntity.origin );
211 
212  if ( le->leFlags & LEF_TUMBLE ) {
213  vec3_t angles;
214 
215  BG_EvaluateTrajectory( NULL, &le->angles, cg.time, angles );
216  AnglesToAxis( angles, le->refEntity.axis );
217  }
218 
219  trap_R_AddRefEntityToScene( &le->refEntity );
220 
221  // add a blood trail
222  if ( le->leBounceSoundType == LEBS_BLOOD ) {
223  CG_BloodTrail( le );
224  }
225 
226  return;
227  }
228 
229  // if it is in a nodrop zone, remove it
230  // this keeps gibs from waiting at the bottom of pits of death
231  // and floating levels
232  if ( CG_PointContents( trace.endpos, 0 ) & CONTENTS_NODROP ) {
233  CG_FreeLocalEntity( le );
234  return;
235  }
236 
237  // leave a mark
238  CG_FragmentBounceMark( le, &trace );
239 
240  // do a bouncy sound
241  CG_FragmentBounceSound( le, &trace );
242 
243  // reflect the velocity on the trace plane
244  CG_ReflectVelocity( le, &trace );
245 
246  trap_R_AddRefEntityToScene( &le->refEntity );
247 }

479 /*
480 ========================
481 CG_AddStraightBeamFade
482 ========================
483 */
484 static void CG_AddStraightBeamFade( localEntity_t *le ) {
485  refEntity_t *ent; // reference entity stored in the local entity
486  float scale_l; // stores the scale factor for the beam's length
487  float scale_w; // stores the scale factor for the beam's width
488  vec3_t start; // temporary storage for the beam's start point
489  vec3_t direction; // vector used in calculating the shortening of the beam
490  vec4_t color;
491  // set up for quick reference
492  ent = &le->refEntity;
493 
494  // Save the start vector so it can be recovered after having been rendered.
495  VectorCopy( ent->origin, start );
496 
497  scale_l = 1 - (float)( cg.time - le->startTime ) / ( le->endTime - le->startTime );
498  scale_w = 1 - (float)( cg.time - le->startTime ) / ( le->endTime - le->startTime );
499  if (scale_w < 0) {
500  scale_w = 0;
501  }
502  VectorSet(color,scale_l,scale_l,scale_l);
503  VectorScale(color,255,color);
504  color[3] = scale_l * 255;
505  // Set the scaled beam
506  VectorSubtract( ent->origin, ent->oldorigin, direction);
507  VectorScale(direction, scale_l, direction);
508  VectorAdd(ent->oldorigin, direction, ent->origin);
509 
510  CG_DrawLineRGBA(ent->origin,ent->oldorigin,le->radius * scale_w,ent->customShader,color);
511 
512  // Restore the start vector
513  VectorCopy(start, ent->origin);
514 }

517 /*
518 ====================
519 CG_AddZEQExplosion
520 ====================
521 */
522 static void CG_AddZEQExplosion( localEntity_t *le ) {
523  refEntity_t *ent;
524  float c;
525  float RGBfade;
526  float phase;
527  vec3_t tmpAxes[3];
528 
529  ent = &le->refEntity;
530 
531  RGBfade = (float)( cg.time - le->startTime ) / ( le->endTime - le->startTime );
532  if ( RGBfade < 0.5 ) {
533  RGBfade = 1.0;
534  } else {
535  RGBfade = 1.0 - ( RGBfade - 0.5 ) * 2;
536  }
537 
538  ent->shaderRGBA[0] = 0xff * RGBfade;
539  ent->shaderRGBA[1] = 0xff * RGBfade;
540  ent->shaderRGBA[2] = 0xff * RGBfade;
541  ent->shaderRGBA[3] = 0xff * RGBfade;
542 
543  // grow time
544  phase = (float)( cg.time - le->startTime ) / ( le->endTime - le->startTime ) * M_PI / 2.0f;
545  c = 1.0f + (sin(phase));
546 
547  // preserve the full scale
548  VectorCopy(ent->axis[0], tmpAxes[0]);
549  VectorCopy(ent->axis[1], tmpAxes[1]);
550  VectorCopy(ent->axis[2], tmpAxes[2]);
551 
552  // set the current scale
553  VectorScale(ent->axis[0], 1 - c, ent->axis[0]);
554  VectorScale(ent->axis[1], 1 - c, ent->axis[1]);
555  VectorScale(ent->axis[2], 1 - c, ent->axis[2]);
556 
557  if(cg_drawBBox.value){
558  vec3_t mins,maxs;
559  trap_R_ModelBounds( ent->hModel, mins, maxs, ent->frame );
560  VectorScale(mins, ent->radius, mins);
561  VectorScale(maxs, ent->radius, maxs);
562  VectorScale(mins, 1 - c, mins);
563  VectorScale(maxs, 1 - c, maxs);
564  CG_DrawBoundingBox( ent->origin, mins, maxs );
565  }
566 
567  // add the entity
568  trap_R_AddRefEntityToScene( ent );
569 
570  // set the full scale again
571  VectorCopy(tmpAxes[0], ent->axis[0]);
572  VectorCopy(tmpAxes[1], ent->axis[1]);
573  VectorCopy(tmpAxes[2], ent->axis[2]);
574 
575  if(le->light){
576  float light = (float)(cg.time - le->startTime) / (le->endTime - le->startTime);
577  float lightRad;
578  vec3_t color;
579  light = light < 0.5f ? light * 2.0f : 1.0f - (light - 0.5f) * 2.0f;
580  lightRad = le->light * light;
581  VectorScale(le->lightColor,light,color);
582  trap_R_AddLightToScene(ent->origin,lightRad,color);
583  }
584 }
