107 void MSG_WriteBits( msg_t *msg, int value, int bits ) {
108  int i;
109 // FILE* fp;
110 
111  oldsize += bits;
112 
113  // this isn't an exact overflow check, but close enough
114  if ( msg->maxsize - msg->cursize < 4 ) {
115  msg->overflowed = qtrue;
116  return;
117  }
118 
119  if ( bits == 0 || bits < -31 || bits > 32 ) {
120  Com_Error( ERR_DROP, "MSG_WriteBits: bad bits %i", bits );
121  }
122 
123  // check for overflows
124  if ( bits != 32 ) {
125  if ( bits > 0 ) {
126  if ( value > ( ( 1 << bits ) - 1 ) || value < 0 ) {
127  overflows++;
128  }
129  } else {
130  int r;
131 
132  r = 1 << (bits-1);
133 
134  if ( value > r - 1 || value < -r ) {
135  overflows++;
136  }
137  }
138  }
139  if ( bits < 0 ) {
140  bits = -bits;
141  }
142  if (msg->oob) {
143  if(bits==8)
144  {
145  msg->data[msg->cursize] = value;
146  msg->cursize += 1;
147  msg->bit += 8;
148  }
149  else if(bits==16)
150  {
151  short temp = value;
152 
153  CopyLittleShort(&msg->data[msg->cursize], &temp);
154  msg->cursize += 2;
155  msg->bit += 16;
156  }
157  else if(bits==32)
158  {
159  CopyLittleLong(&msg->data[msg->cursize], &value);
160  msg->cursize += 4;
161  msg->bit += 32;
162  }
163  else
164  Com_Error(ERR_DROP, "can't write %d bits", bits);
165  } else {
166 // fp = fopen("c:\\netchan.bin", "a");
167  value &= (0xffffffff>>(32-bits));
168  if (bits&7) {
169  int nbits;
170  nbits = bits&7;
171  for(i=0;i<nbits;i++) {
172  Huff_putBit((value&1), msg->data, &msg->bit);
173  value = (value>>1);
174  }
175  bits = bits - nbits;
176  }
177  if (bits) {
178  for(i=0;i<bits;i+=8) {
179 // fwrite(bp, 1, 1, fp);
180  Huff_offsetTransmit (&msgHuff.compressor, (value&0xff), msg->data, &msg->bit);
181  value = (value>>8);
182  }
183  }
184  msg->cursize = (msg->bit>>3)+1;
185 // fclose(fp);
186  }
187 }

189 int MSG_ReadBits( msg_t *msg, int bits ) {
190  int value;
191  int get;
192  qboolean sgn;
193  int i, nbits;
194 // FILE* fp;
195 
196  value = 0;
197 
198  if ( bits < 0 ) {
199  bits = -bits;
200  sgn = qtrue;
201  } else {
202  sgn = qfalse;
203  }
204 
205  if (msg->oob) {
206  if(bits==8)
207  {
208  value = msg->data[msg->readcount];
209  msg->readcount += 1;
210  msg->bit += 8;
211  }
212  else if(bits==16)
213  {
214  short temp;
215 
216  CopyLittleShort(&temp, &msg->data[msg->readcount]);
217  value = temp;
218  msg->readcount += 2;
219  msg->bit += 16;
220  }
221  else if(bits==32)
222  {
223  CopyLittleLong(&value, &msg->data[msg->readcount]);
224  msg->readcount += 4;
225  msg->bit += 32;
226  }
227  else
228  Com_Error(ERR_DROP, "can't read %d bits", bits);
229  } else {
230  nbits = 0;
231  if (bits&7) {
232  nbits = bits&7;
233  for(i=0;i<nbits;i++) {
234  value |= (Huff_getBit(msg->data, &msg->bit)<<i);
235  }
236  bits = bits - nbits;
237  }
238  if (bits) {
239 // fp = fopen("c:\\netchan.bin", "a");
240  for(i=0;i<bits;i+=8) {
241  Huff_offsetReceive (msgHuff.decompressor.tree, &get, msg->data, &msg->bit);
242 // fwrite(&get, 1, 1, fp);
243  value |= (get<<(i+nbits));
244  }
245 // fclose(fp);
246  }
247  msg->readcount = (msg->bit>>3)+1;
248  }
249  if ( sgn ) {
250  if ( value & ( 1 << ( bits - 1 ) ) ) {
251  value |= -1 ^ ( ( 1 << bits ) - 1 );
252  }
253  }
254 
255  return value;
256 }

539 // a string hasher which gives the same hash value even if the
540 // string is later modified via the legacy MSG read/write code
541 int MSG_HashKey(const char *string, int maxlen) {
542  int hash, i;
543 
544  hash = 0;
545  for (i = 0; i < maxlen && string[i] != '\0'; i++) {
546  if (string[i] & 0x80 || string[i] == '%')
547  hash += '.' * (119 + i);
548  else
549  hash += string[i] * (119 + i);
550  }
551  hash = (hash ^ (hash >> 10) ^ (hash >> 20));
552  return hash;
553 }

930

932 Can delta from either a baseline or a previous packet_entity
933 If to is NULL, a remove entity update will be sent
934 If force is not set, then nothing at all will be generated if the entity is
935 identical, under the assumption that the in-order delta code will catch it.
936 ==================
937 */
938 void MSG_WriteDeltaEntity( msg_t *msg, struct entityState_s *from, struct entityState_s *to,
939  qboolean force ) {
940  int i, lc;

944  float fullFloat;
945  int *fromF, *toF;
946 
947  numFields = ARRAY_LEN( entityStateFields );
948 
949  // all fields should be 32 bits to avoid any compiler packing issues
950  // the "number" field is not part of the field list
951  // if this assert fails, someone added a field to the entityState_t
952  // struct without updating the message fields
953  assert( numFields + 1 == sizeof( *from )/4 );
954 
955  // a NULL to is a delta remove message
956  if ( to == NULL ) {
957  if ( from == NULL ) {
958  return;
959  }
960  MSG_WriteBits( msg, from->number, GENTITYNUM_BITS );
961  MSG_WriteBits( msg, 1, 1 );
962  return;
963  }
964 
965  if ( to->number < 0 || to->number >= MAX_GENTITIES ) {
966  Com_Error (ERR_FATAL, "MSG_WriteDeltaEntity: Bad entity number: %i", to->number );
967  }

969  lc = 0;
970  // build the change vector as bytes so it is endien independent
971  for ( i = 0, field = entityStateFields ; i < numFields ; i++, field++ ) {
972  fromF = (int *)( (byte *)from + field->offset );
973  toF = (int *)( (byte *)to + field->offset );
974  if ( *fromF != *toF ) {
975  lc = i+1;
976  }
977  }

980  // nothing at all changed
981  if ( !force ) {
982  return; // nothing at all
983  }
984  // write two bits for no change
985  MSG_WriteBits( msg, to->number, GENTITYNUM_BITS );
986  MSG_WriteBits( msg, 0, 1 ); // not removed
987  MSG_WriteBits( msg, 0, 1 ); // no delta
988  return;

993  MSG_WriteBits( msg, 1, 1 ); // we have a delta

996

998

1001  toF = (int *)( (byte *)to + field->offset );
1002

1014
