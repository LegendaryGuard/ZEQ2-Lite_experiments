89 /*
90 =================
91 InGame_Event
92 =================
93 */
94 void InGame_Event( void *ptr, int notification ) {
95  if( notification != QM_ACTIVATED ) {
96  return;
97  }
98 
99  switch( ((menucommon_s*)ptr)->id ) {
100  case ID_TEAM:
101  UI_TeamMainMenu();
102  break;
103  case ID_SETUP:
104  UI_SystemSettingsMenu();
105  break;
106 
107  case ID_PLAYER:
108  UI_PlayerModelMenu();
109  break;
110 
111  case ID_CAMERA:
112  UI_CameraMenu();
113  break;
114 
115  case ID_LEAVEARENA:
116  trap_Cmd_ExecuteText( EXEC_APPEND, "disconnect\n" );
117  break;
118 
119  case ID_RESTART:
120  UI_ConfirmMenu( "RESTART GAME?", 0, InGame_RestartAction );
121  break;
122 
123  case ID_QUIT:
124  UI_ConfirmMenu( "EXIT GAME?", 0, InGame_QuitAction );
125  break;
126 
127  }
128 }
