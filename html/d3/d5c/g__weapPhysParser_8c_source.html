14 =======================
15 G_weapPhys_StoreBuffer
16 =======================
17 Copies the contents in the buffer to a client's weapon
18 configuration, converting filestrings into qhandle_t in
19 the process.
20 */
21 void G_weapPhys_StoreBuffer(int clientNum, int weaponNum) {
22  g_userWeapon_t *dest;
23  g_userWeaponParseBuffer_t *src;
24 
25  src = &g_weapPhysBuffer;
26  dest = G_FindUserWeaponData( clientNum, weaponNum + 1);
27  memset( dest, 0, sizeof(g_userWeapon_t));
28 
29  // Size and form of buffer and storage is equal, so just
30  // copy the memory.
31  memcpy( dest, src, sizeof(g_userWeapon_t));
32 
33  // Indicate the weapon as charged if there is a chargeReadyPct
34  // or chargeTime set to something other than 0.
35  if ( src->costs_chargeReady || src->costs_chargeTime ) {
36  dest->general_bitflags |= WPF_NEEDSCHARGE;
37  }
38 
39  // If nrShots is set to < 1, internally it must be represented by 1 instead.
40  if ( dest->firing_nrShots < 1 ) dest->firing_nrShots = 1;
41 
42  // Handle the presence of alternate fire weapons through the
43  // piggybacked WPF_ALTWEAPONPRESENT flag.
44 
45  // NOTE: The WPF_ALTWEAPONPRESENT flag will only be enabled in src if src is the
46  // buffer for a secondary fire configuration. In this case it is implicitly
47  // safe to subtract ALTWEAPON_OFFSET to obtain the primary fire index
48  // of the weapon!
49  if ( src->general_bitflags & WPF_ALTWEAPONPRESENT ) {
50  src->general_bitflags &= ~WPF_ALTWEAPONPRESENT;
51  dest = G_FindUserWeaponData( clientNum, weaponNum - ALTWEAPON_OFFSET + 1);
52  dest->general_bitflags |= WPF_ALTWEAPONPRESENT;
53  }
54 }

57 ======================
58 G_weapPhys_ParseType
59 ======================
60 Parses 'type' field.
61 Syntax:
62 'type' '=' "enumstring"
63 */
64 qboolean G_weapPhys_ParseType( g_weapPhysParser_t *parser, g_weapPhysCategoryIndex_t category, int field ) {
65  g_weapPhysToken_t *token = &parser->token;
66  g_weapPhysScanner_t *scanner = &parser->scanner;
67 
68  // NOTE: The sequences of the strings in these stringtables are dependant on the
69  // sequences of enumeration types in g_userweapons.h !!!
70  char *Physics_strings[] = { "Missile", "Beam", "Hitscan", "Trigger", "None", "" };
71  char *Detonation_strings[] = { "Ki", "" };
72  char *Trajectory_strings[] = { "LineOfSight", "ProxBomb", "Guided", "Homing", "Arch", "Drunken", "Cylinder", "" };
73 
74  int i;
75 
76  if ( token->tokenSym != TOKEN_STRING ) {
77  G_weapPhys_ErrorHandle( ERROR_STRING_EXPECTED, scanner, token->stringval, NULL );
78  return qfalse;
79  }
80 
81  switch ( category ) {
82 
83  case CAT_PHYSICS:
84  for ( i = 0; Q_stricmp( Physics_strings[i], token->stringval ); i++ ) {
85  if ( !strcmp( Physics_strings[i], "" )) {
86  G_weapPhys_ErrorHandle( ERROR_UNKNOWN_ENUMTYPE, scanner, token->stringval, NULL );
87  return qfalse;
88  }
89  }
90  g_weapPhysBuffer.general_type = i;
91  break;
92 
93  case CAT_DETONATION:
94  for ( i = 0; Q_stricmp( Detonation_strings[i], token->stringval ); i++ ) {
95  if ( !strcmp( Detonation_strings[i], "" )) {
96  G_weapPhys_ErrorHandle( ERROR_UNKNOWN_ENUMTYPE, scanner, token->stringval, NULL );
97  return qfalse;
98  }
99  }
100  break;
101 
102  case CAT_TRAJECTORY:
103  for ( i = 0; Q_stricmp( Trajectory_strings[i], token->stringval ); i++ ) {
104  if ( !strcmp( Trajectory_strings[i], "" )) {
105  G_weapPhys_ErrorHandle( ERROR_UNKNOWN_ENUMTYPE, scanner, token->stringval, NULL );
106  return qfalse;
107  }
108  }
109  g_weapPhysBuffer.homing_type = i;
110  break;
111 
112  default:
113  G_weapPhys_ErrorHandle( ERROR_FIELD_NOT_IN_CATEGORY, scanner, g_weapPhysFields[field].fieldname, g_weapPhysCategories[category] );
114  return qfalse;
115  break;
116  }
117 
118  if ( !G_weapPhys_NextSym( scanner, token ) ) {
119  if ( token->tokenSym == TOKEN_EOF ) {
120  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
121  }
122  return qfalse;
123  }
124 
125  return qtrue;
126 }

223 qboolean G_weapPhys_ParseRadius( g_weapPhysParser_t *parser, g_weapPhysCategoryIndex_t category, int field ) {
224  g_weapPhysToken_t *token = &parser->token;
225  g_weapPhysScanner_t *scanner = &parser->scanner;
226 
227  if ( (category != CAT_PHYSICS) && (category != CAT_DETONATION) ) {

229  return qfalse;

231 
232  if ( token->tokenSym != TOKEN_INTEGER ) {
233  G_weapPhys_ErrorHandle( ERROR_INTEGER_EXPECTED, scanner, token->stringval, NULL );
234  return qfalse;
235  }
236 
237  switch ( category ) {
238  case CAT_PHYSICS:
239  g_weapPhysBuffer.physics_radius = token->intval;
240  break;
241  case CAT_DETONATION:
242  g_weapPhysBuffer.damage_radius = token->intval;
243  break;
244  default:
245  // shouldn't be able to happen
246  break;
247  }
248 
249  if ( !G_weapPhys_NextSym( scanner, token ) ) {
250  if ( token->tokenSym == TOKEN_EOF ) {
251  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
252  }
253  return qfalse;
254  }
255 
256  if ( token->tokenSym == TOKEN_PLUS ) {
257 
258  if ( !G_weapPhys_NextSym( scanner, token ) ) {
259  if ( token->tokenSym == TOKEN_EOF ) {
260  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
261  }
262  return qfalse;
263  }
264 
265  if ( token->tokenSym != TOKEN_INTEGER ) {
266  G_weapPhys_ErrorHandle( ERROR_INTEGER_EXPECTED, scanner, token->stringval, NULL );
267  return qfalse;
268  }
269 
270  switch ( category ) {
271  case CAT_PHYSICS:
272  g_weapPhysBuffer.physics_radiusMultiplier = token->intval;
273  break;
274  case CAT_DETONATION:
275  g_weapPhysBuffer.damage_radiusMultiplier = token->intval;

287  }
288  } else {
289  // No + multiplier means we reset any old multiplier, to keep it from remaining
290  // as a ghost.
291

310 =======================
311 G_weapPhys_ParseRange
312 =======================
313 Parses 'range' field.
314 Syntax (Physics):
315  'range' '=' ( <int> | ( '[' <int> <int> ']'))
316 Syntax (Trajectory):
317  'range' '=' <int>
318 */
319 qboolean G_weapPhys_ParseRange( g_weapPhysParser_t *parser, g_weapPhysCategoryIndex_t category, int field ) {
320  g_weapPhysToken_t *token = &parser->token;
321  g_weapPhysScanner_t *scanner = &parser->scanner;
322 
323  int rng_start, rng_end;
324 
325  if ( (category != CAT_PHYSICS) && (category != CAT_TRAJECTORY) ) {
326  G_weapPhys_ErrorHandle( ERROR_FIELD_NOT_IN_CATEGORY, scanner, g_weapPhysFields[field].fieldname, g_weapPhysCategories[category] );
327  return qfalse;
328  }
329 
330 
331 
332  switch ( category ) {
333  case CAT_PHYSICS:
334  switch ( token->tokenSym ) {
335  case TOKEN_INTEGER:
336  rng_start = rng_end = token->intval;
337  break;
338 
339  case TOKEN_OPENRANGE:
340 
341  if ( !G_weapPhys_NextSym( scanner, token ) ) {
342  if ( token->tokenSym == TOKEN_EOF ) {
343  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
344  }
345  return qfalse;
346  }
347 
348  if ( token->tokenSym != TOKEN_INTEGER ) {
349  G_weapPhys_ErrorHandle( ERROR_UNEXPECTED_SYMBOL, scanner, token->stringval, NULL );
350  return qfalse;
351  } else {
352  rng_start = token->intval;
353  }
354 
355  if ( !G_weapPhys_NextSym( scanner, token ) ) {
356  if ( token->tokenSym == TOKEN_EOF ) {
357  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
358  }
359  return qfalse;
360  }
361 
362  if ( token->tokenSym != TOKEN_INTEGER ) {
363  G_weapPhys_ErrorHandle( ERROR_UNEXPECTED_SYMBOL, scanner, token->stringval, NULL );
364  return qfalse;
365  } else {
366  rng_end = token->intval;
367  }
368 
369  if ( !G_weapPhys_NextSym( scanner, token ) ) {
370  if ( token->tokenSym == TOKEN_EOF ) {
371  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
372  }
373  return qfalse;
374  }
375 
376  if ( token->tokenSym != TOKEN_CLOSERANGE ) {
377  G_weapPhys_ErrorHandle( ERROR_UNEXPECTED_SYMBOL, scanner, token->stringval, "]" );
378  return qfalse;
379  }
380  break;
381 
382  default:
383  G_weapPhys_ErrorHandle( ERROR_UNEXPECTED_SYMBOL, scanner, token->stringval, NULL );
384  return qfalse;
385  break;
386  }
387 
388  g_weapPhysBuffer.physics_range_min = rng_start;
389  g_weapPhysBuffer.physics_range_max = rng_end;
390  break;
391 
392  case CAT_TRAJECTORY:
393  if ( token->tokenSym != TOKEN_INTEGER ) {
394  G_weapPhys_ErrorHandle( ERROR_INTEGER_EXPECTED, scanner, token->stringval, NULL );
395  return qfalse;
396  }
397 
398  g_weapPhysBuffer.homing_range = token->intval;
399  break;
400 
401  default:
402  // shouldn't be able to happen
403  break;
404  }
405 
406  if ( !G_weapPhys_NextSym( scanner, token ) ) {
407  if ( token->tokenSym == TOKEN_EOF ) {
408  G_weapPhys_ErrorHandle( ERROR_PREMATURE_EOF, scanner, NULL, NULL );
409  }
410  return qfalse;
411  }
412 
413  return qtrue;
414 }
