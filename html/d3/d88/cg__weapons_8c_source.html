49 static void CG_AddPlayerWeaponCharge(refEntity_t* parent,cg_userWeapon_t* weaponGraphics,refEntity_t* charge,float progress){
50  float scale = weaponGraphics->chargeStartsize;
51  float lightScale = weaponGraphics->chargeDlightStartRadius;
52  VectorCopy(parent->lightingOrigin,charge->lightingOrigin);
53  charge->shadowPlane = parent->shadowPlane;
54  charge->renderfx = parent->renderfx;
55  if(weaponGraphics->chargeGrowth){
56  if(weaponGraphics->chargeEndPct <= progress){
57  scale = weaponGraphics->chargeEndsize;
58  lightScale = weaponGraphics->chargeDlightEndRadius;
59  }
60  else{
61  float percentRange = weaponGraphics->chargeEndPct - weaponGraphics->chargeStartPct;
62  float sizeRange = weaponGraphics->chargeEndsize - weaponGraphics->chargeStartsize;
63  float percent = progress - weaponGraphics->chargeStartPct;
64  float size = percent / percentRange * sizeRange;
65  scale = size + weaponGraphics->chargeStartsize;
66  sizeRange = weaponGraphics->chargeDlightEndRadius - weaponGraphics->chargeDlightStartRadius;
67  size = percent / percentRange * sizeRange;
68  lightScale = size + weaponGraphics->chargeDlightStartRadius;
69  }
70  }
71  if(!weaponGraphics->chargeModel || !weaponGraphics->chargeSkin){
72  CG_SetWeaponSprite(charge,scale,weaponGraphics->chargeShader);
73  }
74  else{
75  CG_SetWeaponModel(charge,scale,weaponGraphics->chargeModel,weaponGraphics->chargeSkin);
76  }
77  trap_R_AddRefEntityToScene(charge);
78  if(lightScale){
79  trap_R_AddLightToScene(charge->origin,100 * lightScale,weaponGraphics->chargeDlightColor);
80  }
81 }

113 void CG_AddPlayerWeapon(refEntity_t* parent,centity_t* client){
114  entityState_t* ent = &client->currentState;
115  weaponstate_t weaponState = ent->weaponstate;
116  cg_userWeapon_t* weaponGraphics;
117  orientation_t orient;
118  refEntity_t refEnt;
119  qboolean wasPVS = CG_FrameHist_WasInPVS(ent->number);
120  qboolean newNR = CG_FrameHist_IsWeaponNr(ent->number) != CG_FrameHist_WasWeaponNr(ent->number);
121  qboolean newState = CG_FrameHist_IsWeaponState(ent->number) != CG_FrameHist_WasWeaponState(ent->number);
122  if(weaponState == WEAPON_CHARGING || weaponState == WEAPON_ALTCHARGING){
123  int newLerp;
124  float backLerp;
125  float lerp;
126  if(weaponState == WEAPON_CHARGING){
127  weaponGraphics = CG_FindUserWeaponGraphics(ent->clientNum,ent->weapon);
128  newLerp = ent->charge1.chBase;
129  backLerp = client->lerpPrim;
130  client->lerpPrim = newLerp < backLerp ? newLerp : (client->lerpPrim + newLerp) / 2.0f;
131  lerp = client->lerpPrim;
132  client->lerpSec = 0;
133  }
134  else{
135  weaponGraphics = CG_FindUserWeaponGraphics(ent->clientNum, ent->weapon + ALTWEAPON_OFFSET );
136  newLerp = ent->charge2.chBase;
137  backLerp = client->lerpSec;
138  client->lerpSec = newLerp < backLerp ? newLerp : (client->lerpSec + newLerp) / 2.0f;
139  lerp = client->lerpSec;
140  client->lerpPrim = 0;
141  }
142  if(lerp > weaponGraphics->chargeStartPct){
143  if(CG_GetTagOrientationFromPlayerEntity(client,weaponGraphics->chargeTag[0],&orient)){
144  memset(&refEnt,0,sizeof(refEnt));
145  if(VectorLength(weaponGraphics->chargeSpin)){
146  vec3_t tempAngles;
147  vec3_t lerpAxis[3];
148  vec3_t tempAxis[3];
149  VectorCopy(orient.origin,refEnt.origin);
150  VectorSet(tempAngles,cg.time / 4.0f,cg.time / 4.0f,cg.time / 4.0f);
151  VectorPieceWiseMultiply(tempAngles,weaponGraphics->chargeSpin,tempAngles);
152  AnglesToAxis(tempAngles,lerpAxis);
153  AxisClear(refEnt.axis);
154  MatrixMultiply(refEnt.axis,lerpAxis,tempAxis);
155  MatrixMultiply(tempAxis,orient.axis,refEnt.axis);
156  }
157  else{
158  VectorCopy(orient.origin,refEnt.origin);
159  AxisCopy(orient.axis,refEnt.axis);
160  }
161  CG_AddPlayerWeaponCharge(parent,weaponGraphics,&refEnt,lerp);
162  }
163  }
164  if(weaponGraphics->chargeLoopSound){
165  trap_S_AddLoopingSound(ent->number,client->lerpOrigin,vec3_origin,weaponGraphics->chargeLoopSound);
166  }
167  CG_AddPlayerWeaponChargeVoices(client,weaponGraphics,lerp,backLerp);
168  if(weaponGraphics->chargeParticleSystem[0]){
169  // If the entity wasn't previously in the PVS, if the weapon nr switched, or if the weaponstate switched
170  // we need to start a new system
171  if(!wasPVS || newNR || newState){
172  PSys_SpawnCachedSystem(weaponGraphics->chargeParticleSystem,client->lerpOrigin,NULL,client,weaponGraphics->chargeTag[0],qfalse,qtrue);
173  }
174  }
175  }
176  else if(weaponState == WEAPON_GUIDING || weaponState == WEAPON_ALTGUIDING || weaponState == WEAPON_FIRING || weaponState == WEAPON_ALTFIRING){
177  int index = weaponState == WEAPON_GUIDING || weaponState == WEAPON_FIRING ? ent->weapon : ent->weapon + ALTWEAPON_OFFSET;
178  weaponGraphics = CG_FindUserWeaponGraphics(ent->clientNum,index);
179  if(CG_GetTagOrientationFromPlayerEntity(client,weaponGraphics->chargeTag[0],&orient)){
180  memset(&refEnt,0,sizeof(refEnt));
181  VectorCopy(orient.origin,refEnt.origin);
182  AxisCopy(orient.axis,refEnt.axis);
183  CG_AddPlayerWeaponFlash(parent,weaponGraphics,&refEnt,ent->attackPowerTotal,ent->attackPowerCurrent);
184  }
185  if(weaponGraphics->firingSound){
186  trap_S_AddLoopingSound(ent->number,client->lerpOrigin,vec3_origin,weaponGraphics->firingSound);
187  }
188  if(weaponGraphics->firingParticleSystem[0]){
189  // If the entity wasn't previously in the PVS, if the weapon nr switched, or if the weaponstate switched
190  // we need to start a new system
191  if(!wasPVS || newNR || newState){
192  PSys_SpawnCachedSystem(weaponGraphics->firingParticleSystem,client->lerpOrigin,NULL,client,weaponGraphics->chargeTag[0],qfalse,qtrue);
193  }
194  }
195  }
196 }

202 static void CG_DrawWeaponSelectHorCenterBar(void){
203  int i;
204  int statBits = cg.snap->ps.stats[stSkills];
205  float* color = CG_FadeColor(cg.weaponSelectTime,WEAPON_SELECT_TIME,200);
206  vec2_t position = {70,418};
207  cg_userWeapon_t* weaponInfo;
208  cg_userWeapon_t* alt_weaponInfo;
209  if(!color){
210  cg.drawWeaponBar = 0;
211  return;
212  }
213  trap_R_SetColor(color);
214  for(i=1;i<MAX_PLAYERWEAPONS;++i){
215  qboolean exists = statBits & (1 << i);
216  qboolean usable = cg.snap->ps.powerups[PW_SKILLS] & (1 << (i-1));
217  if(!exists || !usable){continue;}
218  weaponInfo = CG_FindUserWeaponGraphics(cg.snap->ps.clientNum,i);
219  CG_DrawPic(qfalse,position[0],position[1],24,24,weaponInfo->weaponIcon);
220  if(i == cg.weaponSelect){CG_DrawPic(qfalse,position[0] - 4,position[1] - 4, 32,32,cgs.media.selectShader);}
221  position[0] += 30;
222  }
223  weaponInfo = CG_FindUserWeaponGraphics(cg.snap->ps.clientNum,cg.weaponSelect);
224  alt_weaponInfo = CG_FindUserWeaponGraphics(cg.snap->ps.clientNum,ALTWEAPON_OFFSET + cg.weaponSelect);
225  if(weaponInfo->weaponName[0]){
226  // Need a little more leeway
227  char name[MAX_WEAPONNAME * 2 + 4];
228  Com_sprintf(name,sizeof(name),"%s",weaponInfo->weaponName);
229  if(alt_weaponInfo->weaponName[0]){
230  strcat(name,va(" / %s",alt_weaponInfo->weaponName));
231  }
232  CG_DrawSmallStringColor(6,position[1] - 30,name,color);
233  }
234  trap_R_SetColor(NULL);
235 }

272 void CG_FireWeapon(centity_t* client,qboolean altFire){
273  cg_userWeapon_t* weaponGraphics;
274  entityState_t* ent = &client->currentState;
275  int maxSounds = 0;
276  int maxVoices = 0;
277  int weapNr;
278  if(ent->weapon == WP_NONE){return;}
279  weapNr = altFire ? ent->weapon + ALTWEAPON_OFFSET : ent->weapon;
280  weaponGraphics = CG_FindUserWeaponGraphics(ent->clientNum, weapNr );
281  // mark the entity as muzzle flashing, so when it is added it will
282  // append the flash to the weapon model.
283  client->muzzleFlashTime = cg.time;
284  if(weaponGraphics->flashParticleSystem[0]){
285  PSys_SpawnCachedSystem(weaponGraphics->flashParticleSystem,client->lerpOrigin,NULL,client,weaponGraphics->chargeTag[0],qfalse,qfalse);
286  }
287  for(;maxSounds<MAX_FLASH_SOUNDS;++maxSounds){
288  if(!weaponGraphics->flashSound[maxSounds]){break;}
289  }
290  if(maxSounds > 0){
291  trap_S_StartSound(NULL,ent->number,CHAN_WEAPON,weaponGraphics->flashSound[rand()%maxSounds]);
292  }
293  for(;maxVoices<MAX_FLASH_VOICES;++maxVoices){
294  if(!weaponGraphics->voiceSound[maxVoices]){break;}
295  }
296  if(maxVoices > 0){
297  trap_S_StartSound(NULL,ent->number,CHAN_WEAPON,weaponGraphics->voiceSound[rand()%maxVoices]);
298  }
299 }
