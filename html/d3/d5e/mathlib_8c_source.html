45  float d;
46 
47  // this rotate and negate guarantees a vector
48  // not colinear with the original
49  right[1] = -forward[0];

51  right[0] = forward[2];

59 void AnglesToMatrix( vec3_t in, vec3_t out[3] ) {
60  vec_t sr, sp, sy, cr, cp, cy;
61 
62  sy = sin(in[2]);
63  cy = cos(in[2]);
64  sp = sin(in[1]);
65  cp = cos(in[1]);
66  sr = sin(in[0]);
67  cr = cos(in[0]);
68 
69  out[0][0] = cp*cy;
70  out[1][0] = cp*sy;
71  out[2][0] = -sp;
72  out[0][1] = sr*sp*cy+cr*-sy;
73  out[1][1] = sr*sp*sy+cr*cy;
74  out[2][1] = sr*cp;
75  out[0][2] = (cr*sp*cy+-sr*-sy);
76  out[1][2] = (cr*sp*sy+-sr*cy);
77  out[2][2] = cr*cp;
78 }

92 void MatrixInvert(vec3_t m[3], vec3_t inv[3])
93 {
94  int i,j;
95  vec3_t cofactors[3];
96  float determinant;
97 
98  cofactors[0][0]= m[1][1] * m[2][2] - m[2][1] * m[1][2];
99  cofactors[0][1]= -(m[1][0] * m[2][2] - m[1][2] * m[2][0]);
100  cofactors[0][2]= m[1][0] * m[2][1] - m[2][0] * m[1][1];
101  cofactors[1][0]= -(m[0][1] * m[2][2] - m[0][2] * m[2][1]);
102  cofactors[1][1]= m[0][0] * m[2][2] - m[0][2] * m[2][0];
103  cofactors[1][2]= -(m[0][0] * m[2][1] - m[2][0] * m[0][1]);
104  cofactors[2][0]= m[0][1] * m[1][2] - m[0][2] * m[1][1];
105  cofactors[2][1]= -(m[0][0] * m[1][2] - m[1][0] * m[0][2]);
106  cofactors[2][2]= m[0][0] * m[1][1] - m[1][0] * m[0][1];
107 
108  determinant = m[0][0] * cofactors[0][0] + m[0][1] * cofactors[0][1] + m[0][2] * m[0][2];
109 
110  for(i=0;i<3;i++)
111  {
112  for(j=0;j<3;j++)
113  {
114  inv[i][j] = cofactors[j][i]/determinant;
115  }
116  }
117 }

244 void ProjectPointOnPlane( vec3_t dst, const vec3_t p, const vec3_t normal ) {
245  float d;
246  vec3_t n;
247  float inv_denom;
248 
249  inv_denom = 1.0F / DotProduct( normal, normal );
250 
251  d = DotProduct( normal, p ) * inv_denom;
252 
253  n[0] = normal[0] * inv_denom;
254  n[1] = normal[1] * inv_denom;
255  n[2] = normal[2] * inv_denom;
256 
257  dst[0] = p[0] - d * n[0];
258  dst[1] = p[1] - d * n[1];
259  dst[2] = p[2] - d * n[2];

263 ** assumes "src" is normalized
264 */
265 void PerpendicularVector( vec3_t dst, const vec3_t src ) {
266  int pos;
267  int i;
268  float minelem = 1.0F;
269  vec3_t tempvec;
270 
271  /*
272  ** find the smallest magnitude axially aligned vector
273  */
274  for ( pos = 0, i = 0; i < 3; i++ ) {
275  if ( fabs( src[i] ) < minelem ) {
276  pos = i;
277  minelem = fabs( src[i] );
278  }
279  }
280  tempvec[0] = tempvec[1] = tempvec[2] = 0.0F;
281  tempvec[pos] = 1.0F;
282 
283  /*
284  ** project the point onto the plane defined by src
285  */
286  ProjectPointOnPlane( dst, tempvec, src );
287 
288  /*
289  ** normalize the result
290  */
291  VectorNormalize( dst, dst );
292 }
