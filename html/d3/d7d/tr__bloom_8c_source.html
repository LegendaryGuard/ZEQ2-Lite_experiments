91  int x = 0;
92  int y = 0;
93  x = 0;

95  width += x;
96  height += y;

98  texWidth += texX;
99  texHeight += texY;

101  qglBegin( GL_QUADS );
102  qglTexCoord2f( texX, texHeight );

105  qglTexCoord2f( texX, texY );

117 /*
118 =================
119 R_Bloom_InitTextures
120 =================
121 */
122 static void R_Bloom_InitTextures( void )
123 {
124  byte *data;
125 
126  // find closer power of 2 to screen size
127  for (bloom.screen.width = 1;bloom.screen.width< glConfig.vidWidth;bloom.screen.width *= 2);
128  for (bloom.screen.height = 1;bloom.screen.height < glConfig.vidHeight;bloom.screen.height *= 2);
129 
130  bloom.screen.readW = glConfig.vidWidth / (float)bloom.screen.width;
131  bloom.screen.readH = glConfig.vidHeight / (float)bloom.screen.height;
132 
133  // find closer power of 2 to effect size
134  bloom.work.width = r_bloom_sample_size->integer;
135  bloom.work.height = bloom.work.width * ( glConfig.vidWidth / glConfig.vidHeight );
136 
137  for (bloom.effect.width = 1;bloom.effect.width < bloom.work.width;bloom.effect.width *= 2);
138  for (bloom.effect.height = 1;bloom.effect.height < bloom.work.height;bloom.effect.height *= 2);
139 
140  bloom.effect.readW = bloom.work.width / (float)bloom.effect.width;
141  bloom.effect.readH = bloom.work.height / (float)bloom.effect.height;
142 
143 
144  // disable blooms if we can't handle a texture of that size
145  if( bloom.screen.width > glConfig.maxTextureSize ||
146  bloom.screen.height > glConfig.maxTextureSize ||
147  bloom.effect.width > glConfig.maxTextureSize ||
148  bloom.effect.height > glConfig.maxTextureSize ||
149  bloom.work.width > glConfig.vidWidth ||
150  bloom.work.height > glConfig.vidHeight
151  ) {
152  ri.Cvar_Set( "r_bloom", "0" );
153  Com_Printf( S_COLOR_YELLOW"WARNING: 'R_InitBloomTextures' too high resolution for light bloom, effect disabled\n" );
154  return;
155  }
156 
157  data = ri.Hunk_AllocateTempMemory( bloom.screen.width * bloom.screen.height * 4 );
158  Com_Memset( data, 0, bloom.screen.width * bloom.screen.height * 4 );
159  bloom.screen.texture = R_CreateImage( "***bloom screen texture***", data, bloom.screen.width, bloom.screen.height, qfalse, qfalse, GL_CLAMP_TO_EDGE);
160  ri.Hunk_FreeTempMemory( data );
161 
162  data = ri.Hunk_AllocateTempMemory( bloom.effect.width * bloom.effect.height * 4 );
163  Com_Memset( data, 0, bloom.effect.width * bloom.effect.height * 4 );
164  bloom.effect.texture = R_CreateImage( "***bloom effect texture***", data, bloom.effect.width, bloom.effect.height, qfalse, qfalse, GL_CLAMP_TO_EDGE);
165  ri.Hunk_FreeTempMemory( data );
166  bloom.started = qtrue;
167 }

200 =================
201 R_Bloom_GeneratexDiamonds
202 =================
203 */
204 static void R_Bloom_WarsowEffect( void )
205 {
206  int i, j, k;
207  float intensity, scale, *diamond;
208 
209 
210  qglColor4f( 1.0f, 1.0f, 1.0f, 1.0f );
211  //Take the backup texture and downscale it
212  GL_Bind( bloom.screen.texture );
213  GL_State( GLS_DEPTHTEST_DISABLE | GLS_SRCBLEND_ONE | GLS_DSTBLEND_ZERO );
214  R_Bloom_Quad( bloom.work.width, bloom.work.height, 0, 0, bloom.screen.readW, bloom.screen.readH );
215  //Copy downscaled framebuffer into a texture
216  GL_Bind( bloom.effect.texture );
217  qglCopyTexSubImage2D( GL_TEXTURE_2D, 0, 0, 0, 0, 0, bloom.work.width, bloom.work.height );
218  // darkening passes with repeated filter
219  if( r_bloom_darken->integer ) {
220  int i;
221  GL_State( GLS_DEPTHTEST_DISABLE | GLS_SRCBLEND_DST_COLOR | GLS_DSTBLEND_ZERO );
222 
223  for( i = 0; i < r_bloom_darken->integer; i++ ) {
224  R_Bloom_Quad( bloom.work.width, bloom.work.height,
225  0, 0,
226  bloom.effect.readW, bloom.effect.readH );
227  }
228  qglCopyTexSubImage2D( GL_TEXTURE_2D, 0, 0, 0, 0, 0, bloom.work.width, bloom.work.height );
229  }
230  /* Copy the result to the effect texture */
231  GL_Bind( bloom.effect.texture );
232  qglCopyTexSubImage2D( GL_TEXTURE_2D, 0, 0, 0, 0, 0, bloom.work.width, bloom.work.height );
233 
234  // bluring passes, warsow uses a repeated semi blend on a selectable diamond grid
235  qglColor4f( 1.0f, 1.0f, 1.0f, 1.0f );
236  GL_State( GLS_DEPTHTEST_DISABLE | GLS_SRCBLEND_ONE | GLS_DSTBLEND_ONE_MINUS_SRC_COLOR );
237  if( r_bloom_diamond_size->integer > 7 || r_bloom_diamond_size->integer <= 3 ) {
238  if( r_bloom_diamond_size->integer != 8 )
239  ri.Cvar_Set( "r_bloom_diamond_size", "8" );
240  } else if( r_bloom_diamond_size->integer > 5 ) {
241  if( r_bloom_diamond_size->integer != 6 )
242  ri.Cvar_Set( "r_bloom_diamond_size", "6" );
243  } else if( r_bloom_diamond_size->integer > 3 ) {
244  if( r_bloom_diamond_size->integer != 4 )
245  ri.Cvar_Set( "r_bloom_diamond_size", "4" );
246  }
247 
248  switch( r_bloom_diamond_size->integer ) {
249  case 4:
250  k = 2;
251  diamond = &Diamond4x[0][0];
252  scale = r_bloom_intensity->value * 0.8f;
253  break;
254  case 6:
255  k = 3;
256  diamond = &Diamond6x[0][0];
257  scale = r_bloom_intensity->value * 0.5f;
258  break;
259  default:
260 // case 8:
261  k = 4;
262  diamond = &Diamond8x[0][0];
263  scale = r_bloom_intensity->value * 0.3f;
264  break;
265  }
266 
267  for( i = 0; i < r_bloom_diamond_size->integer; i++ ) {
268  for( j = 0; j < r_bloom_diamond_size->integer; j++, diamond++ ) {
269  float x, y;
270  intensity = *diamond * scale;
271  if( intensity < 0.01f )
272  continue;
273  qglColor4f( intensity, intensity, intensity, 1.0 );
274  x = (i - k) * ( 2 / 640.0f ) * bloom.effect.readW;
275  y = (j - k) * ( 2 / 480.0f ) * bloom.effect.readH;
276 
277  R_Bloom_Quad( bloom.work.width, bloom.work.height, x, y, bloom.effect.readW, bloom.effect.readH );
278  }
279  }
280  qglCopyTexSubImage2D( GL_TEXTURE_2D, 0, 0, 0, 0, 0, bloom.work.width, bloom.work.height );
281 }
