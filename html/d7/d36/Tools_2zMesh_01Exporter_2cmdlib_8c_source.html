225 int LoadFileBlock( const char *filename, void **bufferptr ) {

229

235  nAllocSize += MEM_BLOCKSIZE - nBlock;

237  buffer = malloc (nAllocSize+1);
238  memset(buffer, 0, nAllocSize+1);
