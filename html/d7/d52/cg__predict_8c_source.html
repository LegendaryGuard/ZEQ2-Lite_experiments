38 ====================
39 CG_BuildSolidList
40 
41 When a new cg.snap has been set, this function builds a sublist
42 of the entities that are actually solid, to make for more
43 efficient collision detection
44 ====================
45 */
46 void CG_BuildSolidList( void ) {
47  int i;
48  centity_t *cent;
49  snapshot_t *snap;
50  entityState_t *ent;
51 
52  cg_numSolidEntities = 0;
53  cg_numTriggerEntities = 0;
54 
55  if ( cg.nextSnap && !cg.nextFrameTeleport && !cg.thisFrameTeleport ) {
56  snap = cg.nextSnap;
57  } else {
58  snap = cg.snap;
59  }
60 
61  for ( i = 0 ; i < snap->numEntities ; i++ ) {
62  cent = &cg_entities[ snap->entities[ i ].number ];
63  ent = &cent->currentState;
64 
65  if ( ent->eType == ET_ITEM || ent->eType == ET_PUSH_TRIGGER || ent->eType == ET_TELEPORT_TRIGGER ) {
66  cg_triggerEntities[cg_numTriggerEntities] = cent;
67  cg_numTriggerEntities++;
68  continue;
69  }
70 
71  if ( cent->nextState.solid ) {
72  cg_solidEntities[cg_numSolidEntities] = cent;
73  cg_numSolidEntities++;
74  continue;
75  }
76  }
77 }

85 static void CG_ClipMoveToEntities ( const vec3_t start, const vec3_t mins, const vec3_t maxs, const vec3_t end,
86  int skipNumber, int mask, trace_t *tr ) {
87  int i, x, zd, zu;
88  trace_t trace;
89  entityState_t *ent;
90  clipHandle_t cmodel;
91  vec3_t bmins, bmaxs;
92  vec3_t origin, angles;
93  centity_t *cent;
94 
95  for ( i = 0 ; i < cg_numSolidEntities ; i++ ) {
96  cent = cg_solidEntities[ i ];
97  ent = &cent->currentState;
98 
99  if ( ent->number == skipNumber ) {
100  continue;
101  }
102 
103  if ( ent->solid == SOLID_BMODEL ) {
104  // special value for bmodel
105  cmodel = trap_CM_InlineModel( ent->modelindex );
106  VectorCopy( cent->lerpAngles, angles );
107  BG_EvaluateTrajectory( &cent->currentState, &cent->currentState.pos, cg.physicsTime, origin );
108  } else {
109  // encoded bbox
110  x = (ent->solid & 255);
111  zd = ((ent->solid>>8) & 255);
112  zu = ((ent->solid>>16) & 255) - 32;
113 
114  bmins[0] = bmins[1] = -x;
115  bmaxs[0] = bmaxs[1] = x;
116  bmins[2] = -zd;
117  bmaxs[2] = zu;
118 
119  cmodel = trap_CM_TempBoxModel( bmins, bmaxs );
120  VectorCopy( vec3_origin, angles );
121  VectorCopy( cent->lerpOrigin, origin );
122  }
123 
124 
125  trap_CM_TransformedBoxTrace ( &trace, start, end,
126  mins, maxs, cmodel, mask, origin, angles);
127 
128  if (trace.allsolid || trace.fraction < tr->fraction) {
129  trace.entityNum = ent->number;
130  *tr = trace;
131  } else if (trace.startsolid) {
132  tr->startsolid = qtrue;
133  }
134  if ( tr->allsolid ) {
135  return;
136  }
137  }
138 }

214 /*
215 ========================
216 CG_InterpolatePlayerState
217 
218 Generates cg.predictedPlayerState by interpolating between
219 cg.snap->player_state and cg.nextFrame->player_state
220 ========================
221 */
222 static void CG_InterpolatePlayerState(qboolean grabAngles) {
223  float f;
224  int i;
225  playerState_t *out;
226  snapshot_t *prev, *next;
227  out = &cg.predictedPlayerState;
228  prev = cg.snap;
229  next = cg.nextSnap;
230  *out = cg.snap->ps;
231  // if we are still allowing local input, short circuit the view angles
232  if(grabAngles){
233  usercmd_t cmd;
234  int cmdNum;
235  cmdNum = trap_GetCurrentCmdNumber();
236  trap_GetUserCmd(cmdNum,&cmd);
237  PM_UpdateViewAngles(out,&cmd);
238  }
239  if(cg.nextFrameTeleport){return;}
240  if(!next || next->serverTime <= prev->serverTime){return;}
241  f = (float)( cg.time - prev->serverTime ) / ( next->serverTime - prev->serverTime );
242  i = next->ps.bobCycle;
243  if(i<prev->ps.bobCycle){i += 256;}
244  out->bobCycle = prev->ps.bobCycle + f * ( i - prev->ps.bobCycle );
245  for(i=0;i<3;i++){
246  out->origin[i] = prev->ps.origin[i] + f * (next->ps.origin[i] - prev->ps.origin[i]);
247  if(!grabAngles){
248  out->viewangles[i] = LerpAngle(prev->ps.viewangles[i],next->ps.viewangles[i],f);
249  }
250  out->velocity[i] = prev->ps.velocity[i] + f * (next->ps.velocity[i] - prev->ps.velocity[i]);
251  }
252 }

271 static void CG_TouchTriggerPrediction( void ) {
272  int i;
273  trace_t trace;
274  entityState_t *ent;
275  clipHandle_t cmodel;
276  centity_t *cent;
277  qboolean spectator;
278 
279  // dead clients don't activate triggers
280  if ( cg.predictedPlayerState.powerLevel[plFatigue] <= 0 ) {
281  return;
282  }
283 
284  spectator = ( cg.predictedPlayerState.pm_type == PM_SPECTATOR );
285 
286  if ( cg.predictedPlayerState.pm_type != PM_NORMAL && !spectator ) {
287  return;
288  }
289 
290  for ( i = 0 ; i < cg_numTriggerEntities ; i++ ) {
291  cent = cg_triggerEntities[ i ];
292  ent = &cent->currentState;
293 
294  if ( ent->eType == ET_ITEM && !spectator ) {
295  CG_TouchItem( cent );
296  continue;
297  }
298 
299  if ( ent->solid != SOLID_BMODEL ) {
300  continue;
301  }
302 
303  cmodel = trap_CM_InlineModel( ent->modelindex );
304  if ( !cmodel ) {
305  continue;
306  }
307 
308  trap_CM_BoxTrace( &trace, cg.predictedPlayerState.origin, cg.predictedPlayerState.origin,
309  cg_pmove.mins, cg_pmove.maxs, cmodel, -1 );
310 
311  if ( !trace.startsolid ) {
312  continue;
313  }
314 
315  if ( ent->eType == ET_TELEPORT_TRIGGER ) {
316  cg.hyperspace = qtrue;
317  } else if ( ent->eType == ET_PUSH_TRIGGER ) {
318  BG_TouchJumpPad( &cg.predictedPlayerState, ent );
319  }
320  }
321 
322  // if we didn't touch a jump pad this pmove frame
323  if ( cg.predictedPlayerState.jumppad_frame != cg.predictedPlayerState.pmove_framecount ) {
324  cg.predictedPlayerState.jumppad_frame = 0;
325  cg.predictedPlayerState.jumppad_ent = 0;
326  }
327 }

331 /*
332 =================
333 CG_PredictPlayerState
334 
335 Generates cg.predictedPlayerState for the current cg.time
336 cg.predictedPlayerState is guaranteed to be valid after exiting.
337 
338 For demo playback, this will be an interpolation between two valid
339 playerState_t.
340 
341 For normal gameplay, it will be the result of predicted usercmd_t on
342 top of the most recent playerState_t received from the server.
343 
344 Each new snapshot will usually have one or more new usercmd over the last,
345 but we simulate all unacknowledged commands each time, not just the new ones.
346 This means that on an internet connection, quite a few pmoves may be issued
347 each frame.
348 
349 OPTIMIZE: don't re-simulate unless the newly arrived snapshot playerState_t
350 differs from the predicted one. Would require saving all intermediate
351 playerState_t during prediction.
352 
353 We detect prediction errors and allow them to be decayed off over several frames
354 to ease the jerk.
355 =================
356 */
357 void CG_PredictPlayerState( void ) {
358  int cmdNum, current;
359  playerState_t oldPlayerState;
360  qboolean moved;
361  usercmd_t oldestCmd;
362  usercmd_t latestCmd;
363 
364  cg.hyperspace = qfalse;
365  if(!cg.validPPS){
366  cg.validPPS = qtrue;
367  cg.predictedPlayerState = cg.snap->ps;
368  }
369  if(cg.demoPlayback || (cg.snap->ps.pm_flags & PMF_FOLLOW) || cg.snap->ps.lockedTarget > 0) {
370  CG_InterpolatePlayerState(qfalse);
371  return;
372  }
373  cg_pmove.ps = &cg.predictedPlayerState;
374  cg_pmove.trace = CG_Trace;
375  cg_pmove.pointcontents = CG_PointContents;
376  cg_pmove.tracemask = MASK_PLAYERSOLID;
377  if(cg.snap->ps.persistant[PERS_TEAM] == TEAM_SPECTATOR ) {
378  cg_pmove.tracemask &= ~CONTENTS_BODY;
379  }
380  cg_pmove.noFootsteps = ( cgs.dmflags & DF_NO_FOOTSTEPS ) > 0;
381  oldPlayerState = cg.predictedPlayerState;
382  current = trap_GetCurrentCmdNumber();
383  // if we don't have the commands right after the snapshot, we
384  // can't accurately predict a current position, so just freeze at
385  // the last good position we had
386  cmdNum = current - CMD_BACKUP + 1;
387  trap_GetUserCmd( cmdNum, &oldestCmd );
388  if (oldestCmd.serverTime > cg.snap->ps.commandTime && oldestCmd.serverTime < cg.time){
389  if(cg_showmiss.integer){
390  CG_Printf ("exceeded PACKET_BACKUP on commands\n");
391  }
392  return;
393  }
394  // get the latest command so we can know which commands are from previous map_restarts
395  trap_GetUserCmd( current, &latestCmd );
396  // get the most recent information we have, even if
397  // the server time is beyond our current cg.time,
398  // because predicted player positions are going to
399  // be ahead of everything else anyway
400  if ( cg.nextSnap && !cg.nextFrameTeleport && !cg.thisFrameTeleport ) {
401  cg.predictedPlayerState = cg.nextSnap->ps;
402  cg.physicsTime = cg.nextSnap->serverTime;
403  } else {
404  cg.predictedPlayerState = cg.snap->ps;
405  cg.physicsTime = cg.snap->serverTime;
406  }
407 
408  if ( pmove_msec.integer < 8 ) {
409  trap_Cvar_Set("pmove_msec", "8");
410  }
411  else if (pmove_msec.integer > 33) {
412  trap_Cvar_Set("pmove_msec", "33");
413  }
414 
415  cg_pmove.pmove_fixed = pmove_fixed.integer;// | cg_pmove_fixed.integer;
416  cg_pmove.pmove_msec = pmove_msec.integer;
417 
418  // run cmds
419  moved = qfalse;
420  for ( cmdNum = current - CMD_BACKUP + 1 ; cmdNum <= current ; cmdNum++ ) {
421  // get the command
422  trap_GetUserCmd( cmdNum, &cg_pmove.cmd );
423 
424  if ( cg_pmove.pmove_fixed ) {
425  PM_UpdateViewAngles( cg_pmove.ps, &cg_pmove.cmd );
426  }
427 
428  // don't do anything if the time is before the snapshot player time
429  if ( cg_pmove.cmd.serverTime <= cg.predictedPlayerState.commandTime ) {
430  continue;
431  }
432 
433  // don't do anything if the command was from a previous map_restart
434  if ( cg_pmove.cmd.serverTime > latestCmd.serverTime ) {
435  continue;
436  }
437 
438  // check for a prediction error from last frame
439  // on a lan, this will often be the exact value
440  // from the snapshot, but on a wan we will have
441  // to predict several commands to get to the point
442  // we want to compare
443  if ( cg.predictedPlayerState.commandTime == oldPlayerState.commandTime ) {
444  vec3_t delta;
445  float len;
446 
447  if ( cg.thisFrameTeleport ) {
448  // a teleport will not cause an error decay
449  VectorClear( cg.predictedError );
450  if ( cg_showmiss.integer ) {
451  CG_Printf( "PredictionTeleport\n" );
452  }
453  cg.thisFrameTeleport = qfalse;
454  } else {
455  vec3_t adjusted;
456  CG_AdjustPositionForMover( cg.predictedPlayerState.origin,
457  cg.predictedPlayerState.groundEntityNum, cg.physicsTime, cg.oldTime, adjusted );
458 
459  if ( cg_showmiss.integer ) {
460  if (!VectorCompare( oldPlayerState.origin, adjusted )) {
461  CG_Printf("prediction error\n");
462  }
463  }
464  VectorSubtract( oldPlayerState.origin, adjusted, delta );
465  len = VectorLength( delta );
466  if ( len > 0.1 ) {
467  if ( cg_showmiss.integer ) {
468  CG_Printf("Prediction miss: %f\n", len);
469  }
470  if ( cg_errorDecay.integer ) {
471  int t;
472  float f;
473 
474  t = cg.time - cg.predictedErrorTime;
475  f = ( cg_errorDecay.value - t ) / cg_errorDecay.value;
476  if ( f < 0 ) {
477  f = 0;
478  }
479  if ( f > 0 && cg_showmiss.integer ) {
480  CG_Printf("Double prediction decay: %f\n", f);
481  }
482  VectorScale( cg.predictedError, f, cg.predictedError );
483  } else {
484  VectorClear( cg.predictedError );
485  }
486  VectorAdd( delta, cg.predictedError, cg.predictedError );
487  cg.predictedErrorTime = cg.oldTime;
488  }
489  }
490  }
491  // don't predict gauntlet firing, which is only supposed to happen
492  // when it actually inflicts damage
493  cg_pmove.gauntletHit = qfalse;
494  if ( cg_pmove.pmove_fixed ) {
495  cg_pmove.cmd.serverTime = ((cg_pmove.cmd.serverTime + pmove_msec.integer-1) / pmove_msec.integer) * pmove_msec.integer;
496  }
497  Pmove (&cg_pmove);
498  moved = qtrue;
499 
500  // add push trigger movement effects
501  CG_TouchTriggerPrediction();
502 
503  // check for predictable events that changed from previous predictions
504  //CG_CheckChangedPredictableEvents(&cg.predictedPlayerState);
505  }
506 
507  if ( cg_showmiss.integer > 1 ) {
508  CG_Printf( "[%i : %i] ", cg_pmove.cmd.serverTime, cg.time );
509  }
510 
511  if ( !moved ) {
512  if ( cg_showmiss.integer ) {
513  CG_Printf( "not moved\n" );
514  }
515  return;
516  }
517 
518  // adjust for the movement of the groundentity
519  CG_AdjustPositionForMover(cg.predictedPlayerState.origin,cg.predictedPlayerState.groundEntityNum,cg.physicsTime, cg.time, cg.predictedPlayerState.origin);
520  if(cg_showmiss.integer){
521  if(cg.predictedPlayerState.eventSequence > oldPlayerState.eventSequence + MAX_PS_EVENTS){
522  CG_Printf("WARNING: dropped event\n");
523  }
524  }
525  // fire events and other transition triggered things
526 
527  CG_TransitionPlayerState(&cg.predictedPlayerState,&oldPlayerState);
528  if ( cg_showmiss.integer ) {
529  if (cg.eventSequence > cg.predictedPlayerState.eventSequence) {
530  CG_Printf("WARNING: double event\n");
531  cg.eventSequence = cg.predictedPlayerState.eventSequence;
532  }
533  }
534  if(cg.snap->ps.lockedTarget > 0){
535  CG_InterpolatePlayerState(qfalse);
536  return;
537  }
538 }
