183 
184  case TEAM_BLUE: // CTF
185  if( teamgame.blueStatus != status ) {
186  teamgame.blueStatus = status;

188  }
189  break;
190

193  teamgame.flagStatus = status;
194  modified = qtrue;
195  }
196  break;
197  }
198 
199  if( modified ) {

201 
202  if( g_gametype.integer == GT_CTF ) {
203  st[0] = ctfFlagStatusRemap[teamgame.redStatus];
204  st[1] = ctfFlagStatusRemap[teamgame.blueStatus];
205  st[2] = 0;
206  }
207  else { // GT_1FCTF
208  st[0] = oneFlagStatusRemap[teamgame.flagStatus];
209  st[1] = 0;
210  }
211 
212  trap_SetConfigstring( CS_FLAGSTATUS, st );

343 ===========
344 Team_GetLocation
345 
346 Report a location for the player. Uses placed nearby target_location entities
347 ============
348 */
349 gentity_t *Team_GetLocation(gentity_t *ent)
350 {
351  gentity_t *eloc, *best;
352  float bestlen, len;
353  vec3_t origin;
354 
355  best = NULL;
356  bestlen = 3*8192.0*8192.0;
357 
358  VectorCopy( ent->r.currentOrigin, origin );
359 
360  for (eloc = level.locationHead; eloc; eloc = eloc->nextTrain) {
361  len = ( origin[0] - eloc->r.currentOrigin[0] ) * ( origin[0] - eloc->r.currentOrigin[0] )
362  + ( origin[1] - eloc->r.currentOrigin[1] ) * ( origin[1] - eloc->r.currentOrigin[1] )
363  + ( origin[2] - eloc->r.currentOrigin[2] ) * ( origin[2] - eloc->r.currentOrigin[2] );
364 
365  if ( len > bestlen ) {
366  continue;
367  }
368 
369  if ( !trap_InPVS( origin, eloc->r.currentOrigin ) ) {
370  continue;
371  }
372 
373  bestlen = len;
374  best = eloc;
375  }
376 
377  return best;
378 }

414 SelectRandomTeamSpawnPoint
415 
416 go to a random point that doesn't telefrag

419 #define MAX_TEAM_SPAWN_POINTS 32
420 gentity_t *SelectRandomTeamSpawnPoint( int teamstate, team_t team ) {
421  gentity_t *spot;
422  int count;
423  int selection;
424  gentity_t *spots[MAX_TEAM_SPAWN_POINTS];
425  char *classname;
426 
427  if (teamstate == TEAM_BEGIN) {
428  if (team == TEAM_RED)
429  classname = "team_CTF_redplayer";
430  else if (team == TEAM_BLUE)
431  classname = "team_CTF_blueplayer";
432  else
433  return NULL;
434  } else {
435  if (team == TEAM_RED)
436  classname = "team_CTF_redspawn";
437  else if (team == TEAM_BLUE)
438  classname = "team_CTF_bluespawn";
439  else
440  return NULL;
441  }
442  count = 0;
443 
444  spot = NULL;
445 
446  while ((spot = G_Find (spot, FOFS(classname), classname)) != NULL) {
447  if ( SpotWouldTelefrag( spot ) ) {
448  continue;
449  }
450  spots[ count ] = spot;
451  if (++count == MAX_TEAM_SPAWN_POINTS)
452  break;
453  }
454 
455  if ( !count ) { // no spots that won't telefrag
456  return G_Find( NULL, FOFS(classname), classname);
457  }
458 
459  selection = rand() % count;
460  return spots[ selection ];
461 }

494 ==================
495 TeamplayLocationsMessage
496 
497 Format:
498  clientNum location powerLevel armor weapon powerups

500 ==================
501 */
502 void TeamplayInfoMessage( gentity_t *ent ) {
503  char entry[1024];
504  char string[8192];
505  int stringlength;
506  int i, j;
507  gentity_t *player;
508  int cnt;
509  int h, a;
510  int clients[TEAM_MAXOVERLAY];
511 
512  if ( ! ent->client->pers.teamInfo )
513  return;
514 
515  // figure out what client should be on the display
516  // we are limited to 8, but we want to use the top eight players
517  // but in client order (so they don't keep changing position on the overlay)
518  for (i = 0, cnt = 0; i < g_maxclients.integer && cnt < TEAM_MAXOVERLAY; i++) {
519  player = g_entities + level.sortedClients[i];
520  if (player->inuse && player->client->sess.sessionTeam ==
521  ent->client->sess.sessionTeam ) {
522  clients[cnt++] = level.sortedClients[i];
523  }
524  }
525 
526  // We have the top eight players, sort them by clientNum
527  qsort( clients, cnt, sizeof( clients[0] ), SortClients );
528 
529  // send the latest information on all clients
530  string[0] = 0;
531  stringlength = 0;
532 
533  for (i = 0, cnt = 0; i < g_maxclients.integer && cnt < TEAM_MAXOVERLAY; i++) {
534  player = g_entities + i;
535  if (player->inuse && player->client->sess.sessionTeam ==
536  ent->client->sess.sessionTeam ) {
537 
538  h = player->client->ps.powerLevel[plCurrent];
539  //a = player->client->ps.stats[STAT_ARMOR];
540  a = 0;
541  if (h < 0) h = 0;
542  if (a < 0) a = 0;
543 
544  Com_sprintf (entry, sizeof(entry),
545  " %i %i %i %i %i %i",
546 // level.sortedClients[i], player->client->pers.teamState.location, h, a,
547  i, player->client->pers.teamState.location, h, a,
548  player->client->ps.weapon, player->s.powerups);
549  j = strlen(entry);
550  if (stringlength + j >= sizeof(string))
551  break;
552  strcpy (string + stringlength, entry);
553  stringlength += j;
554  cnt++;
555  }
556  }
557 
558  trap_SendServerCommand( ent-g_entities, va("tinfo %i %s", cnt, string) );
559 }
