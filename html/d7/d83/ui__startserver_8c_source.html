111 /*
112 =================
113 GametypeBits
114 =================
115 */
116 static int GametypeBits( char *string ) {
117  int bits;
118  char *p;
119  char *token;
120 
121  bits = 0;
122  p = string;
123  while( 1 ) {
124  token = COM_ParseExt( &p, qfalse );
125  if( token[0] == 0 ) {
126  break;
127  }
128 
129  if( Q_stricmp( token, "ffa" ) == 0 ) {
130  bits |= 1 << GT_FFA;
131  continue;
132  }
133 
134  if( Q_stricmp( token, "tourney" ) == 0 ) {
135  bits |= 1 << GT_TOURNAMENT;
136  continue;
137  }
138  if( Q_stricmp( token, "struggle" ) == 0 ) {
139  bits |= 1 << GT_STRUGGLE;
140  continue;
141  }
142  if( Q_stricmp( token, "single" ) == 0 ) {
143  bits |= 1 << GT_SINGLE_PLAYER;
144  continue;
145  }
146 
147  if( Q_stricmp( token, "team" ) == 0 ) {
148  bits |= 1 << GT_TEAM;
149  continue;
150  }
151 
152  if( Q_stricmp( token, "ctf" ) == 0 ) {
153  bits |= 1 << GT_CTF;
154  continue;
155  }
156  }
157 
158  return bits;
159 }

195 /*
196 =================
197 StartServer_Update
198 =================
199 */
200 static void StartServer_Update( void ) {
201  int i;
202  int top;
203  static char picname[MAX_MAPSPERPAGE][64];
204  const char *info;
205  char mapname[MAX_NAMELENGTH];
206 
207  top = s_startserver.page*MAX_MAPSPERPAGE;
208 
209  for (i=0; i<MAX_MAPSPERPAGE; i++)
210  {
211  if (top+i >= s_startserver.nummaps)
212  break;
213 
214  info = UI_GetArenaInfoByNumber( s_startserver.maplist[ top + i ]);
215  Q_strncpyz( mapname, Info_ValueForKey( info, "map"), MAX_NAMELENGTH );
216  Q_strupr( mapname );
217 
218  Com_sprintf( picname[i], sizeof(picname[i]), "maps/%s", mapname );
219 
220  s_startserver.mappics[i].generic.flags &= ~QMF_HIGHLIGHT;
221  s_startserver.mappics[i].generic.name = picname[i];
222  s_startserver.mappics[i].shader = 0;
223 
224  // reset
225  s_startserver.mapbuttons[i].generic.flags |= QMF_PULSEIFFOCUS;
226  s_startserver.mapbuttons[i].generic.flags &= ~QMF_INACTIVE;
227  }
228 
229  for (; i<MAX_MAPSPERPAGE; i++)
230  {
231  s_startserver.mappics[i].generic.flags &= ~QMF_HIGHLIGHT;
232  s_startserver.mappics[i].generic.name = NULL;
233  s_startserver.mappics[i].shader = 0;
234 
235  // disable
236  s_startserver.mapbuttons[i].generic.flags &= ~QMF_PULSEIFFOCUS;
237  s_startserver.mapbuttons[i].generic.flags |= QMF_INACTIVE;
238  }
239 
240 
241  // no servers to start
242  if( !s_startserver.nummaps ) {
243  s_startserver.next.generic.flags |= QMF_INACTIVE;
244 
245  // set the map name
246  strcpy( s_startserver.mapname.string, "NO MAPS FOUND" );
247  }
248  else {
249  // set the highlight
250  s_startserver.next.generic.flags &= ~QMF_INACTIVE;
251  i = s_startserver.currentmap - top;
252  if ( i >=0 && i < MAX_MAPSPERPAGE )
253  {
254  s_startserver.mappics[i].generic.flags |= QMF_HIGHLIGHT;
255  s_startserver.mapbuttons[i].generic.flags &= ~QMF_PULSEIFFOCUS;
256  }
257 
258  // set the map name
259  info = UI_GetArenaInfoByNumber( s_startserver.maplist[ s_startserver.currentmap ]);
260  Q_strncpyz( s_startserver.mapname.string, Info_ValueForKey( info, "map" ), MAX_NAMELENGTH);
261  }
262 
263  Q_strupr( s_startserver.mapname.string );
264 }

282 /*
283 =================
284 StartServer_GametypeEvent
285 =================
286 */
287 static void StartServer_GametypeEvent( void* ptr, int event ) {
288  int i;
289  int count;
290  int gamebits;
291  int matchbits;
292  const char *info;
293 
294  if( event != QM_ACTIVATED) {
295  return;
296  }
297 
298  count = UI_GetNumArenas();
299  s_startserver.nummaps = 0;
300  matchbits = 1 << gametype_remap[s_startserver.gametype.curvalue];
301  if( gametype_remap[s_startserver.gametype.curvalue] == GT_FFA ) {
302  matchbits |= ( 1 << GT_SINGLE_PLAYER );
303  }
304  for( i = 0; i < count; i++ ) {
305  info = UI_GetArenaInfoByNumber( i );
306 
307  gamebits = GametypeBits( Info_ValueForKey( info, "type") );
308  if( !( gamebits & matchbits ) ) {
309  continue;
310  }
311 
312  s_startserver.maplist[ s_startserver.nummaps ] = i;
313  s_startserver.nummaps++;
314  }
315  s_startserver.maxpages = (s_startserver.nummaps + MAX_MAPSPERPAGE-1)/MAX_MAPSPERPAGE;
316  s_startserver.page = 0;
317  s_startserver.currentmap = 0;
318 
319  StartServer_Update();
320 }

323 /*
324 =================
325 StartServer_MenuEvent
326 =================
327 */
328 static void StartServer_MenuEvent( void* ptr, int event ) {
329  if( event != QM_ACTIVATED ) {
330  return;
331  }
332 
333  switch( ((menucommon_s*)ptr)->id ) {
334  case ID_PREVPAGE:
335  if( s_startserver.page > 0 ) {
336  s_startserver.page--;
337  StartServer_Update();
338  }
339  break;
340 
341  case ID_NEXTPAGE:
342  if( s_startserver.page < s_startserver.maxpages - 1 ) {
343  s_startserver.page++;
344  StartServer_Update();
345  }
346  break;
347 
348  case ID_STARTSERVERNEXT:
349  trap_Cvar_SetValue( "g_gameType", gametype_remap[s_startserver.gametype.curvalue] );
350  StartServer_Start();
351  break;
352 
353  case ID_STARTSERVERBACK:
354  UI_PopMenu();
355  break;
356  }
357 }

379 
380  if( b->generic.name && !b->shader ) {
381  b->shader = trap_R_RegisterShaderNoMip( b->generic.name );
382  if( !b->shader && b->errorpic ) {
383  b->shader = trap_R_RegisterShaderNoMip( b->errorpic );
384  }
385  }
386 
387  if( b->focuspic && !b->focusshader ) {
388  b->focusshader = trap_R_RegisterShaderNoMip( b->focuspic );
389  }
390 
391  x = b->generic.x;
392  y = b->generic.y;
393  w = b->width;
394  h = b->height;
395  if( b->shader ) {
396  UI_DrawHandlePic( x, y, w, h, b->shader );
397  }
398 
399  x = b->generic.x;
400  y = b->generic.y + b->height;
401  UI_FillRect( x, y, b->width, 28, colorBlack );
402 
403  x += b->width / 2;
404  y += 4;
405  n = s_startserver.page * MAX_MAPSPERPAGE + b->generic.id - ID_PICTURES;
406 
407  info = UI_GetArenaInfoByNumber( s_startserver.maplist[ n ]);
408  UI_DrawString( x, y, Info_ValueForKey( info, "map" ), UI_CENTER|UI_SMALLFONT|UI_DROPSHADOW, color_orange );
409 
410  x = b->generic.x;
411  y = b->generic.y;
412  w = b->width;
413  h = b->height + 28;
414  if( b->generic.flags & QMF_HIGHLIGHT ) {
415  UI_DrawHandlePic( x, y, w, h, b->focusshader );
416  }
417 }
