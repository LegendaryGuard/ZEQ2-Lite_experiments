28 ================
29 G_RankRunFrame
30 ================
31 */
32 void G_RankRunFrame()
33 {
34  gentity_t* ent;
35  gentity_t* ent2;
36  grank_status_t old_status;
37  grank_status_t status;
38  int time;
39  int i;
40  int j;
41 
42  if( !trap_RankCheckInit() )
43  {
44  trap_RankBegin( GR_GAMEKEY );
45  }
46 
47  trap_RankPoll();
48 
49  if( trap_RankActive() )
50  {
51  for( i = 0; i < level.maxclients; i++ )
52  {
53  ent = &(g_entities[i]);
54  if ( !ent->inuse )
55  continue;
56  if ( ent->client == NULL )
57  continue;
58  if ( ent->r.svFlags & SVF_BOT)
59  {
60  // no bots in ranked games
61  trap_SendConsoleCommand( EXEC_INSERT, va("kick %s\n",
62  ent->client->pers.netname) );
63  continue;
64  }
65 
66  old_status = ent->client->client_status;
67  status = trap_RankUserStatus( i );
68 
69  if( ent->client->client_status != status )
70  {
71  // inform client of current status
72  // not needed for client side log in
73  trap_SendServerCommand( i, va("rank_status %i\n",status) );
74  if ( i == 0 )
75  {
76  int j = 0;
77  }
78  ent->client->client_status = status;
79  }
80 
81  switch( status )
82  {
83  case QGR_STATUS_NEW:
84  case QGR_STATUS_SPECTATOR:
85  if( ent->client->sess.sessionTeam != TEAM_SPECTATOR )
86  {
87  ent->client->sess.sessionTeam = TEAM_SPECTATOR;
88  ent->client->sess.spectatorState = SPECTATOR_FREE;
89  ClientSpawn( ent );
90  // make sure by now CS_GRAND rankingsGameID is ready
91  trap_SendServerCommand( i, va("rank_status %i\n",status) );
92  trap_SendServerCommand( i, "rank_menu\n" );
93  }
94  break;
95  case QGR_STATUS_NO_USER:
96  case QGR_STATUS_BAD_PASSWORD:
97  case QGR_STATUS_TIMEOUT:
98  case QGR_STATUS_NO_MEMBERSHIP:
99  case QGR_STATUS_INVALIDUSER:
100  case QGR_STATUS_ERROR:
101  if( (ent->r.svFlags & SVF_BOT) == 0 )
102  {
103  trap_RankUserReset( ent->s.clientNum );
104  }
105  break;
106  case QGR_STATUS_ACTIVE:
107  if( (ent->client->sess.sessionTeam == TEAM_SPECTATOR) &&
108  (g_gametype.integer < GT_TEAM) )
109  {
110  SetTeam( ent, "free" );
111  }
112 
113  if( old_status != QGR_STATUS_ACTIVE )
114  {
115  // player has just become active
116  for( j = 0; j < level.maxclients; j++ )
117  {
118  ent2 = &(g_entities[j]);
119  if ( !ent2->inuse )
120  continue;
121  if ( ent2->client == NULL )
122  continue;
123  if ( ent2->r.svFlags & SVF_BOT)
124  continue;
125 
126  if( (i != j) && (trap_RankUserStatus( j ) == QGR_STATUS_ACTIVE) )
127  {
128  trap_RankReportInt( i, j, QGR_KEY_PLAYED_WITH, 1, 0 );
129  }
130 
131  // send current scores so the player's rank will show
132  // up under the crosshair immediately
133  DeathmatchScoreboardMessage( ent2 );
134  }
135  }
136  break;
137  default:
138  break;
139  }
140  }
141 
142  // don't let ranked games last forever
143  if( ((g_fraglimit.integer == 0) || (g_fraglimit.integer > 100)) &&
144  ((g_timelimit.integer == 0) || (g_timelimit.integer > 1000)) )
145  {
146  trap_Cvar_Set( "timelimit", "1000" );
147  }
148  }
149 
150  // tell time to clients so they can show current match rating
151  if( level.intermissiontime == 0 )
152  {
153  for( i = 0; i < level.maxclients; i++ )
154  {
155  ent = &(g_entities[i]);
156  if( ent->client == NULL )
157  {
158  continue;
159  }
160 
161  time = (level.time - ent->client->pers.enterTime) / 1000;
162  ent->client->ps.persistant[PERS_MATCH_TIME] = time;
163  }
164  }
165 }

175 ================
176 G_RankDamage
177 ================
178 */
179 void G_RankDamage( int self, int attacker, int damage, int means_of_death )
180 {
181  // state information to avoid counting each shotgun pellet as a hit
182  static int last_framenum = -1;
183  static int last_self = -1;
184  static int last_attacker = -1;
185  static int last_means_of_death = MOD_UNKNOWN;
186 
187  qboolean new_hit;
188  int splash;
189  int key_hit;
190  int key_damage;
191  int key_splash;
192 
193  if( level.warmupTime != 0 )
194  {
195  // no reports during warmup period
196  return;
197  }
198 
199  new_hit = (level.framenum != last_framenum) ||
200  (self != last_self) ||
201  (attacker != last_attacker) ||
202  (means_of_death != last_means_of_death);
203 
204  // update state information
205  last_framenum = level.framenum;
206  last_self = self;
207  last_attacker = attacker;
208  last_means_of_death = means_of_death;
209 
210  // the gauntlet only "fires" when it actually hits something
211  if( (attacker != ENTITYNUM_WORLD) && (attacker != self) &&
212  (means_of_death == MOD_GAUNTLET) &&
213  (g_entities[attacker].client) )
214  {
215  trap_RankReportInt( attacker, -1, QGR_KEY_SHOT_FIRED_GAUNTLET, 1, 1 );
216  }
217 
218  // don't track hazard damage, just deaths
219  switch( means_of_death )
220  {
221  case MOD_CRUSH:
222  case MOD_TELEFRAG:
223  case MOD_FALLING:
224  case MOD_TRIGGER_HURT:
225  return;
226  default:
227  break;
228  }
229  splash = 0;
230  key_splash = -1;
231 
232  // hit, damage, and splash taken
233  key_hit = QGR_KEY_HIT_TAKEN_UNKNOWN;
234  key_damage = QGR_KEY_DAMAGE_TAKEN_UNKNOWN;
235  break;
236 
237  // report general and specific hit taken
238  if( new_hit )
239  {
240  trap_RankReportInt( self, -1, QGR_KEY_HIT_TAKEN, 1, 1 );
241  trap_RankReportInt( self, -1, key_hit, 1, 1 );
242  }
243 
244  // report general and specific damage taken
245  trap_RankReportInt( self, -1, QGR_KEY_DAMAGE_TAKEN, damage, 1 );
246  trap_RankReportInt( self, -1, key_damage, damage, 1 );
247 
248  // report general and specific splash taken
249  if( splash != 0 )
250  {
251  trap_RankReportInt( self, -1, QGR_KEY_SPLASH_TAKEN, splash, 1 );
252  trap_RankReportInt( self, -1, key_splash, splash, 1 );
253  }
254 
255  // hit, damage, and splash given
256  if( (attacker != ENTITYNUM_WORLD) && (attacker != self) )
257  {
258  key_hit = QGR_KEY_HIT_GIVEN_UNKNOWN;
259  key_damage = QGR_KEY_DAMAGE_GIVEN_UNKNOWN;
260  break;
261  // report general and specific hit given
262  // jwu 8/26/00
263  // had a case where attacker is 245 which is grnadeshooter attacker is
264  // g_entities index not necessarilly clientnum
265  if (g_entities[attacker].client) {
266  if( new_hit )
267  {
268  trap_RankReportInt( attacker, -1, QGR_KEY_HIT_GIVEN, 1, 1 );
269  trap_RankReportInt( attacker, -1, key_hit, 1, 1 );
270  }
271 
272  // report general and specific damage given
273  trap_RankReportInt( attacker, -1, QGR_KEY_DAMAGE_GIVEN, damage, 1 );
274  trap_RankReportInt( attacker, -1, key_damage, damage, 1 );
275 
276  // report general and specific splash given
277  if( splash != 0 )
278  {
279  trap_RankReportInt( attacker, -1, QGR_KEY_SPLASH_GIVEN, splash, 1 );
280  trap_RankReportInt( attacker, -1, key_splash, splash, 1 );
281  }
282  }
283  }
284 
285  // friendly fire
286  if( (attacker != self) &&
287  OnSameTeam( &(g_entities[self]), &(g_entities[attacker])) &&
288  (g_entities[attacker].client) )
289  {
290  // report teammate hit
291  if( new_hit )
292  {
293  trap_RankReportInt( self, -1, QGR_KEY_TEAMMATE_HIT_TAKEN, 1, 1 );
294  trap_RankReportInt( attacker, -1, QGR_KEY_TEAMMATE_HIT_GIVEN, 1,
295  1 );
296  }
297 
298  // report teammate damage
299  trap_RankReportInt( self, -1, QGR_KEY_TEAMMATE_DAMAGE_TAKEN, damage,
300  1 );
301  trap_RankReportInt( attacker, -1, QGR_KEY_TEAMMATE_DAMAGE_GIVEN,
302  damage, 1 );
303 
304  // report teammate splash
305  if( splash != 0 )
306  {
307  trap_RankReportInt( self, -1, QGR_KEY_TEAMMATE_SPLASH_TAKEN,
308  splash, 1 );
309  trap_RankReportInt( attacker, -1, QGR_KEY_TEAMMATE_SPLASH_GIVEN,
310  splash, 1 );
311  }
312  }
313 }
