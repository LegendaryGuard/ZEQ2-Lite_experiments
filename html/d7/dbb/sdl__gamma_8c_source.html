38 {
39  Uint16 table[3][256];
40  int i, j;
41 
42  if( !glConfig.deviceSupportsGamma || r_ignorehwgamma->integer )
43  return;
44 
45  for (i = 0; i < 256; i++)
46  {
47  table[0][i] = ( ( ( Uint16 ) red[i] ) << 8 ) | red[i];
48  table[1][i] = ( ( ( Uint16 ) green[i] ) << 8 ) | green[i];
49  table[2][i] = ( ( ( Uint16 ) blue[i] ) << 8 ) | blue[i];
50  }

57  OSVERSIONINFO vinfo;

62  {

69  table[ j ][ i ] = ( 128 + i ) << 8;
