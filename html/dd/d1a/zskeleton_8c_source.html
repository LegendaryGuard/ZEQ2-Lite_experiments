113 void LoadSMD(smdModel_t *smd, char modelName[32], char fileName[32]) {
114  int smdVer,i;
115  char smdFileName[512];
116 
117  strcpy(smdFileName,fileName);
118  // load file and check version
119  StripExtension(smdFileName);
120  DefaultExtension(smdFileName,".smd");
121  LoadScriptFile(smdFileName);
122  MatchToken("version");
123  ParseInt(&smdVer);
124  if(smdVer != SMD_VERSION) {
125  Error("LoadSMD: expected version %i, found version %i in %s\n",SMD_VERSION,smdVer,smdFileName);
126  }
127 
128  // set the model name
129  strncpy(smd->name,modelName,sizeof(smd->name));
130 
131  // parse through the file
132  while(GetToken(qtrue)) {
133  // process the next section
134  if(!strcmp(token,"nodes")) {
135  // bone names and hierarchy
136  LoadSMD_ParseNodes(smd);
137  } else if(!strcmp(token,"skeleton")) {
138  // animation data
139  LoadSMD_ParseSkeleton(smd);
140  } else if(!strcmp(token,"triangles")) {
141  // mesh data (optional)
142  LoadSMD_ParseTriangles(smd);
143  } else if(!strcmp(token,"end")) {
144  break;
145  } else {
146  if(config.debug) {
147  printf("LoadSMD: unknown section \"%s\" in %s, skipping\n",token,smdFileName);
148  }
149  break;
150  }
151  }
152  // children inherit parent's bone flags
153  for(i = 0; i < smd->numBones; i++) {
154  if((smd->bones[i].flags == 0) && (smd->bones[i].parent >= 0) && (smd->bones[smd->bones[i].parent].flags != 0)) {
155  smd->bones[i].flags = smd->bones[smd->bones[i].parent].flags;
156  }
157  }

223 void LoadSMD_ParseSkeleton(smdModel_t *smd) {
224  int numFrames = 0;
225 
226  if(config.verbose && !config.debug) {
227  printf(" frames\r");
228  }
229 
230  // allocate space and zero it out
231  smd->frames = (smdFrame_t *)calloc(MAX_FRAMES,sizeof(smdFrame_t) + (sizeof(smdBoneFrame_t) * smd->numBones));
232 
233  while(1) {
234  int i,frameNum = -1;
235  smdBoneFrame_t *bones;
236  smdFrame_t *frame;
237 
238  GetToken(qtrue); // "time"
239  if(!strcmp(token,"end")) {
240  break;
241  }
242 
243  if(numFrames >= MAX_FRAMES) {
244  Error("LoadSMD_ParseSkeleton: exceeded maximum of %i frames\n",MAX_FRAMES);
245  }
246 
247  ParseInt(&frameNum); // frame number
248  if(frameNum < 0) {
249  Error("LoadSMD_ParseSkeleton: invalid frame index %i\n",frameNum);
250  }
251 
252  // make sure space for the bones is properly allocated
253  frame = &smd->frames[frameNum];
254  frame->bones = (smdBoneFrame_t *)realloc(frame->bones,sizeof(smdBoneFrame_t) * smd->numBones);
255  bones = frame->bones;
256 
257  // clear bounding box for frame
258  ClearBounds(frame->bounds[0],frame->bounds[1]);
259 
260  for(i = 0; i < smd->numBones; i++) {
261  int boneNum,parent;
262  float rotation[3];
263  smdBoneFrame_t temp;
264 
265  GetToken(qtrue); // bone number
266  if(!strcmp(token,"time")) {
267  // FIXME: smd spec allows lerping of bones between unkeyed frames
268  Error("LoadSMD_ParseSkeleton: missing bone keys in frame %i\n",frameNum);
269  }
270 
271  boneNum = atoi(token);
272  if(boneNum < 0) {
273  Error("LoadSMD_ParseSkeleton: frame %i has reference to invalid bone index %i\n",frameNum,boneNum);
274  }
275 
276  Parse1DMatrix(3,temp.origin); // bone position
277  Parse1DMatrix(3,rotation); // bone rotation
278 
279  // calculate bone matrix
280  parent = smd->bones[boneNum].parent;
281  AnglesToMatrix(rotation,temp.matrix);
282  if(parent >= 0) {
283  // FIXME: assumes bones listed in order, smd spec allows bones to be in any order
284  MatrixMultiply(bones[parent].matrix,temp.matrix,bones[boneNum].matrix);
285  VectorRotate(temp.origin,bones[parent].matrix,bones[boneNum].origin);
286  VectorAdd(bones[parent].origin,bones[boneNum].origin,bones[boneNum].origin);
287  } else {
288  // this is a root bone, copy as-is, child of world
289  VectorCopy(temp.origin,bones[boneNum].origin);
290  VectorCopy(temp.matrix[0],bones[boneNum].matrix[0]);
291  VectorCopy(temp.matrix[1],bones[boneNum].matrix[1]);
292  VectorCopy(temp.matrix[2],bones[boneNum].matrix[2]);
293  }
294  }
295 
296  numFrames++;
297  }
298 
299  // update the counter and resize array to save heap space
300  smd->numFrames = numFrames;
301  smd->frames = (smdFrame_t *)realloc(smd->frames,(sizeof(smdFrame_t) + (sizeof(smdBoneFrame_t) * smd->numBones)) * smd->numFrames);
302 
303  if(config.verbose && !config.debug) {
304  printf(" \n");
305  }

308 void LoadSMD_ParseTriangles(smdModel_t *smd) {
309  int i,j,k,numSurfs = 0;
310  smdSurface_t *surfs;
311  smdBoneFrame_t *bones;
312 
313  if(config.verbose && !config.debug) {
314  printf(" meshes\r");
315  }
316 
317  // allocate space and zero it out
318  smd->surfs = (smdSurface_t *)calloc(1,sizeof(smdSurface_t) + (sizeof(smdTriangle_t) * MAX_TRIS) + (sizeof(smdVertex_t) * MAX_VERTS) + (sizeof(int) * MAX_VERTS));
319 
320  // reduce typing, my hands hurt...
321  surfs = smd->surfs;
322  bones = smd->frames[0].bones;
323 
324  while(1) {
325  int found = -1;
326  int curSurf = -1,vertIndex = -1;
327  smdVertex_t *verts;
328  smdTriangle_t *tris;
329  smdVertex_t temp;
330 
331  GetToken(qtrue); // surface name
332  if(!strcmp(token,"end")) {
333  break;
334  }
335 
336  // look for surf name to see if it exists
337  for(i = 0; i < numSurfs; i++) {
338  if(!strcmp(token,surfs[i].name)) {
339  found = i;
340  break;
341  }
342  }
343  // add surf if it wasn't found
344  if(found < 0) {
345  if(numSurfs >= MAX_SURFS) {
346  Error("LoadSMD_ParseTriangles: exceeded maximum of %i surfaces\n",MAX_SURFS);
347  }
348 
349  curSurf = numSurfs;
350  // expand surf data as we go
351  smd->surfs = (smdSurface_t *)realloc(smd->surfs,(sizeof(smdSurface_t) + (sizeof(smdTriangle_t) * MAX_TRIS) + (sizeof(smdVertex_t) * MAX_VERTS) + (sizeof(int) * MAX_VERTS)) * (numSurfs + 1));
352  surfs = smd->surfs;
353  strncpy(surfs[curSurf].name,token,sizeof(surfs[curSurf].name));
354  surfs[curSurf].id = ++numSurfs;
355  surfs[curSurf].numTris = 0;
356  surfs[curSurf].numVerts = 0;
357  surfs[curSurf].minLod = 0;
358  surfs[curSurf].tris = (smdTriangle_t *)realloc(surfs[curSurf].tris,sizeof(smdTriangle_t) * MAX_TRIS);
359  surfs[curSurf].verts = (smdVertex_t *)realloc(surfs[curSurf].verts,sizeof(smdVertex_t) * MAX_VERTS);
360  surfs[curSurf].map = (int *)realloc(surfs[curSurf].map,sizeof(int) * MAX_VERTS);
361  } else {
362  curSurf = found;
363 
364  if(surfs[curSurf].numTris >= MAX_TRIS) {
365  Error("LoadSMD_ParseTriangles: exceeded maximum of %i triangles per surface\n",MAX_TRIS);
366  }
367  }
368  tris = surfs[curSurf].tris;
369  verts = surfs[curSurf].verts;
370 
371  // parse tri with vert indexes
372  for(i = 0; i < 3; i++) {
373  GetToken(qtrue); // reference bone
374  temp.parent = atoi(token);
375  if(temp.parent < 0) {
376  Error("LoadSMD_ParseTriangles: invalid reference bone %i for vertex\n",temp.parent);
377  }
378 
379  Parse1DMatrix(3,temp.origin); // vertex position
380  Parse1DMatrix(3,temp.normal); // vertex normal
381  Parse1DMatrix(2,temp.texCoords); // vertex texture coords
382 
383  // half-life 2 smd explicitly lists each influence per vertex
384  if(GetToken(qfalse)) {
385  temp.numWeights = atoi(token); // number of influences
386  if(temp.numWeights < 1) {
387  Error("LoadSMD_ParseTriangles: invalid number of influences (%i) for vertex\n",temp.numWeights);
388  } else if(temp.numWeights > MAX_WEIGHTS) {
389  Error("LoadSMD_ParseTriangles: exceeded maximum of %i influences per vertex\n",MAX_WEIGHTS);
390  }
391 
392  for(j = 0; j < temp.numWeights; j++) {
393  ParseInt(&temp.weights[j].bone); // bone index
394  if(temp.weights[j].bone < 0) {
395  Error("LoadSMD_ParseTriangles: invalid bone index %i for influence %i on vertex\n",temp.weights[j].bone,j);
396  }
397  Parse1DMatrix(1,&temp.weights[j].weight); // influence weight
398  if(temp.weights[j].weight < 0) {
399  Error("LoadSMD_ParseTriangles: invalid weight %f for influence %i on vertex\n",temp.weights[j].weight,j);
400  }
401  }
402  } else {
403  // half-life 1 smd implies only 1 influence (the reference bone)
404  temp.numWeights = 1;
405  temp.weights[0].bone = temp.parent;
406  temp.weights[0].weight = 1.0f;
407  }
408 
409  // check for redundant vertices
410  found = -1;
411  for(j = 0; j < surfs[curSurf].numVerts; j++) {
412  if( (temp.parent != verts[j].parent) ||
413  (temp.numWeights != verts[j].numWeights) ||
414  !VectorCompare(temp.origin,verts[j].origin) ||
415  !VectorCompare(temp.normal,verts[j].normal) ||
416  (fabs(temp.texCoords[0] - verts[j].texCoords[0]) > 0.0001) ||
417  (fabs(temp.texCoords[1] - verts[j].texCoords[1]) > 0.0001) ) {
418  continue;
419  }
420  for(k = 0; k < temp.numWeights; k++) {
421  if( (temp.weights[k].bone != verts[j].weights[k].bone) ||
422  (fabs(temp.weights[k].weight - verts[j].weights[k].weight) > 0.0001) ) {
423  found = 0;
424  break;
425  } else {
426  found = 1;
427  }
428  }
429  if(!found) {
430  found = -1;
431  continue;
432  } else {
433  found = j;
434  break;
435  }
436  }
437 
438  // add vertex if it wasn't found
439  if(found < 0) {
440  if(surfs[curSurf].numVerts >= MAX_VERTS) {
441  Error("LoadSMD_ParseTriangles: exceeded maximum of %i vertices per surface\n",MAX_VERTS);
442  }
443 
444  vertIndex = surfs[curSurf].numVerts;
445 
446  VectorCopy(temp.origin,verts[vertIndex].origin);
447  VectorCopy(temp.normal,verts[vertIndex].normal);
448  verts[vertIndex].texCoords[0] = temp.texCoords[0];
449  verts[vertIndex].texCoords[1] = temp.texCoords[1];
450  verts[vertIndex].parent = temp.parent;
451  verts[vertIndex].numWeights = temp.numWeights;
452 
453  // calculate influence offsets
454  for(j = 0; j < verts[vertIndex].numWeights; j++) {
455  int boneNum;
456  float tempOrg[3];
457 
458  verts[vertIndex].weights[j].bone = temp.weights[j].bone;
459  verts[vertIndex].weights[j].weight = temp.weights[j].weight;
460 
461  boneNum = verts[vertIndex].weights[j].bone;
462  VectorSubtract(verts[vertIndex].origin,bones[boneNum].origin,tempOrg);
463  VectorInverseRotate(tempOrg,bones[boneNum].matrix,verts[vertIndex].weights[j].offset);
464  }
465 
466  surfs[curSurf].map[vertIndex] = vertIndex;
467  surfs[curSurf].numVerts++;
468  surfs[curSurf].minLod = surfs[curSurf].numVerts;
469  } else {
470  vertIndex = found;
471  }
472 
473  // update vertex index
474  tris[surfs[curSurf].numTris].verts[i] = vertIndex;
475  }
476  // add triangle and update counter
477  tris[surfs[curSurf].numTris].surf = surfs[curSurf].id;
478  smd->surfs[curSurf].numTris++;
479  }
480 
481  // update counter
482  smd->numSurfs = numSurfs;
483 
484  if(verbose && !debug) {
485  printf(" \n");
486  }
