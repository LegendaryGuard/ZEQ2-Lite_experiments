46  verts[ 1 ].st[ 0 ] = 1;
47  verts[ 1 ].st[ 1 ] = 0;
48  Vector4Copy( color, verts[ 1 ].modulate );

51  verts[ 2 ].st[ 0 ] = 0;
52  verts[ 2 ].st[ 1 ] = 0;
53  Vector4Copy( color, verts[ 2 ].modulate );
54

56  verts[ 3 ].st[ 0 ] = 0;
57  verts[ 3 ].st[ 1 ] = 1;
58  Vector4Copy( color, verts[ 3 ].modulate );
59

61

489 void CG_TrailFunc_SpiralBeam( centity_t *ent ) {

503

506 
507  if (!weaponGraphics->missileTrailSpiralShader) {
508  return;
509  }
510 
511  startTime = ent->trailTime;
512  ent->trailTime = cg.time;
513  t = startTime + step;
514 
515  // if object (e.g. grenade) is stationary, don't show tail
516  if ( es->pos.trType == TR_STATIONARY ) {
517  return;
518  }
519 
520  CG_TrailFunc_SpiralBeam_Helper ( es, ent, startTime, lastPos);
521

523

529  le->leType = LE_FADE_ALPHA;

531  le->endTime = t + 15000;
