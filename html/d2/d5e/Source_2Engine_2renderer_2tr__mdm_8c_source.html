70 static float ProjectRadius( float r, vec3_t location ) {
71  float pr;
72  float dist;
73  float c;
74  vec3_t p;
75  float projected[4];
76 
77  c = DotProduct( tr.viewParms.or.axis[0], tr.viewParms.or.origin );
78  dist = DotProduct( tr.viewParms.or.axis[0], location ) - c;
79 
80  if ( dist <= 0 ) {
81  return 0;
82  }
83 
84  p[0] = 0;
85  p[1] = Q_fabs( r );
86  p[2] = -dist;
87 
88  projected[0] = p[0] * tr.viewParms.projectionMatrix[0] +
89  p[1] * tr.viewParms.projectionMatrix[4] +
90  p[2] * tr.viewParms.projectionMatrix[8] +
91  tr.viewParms.projectionMatrix[12];
92 
93  projected[1] = p[0] * tr.viewParms.projectionMatrix[1] +
94  p[1] * tr.viewParms.projectionMatrix[5] +
95  p[2] * tr.viewParms.projectionMatrix[9] +
96  tr.viewParms.projectionMatrix[13];
97 
98  projected[2] = p[0] * tr.viewParms.projectionMatrix[2] +
99  p[1] * tr.viewParms.projectionMatrix[6] +
100  p[2] * tr.viewParms.projectionMatrix[10] +
101  tr.viewParms.projectionMatrix[14];
102 
103  projected[3] = p[0] * tr.viewParms.projectionMatrix[3] +
104  p[1] * tr.viewParms.projectionMatrix[7] +
105  p[2] * tr.viewParms.projectionMatrix[11] +
106  tr.viewParms.projectionMatrix[15];
107 
108 
109  pr = projected[1] / projected[3];
110 
111  if ( pr > 1.0f ) {
112  pr = 1.0f;
113  }
114 
115  return pr;
116 }

123 static int R_CullModel( trRefEntity_t *ent ) {
124  vec3_t bounds[2];
125  mdxHeader_t *oldFrameHeader, *newFrameHeader;
126  mdxFrame_t *oldFrame, *newFrame;
127  int i;
128 
129  newFrameHeader = R_GetModelByHandle( ent->e.frameModel )->mdx;
130  oldFrameHeader = R_GetModelByHandle( ent->e.oldframeModel )->mdx;
131 
132  if ( !newFrameHeader || !oldFrameHeader ) {
133  return CULL_OUT;
134  }
135 
136  // compute frame pointers
137  newFrame = ( mdxFrame_t * )( ( byte * ) newFrameHeader + newFrameHeader->ofsFrames +
138  ent->e.frame * (int) ( sizeof( mdxBoneFrameCompressed_t ) ) * newFrameHeader->numBones +
139  ent->e.frame * sizeof( mdxFrame_t ) );
140  oldFrame = ( mdxFrame_t * )( ( byte * ) oldFrameHeader + oldFrameHeader->ofsFrames +
141  ent->e.oldframe * (int) ( sizeof( mdxBoneFrameCompressed_t ) ) * oldFrameHeader->numBones +
142  ent->e.oldframe * sizeof( mdxFrame_t ) );
143 
144  // cull bounding sphere ONLY if this is not an upscaled entity
145  if ( !ent->e.nonNormalizedAxes ) {
146  if ( ent->e.frame == ent->e.oldframe && ent->e.frameModel == ent->e.oldframeModel ) {
147  switch ( R_CullLocalPointAndRadius( newFrame->localOrigin, newFrame->radius ) )
148  {
149  case CULL_OUT:
150  tr.pc.c_sphere_cull_md3_out++;
151  return CULL_OUT;
152 
153  case CULL_IN:
154  tr.pc.c_sphere_cull_md3_in++;
155  return CULL_IN;
156 
157  case CULL_CLIP:
158  tr.pc.c_sphere_cull_md3_clip++;
159  break;
160  }
161  } else
162  {
163  int sphereCull, sphereCullB;
164 
165  sphereCull = R_CullLocalPointAndRadius( newFrame->localOrigin, newFrame->radius );
166  if ( newFrame == oldFrame ) {
167  sphereCullB = sphereCull;
168  } else {
169  sphereCullB = R_CullLocalPointAndRadius( oldFrame->localOrigin, oldFrame->radius );
170  }
171 
172  if ( sphereCull == sphereCullB ) {
173  if ( sphereCull == CULL_OUT ) {
174  tr.pc.c_sphere_cull_md3_out++;
175  return CULL_OUT;
176  } else if ( sphereCull == CULL_IN ) {
177  tr.pc.c_sphere_cull_md3_in++;
178  return CULL_IN;
179  } else
180  {
181  tr.pc.c_sphere_cull_md3_clip++;
182  }
183  }
184  }
185  }
186 
187  // calculate a bounding box in the current coordinate system
188  for ( i = 0 ; i < 3 ; i++ ) {
189  bounds[0][i] = oldFrame->bounds[0][i] < newFrame->bounds[0][i] ? oldFrame->bounds[0][i] : newFrame->bounds[0][i];
190  bounds[1][i] = oldFrame->bounds[1][i] > newFrame->bounds[1][i] ? oldFrame->bounds[1][i] : newFrame->bounds[1][i];
191  }
192 
193  switch ( R_CullLocalBox( bounds ) )
194  {
195  case CULL_IN:
196  tr.pc.c_box_cull_md3_in++;
197  return CULL_IN;
198  case CULL_CLIP:
199  tr.pc.c_box_cull_md3_clip++;
200  return CULL_CLIP;
201  case CULL_OUT:
202  default:
203  tr.pc.c_box_cull_md3_out++;
204  return CULL_OUT;
205  }
206 }

214 static float R_CalcMDMLod( refEntity_t *refent, vec3_t origin, float radius, float modelBias, float modelScale ) {
215  float flod, lodScale;
216  float projectedRadius;
217 
218  // compute projected bounding sphere and use that as a criteria for selecting LOD
219 
220  projectedRadius = ProjectRadius( radius, origin );
221  if ( projectedRadius != 0 ) {
222 
223 // ri.Printf (PRINT_ALL, "projected radius: %f\n", projectedRadius);
224 
225  lodScale = r_lodscale->value; // fudge factor since MDS uses a much smoother method of LOD
226  flod = projectedRadius * lodScale * modelScale;
227  } else
228  {
229  // object intersects near view plane, e.g. view weapon
230  flod = 1.0f;
231  }
232 
233  if ( refent->reFlags & REFLAG_FORCE_LOD ) {
234  flod *= 0.5;
235  }
236 //----(SA) like reflag_force_lod, but separate for the moment
237  if ( refent->reFlags & REFLAG_DEAD_LOD ) {
238  flod *= 0.8;
239  }
240 
241  flod -= modelBias;
242 
243  if ( flod < 0.0 ) {
244  flod = 0.0;
245  } else if ( flod > 1.0f ) {
246  flod = 1.0f;
247  }
248 
249  return flod;
250 }

501 static void InglesToAxis( int ingles[ 3 ], vec3_t axis[ 3 ] ) {
502  // get sine/cosines for angles

504  cy = COS_TABLE( ingles[ YAW ] & 65535 );
505  sp = SIN_TABLE( ingles[ PITCH ] & 65535 );
506  cp = COS_TABLE( ingles[ PITCH ] & 65535 );
507  sr = SIN_TABLE( ingles[ ROLL ] & 65535 );
508  cr = COS_TABLE( ingles[ ROLL ] & 65535 );
509 
510  // calculate axis vecs
511  axis[ 0 ][ 0 ] = cp * cy;
512  axis[ 0 ][ 1 ] = cp * sy;
513  axis[ 0 ][ 2 ] = -sp;
514 
515  axis[ 1 ][ 0 ] = sr * sp * cy + cr * -sy;
516  axis[ 1 ][ 1 ] = sr * sp * sy + cr * cy;
517  axis[ 1 ][ 2 ] = sr * cp;
