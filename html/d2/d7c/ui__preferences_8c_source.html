145 static void Preferences_Event( void* ptr, int notification ) {
146  if(notification != QM_ACTIVATED){
147  return;
148  }
149  switch(((menucommon_s*)ptr)->id){
150  case ID_CROSSHAIR:
151  s_preferences.crosshair.curvalue++;
152  if( s_preferences.crosshair.curvalue == NUM_CROSSHAIRS ) {
153  s_preferences.crosshair.curvalue = 0;
154  }
155  trap_Cvar_SetValue( "cg_drawCrosshair", s_preferences.crosshair.curvalue );
156  break;
157  case ID_CROSSHAIRSIZE:
158  trap_Cvar_SetValue( "cg_crosshairSize", s_preferences.crosshairSize.curvalue);
159  break;
160  case ID_CROSSHAIRNAMES:
161  trap_Cvar_SetValue( "cg_drawCrosshairNames", s_preferences.crosshairNames.curvalue);
162  break;
163  case ID_CAMERASTYLE:
164  trap_Cvar_SetValue( "cg_thirdPersonCamera", s_preferences.camerastyle.curvalue );
165  break;
166  case ID_BEAMCONTROL:
167  trap_Cvar_SetValue( "cg_beamControl", s_preferences.beamcontrol.curvalue );
168  break;
169  case ID_BEAMDETAIL:
170  trap_Cvar_SetValue( "r_beamDetail", s_preferences.beamdetail.curvalue);
171  break;
172  case ID_PARTICLESQUALITY:
173  trap_Cvar_SetValue( "cg_particlesQuality", s_preferences.particlesQuality.curvalue );
174  break;
175  case ID_PARTICLESOPTIMISE:
176  trap_Cvar_SetValue( "cg_particlesStop", s_preferences.particlesOptimise.curvalue );
177  break;
178  case ID_PORTALS:
179  trap_Cvar_SetValue( "r_portals", s_preferences.portals.curvalue );
180  break;
181  case ID_OUTLINES:
182  trap_Cvar_SetValue( "r_outlines", s_preferences.outlines.curvalue );
183  break;
184  case ID_FLIGHT:
185  trap_Cvar_SetValue( "cg_advancedFlight", s_preferences.flight.curvalue );
186  break;
187  case ID_BLOOMQUALITY:
188  switch ( s_preferences.bloomQuality.curvalue )
189  {
190  case 0:
191  trap_Cvar_SetValue( "r_bloom_sample_size", 0 );
192  trap_Cvar_SetValue( "r_bloom", 0 );
193  break;
194  case 1:
195  trap_Cvar_SetValue( "r_bloom_sample_size", 64 );
196  trap_Cvar_SetValue( "r_bloom", 1 );
197  break;
198  case 2:
199  trap_Cvar_SetValue( "r_bloom_sample_size", 128 );
200  trap_Cvar_SetValue( "r_bloom", 1 );
201  break;
202  case 3:
203  trap_Cvar_SetValue( "r_bloom_sample_size", 256 );
204  trap_Cvar_SetValue( "r_bloom", 1 );
205  break;
206  case 4:
207  trap_Cvar_SetValue( "r_bloom_sample_size", 512 );
208  trap_Cvar_SetValue( "r_bloom", 1 );
209  break;
210  }
211  break;
212  case ID_BLOOMINTENSITY:
213  trap_Cvar_SetValue( "r_bloom_intensity", s_preferences.bloomIntensity.curvalue);
214  break;
215  case ID_BLOOMDARKEN:
216  switch ( s_preferences.bloomDarken.curvalue )
217  {
218  case 0:
219  trap_Cvar_SetValue( "r_bloom_darken", 16 );
220  break;
221  case 1:
222  trap_Cvar_SetValue( "r_bloom_darken", 32 );
223  break;
224  case 2:
225  trap_Cvar_SetValue( "r_bloom_darken", 64 );
226  break;
227  }
228  break;
229  case ID_BLOOMALPHA:
230  trap_Cvar_SetValue( "r_bloom_alpha", s_preferences.bloomAlpha.curvalue);
231  break;
232  case ID_CONTROLS:
233  UI_PopMenu();
234  UI_ControlsMenu();
235  break;
236  case ID_SYSTEM:
237  UI_PopMenu();
238  UI_SystemSettingsMenu();
239  break;
240  case ID_GENERAL:
241  break;
242  case ID_BACK2:
243  UI_PopMenu();
244  break;
245  }

250 static void Crosshair_Draw( void *self ) {
251  menulist_s *s;
252  float *color;
253  int x, y;
254  int style;
255  int crosshairSizeImage;
256  qboolean focus;
257  crosshairSizeImage = trap_Cvar_VariableValue( "cg_crosshairSize" ) * 8 + 8;
258  s = (menulist_s *)self;
259  x = s->generic.x;
260  y = s->generic.y;
261  style = UI_TINYFONT|UI_DROPSHADOW;
262  focus = (s->generic.parent->cursor == s->generic.menuPosition);
263  if(s->generic.flags & QMF_GRAYED){
264  color = text_color_disabled;
265  }
266  else if(focus){
267  color = text_color_highlight;
268  style |= UI_PULSE;
269  }
270  else if(s->generic.flags & QMF_BLINK){
271  color = text_color_highlight;
272  style |= UI_BLINK;
273  }
274  else{
275  color = text_color_normal;
276  }
277  if(focus){
278  // draw cursor
279  UI_FillRect( s->generic.left, s->generic.top, s->generic.right-s->generic.left+1, s->generic.bottom-s->generic.top+1, listbar_color );
280  UI_DrawChar( x, y, 13, UI_CENTER|UI_BLINK|UI_SMALLFONT|UI_DROPSHADOW, color);
281  }
282  UI_DrawString( x - SMALLCHAR_WIDTH, y, s->generic.name, style|UI_RIGHT, color );
283  if(!s->curvalue){return;}
284  UI_DrawHandlePic( x + SMALLCHAR_WIDTH + 8 - 0.5f * crosshairSizeImage, y + 6 - 0.5f * crosshairSizeImage, crosshairSizeImage, crosshairSizeImage, s_preferences.crosshairShader[s->curvalue] );
285 }
