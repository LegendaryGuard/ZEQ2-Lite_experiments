154 void RB_ShadowTessEnd( void ) {
155  int i;
156  int numTris;
157  vec3_t lightDir;
158  GLboolean rgba[4];
159 
160  // we can only do this if we have enough space in the vertex buffers
161  if ( tess.numVertexes >= SHADER_MAX_VERTEXES / 2 ) {
162  return;
163  }
164 
165  if ( glConfig.stencilBits < 4 ) {
166  return;
167  }
168 
169  VectorCopy( backEnd.currentEntity->lightDir, lightDir );
170 
171  // project vertexes away from light direction
172  for ( i = 0 ; i < tess.numVertexes ; i++ ) {
173  VectorMA( tess.xyz[i], -512, lightDir, tess.xyz[i+tess.numVertexes] );
174  }
175 
176  // decide which triangles face the light
177  Com_Memset( numEdgeDefs, 0, 4 * tess.numVertexes );
178 
179  numTris = tess.numIndexes / 3;
180  for ( i = 0 ; i < numTris ; i++ ) {
181  int i1, i2, i3;
182  vec3_t d1, d2, normal;
183  float *v1, *v2, *v3;
184  float d;
185 
186  i1 = tess.indexes[ i*3 + 0 ];
187  i2 = tess.indexes[ i*3 + 1 ];
188  i3 = tess.indexes[ i*3 + 2 ];
189 
190  v1 = tess.xyz[ i1 ];
191  v2 = tess.xyz[ i2 ];
192  v3 = tess.xyz[ i3 ];
193 
194  VectorSubtract( v2, v1, d1 );
195  VectorSubtract( v3, v1, d2 );
196  CrossProduct( d1, d2, normal );
197 
198  d = DotProduct( normal, lightDir );
199  if ( d > 0 ) {
200  facing[ i ] = 1;
201  } else {
202  facing[ i ] = 0;
203  }
204 
205  // create the edges
206  R_AddEdgeDef( i1, i2, facing[ i ] );
207  R_AddEdgeDef( i2, i3, facing[ i ] );
208  R_AddEdgeDef( i3, i1, facing[ i ] );
209  }
210 
211  // draw the silhouette edges
212 
213  GL_Bind( tr.whiteImage );
214  qglEnable( GL_CULL_FACE );
215  GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ZERO );
216  qglColor3f( 0.2f, 0.2f, 0.2f );

220  qglColorMask( GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE );

308 void RB_ProjectionShadowDeform( void ) {
309  float *xyz;
310  int i;
311  float h;
312  vec3_t ground;
313  vec3_t light;
314  float groundDist;
315  float d;
316  vec3_t lightDir;
317 
318  xyz = ( float * ) tess.xyz;
319 
320  ground[0] = backEnd.or.axis[0][2];
321  ground[1] = backEnd.or.axis[1][2];
322  ground[2] = backEnd.or.axis[2][2];
323 
324  groundDist = backEnd.or.origin[2] - backEnd.currentEntity->e.shadowPlane;
325 
326  VectorCopy( backEnd.currentEntity->lightDir, lightDir );
327  d = DotProduct( lightDir, ground );
328  // don't let the shadows get too long or go negative
329  if ( d < 0.5 ) {
330  VectorMA( lightDir, (0.5 - d), ground, lightDir );
331  d = DotProduct( lightDir, ground );
332  }
333  d = 1.0 / d;
334 
335  light[0] = lightDir[0] * d;
336  light[1] = lightDir[1] * d;
337  light[2] = lightDir[2] * d;
338 
339  for ( i = 0; i < tess.numVertexes; i++, xyz += 4 ) {
340  h = DotProduct( xyz, ground ) + groundDist;
341 
342  xyz[0] -= light[0] * h;
343  xyz[1] -= light[1] * h;
344  xyz[2] -= light[2] * h;
345  }
346 }
