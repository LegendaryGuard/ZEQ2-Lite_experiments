68 void R_LoadJPG(const char *filename, unsigned char **pic, int *width, int *height)
69 {
70  /* This struct contains the JPEG decompression parameters and pointers to
71  * working space (which is allocated as needed by the JPEG library).
72  */
73  struct jpeg_decompress_struct cinfo = {NULL};
74  /* We use our private extension JPEG error handler.
75  * Note that this struct must live as long as the main JPEG parameter
76  * struct, to avoid dangling-pointer problems.
77  */
78  /* This struct represents a JPEG error handler. It is declared separately
79  * because applications often want to supply a specialized error handler
80  * (see the second half of this file for an example). But here we just
81  * take the easy way out and use the standard error handler, which will
82  * print a message on stderr and call exit() if compression fails.
83  * Note that this struct must live as long as the main JPEG parameter
84  * struct, to avoid dangling-pointer problems.
85  */
86  struct jpeg_error_mgr jerr;
87  /* More stuff */
88  JSAMPARRAY buffer; /* Output row buffer */
89  unsigned int row_stride; /* physical row width in output buffer */
90  unsigned int pixelcount, memcount;
91  unsigned int sindex, dindex;
92  byte *out;
93  int len;
94  union {
95  byte *b;
96  void *v;
97  } fbuffer;
98  byte *buf;
99 
100  /* In this example we want to open the input file before doing anything else,
101  * so that the setjmp() error recovery below can assume the file is open.
102  * VERY IMPORTANT: use "b" option to fopen() if you are on a machine that
103  * requires it in order to read binary files.
104  */
105 
106  len = ri.FS_ReadFile ( ( char * ) filename, &fbuffer.v);
107  if (!fbuffer.b || len < 0) {
108  return;
109  }
110 
111  /* Step 1: allocate and initialize JPEG decompression object */
112 
113  /* We have to set up the error handler first, in case the initialization
114  * step fails. (Unlikely, but it could happen if you are out of memory.)
115  * This routine fills in the contents of struct jerr, and returns jerr's
116  * address which we place into the link field in cinfo.
117  */
118  cinfo.err = jpeg_std_error(&jerr);
119  cinfo.err->error_exit = R_JPGErrorExit;
120  cinfo.err->output_message = R_JPGOutputMessage;
121 
122  /* Now we can initialize the JPEG decompression object. */
123  jpeg_create_decompress(&cinfo);
124 
125  /* Step 2: specify data source (eg, a file) */
126 
127  jpeg_mem_src(&cinfo, fbuffer.b, len);
128 
129  /* Step 3: read file parameters with jpeg_read_header() */
130 
131  (void) jpeg_read_header(&cinfo, TRUE);
132  /* We can ignore the return value from jpeg_read_header since
133  * (a) suspension is not possible with the stdio data source, and
134  * (b) we passed TRUE to reject a tables-only JPEG file as an error.
135  * See libjpeg.doc for more info.
136  */
137 
138  /* Step 4: set parameters for decompression */
139 
140  /*
141  * Make sure it always converts images to RGB color space. This will
142  * automatically convert 8-bit greyscale images to RGB as well.
143  */
144  cinfo.out_color_space = JCS_RGB;
145 
146  /* Step 5: Start decompressor */
147 
148  (void) jpeg_start_decompress(&cinfo);
149  /* We can ignore the return value since suspension is not possible
150  * with the stdio data source.
151  */
152 
153  /* We may need to do some setup of our own at this point before reading
154  * the data. After jpeg_start_decompress() we have the correct scaled
155  * output image dimensions available, as well as the output colormap
156  * if we asked for color quantization.
157  * In this example, we need to make an output work buffer of the right size.
158  */
159  /* JSAMPLEs per row in output buffer */
160 
161  pixelcount = cinfo.output_width * cinfo.output_height;
162 
163  if(!cinfo.output_width || !cinfo.output_height
164  || ((pixelcount * 4) / cinfo.output_width) / 4 != cinfo.output_height
165  || pixelcount > 0x1FFFFFFF || cinfo.output_components != 3
166  )
167  {
168  // Free the memory to make sure we don't leak memory
169  ri.FS_FreeFile (fbuffer.v);
170  jpeg_destroy_decompress(&cinfo);
171 
172  ri.Error(ERR_DROP, "LoadJPG: %s has an invalid image format: %dx%d*4=%d, components: %d", filename,
173  cinfo.output_width, cinfo.output_height, pixelcount * 4, cinfo.output_components);
174  }
175 
176  memcount = pixelcount * 4;
177  row_stride = cinfo.output_width * cinfo.output_components;
178 
179  out = ri.Malloc(memcount);
180 
181  *width = cinfo.output_width;
182  *height = cinfo.output_height;
183 
184  /* Step 6: while (scan lines remain to be read) */
185  /* jpeg_read_scanlines(...); */
186 
187  /* Here we use the library's state variable cinfo.output_scanline as the
188  * loop counter, so that we don't have to keep track ourselves.
189  */
190  while (cinfo.output_scanline < cinfo.output_height) {
191  /* jpeg_read_scanlines expects an array of pointers to scanlines.
192  * Here the array is only one element long, but you could ask for
193  * more than one scanline at a time if that's more convenient.
194  */
195  buf = ((out+(row_stride*cinfo.output_scanline)));
196  buffer = &buf;
197  (void) jpeg_read_scanlines(&cinfo, buffer, 1);
198  }
199 
200  buf = out;
201 
202  // Expand from RGB to RGBA
203  sindex = pixelcount * cinfo.output_components;
204  dindex = memcount;
205 
206  do
207  {
208  buf[--dindex] = 255;
209  buf[--dindex] = buf[--sindex];
210  buf[--dindex] = buf[--sindex];
211  buf[--dindex] = buf[--sindex];
212  } while(sindex);
213 
214  *pic = out;
215 
216  /* Step 7: Finish decompression */
217 
218  jpeg_finish_decompress(&cinfo);
219  /* We can ignore the return value since suspension is not possible
220  * with the stdio data source.
221  */
222 
223  /* Step 8: Release JPEG decompression object */
224 
225  /* This is an important step since it will release a good deal of memory. */
226  jpeg_destroy_decompress(&cinfo);
227 
228  /* After finish_decompress, we can close the input file.
229  * Here we postpone it until after no more JPEG errors are possible,
230  * so as to simplify the setjmp error logic above. (Actually, I don't
231  * think that jpeg_destroy can do an error exit, but why assume anything...)
232  */
233  ri.FS_FreeFile (fbuffer.v);
234 
235  /* At this point you may want to check to see whether any corrupt-data
236  * warnings occurred (test whether jerr.pub.num_warnings is nonzero).
237  */
238 
239  /* And we're done! */
240 }

270 /*
271  * Empty the output buffer --- called whenever buffer fills up.
272  *
273  * In typical applications, this should write the entire output buffer
274  * (ignoring the current state of next_output_byte & free_in_buffer),
275  * reset the pointer & count to the start of the buffer, and return TRUE
276  * indicating that the buffer has been dumped.
277  *
278  * In applications that need to be able to suspend compression due to output
279  * overrun, a FALSE return indicates that the buffer cannot be emptied now.
280  * In this situation, the compressor will return to its caller (possibly with
281  * an indication that it has not accepted all the supplied scanlines). The
282  * application should resume compression after it has made more room in the
283  * output buffer. Note that there are substantial restrictions on the use of
284  * suspension --- see the documentation.
285  *
286  * When suspending, the compressor will back up to a convenient restart point
287  * (typically the start of the current MCU). next_output_byte & free_in_buffer
288  * indicate where the restart point will be if the current call returns FALSE.
289  * Data beyond this point will be regenerated after resumption, so do not
290  * write it out when emptying the buffer externally.
291  */
292

298  jpeg_destroy_compress(cinfo);
299

303

327 static void
328 jpegDest (j_compress_ptr cinfo, byte* outfile, int size)
329 {
330  my_dest_ptr dest;
331 
332  /* The destination object is made permanent so that multiple JPEG images
333  * can be written to the same file without re-executing jpeg_stdio_dest.
334  * This makes it dangerous to use this manager and a different destination
335  * manager serially with the same JPEG object, because their private object
336  * sizes may be different. Caveat programmer.
337  */
338  if (cinfo->dest == NULL) { /* first time for this JPEG object? */
339  cinfo->dest = (struct jpeg_destination_mgr *)
340  (*cinfo->mem->alloc_small) ((j_common_ptr) cinfo, JPOOL_PERMANENT,
341  sizeof(my_destination_mgr));
342  }
343 
344  dest = (my_dest_ptr) cinfo->dest;
345  dest->pub.init_destination = init_destination;
346  dest->pub.empty_output_buffer = empty_output_buffer;
347  dest->pub.term_destination = term_destination;
348  dest->outfile = outfile;
349  dest->size = size;
350 }

353 =================
354 SaveJPGToBuffer
355 
356 Encodes JPEG from image in image_buffer and writes to buffer.
357 Expects RGB input data
358 =================

360 size_t RE_SaveJPGToBuffer(byte *buffer, size_t bufSize, int quality,
361  int image_width, int image_height, byte *image_buffer, int padding)
362 {
363  struct jpeg_compress_struct cinfo;
364  struct jpeg_error_mgr jerr;
365  JSAMPROW row_pointer[1]; /* pointer to JSAMPLE row[s] */
366  my_dest_ptr dest;
367  int row_stride; /* physical row width in image buffer */
368  size_t outcount;
369 
370  /* Step 1: allocate and initialize JPEG compression object */
371  cinfo.err = jpeg_std_error(&jerr);
372  cinfo.err->error_exit = R_JPGErrorExit;
373  cinfo.err->output_message = R_JPGOutputMessage;
374 
375  /* Now we can initialize the JPEG compression object. */
376  jpeg_create_compress(&cinfo);
377 
378  /* Step 2: specify data destination (eg, a file) */
379  /* Note: steps 2 and 3 can be done in either order. */
380  jpegDest(&cinfo, buffer, bufSize);
381 
382  /* Step 3: set parameters for compression */
383  cinfo.image_width = image_width; /* image width and height, in pixels */
384  cinfo.image_height = image_height;
385  cinfo.input_components = 3; /* # of color components per pixel */
386  cinfo.in_color_space = JCS_RGB; /* colorspace of input image */
387 
388  jpeg_set_defaults(&cinfo);
389  jpeg_set_quality(&cinfo, quality, TRUE /* limit to baseline-JPEG values */);
390  /* If quality is set high, disable chroma subsampling */
391  if (quality >= 85) {
392  cinfo.comp_info[0].h_samp_factor = 1;
393  cinfo.comp_info[0].v_samp_factor = 1;
394  }
395 
396  /* Step 4: Start compressor */
397  jpeg_start_compress(&cinfo, TRUE);
398 
399  /* Step 5: while (scan lines remain to be written) */
400  /* jpeg_write_scanlines(...); */
401  row_stride = image_width * cinfo.input_components + padding; /* JSAMPLEs per row in image_buffer */
402 
403  while (cinfo.next_scanline < cinfo.image_height) {
404  /* jpeg_write_scanlines expects an array of pointers to scanlines.
405  * Here the array is only one element long, but you could pass
406  * more than one scanline at a time if that's more convenient.
407  */
408  row_pointer[0] = &image_buffer[((cinfo.image_height-1)*row_stride)-cinfo.next_scanline * row_stride];
409  (void) jpeg_write_scanlines(&cinfo, row_pointer, 1);
410  }
411 
412  /* Step 6: Finish compression */
413  jpeg_finish_compress(&cinfo);
414 
415  dest = (my_dest_ptr) cinfo.dest;
416  outcount = dest->size - dest->pub.free_in_buffer;
417 
418  /* Step 7: release JPEG compression object */
419  jpeg_destroy_compress(&cinfo);
420 
421  /* And we're done! */
422  return outcount;
423 }
