44 static void HSVtoRGB( float h, float s, float v, float rgb[3] )
45 {
46  int i;
47  float f;
48  float p, q, t;
49 
50  h *= 5;
51 
52  i = floor( h );
53  f = h - i;
54 
55  p = v * ( 1 - s );
56  q = v * ( 1 - s * f );
57  t = v * ( 1 - s * ( 1 - f ) );
58 
59  switch ( i )
60  {
61  case 0:
62  rgb[0] = v;
63  rgb[1] = t;
64  rgb[2] = p;
65  break;
66  case 1:
67  rgb[0] = q;
68  rgb[1] = v;
69  rgb[2] = p;
70  break;
71  case 2:
72  rgb[0] = p;
73  rgb[1] = v;
74  rgb[2] = t;
75  break;
76  case 3:
77  rgb[0] = p;
78  rgb[1] = q;
79  rgb[2] = v;
80  break;
81  case 4:
82  rgb[0] = t;
83  rgb[1] = p;
84  rgb[2] = v;
85  break;
86  case 5:
87  rgb[0] = v;
88  rgb[1] = p;
89  rgb[2] = q;
90  break;
91  }
92 }

100 static void R_ColorShiftLightingBytes( byte in[4], byte out[4] ) {
101  int shift, r, g, b;
102 
103  // shift the color data based on overbright range
104  shift = r_mapOverBrightBits->integer - tr.overbrightBits;
105 
106  // shift the data based on overbright range
107  r = in[0] << shift;
108  g = in[1] << shift;
109  b = in[2] << shift;
110 
111  // normalize by color instead of saturating to white
112  if ( ( r | g | b ) > 255 ) {
113  int max;
114 
115  max = r > g ? r : g;
116  max = max > b ? max : b;
117  r = r * 255 / max;
118  g = g * 255 / max;
119  b = b * 255 / max;
120  }
121 
122  out[0] = r;
123  out[1] = g;
124  out[2] = b;
125  out[3] = in[3];
126 }

134 #define LIGHTMAP_SIZE 128
135 static void R_LoadLightmaps( lump_t *l ) {
136  byte *buf, *buf_p;
137  int len;
138  byte image[LIGHTMAP_SIZE*LIGHTMAP_SIZE*4];
139  int i, j;
140  float maxIntensity = 0;
141  double sumIntensity = 0;
142 
143  len = l->filelen;
144  if ( !len ) {
145  return;
146  }
147  buf = fileBase + l->fileofs;
148 
149  // we are about to upload textures
150  R_SyncRenderThread();
151 
152  // create all the lightmaps
153  tr.numLightmaps = len / (LIGHTMAP_SIZE * LIGHTMAP_SIZE * 3);
154  if ( tr.numLightmaps == 1 ) {
155  //FIXME: HACK: maps with only one lightmap turn up fullbright for some reason.
156  //this avoids this, but isn't the correct solution.
157  tr.numLightmaps++;
158  }
159 
160  // if we are in r_vertexLight mode, we don't need the lightmaps at all
161  if ( r_vertexLight->integer || glConfig.hardwareType == GLHW_PERMEDIA2 ) {
162  return;
163  }
164 
165  tr.lightmaps = ri.Hunk_Alloc( tr.numLightmaps * sizeof(image_t *), h_low );
166  for ( i = 0 ; i < tr.numLightmaps ; i++ ) {
167  // expand the 24 bit on-disk to 32 bit
168  buf_p = buf + i * LIGHTMAP_SIZE*LIGHTMAP_SIZE * 3;
169 
170  if ( r_lightmap->integer == 2 )
171  { // color code by intensity as development tool (FIXME: check range)
172  for ( j = 0; j < LIGHTMAP_SIZE * LIGHTMAP_SIZE; j++ )
173  {
174  float r = buf_p[j*3+0];
175  float g = buf_p[j*3+1];
176  float b = buf_p[j*3+2];
177  float intensity;
178  float out[3] = {0.0, 0.0, 0.0};
179 
180  intensity = 0.33f * r + 0.685f * g + 0.063f * b;
181 
182  if ( intensity > 255 )
183  intensity = 1.0f;
184  else
185  intensity /= 255.0f;
186 
187  if ( intensity > maxIntensity )
188  maxIntensity = intensity;
189 
190  HSVtoRGB( intensity, 1.00, 0.50, out );
191 
192  image[j*4+0] = out[0] * 255;
193  image[j*4+1] = out[1] * 255;
194  image[j*4+2] = out[2] * 255;
195  image[j*4+3] = 255;
196 
197  sumIntensity += intensity;
198  }
199  } else {
200  for ( j = 0 ; j < LIGHTMAP_SIZE * LIGHTMAP_SIZE; j++ ) {
201  R_ColorShiftLightingBytes( &buf_p[j*3], &image[j*4] );
202  image[j*4+3] = 255;
203  }
204  }
205  tr.lightmaps[i] = R_CreateImage( va("*lightmap%d",i), image,
206  LIGHTMAP_SIZE, LIGHTMAP_SIZE, qfalse, qfalse, GL_CLAMP_TO_EDGE );
207  }
208 
209  if ( r_lightmap->integer == 2 ) {
210  ri.Printf( PRINT_ALL, "Brightest lightmap value: %d\n", ( int ) ( maxIntensity * 255 ) );
211  }
212 }

304 static void ParseFace( dsurface_t *ds, drawVert_t *verts, msurface_t *surf, int *indexes ) {
305  int i, j;
306  srfSurfaceFace_t *cv;
307  int numPoints, numIndexes;
308  int lightmapNum;
309  int sfaceSize, ofsIndexes;
310 
311  lightmapNum = LittleLong( ds->lightmapNum );
312 
313  // get fog volume
314  surf->fogIndex = LittleLong( ds->fogNum ) + 1;
315 
316  // get shader value
317  surf->shader = ShaderForShaderNum( ds->shaderNum, lightmapNum );
318  if ( r_singleShader->integer && !surf->shader->isSky ) {
319  surf->shader = tr.defaultShader;
320  }
321 
322  numPoints = LittleLong( ds->numVerts );
323  if (numPoints > MAX_FACE_POINTS) {
324  ri.Printf( PRINT_WARNING, "WARNING: MAX_FACE_POINTS exceeded: %i\n", numPoints);
325  numPoints = MAX_FACE_POINTS;
326  surf->shader = tr.defaultShader;
327  }
328 
329  numIndexes = LittleLong( ds->numIndexes );
330 
331  // create the srfSurfaceFace_t
332  sfaceSize = ( size_t ) &((srfSurfaceFace_t *)0)->points[numPoints];
333  ofsIndexes = sfaceSize;
334  sfaceSize += sizeof( int ) * numIndexes;
335 
336  cv = ri.Hunk_Alloc( sfaceSize, h_low );
337  cv->surfaceType = SF_FACE;
338  cv->numPoints = numPoints;
339  cv->numIndices = numIndexes;
340  cv->ofsIndices = ofsIndexes;
341 
342  verts += LittleLong( ds->firstVert );
343  for ( i = 0 ; i < numPoints ; i++ ) {
344  for ( j = 0 ; j < 3 ; j++ ) {
345  cv->points[i][j] = LittleFloat( verts[i].xyz[j] );
346  }
347  for ( j = 0 ; j < 2 ; j++ ) {
348  cv->points[i][3+j] = LittleFloat( verts[i].st[j] );
349  cv->points[i][5+j] = LittleFloat( verts[i].lightmap[j] );
350  }
351  R_ColorShiftLightingBytes( verts[i].color, (byte *)&cv->points[i][7] );
352  }
353 
354  indexes += LittleLong( ds->firstIndex );
355  for ( i = 0 ; i < numIndexes ; i++ ) {
356  ((int *)((byte *)cv + cv->ofsIndices ))[i] = LittleLong( indexes[ i ] );
357  }
358 
359  // take the plane information from the lightmap vector
360  for ( i = 0 ; i < 3 ; i++ ) {
361  cv->plane.normal[i] = LittleFloat( ds->lightmapVecs[2][i] );
362  }
363  cv->plane.dist = DotProduct( cv->points[0], cv->plane.normal );
364  SetPlaneSignbits( &cv->plane );
365  cv->plane.type = PlaneTypeForNormal( cv->plane.normal );
366 
367  surf->data = (surfaceType_t *)cv;
368 }

376 static void ParseMesh ( dsurface_t *ds, drawVert_t *verts, msurface_t *surf ) {
377  srfGridMesh_t *grid;
378  int i, j;
379  int width, height, numPoints;
380  drawVert_t points[MAX_PATCH_SIZE*MAX_PATCH_SIZE];
381  int lightmapNum;
382  vec3_t bounds[2];
383  vec3_t tmpVec;
384  static surfaceType_t skipData = SF_SKIP;
385 
386  lightmapNum = LittleLong( ds->lightmapNum );
387 
388  // get fog volume
389  surf->fogIndex = LittleLong( ds->fogNum ) + 1;
390 
391  // get shader value
392  surf->shader = ShaderForShaderNum( ds->shaderNum, lightmapNum );
393  if ( r_singleShader->integer && !surf->shader->isSky ) {
394  surf->shader = tr.defaultShader;
395  }
396 
397  // we may have a nodraw surface, because they might still need to
398  // be around for movement clipping
399  if ( s_worldData.shaders[ LittleLong( ds->shaderNum ) ].surfaceFlags & SURF_NODRAW ) {
400  surf->data = &skipData;
401  return;
402  }
403 
404  width = LittleLong( ds->patchWidth );
405  height = LittleLong( ds->patchHeight );
406 
407  verts += LittleLong( ds->firstVert );
408  numPoints = width * height;
409  for ( i = 0 ; i < numPoints ; i++ ) {
410  for ( j = 0 ; j < 3 ; j++ ) {
411  points[i].xyz[j] = LittleFloat( verts[i].xyz[j] );
412  points[i].normal[j] = LittleFloat( verts[i].normal[j] );
413  }
414  for ( j = 0 ; j < 2 ; j++ ) {
415  points[i].st[j] = LittleFloat( verts[i].st[j] );
416  points[i].lightmap[j] = LittleFloat( verts[i].lightmap[j] );
417  }
418  R_ColorShiftLightingBytes( verts[i].color, points[i].color );
419  }
420 
421  // pre-tesseleate
422  grid = R_SubdividePatchToGrid( width, height, points );
423  surf->data = (surfaceType_t *)grid;
424 
425  // copy the level of detail origin, which is the center
426  // of the group of all curves that must subdivide the same
427  // to avoid cracking
428  for ( i = 0 ; i < 3 ; i++ ) {
429  bounds[0][i] = LittleFloat( ds->lightmapVecs[0][i] );
430  bounds[1][i] = LittleFloat( ds->lightmapVecs[1][i] );
431  }
432  VectorAdd( bounds[0], bounds[1], bounds[1] );
433  VectorScale( bounds[1], 0.5f, grid->lodOrigin );
434  VectorSubtract( bounds[0], grid->lodOrigin, tmpVec );
435  grid->lodRadius = VectorLength( tmpVec );
436 }

573 
574 NOTE: never sync LoD through grid edges with merged points!
575 
576 FIXME: write generalized version that also avoids cracks between a patch and one that meets half way?
577 =================
578 */
579 void R_FixSharedVertexLodError_r( int start, srfGridMesh_t *grid1 ) {

581  srfGridMesh_t *grid2;
582 
583  for ( j = start; j < s_worldData.numsurfaces; j++ ) {

586  // if this surface is not a grid
587  if ( grid2->surfaceType != SF_GRID ) continue;
588  // if the LOD errors are already fixed for this patch
589  if ( grid2->lodFixed == 2 ) continue;
590  // grids in the same LOD group should have the exact same lod radius
591  if ( grid1->lodRadius != grid2->lodRadius ) continue;
592  // grids in the same LOD group should have the exact same lod origin
593  if ( grid1->lodOrigin[0] != grid2->lodOrigin[0] ) continue;
594  if ( grid1->lodOrigin[1] != grid2->lodOrigin[1] ) continue;
595  if ( grid1->lodOrigin[2] != grid2->lodOrigin[2] ) continue;
596  //
597  touch = qfalse;
598  for (n = 0; n < 2; n++) {
599  //
600  if (n) offset1 = (grid1->height-1) * grid1->width;
601  else offset1 = 0;
602  if (R_MergedWidthPoints(grid1, offset1)) continue;
603  for (k = 1; k < grid1->width-1; k++) {
604  for (m = 0; m < 2; m++) {
605 
606  if (m) offset2 = (grid2->height-1) * grid2->width;
607  else offset2 = 0;
608  if (R_MergedWidthPoints(grid2, offset2)) continue;
609  for ( l = 1; l < grid2->width-1; l++) {
610  //
611  if ( fabs(grid1->verts[k + offset1].xyz[0] - grid2->verts[l + offset2].xyz[0]) > .1) continue;
612  if ( fabs(grid1->verts[k + offset1].xyz[1] - grid2->verts[l + offset2].xyz[1]) > .1) continue;
613  if ( fabs(grid1->verts[k + offset1].xyz[2] - grid2->verts[l + offset2].xyz[2]) > .1) continue;
614  // ok the points are equal and should have the same lod error
615  grid2->widthLodError[l] = grid1->widthLodError[k];
616  touch = qtrue;
617  }
618  }
619  for (m = 0; m < 2; m++) {
620 
621  if (m) offset2 = grid2->width-1;
622  else offset2 = 0;
623  if (R_MergedHeightPoints(grid2, offset2)) continue;
624  for ( l = 1; l < grid2->height-1; l++) {
625  //
626  if ( fabs(grid1->verts[k + offset1].xyz[0] - grid2->verts[grid2->width * l + offset2].xyz[0]) > .1) continue;
627  if ( fabs(grid1->verts[k + offset1].xyz[1] - grid2->verts[grid2->width * l + offset2].xyz[1]) > .1) continue;
628  if ( fabs(grid1->verts[k + offset1].xyz[2] - grid2->verts[grid2->width * l + offset2].xyz[2]) > .1) continue;
629  // ok the points are equal and should have the same lod error
630  grid2->heightLodError[l] = grid1->widthLodError[k];
631  touch = qtrue;
632  }
633  }
634  }
635  }
636  for (n = 0; n < 2; n++) {
637  //
638  if (n) offset1 = grid1->width-1;
639  else offset1 = 0;
640  if (R_MergedHeightPoints(grid1, offset1)) continue;
641  for (k = 1; k < grid1->height-1; k++) {
642  for (m = 0; m < 2; m++) {
643
