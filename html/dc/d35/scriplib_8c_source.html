105 qboolean EndOfScript (qboolean crossline) {
106  if (!crossline)
107  Error ("Line %i is incomplete\n",scriptline);
108 
109  if (!strcmp (script->filename, "memory buffer")) {
110  endofscript = qtrue;
111  return qfalse;
112  }
113 
114  free (script->buffer);
115  if (script == scriptstack+1) {
116  endofscript = qtrue;
117  return qfalse;
118  }
119  script--;
120  scriptline = script->line;

125 /*
126 ==============
127 GetToken
128 ==============
129 */
130 qboolean GetToken (qboolean crossline) {
131  char *token_p;
132 
133  if (tokenready) {
134  // is a token already waiting?
135  tokenready = qfalse;
136  return qtrue;
137  }
138 
139  if (script->script_p >= script->end_p)
140  return EndOfScript (crossline);
141 
142 //
143 // skip space
144 //
145 skipspace:
146  while (*script->script_p <= 32) {
147  if (script->script_p >= script->end_p)
148  return EndOfScript (crossline);
149  if (*script->script_p++ == '\n') {
150  if (!crossline)
151  Error ("Line %i is incomplete\n",scriptline);
152  scriptline = script->line++;
153  }
154  }
155 
156  if (script->script_p >= script->end_p)
157  return EndOfScript (crossline);
158 
159  // ; # // comments
160  if (*script->script_p == ';' || *script->script_p == '#'
161  || ( script->script_p[0] == '/' && script->script_p[1] == '/') ) {
162  if (!crossline)
163  Error ("Line %i is incomplete\n",scriptline);
164  while (*script->script_p++ != '\n')
165  if (script->script_p >= script->end_p)
166  return EndOfScript (crossline);
167  scriptline = script->line++;
168  goto skipspace;
169  }
170 
171  // /* */ comments
172  if (script->script_p[0] == '/' && script->script_p[1] == '*') {
173  if (!crossline)
174  Error ("Line %i is incomplete\n",scriptline);
175  script->script_p+=2;
176  while (script->script_p[0] != '*' && script->script_p[1] != '/') {
177  if ( *script->script_p == '\n' ) {
178  scriptline = script->line++;
179  }
180  script->script_p++;
181  if (script->script_p >= script->end_p)
182  return EndOfScript (crossline);
183  }
184  script->script_p += 2;
185  goto skipspace;
186  }
187 
188 //
189 // copy token
190 //
191  token_p = token;
192 
193  if (*script->script_p == '"') {
194  // quoted token
195  script->script_p++;
196  while (*script->script_p != '"') {
197  *token_p++ = *script->script_p++;
198  if (script->script_p == script->end_p)
199  break;
200  if (token_p == &token[MAXTOKEN])
201  Error ("Token too large on line %i\n",scriptline);
202  }
203  script->script_p++;
204  } else {
205  // regular token
206  while ( *script->script_p > 32 && *script->script_p != ';') {
207  *token_p++ = *script->script_p++;
208  if (script->script_p == script->end_p)
209  break;
210  if (token_p == &token[MAXTOKEN])
211  Error ("Token too large on line %i\n",scriptline);
212  }
213  }
214 
215  *token_p = 0;
216 
217  return qtrue;
218 }
