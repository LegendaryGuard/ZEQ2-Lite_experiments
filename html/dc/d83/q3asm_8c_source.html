267 static void hashtable_add (hashtable_t *H, int hashvalue, void *datum)
268 {
269  hashchain_t *hc, **hb;
270 
271  hashvalue = (abs(hashvalue) % H->buckets);
272  hb = &(H->table[hashvalue]);
273  if (*hb == 0)
274  {
275  /* Empty bucket. Create new one. */
276  *hb = calloc(1, sizeof(**hb));
277  hc = *hb;
278  }
279  else
280  {
281  /* Get hc to point to last node in chain. */
282  for (hc = *hb; hc && hc->next; hc = hc->next);
283  hc->next = calloc(1, sizeof(*hc));
284  hc = hc->next;
285  }
286  hc->data = datum;
287  hc->next = 0;
288  return;
289 }

426 /*
427  Problem:
428  BYTE values are specified as signed decimal string. A properly functional
429  atoip() will cap large signed values at 0x7FFFFFFF. Negative word values are
430  often specified as very large decimal values by lcc. Therefore, values that
431  should be between 0x7FFFFFFF and 0xFFFFFFFF come out as 0x7FFFFFFF when using
432  atoi(). Bad.
433 
434  This function is one big evil hack to work around this problem.
435 */
436 static int atoiNoCap (const char *s)
437 {
438  INT64 l;
439  union {
440  unsigned int u;
441  signed int i;
442  } retval;
443 
444  l = atoi64(s);
445  /* Now smash to signed 32 bits accordingly. */
446  if (l < 0) {
447  retval.i = (int)l;
448  } else {
449  retval.u = (unsigned int)l;
450  }
451  return retval.i; /* <- union hackage. I feel dirty with this. -PH */
452 }

457 =============
458 HashString
459 =============
460 */
461 /* Default hash function of Kazlib 1.19, slightly modified. */
462 static unsigned int HashString (const char *key)
463 {
464  static unsigned long randbox[] = {
465  0x49848f1bU, 0xe6255dbaU, 0x36da5bdcU, 0x47bf94e9U,
466  0x8cbcce22U, 0x559fc06aU, 0xd268f536U, 0xe10af79aU,
467  0xc1af4d69U, 0x1d2917b5U, 0xec4c304dU, 0x9ee5016cU,
468  0x69232f74U, 0xfead7bb3U, 0xe9089ab6U, 0xf012f6aeU,
469  };
470 
471  const char *str = key;
472  unsigned int acc = 0;
473 
474  while (*str) {
475  acc ^= randbox[(*str + acc) & 0xf];
476  acc = (acc << 1) | (acc >> 31);
477  acc &= 0xffffffffU;
478  acc ^= randbox[((*str++ >> 4) + acc) & 0xf];
479  acc = (acc << 2) | (acc >> 30);
480  acc &= 0xffffffffU;
481  }
482  return abs(acc);
483 }

539 static void DefineSymbol( char *sym, int value ) {
540  /* Hand optimization by PhaethonH */
541  symbol_t *s;
542  char expanded[MAX_LINE_LENGTH];
543  int hash;
544 
545  if ( passNumber == 1 ) {
546  return;
547  }
548 
549  // add the file prefix to local symbols to guarantee unique
550  if ( sym[0] == '$' ) {
551  sprintf( expanded, "%s_%i", sym, currentFileIndex );
552  sym = expanded;
553  }
554 
555  hash = HashString( sym );
556 
557  if (hashtable_symbol_exists(symtable, hash, sym)) {
558  CodeError( "Multiple definitions for %s\n", sym );
559  return;
560  }
561 
562  s = malloc( sizeof( *s ) );
563  s->next = NULL;
564  s->name = copystring( sym );
565  s->hash = hash;
566  s->value = value;
567  s->segment = currentSegment;
568 
569  hashtable_add(symtable, hash, s);

571 /*
572  Hash table lookup already speeds up symbol lookup enormously.
573  We postpone sorting until end of pass 0.
574  Since we're not doing the insertion sort, lastSymbol should always
575  wind up pointing to the end of list.
576  This allows constant time for adding to the list.
577  -PH
578 */
579  if (symbols == 0) {
580  lastSymbol = symbols = s;
581  } else {
582  lastSymbol->next = s;
583  lastSymbol = s;
584  }
585 }

595 static int LookupSymbol( char *sym ) {
596  symbol_t *s;
597  char expanded[MAX_LINE_LENGTH];
598  int hash;
599  hashchain_t *hc;
600

602  return 0;

607  sprintf( expanded, "%s_%i", sym, currentFileIndex );
608  sym = expanded;
609  }
610 
611  hash = HashString( sym );
612 
613 /*
614  Hand optimization by PhaethonH
615 
616  Using a hash table with chain/bucket for lookups alone sped up q3asm by almost 3x for me.
617  -PH
618 */
619  for (hc = hashtable_get(symtable, hash); hc; hc = hc->next) {

627  passNumber = 0;
628  DefineSymbol( sym, 0 ); // so more errors aren't printed
629  passNumber = 1;
630  return 0;

643 static char *ExtractLine( char *data ) {
644 /* Goal:
645  Given a string `data', extract one text line into buffer `lineBuffer' that
646  is no longer than MAX_LINE_LENGTH characters long. Return value is
647  remainder of `data' that isn't part of `lineBuffer'.
648  -PH

652 
653  currentFileLine++;

656  token[0] = 0;
657  *lineBuffer = 0;

660  if (!*q) {
661  return NULL;
662  }
663 
664  for ( ; !((*p == 0) || (*p == '\n')); p++) /* nop */ ;
665

667  CodeError( "MAX_LINE_LENGTH" );
668  return data;
669  }
670 
671  memcpy( lineBuffer, data, (p - data) );
672  lineBuffer[(p - data)] = 0;
673  p += (*p == '\n') ? 1 : 0; /* Skip over final newline. */
674  return p;
675 }

727 /*
728 ==============
729 ParseExpression
730 ==============
731 */
732 static int ParseExpression(void) {
733  /* Hand optimization, PhaethonH */
734  int i, j;
735  char sym[MAX_LINE_LENGTH];
736  int v;
737 
738  /* Skip over a leading minus. */
739  for ( i = ((token[0] == '-') ? 1 : 0) ; i < MAX_LINE_LENGTH ; i++ ) {
740  if ( token[i] == '+' || token[i] == '-' || token[i] == 0 ) {
741  break;
742  }
743  }
744 
745  memcpy( sym, token, i );
746  sym[i] = 0;
747 
748  switch (*sym) { /* Resolve depending on first character. */
749 /* Optimizing compilers can convert cases into "calculated jumps". I think these are faster. -PH */
750  case '-':
751  case '0': case '1': case '2': case '3': case '4':
752  case '5': case '6': case '7': case '8': case '9':
753  v = atoiNoCap(sym);
754  break;
755  default:
756  v = LookupSymbol(sym);
757  break;
758  }
759 
760  // parse add / subtract offsets
761  while ( token[i] != 0 ) {
762  for ( j = i + 1 ; j < MAX_LINE_LENGTH ; j++ ) {
763  if ( token[j] == '+' || token[j] == '-' || token[j] == 0 ) {
764  break;
765  }
766  }
767 
768  memcpy( sym, token+i+1, j-i-1 );
769  sym[j-i-1] = 0;
770 
771  switch (token[i]) {
772  case '+':
773  v += atoiNoCap(sym);
774  break;
775  case '-':
776  v -= atoiNoCap(sym);
777  break;
778  }
779 
780  i = j;
781  }
782 
783  return v;
784 }

1166 
1167  Parse();

1169  return;
1170  }
1171 
1172  hash = HashString( token );
1173 
1174 /*
1175  Opcode search using hash table.
1176  Since the opcodes stays mostly fixed, this may benefit even more from a tree.
1177  Always with the tree :)

1179 */
1180  for (hc = hashtable_get(optable, hash); hc; hc = hc->next) {
1181  op = (sourceOps_t*)(hc->data);
1182  i = op - sourceOps;
1183  if ((hash == opcodesHash[i]) && (!strcmp(token, op->name))) {
1184  int opcode;
1185  int expression;
1186 
1187  if ( op->opcode == OP_UNDEF ) {
1188  CodeError( "Undefined opcode: %s\n", token );
1189  }
1190  if ( op->opcode == OP_IGNORE ) {
1191  return; // we ignore most conversions
1192  }
1193 
1194  // sign extensions need to check next parm
1195  opcode = op->opcode;
1196  if ( opcode == OP_SEX8 ) {
1197  Parse();
1198  if ( token[0] == '1' ) {
1199  opcode = OP_SEX8;
1200  } else if ( token[0] == '2' ) {
1201  opcode = OP_SEX16;
1202  } else {
1203  CodeError( "Bad sign extension: %s\n", token );
1204  return;
1205  }
1206  }
1207 
1208  // check for expression
1209  Parse();
1210  if ( token[0] && op->opcode != OP_CVIF
1211  && op->opcode != OP_CVFI ) {
1212  expression = ParseExpression();
1213 
1214  // code like this can generate non-dword block copies:
1215  // auto char buf[2] = " ";
1216  // we are just going to round up. This might conceivably
1217  // be incorrect if other initialized chars follow.
1218  if ( opcode == OP_BLOCK_COPY ) {
1219  expression = ( expression + 3 ) & ~3;
1220  }
1221 
1222  EmitByte( &segment[CODESEG], opcode );
1223  EmitInt( &segment[CODESEG], expression );
1224  } else {
1225  EmitByte( &segment[CODESEG], opcode );
1226  }
1227 
1228  instructionCount++;
1229  return;
1230  }
