131 static void mdfour_tail(byte *in, int n)
132 {
133  byte buf[128];

137  m->totalN += n;

158 static void mdfour_update(struct mdfour *md, byte *in, int n)
159 {
160  uint32_t M[16];
161 
162  m = md;
163 
164  if (n == 0) mdfour_tail(in, n);
165 
166  while (n >= 64) {
167  copy64(M, in);
168  mdfour64(M);
169  in += 64;
170  n -= 64;
171  m->totalN += 64;
172  }
173 
174  mdfour_tail(in, n);
175 }

189 {
