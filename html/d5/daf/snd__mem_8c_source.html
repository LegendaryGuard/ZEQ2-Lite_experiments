116 static void ResampleSfx( sfx_t *sfx, int inrate, int inwidth, byte *data, qboolean compressed ) {
117  int outcount;
118  int srcsample;
119  float stepscale;
120  int i;
121  int sample, samplefrac, fracstep;
122  int part;
123  sndBuffer *chunk;
124 
125  stepscale = (float)inrate / dma.speed; // this is usually 0.5, 1, or 2
126 
127  outcount = sfx->soundLength / stepscale;
128  sfx->soundLength = outcount;
129 
130  samplefrac = 0;
131  fracstep = stepscale * 256;
132  chunk = sfx->soundData;
133 
134  for (i=0 ; i<outcount ; i++)
135  {
136  srcsample = samplefrac >> 8;
137  samplefrac += fracstep;
138  if( inwidth == 2 ) {
139  sample = ( ((short *)data)[srcsample] );
140  } else {
141  sample = (int)( (unsigned char)(data[srcsample]) - 128) << 8;
142  }
143  part = (i&(SND_CHUNK_SIZE-1));
144  if (part == 0) {
145  sndBuffer *newchunk;
146  newchunk = SND_malloc();
147  if (chunk == NULL) {
148  sfx->soundData = newchunk;
149  } else {
150  chunk->next = newchunk;
151  }
152  chunk = newchunk;
153  }
154 
155  chunk->sndChunk[part] = sample;
156  }
157 }
