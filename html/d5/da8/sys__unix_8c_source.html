98 /*
99 ================
100 Sys_Milliseconds
101 ================
102 */
103 /* base time in seconds, that's our origin
104  timeval:tv_sec is an int:
105  assuming this wraps every 0x7fffffff - ~68 years since the Epoch (1970) - we're safe till 2038 */
106 unsigned long sys_timeBase = 0;
107 /* current time in ms, using sys_timeBase as origin
108  NOTE: sys_timeBase*1000 + curtime -> ms since the Epoch
109  0x7fffffff ms - ~24 days
110  although timeval:tv_usec is an int, I'm not sure wether it is actually used as an unsigned int
111  (which would affect the wrap period) */
112 int curtime;
113 int Sys_Milliseconds (void)
114 {
115  struct timeval tp;
116 
117  gettimeofday(&tp, NULL);
118 
119  if (!sys_timeBase)
120  {
121  sys_timeBase = tp.tv_sec;
122  return tp.tv_usec/1000;
123  }
124 
125  curtime = (tp.tv_sec - sys_timeBase)*1000 + tp.tv_usec/1000;
126 
127  return curtime;
128 }

233 {
234  FILE *fifo;
235  int result;
236  int fn;
237  struct stat buf;
238 
239  // if file already exists AND is a pipefile, remove it

244  if( result != 0 )
245  return NULL;

251  fcntl( fn, F_SETFL, O_NONBLOCK );
252  }
253

285 /*
286 ==================
287 Sys_ListFilteredFiles
288 ==================
289 */
290 void Sys_ListFilteredFiles( const char *basedir, char *subdirs, char *filter, char **list, int *numfiles )
291 {
292  char search[MAX_OSPATH], newsubdirs[MAX_OSPATH];
293  char filename[MAX_OSPATH];
294  DIR *fdir;
295  struct dirent *d;
296  struct stat st;
297 
298  if ( *numfiles >= MAX_FOUND_FILES - 1 ) {
299  return;
300  }
301 
302  if (strlen(subdirs)) {
303  Com_sprintf( search, sizeof(search), "%s/%s", basedir, subdirs );
304  }
305  else {
306  Com_sprintf( search, sizeof(search), "%s", basedir );
307  }
308 
309  if ((fdir = opendir(search)) == NULL) {
310  return;
311  }
312 
313  while ((d = readdir(fdir)) != NULL) {
314  Com_sprintf(filename, sizeof(filename), "%s/%s", search, d->d_name);
315  if (stat(filename, &st) == -1)
316  continue;
317 
318  if (st.st_mode & S_IFDIR) {
319  if (Q_stricmp(d->d_name, ".") && Q_stricmp(d->d_name, "..")) {
320  if (strlen(subdirs)) {
321  Com_sprintf( newsubdirs, sizeof(newsubdirs), "%s/%s", subdirs, d->d_name);
322  }
323  else {
324  Com_sprintf( newsubdirs, sizeof(newsubdirs), "%s", d->d_name);
325  }
326  Sys_ListFilteredFiles( basedir, newsubdirs, filter, list, numfiles );
327  }
328  }
329  if ( *numfiles >= MAX_FOUND_FILES - 1 ) {
330  break;
331  }
332  Com_sprintf( filename, sizeof(filename), "%s/%s", subdirs, d->d_name );
333  if (!Com_FilterPath( filter, filename, qfalse ))
334  continue;
335  list[ *numfiles ] = CopyString( filename );
336  (*numfiles)++;
337  }
338 
339  closedir(fdir);
340 }

342 /*
343 ==================
344 Sys_ListFiles
345 ==================
346 */
347 char **Sys_ListFiles( const char *directory, const char *extension, char *filter, int *numfiles, qboolean wantsubs )
348 {
349  struct dirent *d;
350  DIR *fdir;
351  qboolean dironly = wantsubs;
352  char search[MAX_OSPATH];
353  int nfiles;
354  char **listCopy;
355  char *list[MAX_FOUND_FILES];
356  int i;
357  struct stat st;
358 
359  int extLen;
360 
361  if (filter) {
362 
363  nfiles = 0;
364  Sys_ListFilteredFiles( directory, "", filter, list, &nfiles );
365 
366  list[ nfiles ] = NULL;
367  *numfiles = nfiles;
368 
369  if (!nfiles)
370  return NULL;
371 
372  listCopy = Z_Malloc( ( nfiles + 1 ) * sizeof( *listCopy ) );
373  for ( i = 0 ; i < nfiles ; i++ ) {
374  listCopy[i] = list[i];
375  }
376  listCopy[i] = NULL;
377 
378  return listCopy;
379  }
380 
381  if ( !extension)
382  extension = "";
383 
384  if ( extension[0] == '/' && extension[1] == 0 ) {
385  extension = "";
386  dironly = qtrue;
387  }
388 
389  extLen = strlen( extension );
390 
391  // search
392  nfiles = 0;
393 
394  if ((fdir = opendir(directory)) == NULL) {
395  *numfiles = 0;
396  return NULL;
397  }
398 
399  while ((d = readdir(fdir)) != NULL) {
400  Com_sprintf(search, sizeof(search), "%s/%s", directory, d->d_name);
401  if (stat(search, &st) == -1)
402  continue;
403  if ((dironly && !(st.st_mode & S_IFDIR)) ||
404  (!dironly && (st.st_mode & S_IFDIR)))
405  continue;
406 
407  if (*extension) {
408  if ( strlen( d->d_name ) < extLen ||
409  Q_stricmp(
410  d->d_name + strlen( d->d_name ) - extLen,
411  extension ) ) {
412  continue; // didn't match
413  }
414  }
415 
416  if ( nfiles == MAX_FOUND_FILES - 1 )
417  break;
418  list[ nfiles ] = CopyString( d->d_name );
419  nfiles++;
420  }
421 
422  list[ nfiles ] = NULL;
423 
424  closedir(fdir);
425 
426  // return a copy of the list
427  *numfiles = nfiles;
428 
429  if ( !nfiles ) {
430  return NULL;
431  }
432 
433  listCopy = Z_Malloc( ( nfiles + 1 ) * sizeof( *listCopy ) );
434  for ( i = 0 ; i < nfiles ; i++ ) {
435  listCopy[i] = list[i];
436  }
437  listCopy[i] = NULL;
438 
439  return listCopy;
440 }

462 /*
463 ==================
464 Sys_Sleep
465 
466 Block execution for msec or until input is recieved.
467 ==================
468 */
469 void Sys_Sleep( int msec )
470 {
471  if( msec == 0 )
472  return;
473 
474  if( stdinIsATTY )
475  {
476  fd_set fdset;
477 
478  FD_ZERO(&fdset);
479  FD_SET(STDIN_FILENO, &fdset);
480  if( msec < 0 )
481  {
482  select(STDIN_FILENO + 1, &fdset, NULL, NULL, NULL);
483  }
484  else
485  {
486  struct timeval timeout;
487 
488  timeout.tv_sec = msec/1000;
489  timeout.tv_usec = (msec%1000)*1000;
490  select(STDIN_FILENO + 1, &fdset, NULL, NULL, &timeout);
491  }
492  }
493  else
494  {
495  // With nothing to select() on, we can't wait indefinitely
496  if( msec < 0 )
497  msec = 10;
498 
499  usleep( msec * 1000 );
500  }
501 }

503 /*
504 ==============
505 Sys_ErrorDialog
506 
507 Display an error message
508 ==============
509 */
510 void Sys_ErrorDialog( const char *error )
511 {
512  char buffer[ 1024 ];
513  unsigned int size;
514  int f = -1;
515  const char *homepath = Cvar_VariableString( "fs_homepath" );
516  const char *gamedir = Cvar_VariableString( "fs_game" );
517  const char *fileName = "crashlog.txt";
518  char *ospath = FS_BuildOSPath( homepath, gamedir, fileName );
519 
520  Sys_Print( va( "%s\n", error ) );
521 
522 #ifndef DEDICATED
523  Sys_Dialog( DT_ERROR, va( "%s. See \"%s\" for details.", error, ospath ), "Error" );
524 #endif
525 
526  // Make sure the write path for the crashlog exists...
527  if( FS_CreatePath( ospath ) ) {
528  Com_Printf( "ERROR: couldn't create path '%s' for crash log.\n", ospath );
529  return;
530  }
531 
532  // We might be crashing because we maxed out the Quake MAX_FILE_HANDLES,
533  // which will come through here, so we don't want to recurse forever by
534  // calling FS_FOpenFileWrite()...use the Unix system APIs instead.
535  f = open( ospath, O_CREAT | O_TRUNC | O_WRONLY, 0640 );
536  if( f == -1 )
537  {
538  Com_Printf( "ERROR: couldn't open %s\n", fileName );
539  return;
540  }
541 
542  // We're crashing, so we don't care much if write() or close() fails.
543  while( ( size = CON_LogRead( buffer, sizeof( buffer ) ) ) > 0 ) {
544  if( write( f, buffer, size ) != size ) {
545  Com_Printf( "ERROR: couldn't fully write to %s\n", fileName );
546  break;
547  }
548  }
549 
550  close( f );
551 }

571 /*
572 ==============
573 Sys_AppendToExecBuffer
574 ==============
575 */
576 static void Sys_AppendToExecBuffer( const char *text )
577 {
578  size_t size = sizeof( execBuffer ) - ( execBufferPointer - execBuffer );
579  int length = strlen( text ) + 1;
580 
581  if( length > size || execArgc >= ARRAY_LEN( execArgv ) )
582  return;
583 
584  Q_strncpyz( execBufferPointer, text, size );
585  execArgv[ execArgc++ ] = execBufferPointer;
586 
587  execBufferPointer += length;
588 }

590 /*
591 ==============
592 Sys_Exec
593 ==============
594 */
595 static int Sys_Exec( void )
596 {
597  pid_t pid = fork( );
598 
599  if( pid < 0 )
600  return -1;
601 
602  if( pid )
603  {
604  // Parent
605  int exitCode;
606 
607  wait( &exitCode );
608 
609  return WEXITSTATUS( exitCode );
610  }
611  else
612  {
613  // Child
614  execvp( execArgv[ 0 ], execArgv );
615 
616  // Failed to execute
617  exit( -1 );
618 
619  return -1;
620  }
621 }

702 /*
703 ==============
704 Sys_Dialog
705 
706 Display a *nix dialog box
707 ==============
708 */
709 dialogResult_t Sys_Dialog( dialogType_t type, const char *message, const char *title )
710 {
711  typedef enum
712  {
713  NONE = 0,
714  ZENITY,
715  KDIALOG,
716  XMESSAGE,
717  NUM_DIALOG_PROGRAMS
718  } dialogCommandType_t;
719  typedef void (*dialogCommandBuilder_t)( dialogType_t, const char *, const char * );
720 
721  const char *session = getenv( "DESKTOP_SESSION" );
722  qboolean tried[ NUM_DIALOG_PROGRAMS ] = { qfalse };
723  dialogCommandBuilder_t commands[ NUM_DIALOG_PROGRAMS ] = { NULL };
724  dialogCommandType_t preferredCommandType = NONE;
725 
726  commands[ ZENITY ] = &Sys_ZenityCommand;
727  commands[ KDIALOG ] = &Sys_KdialogCommand;
728  commands[ XMESSAGE ] = &Sys_XmessageCommand;
729 
730  // This may not be the best way
731  if( !Q_stricmp( session, "gnome" ) )
732  preferredCommandType = ZENITY;
733  else if( !Q_stricmp( session, "kde" ) )
734  preferredCommandType = KDIALOG;
735 
736  while( 1 )
737  {
738  int i;
739 
740  for( i = NONE + 1; i < NUM_DIALOG_PROGRAMS; i++ )
741  {
742  if( preferredCommandType != NONE && preferredCommandType != i )
743  continue;
744 
745  if( !tried[ i ] )
746  {
747  int exitCode;
748 
749  commands[ i ]( type, message, title );
750  exitCode = Sys_Exec( );
751 
752  if( exitCode >= 0 )
753  {
754  switch( type )
755  {
756  case DT_YES_NO: return exitCode ? DR_NO : DR_YES;
757  case DT_OK_CANCEL: return exitCode ? DR_CANCEL : DR_OK;
758  default: return DR_OK;
759  }
760  }
761 
762  tried[ i ] = qtrue;
763 
764  // The preference failed, so start again in order
765  if( preferredCommandType != NONE )
766  {
767  preferredCommandType = NONE;
768  break;
769  }
770  }
771  }
772 
773  for( i = NONE + 1; i < NUM_DIALOG_PROGRAMS; i++ )
774  {
775  if( !tried[ i ] )
776  continue;
777  }
778 
779  break;
780  }
781 
782  Com_DPrintf( S_COLOR_YELLOW "WARNING: failed to show a dialog\n" );
783  return DR_OK;
784 }
