197 /*
198 ** GL_State
199 **
200 ** This routine is responsible for setting the most commonly changed state
201 ** in Q3.
202 */
203 void GL_State( unsigned long stateBits )
204 {
205  unsigned long diff = stateBits ^ glState.glStateBits;
206 
207  if ( !diff )
208  {
209  return;
210  }
211 
212  //
213  // check depthFunc bits
214  //
215  if ( diff & GLS_DEPTHFUNC_EQUAL )
216  {
217  if ( stateBits & GLS_DEPTHFUNC_EQUAL )
218  {
219  qglDepthFunc( GL_EQUAL );
220  }
221  else
222  {
223  qglDepthFunc( GL_LEQUAL );
224  }
225  }
226 
227  //
228  // check blend bits
229  //
230  if ( diff & ( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS ) )
231  {
232  GLenum srcFactor, dstFactor;
233 
234  if ( stateBits & ( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS ) )
235  {
236  switch ( stateBits & GLS_SRCBLEND_BITS )
237  {
238  case GLS_SRCBLEND_ZERO:
239  srcFactor = GL_ZERO;
240  break;
241  case GLS_SRCBLEND_ONE:
242  srcFactor = GL_ONE;
243  break;
244  case GLS_SRCBLEND_DST_COLOR:
245  srcFactor = GL_DST_COLOR;
246  break;
247  case GLS_SRCBLEND_ONE_MINUS_DST_COLOR:
248  srcFactor = GL_ONE_MINUS_DST_COLOR;
249  break;
250  case GLS_SRCBLEND_SRC_ALPHA:
251  srcFactor = GL_SRC_ALPHA;
252  break;
253  case GLS_SRCBLEND_ONE_MINUS_SRC_ALPHA:
254  srcFactor = GL_ONE_MINUS_SRC_ALPHA;
255  break;
256  case GLS_SRCBLEND_DST_ALPHA:
257  srcFactor = GL_DST_ALPHA;
258  break;
259  case GLS_SRCBLEND_ONE_MINUS_DST_ALPHA:
260  srcFactor = GL_ONE_MINUS_DST_ALPHA;
261  break;
262  case GLS_SRCBLEND_ALPHA_SATURATE:
263  srcFactor = GL_SRC_ALPHA_SATURATE;
264  break;
265  default:
266  srcFactor = GL_ONE; // to get warning to shut up
267  ri.Error( ERR_DROP, "GL_State: invalid src blend state bits" );
268  break;
269  }
270 
271  switch ( stateBits & GLS_DSTBLEND_BITS )
272  {
273  case GLS_DSTBLEND_ZERO:
274  dstFactor = GL_ZERO;
275  break;
276  case GLS_DSTBLEND_ONE:
277  dstFactor = GL_ONE;
278  break;
279  case GLS_DSTBLEND_SRC_COLOR:
280  dstFactor = GL_SRC_COLOR;
281  break;
282  case GLS_DSTBLEND_ONE_MINUS_SRC_COLOR:
283  dstFactor = GL_ONE_MINUS_SRC_COLOR;
284  break;
285  case GLS_DSTBLEND_SRC_ALPHA:
286  dstFactor = GL_SRC_ALPHA;
287  break;
288  case GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA:
289  dstFactor = GL_ONE_MINUS_SRC_ALPHA;
290  break;
291  case GLS_DSTBLEND_DST_ALPHA:
292  dstFactor = GL_DST_ALPHA;
293  break;
294  case GLS_DSTBLEND_ONE_MINUS_DST_ALPHA:
295  dstFactor = GL_ONE_MINUS_DST_ALPHA;
296  break;
297  default:
298  dstFactor = GL_ONE; // to get warning to shut up
299  ri.Error( ERR_DROP, "GL_State: invalid dst blend state bits" );
300  break;
301  }
302 
303  qglEnable( GL_BLEND );
304  qglBlendFunc( srcFactor, dstFactor );
305  }
306  else
307  {
308  qglDisable( GL_BLEND );
309  }
310  }
311 
312  //
313  // check depthmask
314  //
315  if ( diff & GLS_DEPTHMASK_TRUE )
316  {
317  if ( stateBits & GLS_DEPTHMASK_TRUE )
318  {
319  qglDepthMask( GL_TRUE );
320  }
321  else
322  {
323  qglDepthMask( GL_FALSE );
324  }
325  }
326 
327  //
328  // fill/line mode
329  //
330  if ( diff & GLS_POLYMODE_LINE )
331  {
332  if ( stateBits & GLS_POLYMODE_LINE )
333  {
334  qglPolygonMode( GL_FRONT_AND_BACK, GL_LINE );
335  }
336  else
337  {
338  qglPolygonMode( GL_FRONT_AND_BACK, GL_FILL );
339  }
340  }
341 
342  //
343  // depthtest
344  //
345  if ( diff & GLS_DEPTHTEST_DISABLE )
346  {
347  if ( stateBits & GLS_DEPTHTEST_DISABLE )
348  {
349  qglDisable( GL_DEPTH_TEST );
350  }
351  else
352  {
353  qglEnable( GL_DEPTH_TEST );
354  }
355  }
356 
357  //
358  // alpha test
359  //
360  if ( diff & GLS_ATEST_BITS )
361  {
362  switch ( stateBits & GLS_ATEST_BITS )
363  {
364  case 0:
365  qglDisable( GL_ALPHA_TEST );
366  break;
367  case GLS_ATEST_GT_0:
368  qglEnable( GL_ALPHA_TEST );
369  qglAlphaFunc( GL_GREATER, 0.0f );
370  break;
371  case GLS_ATEST_LT_80:
372  qglEnable( GL_ALPHA_TEST );
373  qglAlphaFunc( GL_LESS, 0.5f );
374  break;
375  case GLS_ATEST_GE_80:
376  qglEnable( GL_ALPHA_TEST );
377  qglAlphaFunc( GL_GEQUAL, 0.5f );
378  break;
379  default:
380  assert( 0 );
381  break;
382  }
383  }
384 
385  glState.glStateBits = stateBits;
386 }

427 /*
428 =================
429 RB_BeginDrawingView
430 
431 Any mirrored or portaled views have already been drawn, so prepare
432 to actually render the visible surfaces for this view
433 =================
434 */
435 void RB_BeginDrawingView (void) {
436  int clearBits = 0;
437 
438  // sync with gl if needed
439  if ( r_finish->integer == 1 && !glState.finishCalled ) {
440  qglFinish ();
441  glState.finishCalled = qtrue;
442  }
443  if ( r_finish->integer == 0 ) {
444  glState.finishCalled = qtrue;
445  }
446 
447  // we will need to change the projection matrix before drawing
448  // 2D images again
449  backEnd.projection2D = qfalse;
450 
451  // ensures that depth writes are enabled for the depth clear
452  GL_State( GLS_DEFAULT );
453  // clear relevant buffers
454  clearBits = GL_DEPTH_BUFFER_BIT;
455 
456  if ( r_measureOverdraw->integer || r_shadows->integer == 2 )
457  {
458  clearBits |= GL_STENCIL_BUFFER_BIT;
459  }
460  if ( (r_fastsky->integer && !( backEnd.refdef.rdflags & RDF_NOWORLDMODEL)) || r_zfar->value != -1 )
461  {
462  clearBits |= GL_COLOR_BUFFER_BIT; // FIXME: only if sky shaders have been used
463  qglClearColor( 0.0f, 0.3f, 1.0f, 1.0f ); // FIXME: get color of sky
464  }
465  qglClear( clearBits );
466 
467  //
468  // set the modelview matrix for the viewer
469  //
470  SetViewportAndScissor();
471 
472  if ( ( backEnd.refdef.rdflags & RDF_HYPERSPACE ) )
473  {
474  RB_Hyperspace();
475  return;
476  }
477  else
478  {
479  backEnd.isHyperspace = qfalse;
480  }
481 
482  glState.faceCulling = -1; // force face culling to set next time
483 
484  // we will only draw a sun if there was sky rendered in this view
485  backEnd.skyRenderedThisView = qfalse;
486 
487  // clip to the plane of the portal
488  if ( backEnd.viewParms.isPortal ) {
489  float plane[4];
490  double plane2[4];
491 
492  plane[0] = backEnd.viewParms.portalPlane.normal[0];
493  plane[1] = backEnd.viewParms.portalPlane.normal[1];
494  plane[2] = backEnd.viewParms.portalPlane.normal[2];
495  plane[3] = backEnd.viewParms.portalPlane.dist;
496 
497  plane2[0] = DotProduct (backEnd.viewParms.or.axis[0], plane);
498  plane2[1] = DotProduct (backEnd.viewParms.or.axis[1], plane);
499  plane2[2] = DotProduct (backEnd.viewParms.or.axis[2], plane);
500  plane2[3] = DotProduct (plane, backEnd.viewParms.or.origin) - plane[3];
501 
502  qglLoadMatrixf( s_flipMatrix );
503 
504  Matrix4Copy(s_flipMatrix, glState.currentModelViewMatrix);
505  Matrix4Multiply(glState.currentProjectionMatrix, glState.currentModelViewMatrix, glState.currentModelViewProjectionMatrix);
506 
507  qglClipPlane (GL_CLIP_PLANE0, plane2);
508  qglEnable (GL_CLIP_PLANE0);
509  } else {
510  qglDisable (GL_CLIP_PLANE0);
511  }
512 }

517 /*
518 ==================
519 RB_RenderDrawSurfList
520 ==================
521 */
522 void RB_RenderDrawSurfList( drawSurf_t *drawSurfs, int numDrawSurfs ) {
523  shader_t *shader, *oldShader;
524  int fogNum, oldFogNum;
525  int entityNum, oldEntityNum;
526  int dlighted, oldDlighted;
527  qboolean depthRange, oldDepthRange, isCrosshair, wasCrosshair;
528  int i;
529  drawSurf_t *drawSurf;
530  int oldSort;
531  float originalTime;
532 
533  // save original time for entity shader offsets
534  originalTime = backEnd.refdef.floatTime;
535 
536  // clear the z buffer, set the modelview, etc
537  RB_BeginDrawingView ();
538 
539  // draw everything
540  oldEntityNum = -1;
541  backEnd.currentEntity = &tr.worldEntity;
542  oldShader = NULL;
543  oldFogNum = -1;
544  oldDepthRange = qfalse;
545  wasCrosshair = qfalse;
546  oldDlighted = qfalse;
547  oldSort = -1;
548  depthRange = qfalse;
549 
550  backEnd.pc.c_surfaces += numDrawSurfs;
551 
552  for (i = 0, drawSurf = drawSurfs ; i < numDrawSurfs ; i++, drawSurf++) {
553  if ( drawSurf->sort == oldSort ) {
554  // fast path, same as previous sort
555  rb_surfaceTable[ *drawSurf->surface ]( drawSurf->surface );
556  continue;
557  }
558  oldSort = drawSurf->sort;
559  R_DecomposeSort( drawSurf->sort, &entityNum, &shader, &fogNum, &dlighted );
560 
561  //
562  // change the tess parameters if needed
563  // a "entityMergable" shader is a shader that can have surfaces from seperate
564  // entities merged into a single batch, like smoke and blood puff sprites
565  if (shader != oldShader || fogNum != oldFogNum || dlighted != oldDlighted
566  || ( entityNum != oldEntityNum && !shader->entityMergable ) ) {
567  if (oldShader != NULL) {
568  RB_EndSurface();
569  }
570  RB_BeginSurface( shader, fogNum );
571  oldShader = shader;
572  oldFogNum = fogNum;
573  oldDlighted = dlighted;
574  }
575 
576  //
577  // change the modelview matrix if needed
578  //
579  if ( entityNum != oldEntityNum ) {
580  depthRange = isCrosshair = qfalse;
581 
582  if ( entityNum != ENTITYNUM_WORLD ) {
583  backEnd.currentEntity = &backEnd.refdef.entities[entityNum];
584  backEnd.refdef.floatTime = originalTime - backEnd.currentEntity->e.shaderTime;
585  // we have to reset the shaderTime as well otherwise image animations start
586  // from the wrong frame
587  tess.shaderTime = backEnd.refdef.floatTime - tess.shader->timeOffset;
588 
589  // set up the transformation matrix
590  R_RotateForEntity( backEnd.currentEntity, &backEnd.viewParms, &backEnd.or );
591 
592  // set up the dynamic lighting if needed
593  if ( backEnd.currentEntity->needDlights ) {
594  R_TransformDlights( backEnd.refdef.num_dlights, backEnd.refdef.dlights, &backEnd.or );
595  }
596 
597  if(backEnd.currentEntity->e.renderfx & RF_DEPTHHACK)
598  {
599  // hack the depth range to prevent view model from poking into walls
600  depthRange = qtrue;
601 
602  if(backEnd.currentEntity->e.renderfx & RF_CROSSHAIR)
603  isCrosshair = qtrue;
604  }
605  } else {
606  backEnd.currentEntity = &tr.worldEntity;
607  backEnd.refdef.floatTime = originalTime;
608  backEnd.or = backEnd.viewParms.world;
609  // we have to reset the shaderTime as well otherwise image animations on
610  // the world (like water) continue with the wrong frame
611  tess.shaderTime = backEnd.refdef.floatTime - tess.shader->timeOffset;
612  R_TransformDlights( backEnd.refdef.num_dlights, backEnd.refdef.dlights, &backEnd.or );
613  }
614 
615  qglLoadMatrixf( backEnd.or.modelMatrix );
616 
617  Matrix4Copy(backEnd.or.modelMatrix, glState.currentModelViewMatrix);
618  Matrix4Multiply(glState.currentProjectionMatrix, glState.currentModelViewMatrix, glState.currentModelViewProjectionMatrix);
619 
620  //
621  // change depthrange. Also change projection matrix so first person weapon does not look like coming
622  // out of the screen.
623  //
624  if (oldDepthRange != depthRange || wasCrosshair != isCrosshair)
625  {
626  if (depthRange)
627  {
628  if(backEnd.viewParms.stereoFrame != STEREO_CENTER)
629  {
630  if(isCrosshair)
631  {
632  if(oldDepthRange)
633  {
634  // was not a crosshair but now is, change back proj matrix
635  qglMatrixMode(GL_PROJECTION);
636  qglLoadMatrixf(backEnd.viewParms.projectionMatrix);
637  qglMatrixMode(GL_MODELVIEW);
638 
639  Matrix4Copy(backEnd.viewParms.projectionMatrix, glState.currentProjectionMatrix);
640  Matrix4Multiply(glState.currentProjectionMatrix, glState.currentModelViewMatrix, glState.currentModelViewProjectionMatrix);
641  }
642  }
643  else
644  {
645  viewParms_t temp = backEnd.viewParms;
646 
647  R_SetupProjection(&temp, r_znear->value, qfalse);
648 
649  qglMatrixMode(GL_PROJECTION);
650  qglLoadMatrixf(temp.projectionMatrix);
651  qglMatrixMode(GL_MODELVIEW);
652 
653  Matrix4Copy(temp.projectionMatrix, glState.currentProjectionMatrix);
654  Matrix4Multiply(glState.currentProjectionMatrix, glState.currentModelViewMatrix, glState.currentModelViewProjectionMatrix);
655  }
656  }
657 
658  if(!oldDepthRange)
659  qglDepthRange (0, 0.3);
660  }
661  else
662  {
663  if(!wasCrosshair && backEnd.viewParms.stereoFrame != STEREO_CENTER)
664  {
665  qglMatrixMode(GL_PROJECTION);
666  qglLoadMatrixf(backEnd.viewParms.projectionMatrix);
667  qglMatrixMode(GL_MODELVIEW);
668 
669  Matrix4Copy(backEnd.viewParms.projectionMatrix, glState.currentProjectionMatrix);
670  Matrix4Multiply(glState.currentProjectionMatrix, glState.currentModelViewMatrix, glState.currentModelViewProjectionMatrix);
671  }
672 
673  qglDepthRange (0, 1);
674  }
675 
676  oldDepthRange = depthRange;
677  wasCrosshair = isCrosshair;
678  }
679 
680  oldEntityNum = entityNum;
681  }
682 
683  // add the triangles for this surface
684  rb_surfaceTable[ *drawSurf->surface ]( drawSurf->surface );
685  }
686 
687  backEnd.refdef.floatTime = originalTime;
688 
689  // draw the contents of the last shader batch
690  if (oldShader != NULL) {
691  RB_EndSurface();
692  }
693 
694  // go back to the world modelview matrix
695  qglLoadMatrixf( backEnd.viewParms.world.modelMatrix );
696 
697  Matrix4Copy(backEnd.viewParms.world.modelMatrix, glState.currentModelViewMatrix);
698  Matrix4Multiply(glState.currentProjectionMatrix, glState.currentModelViewMatrix, glState.currentModelViewProjectionMatrix);
699 
700  if ( depthRange ) {
701  qglDepthRange (0, 1);
702  }
703 
704 #if 1
705  RB_DrawSun();
706 #endif
707  // darken down any stencil shadows
708  RB_ShadowFinish();
709 
710  // add light flares on lights that aren't obscured
711  RB_RenderFlares();
712 
713  // <-- RiO_MotionBlur
714  RB_MotionBlur();
715  // -->
716 }
