238 /*
239 =================
240 ArenaServers_Compare
241 =================
242 */
243 static int QDECL ArenaServers_Compare( const void *arg1, const void *arg2 ) {
244  float f1;
245  float f2;
246  servernode_t* t1;
247  servernode_t* t2;
248 
249  t1 = (servernode_t *)arg1;
250  t2 = (servernode_t *)arg2;
251 
252  switch( g_sortkey ) {
253  case SORT_HOST:
254  return Q_stricmp( t1->hostname, t2->hostname );
255 
256  case SORT_MAP:
257  return Q_stricmp( t1->mapname, t2->mapname );
258 
259  case SORT_CLIENTS:
260  f1 = t1->maxclients - t1->numclients;
261  if( f1 < 0 ) {
262  f1 = 0;
263  }
264 
265  f2 = t2->maxclients - t2->numclients;
266  if( f2 < 0 ) {
267  f2 = 0;
268  }
269 
270  if( f1 < f2 ) {
271  return 1;
272  }
273  if( f1 == f2 ) {
274  return 0;
275  }
276  return -1;
277 
278  case SORT_PING:
279  if( t1->pingtime < t2->pingtime ) {
280  return -1;
281  }
282  if( t1->pingtime > t2->pingtime ) {
283  return 1;
284  }
285  return Q_stricmp( t1->hostname, t2->hostname );
286  }
287 
288  return 0;
289 }

766 /*
767 =================
768 ArenaServers_DoRefresh
769 =================
770 */
771 static void ArenaServers_DoRefresh( void )
772 {
773  int i;
774  int j;
775  int time;
776  int maxPing;
777  char adrstr[MAX_ADDRESSLENGTH];
778  char info[MAX_INFO_STRING];
779 
780  if (uis.realtime < g_arenaservers.refreshtime)
781  {
782  if (g_servertype != UIAS_FAVORITES) {
783  if (g_servertype == UIAS_LOCAL) {
784  if (!trap_LAN_GetServerCount(AS_LOCAL)) {
785  return;
786  }
787  }
788  if (trap_LAN_GetServerCount(ArenaServers_SourceForLAN()) < 0) {
789  // still waiting for response
790  return;
791  }
792  }
793  }
794 
795  if (uis.realtime < g_arenaservers.nextpingtime)
796  {
797  // wait for time trigger
798  return;
799  }
800 
801  // trigger at 10Hz intervals
802  g_arenaservers.nextpingtime = uis.realtime + 10;
803 
804  // process ping results
805  maxPing = ArenaServers_MaxPing();
806  for (i=0; i<MAX_PINGREQUESTS; i++)
807  {
808  trap_LAN_GetPing( i, adrstr, MAX_ADDRESSLENGTH, &time );
809  if (!adrstr[0])
810  {
811  // ignore empty or pending pings
812  continue;
813  }
814 
815  // find ping result in our local list
816  for (j=0; j<MAX_PINGREQUESTS; j++)
817  if (!Q_stricmp( adrstr, g_arenaservers.pinglist[j].adrstr ))
818  break;
819 
820  if (j < MAX_PINGREQUESTS)
821  {
822  // found it
823  if (!time)
824  {
825  time = uis.realtime - g_arenaservers.pinglist[j].start;
826  if (time < maxPing)
827  {
828  // still waiting
829  continue;
830  }
831  }
832 
833  if (time > maxPing)
834  {
835  // stale it out
836  info[0] = '\0';
837  time = maxPing;
838  }
839  else
840  {
841  trap_LAN_GetPingInfo( i, info, MAX_INFO_STRING );
842  }
843 
844  // insert ping results
845  ArenaServers_Insert( adrstr, info, time );
846 
847  // clear this query from internal list
848  g_arenaservers.pinglist[j].adrstr[0] = '\0';
849  }
850 
851  // clear this query from external list
852  trap_LAN_ClearPing( i );
853  }
854 
855  // get results of servers query
856  // counts can increase as servers respond
857  if (g_servertype == UIAS_FAVORITES) {
858  g_arenaservers.numqueriedservers = g_arenaservers.numfavoriteaddresses;
859  } else {
860  g_arenaservers.numqueriedservers = trap_LAN_GetServerCount(ArenaServers_SourceForLAN());
861  }
862 
863 // if (g_arenaservers.numqueriedservers > g_arenaservers.maxservers)
864 // g_arenaservers.numqueriedservers = g_arenaservers.maxservers;
865 
866  // send ping requests in reasonable bursts
867  // iterate ping through all found servers
868  for (i=0; i<MAX_PINGREQUESTS && g_arenaservers.currentping < g_arenaservers.numqueriedservers; i++)
869  {
870  if (trap_LAN_GetPingQueueCount() >= MAX_PINGREQUESTS)
871  {
872  // ping queue is full
873  break;
874  }
875 
876  // find empty slot
877  for (j=0; j<MAX_PINGREQUESTS; j++)
878  if (!g_arenaservers.pinglist[j].adrstr[0])
879  break;
880 
881  if (j >= MAX_PINGREQUESTS)
882  // no empty slots available yet - wait for timeout
883  break;
884 
885  // get an address to ping
886 
887  if (g_servertype == UIAS_FAVORITES) {
888  strcpy( adrstr, g_arenaservers.favoriteaddresses[g_arenaservers.currentping] );
889  } else {
890  trap_LAN_GetServerAddressString(ArenaServers_SourceForLAN(), g_arenaservers.currentping, adrstr, MAX_ADDRESSLENGTH );
891  }
892 
893  strcpy( g_arenaservers.pinglist[j].adrstr, adrstr );
894  g_arenaservers.pinglist[j].start = uis.realtime;
895 
896  trap_Cmd_ExecuteText( EXEC_NOW, va( "ping %s\n", adrstr ) );
897 
898  // advance to next server
899  g_arenaservers.currentping++;
900  }
901 
902  if (!trap_LAN_GetPingQueueCount())
903  {
904  // all pings completed
905  ArenaServers_StopRefresh();
906  return;
907  }
908 
909  // update the user interface with ping status
910  ArenaServers_UpdateMenu();
911 }
