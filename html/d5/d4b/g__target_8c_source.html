125 /*QUAKED target_speaker (1 0 0) (-8 -8 -8) (8 8 8) looped-on looped-off global activator
126 "noise" wav file to play
127 
128 A global sound will play full volume throughout the level.
129 Activator sounds will play on the player that activated the target.
130 Global and activator sounds can't be combined with looping.
131 Normal sounds play each time the target is used.
132 Looped sounds will be toggled by use functions.
133 Multiple identical looping sounds will just increase volume without any speed cost.
134 "wait" : Seconds between auto triggerings, 0 = don't auto trigger
135 "random" wait variance, default is 0
136 */
137 void Use_Target_Speaker (gentity_t *ent, gentity_t *other, gentity_t *activator) {
138  if (ent->spawnflags & 3) { // looping sound toggles
139  if (ent->s.loopSound)
140  ent->s.loopSound = 0; // turn it off
141  else
142  ent->s.loopSound = ent->noise_index; // start it
143  }else { // normal sound
144  if ( ent->spawnflags & 8 ) {
145  G_AddEvent( activator, EV_GENERAL_SOUND, ent->noise_index );
146  } else if (ent->spawnflags & 4) {
147  G_AddEvent( ent, EV_GLOBAL_SOUND, ent->noise_index );
148  } else {
149  G_AddEvent( ent, EV_GENERAL_SOUND, ent->noise_index );
150  }
151  }
152 }

154 void SP_target_speaker( gentity_t *ent ) {
155  char buffer[MAX_QPATH];
156  char *s;
157 
158  G_SpawnFloat( "wait", "0", &ent->wait );
159  G_SpawnFloat( "random", "0", &ent->random );
160 
161  if ( !G_SpawnString( "noise", "NOSOUND", &s ) ) {
162  G_Error( "target_speaker without a noise key at %s", vtos( ent->s.origin ) );
163  }
164 
165  // force all client reletive sounds to be "activator" speakers that
166  // play on the entity that activates it
167  if ( s[0] == '*' ) {
168  ent->spawnflags |= 8;
169  }
170 
171  if (!strstr( s, ".ogg" )) {
172  Com_sprintf (buffer, sizeof(buffer), "%s.ogg", s );
173  } else {
174  Q_strncpyz( buffer, s, sizeof(buffer) );
175  }
176  ent->noise_index = G_SoundIndex(buffer);
177 
178  // a repeating speaker can be done completely client side
179  ent->s.eType = ET_SPEAKER;
180  ent->s.eventParm = ent->noise_index;
181  ent->s.frame = ent->wait * 10;
182  ent->s.clientNum = ent->random * 10;
183 
184 
185  // check for prestarted looping sound
186  if ( ent->spawnflags & 1 ) {
187  ent->s.loopSound = ent->noise_index;
188  }
189 
190  ent->use = Use_Target_Speaker;
191 
192  if (ent->spawnflags & 4) {
193  ent->r.svFlags |= SVF_BROADCAST;
194  }
195 
196  VectorCopy( ent->s.origin, ent->s.pos.trBase );
197 
198  // must link the entity so we get areas and clusters so
199  // the server can determine who to send updates to
200  trap_LinkEntity( ent );
201 }
