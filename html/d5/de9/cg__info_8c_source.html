69 void CG_LoadingClient( int clientNum ) {
70  const char *info;
71  char *skin;
72  char name[32];
73  char personality[MAX_QPATH];
74  char model[MAX_QPATH];
75  char iconName[MAX_QPATH];
76 
77  info = CG_ConfigString( CS_PLAYERS + clientNum );
78 
79  if ( loadingPlayerIconCount < MAX_LOADING_PLAYER_ICONS ) {
80  Q_strncpyz( name, Info_ValueForKey( info, "n" ), sizeof(name) );
81  Q_strncpyz( model, Info_ValueForKey( info, "model" ), sizeof( model ) );
82  skin = strrchr( model, '/' );
83  if ( skin ) {
84  *skin++ = '\0';
85  } else {
86  skin = "default";
87  }
88 
89  Com_sprintf( iconName, MAX_QPATH, "players//%s/icon_%s.tga", model, skin );
90 
91  strcpy(loadingPlayerNames[loadingPlayerIconCount],name);
92  loadingPlayerIcons[loadingPlayerIconCount] = trap_R_RegisterShaderNoMip( iconName );
93  if ( !loadingPlayerIcons[loadingPlayerIconCount] ) {
94  Com_sprintf( iconName, MAX_QPATH, "players//characters/%s/icon_%s.tga", model, skin );
95  loadingPlayerIcons[loadingPlayerIconCount] = trap_R_RegisterShaderNoMip( iconName );
96  }
97  if ( !loadingPlayerIcons[loadingPlayerIconCount] ) {
98  Com_sprintf( iconName, MAX_QPATH, "players//%s/icon_%s.tga", DEFAULT_MODEL, "default" );
99  loadingPlayerIcons[loadingPlayerIconCount] = trap_R_RegisterShaderNoMip( iconName );
100  }
101  if ( loadingPlayerIcons[loadingPlayerIconCount] ) {
102  loadingPlayerIconCount++;
103  }
104  }
105  Q_strncpyz( personality, Info_ValueForKey( info, "n" ), sizeof(personality) );
106  Q_CleanStr( personality );
107  CG_LoadingString( personality );
108 }
