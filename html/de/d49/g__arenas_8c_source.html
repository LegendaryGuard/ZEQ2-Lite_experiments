40 void UpdateTournamentInfo( void ) {
41  int i;
42  gentity_t *player;
43  int playerClientNum;
44  int n, accuracy, perfect, msglen;
45  char buf[32];
46  char msg[MAX_STRING_CHARS];
47 
48  // find the real player
49  player = NULL;
50  for (i = 0; i < level.maxclients; i++ ) {
51  player = &g_entities[i];
52  if ( !player->inuse ) {
53  continue;
54  }
55  }
56  // this should never happen!
57  if ( !player || i == level.maxclients ) {
58  return;
59  }
60  playerClientNum = i;
61 
62  CalculateRanks();
63 
64  if ( level.clients[playerClientNum].sess.sessionTeam == TEAM_SPECTATOR ) {
65  Com_sprintf( msg, sizeof(msg), "postgame %i %i 0 0 0 0 0 0", level.numNonSpectatorClients, playerClientNum );
66  }
67  msglen = strlen( msg );
68  trap_SendConsoleCommand( EXEC_APPEND, msg );
69 }

99  body->s.event = 0;

101  VectorCopy (ent->r.mins, body->r.mins);
102  VectorCopy (ent->r.maxs, body->r.maxs);
103  VectorCopy (ent->r.absmin, body->r.absmin);
104  VectorCopy (ent->r.absmax, body->r.absmax);
105  body->clipmask = CONTENTS_SOLID | CONTENTS_PLAYERCLIP;
106  body->r.contents = CONTENTS_BODY;
107  body->r.ownerNum = ent->r.ownerNum;
108  body->takedamage = qfalse;
109 
110  VectorSubtract( level.intermission_origin, pad->r.currentOrigin, vec );
111  vectoangles( vec, body->s.apos.trBase );
112  body->s.apos.trBase[PITCH] = 0;
113  body->s.apos.trBase[ROLL] = 0;
114 
115  AngleVectors( body->s.apos.trBase, f, r, u );
116  VectorMA( pad->r.currentOrigin, offset[0], f, vec );
117  VectorMA( vec, offset[1], r, vec );
118  VectorMA( vec, offset[2], u, vec );
119 
120  G_SetOrigin( body, vec );
121 
122  trap_LinkEntity (body);
123 
124  body->count = place;
125 
126  return body;
127 }
