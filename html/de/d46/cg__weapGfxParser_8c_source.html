1029 qboolean CG_weapGfx_ParseDefinition(cg_weapGfxParser_t* parser,char* refname,cg_weapGfxAccessLvls_t* accessLvl){
1030  int i = CG_weapGfx_FindDefinitionRef(parser,refname) - 1;
1031  cg_weapGfxScanner_t* scanner;
1032  cg_weapGfxToken_t* token;
1033  // <-- Incase there IS no last access level from a super class
1034  cg_weapGfxAccessLvls_t lastAccessLvl = LVL_PUBLIC;
1035  scanner = &parser->scanner;
1036  token = &parser->token;
1037  if(i < MAX_DEFINES){
1038  // local declaration
1039  if(parser->definitionRef[i].hasSuper){
1040  if(!CG_weapGfx_IncreaseRecursionDepth()){
1041  return CG_weapGfx_ErrorHandle(ERROR_MAX_RECURSION,scanner,NULL,NULL);
1042  }
1043  if(cg_verboseParse.integer){
1044  CG_Printf("Inheriting superclass '%s'\n",parser->definitionRef[i].supername);
1045  }
1046  if(!CG_weapGfx_ParseDefinition(parser,parser->definitionRef[i].supername,&lastAccessLvl)){
1047  return qfalse;
1048  }
1049  CG_weapGfx_DecreaseRecursionDepth();
1050  }
1051  if(lastAccessLvl == LVL_PRIVATE){
1052  return CG_weapGfx_ErrorHandle(ERROR_INHERITING_PRIVATE,scanner,parser->definitionRef[i].supername,NULL);
1053  }
1054  // Reposition the lexical scanner
1055  scanner->pos = parser->definitionRef[i].scannerPos;
1056  scanner->line = parser->definitionRef[i].scannerLine;
1057  // Check if we're not breaking access level hierarchy
1058  if(accessLvl){
1059  *accessLvl = parser->definitionRef[i].accessLvl;
1060  if(*accessLvl < lastAccessLvl){
1061  return CG_weapGfx_ErrorHandle(ERROR_OVERRIDING_WITH_HIGHER_ACCESS,scanner,parser->definitionRef[i].refname,NULL);
1062  }
1063  }
1064  // Skip the '{' opening brace of the definition block, and align to the first real
1065  // symbol in the block.
1066  if(!CG_weapGfx_CheckPrematureEOF(scanner,token)){return qfalse;}
1067  if(!CG_weapGfx_ParseCategories(parser)){return qfalse;}
1068  }
1069  else{
1070  // imported declaration
1071  // First subtract the offset we added to detect difference between
1072  // an imported and a local definition.
1073  i -= MAX_DEFINES;
1074  if(!CG_weapGfx_IncreaseRecursionDepth()){
1075  return CG_weapGfx_ErrorHandle(ERROR_MAX_RECURSION,scanner,NULL,NULL);
1076  }
1077  if(cg_verboseParse.integer) {
1078  CG_Printf("Importing '%s'\n",refname);
1079  }
1080  if(!CG_weapGfx_ParseRemoteDefinition(parser->importRef[i].filename,parser->importRef[i].defname)){
1081  return qfalse;
1082  }
1083  CG_weapGfx_DecreaseRecursionDepth();
1084  }
1085  return qtrue;
1086 }
