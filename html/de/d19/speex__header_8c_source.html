48 static inline spx_int32_t le_int(spx_int32_t i)
49 {
50 #if !defined(__LITTLE_ENDIAN__) && ( defined(WORDS_BIGENDIAN) || defined(__BIG_ENDIAN__) )
51  spx_uint32_t ui, ret;
52  ui = i;
53  ret = ui>>24;
54  ret |= (ui>>8)&0x0000ff00;
55  ret |= (ui<<8)&0x00ff0000;
56  ret |= (ui<<24);
57  return ret;
58 #else
59  return i;
60 #endif
61 }

86 void speex_init_header(SpeexHeader *header, int rate, int nb_channels, const SpeexMode *m)
87 {
88  int i;
89  const char *h="Speex ";
90  /*
91  strncpy(header->speex_string, "Speex ", 8);
92  strncpy(header->speex_version, SPEEX_VERSION, SPEEX_HEADER_VERSION_LENGTH-1);
93  header->speex_version[SPEEX_HEADER_VERSION_LENGTH-1]=0;
94  */
95  for (i=0;i<8;i++)
96  header->speex_string[i]=h[i];
97  for (i=0;i<SPEEX_HEADER_VERSION_LENGTH-1 && SPEEX_VERSION[i];i++)
98  header->speex_version[i]=SPEEX_VERSION[i];
99  for (;i<SPEEX_HEADER_VERSION_LENGTH;i++)
100  header->speex_version[i]=0;
101 
102  header->speex_version_id = 1;
103  header->header_size = sizeof(SpeexHeader);
104 
105  header->rate = rate;
106  header->mode = m->modeID;
107  header->mode_bitstream_version = m->bitstream_version;
108  if (m->modeID<0)
109  speex_warning("This mode is meant to be used alone");
110  header->nb_channels = nb_channels;
111  header->bitrate = -1;
112  speex_mode_query(m, SPEEX_MODE_FRAME_SIZE, &header->frame_size);
113  header->vbr = 0;
114 
115  header->frames_per_packet = 0;
116  header->extra_headers = 0;
117  header->reserved1 = 0;
118  header->reserved2 = 0;
119 }

121 char *speex_header_to_packet(SpeexHeader *header, int *size)
122 {
123  SpeexHeader *le_header;
124  le_header = (SpeexHeader*)speex_alloc(sizeof(SpeexHeader));
125 
126  SPEEX_COPY(le_header, header, 1);
127 
128  /*Make sure everything is now little-endian*/
129  ENDIAN_SWITCH(le_header->speex_version_id);
130  ENDIAN_SWITCH(le_header->header_size);
131  ENDIAN_SWITCH(le_header->rate);
132  ENDIAN_SWITCH(le_header->mode);
133  ENDIAN_SWITCH(le_header->mode_bitstream_version);
134  ENDIAN_SWITCH(le_header->nb_channels);
135  ENDIAN_SWITCH(le_header->bitrate);
136  ENDIAN_SWITCH(le_header->frame_size);
137  ENDIAN_SWITCH(le_header->vbr);
138  ENDIAN_SWITCH(le_header->frames_per_packet);
139  ENDIAN_SWITCH(le_header->extra_headers);
140 
141  *size = sizeof(SpeexHeader);
142  return (char *)le_header;

145 SpeexHeader *speex_packet_to_header(char *packet, int size)
146 {
147  int i;
148  SpeexHeader *le_header;
149  const char *h = "Speex ";
150  for (i=0;i<8;i++)
151  if (packet[i]!=h[i])
152  {
153  speex_notify("This doesn't look like a Speex file");
154  return NULL;
155  }
156 
157  /*FIXME: Do we allow larger headers?*/
158  if (size < (int)sizeof(SpeexHeader))
159  {
160  speex_notify("Speex header too small");
161  return NULL;
162  }
163 
164  le_header = (SpeexHeader*)speex_alloc(sizeof(SpeexHeader));
165 
166  SPEEX_COPY(le_header, (SpeexHeader*)packet, 1);
167 
168  /*Make sure everything is converted correctly from little-endian*/
169  ENDIAN_SWITCH(le_header->speex_version_id);
170  ENDIAN_SWITCH(le_header->header_size);
171  ENDIAN_SWITCH(le_header->rate);
