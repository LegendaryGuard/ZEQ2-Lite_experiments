40 
41 Returns qtrue if the velocity was clipped in some way
42 ==================

44 #define MAX_CLIP_PLANES 5
45 qboolean PM_SlideMove( qboolean gravity ) {
46  int bumpcount, numbumps;
47  vec3_t dir;
48  float d;
49  int numplanes;
50  vec3_t planes[MAX_CLIP_PLANES];
51  vec3_t primal_velocity;
52  vec3_t clipVelocity;
53  int i, j, k;
54  trace_t trace;
55  vec3_t end;
56  float time_left;
57  float into;
58  vec3_t endVelocity;
59  vec3_t endClipVelocity;
60 
61  numbumps = 4;
62 
63  VectorCopy (pm->ps->velocity, primal_velocity);
64  if(gravity){
65  VectorCopy( pm->ps->velocity, endVelocity );
66  endVelocity[0] -= pml.gravityDirection[0] * pml.frametime;
67  endVelocity[1] -= pml.gravityDirection[1] * pml.frametime;
68  endVelocity[2] -= pml.gravityDirection[2] * pml.frametime;
69  primal_velocity[2] = endVelocity[2];
70  }
71  time_left = pml.frametime;
72 
73  // never turn against the ground plane
74  if ( pml.groundPlane ) {
75  numplanes = 1;
76  VectorCopy( pml.groundTrace.plane.normal, planes[0] );
77  } else {
78  numplanes = 0;
79  }
80 
81  // never turn against original velocity
82  VectorNormalize2( pm->ps->velocity, planes[numplanes] );
83  numplanes++;
84 
85  for ( bumpcount=0 ; bumpcount < numbumps ; bumpcount++ ) {
86 
87  // calculate position we are trying to move to
88  VectorMA( pm->ps->origin, time_left, pm->ps->velocity, end );
89 
90  // see if we can make it there
91  pm->trace ( &trace, pm->ps->origin, pm->mins, pm->maxs, end, pm->ps->clientNum, pm->tracemask);
92 
93  if (trace.allsolid) {
94  // entity is completely trapped in another solid
95  pm->ps->velocity[2] = 0; // don't build up falling damage, but allow sideways acceleration
96  return qtrue;
97  }
98 
99  if (trace.fraction > 0) {
100  // actually covered some distance
101  VectorCopy (trace.endpos, pm->ps->origin);
102  }
103 
104  if (trace.fraction == 1) {
105  break; // moved the entire distance
106  }
107 
108  // save entity for contact
109  PM_AddTouchEnt( trace.entityNum );
110 
111  time_left -= time_left * trace.fraction;
112 
113  if (numplanes >= MAX_CLIP_PLANES) {
114  // this shouldn't really happen
115  VectorClear( pm->ps->velocity );
116  return qtrue;
117  }
118 
119  //
120  // if this is the same plane we hit before, nudge velocity
121  // out along it, which fixes some epsilon issues with
122  // non-axial planes
123  //
124  for ( i = 0 ; i < numplanes ; i++ ) {
125  if ( DotProduct( trace.plane.normal, planes[i] ) > 0.99 ) {
126  VectorAdd( trace.plane.normal, pm->ps->velocity, pm->ps->velocity );
127  break;
128  }
129  }
130  if ( i < numplanes ) {
131  continue;
132  }
133  VectorCopy (trace.plane.normal, planes[numplanes]);
134  numplanes++;
135 
136  //
137  // modify velocity so it parallels all of the clip planes
138  //
139 
140  // find a plane that it enters
141  for ( i = 0 ; i < numplanes ; i++ ) {
142  into = DotProduct( pm->ps->velocity, planes[i] );
143  if ( into >= 0.1 ) {
144  continue; // move doesn't interact with the plane
145  }
146 
147  // see how hard we are hitting things
148  if ( -into > pml.impactSpeed ) {
149  pml.impactSpeed = -into;
150  }
151 
152  // slide along the plane
153  PM_ClipVelocity (pm->ps->velocity, planes[i], clipVelocity, OVERCLIP );
154 
155  // slide along the plane
156  PM_ClipVelocity (endVelocity, planes[i], endClipVelocity, OVERCLIP );
157 
158  // see if there is a second plane that the new move enters
159  for ( j = 0 ; j < numplanes ; j++ ) {
160  if ( j == i ) {
161  continue;
162  }
163  if ( DotProduct( clipVelocity, planes[j] ) >= 0.1 ) {
164  continue; // move doesn't interact with the plane
165  }
166 
167  // try clipping the move to the plane
168  PM_ClipVelocity( clipVelocity, planes[j], clipVelocity, OVERCLIP );
169  PM_ClipVelocity( endClipVelocity, planes[j], endClipVelocity, OVERCLIP );
170 
171  // see if it goes back into the first clip plane
172  if ( DotProduct( clipVelocity, planes[i] ) >= 0 ) {
173  continue;
174  }
175 
176  // slide the original velocity along the crease
177  CrossProduct (planes[i], planes[j], dir);
178  VectorNormalize( dir );
179  d = DotProduct( dir, pm->ps->velocity );
180  VectorScale( dir, d, clipVelocity );
181 
182  CrossProduct (planes[i], planes[j], dir);
183  VectorNormalize( dir );
184  d = DotProduct( dir, endVelocity );
185  VectorScale( dir, d, endClipVelocity );
186 
187  // see if there is a third plane the the new move enters
188  for ( k = 0 ; k < numplanes ; k++ ) {
189  if ( k == i || k == j ) {
190  continue;
191  }
192  if ( DotProduct( clipVelocity, planes[k] ) >= 0.1 ) {
193  continue; // move doesn't interact with the plane
194  }
195 
196  // stop dead at a tripple plane interaction
197  VectorClear( pm->ps->velocity );
198  return qtrue;
199  }
200  }
201 
202  // if we have fixed all interactions, try another move
203  VectorCopy( clipVelocity, pm->ps->velocity );
204  VectorCopy( endClipVelocity, endVelocity );
205  break;
206  }
207  }
208 
209  if ( gravity ) {
210  VectorCopy( endVelocity, pm->ps->velocity );
211  }
212 
213  return ( bumpcount != 0 );
214 }

222 void PM_StepSlideMove( qboolean gravity ) {
223  vec3_t start_o, start_v;
224  vec3_t down_o, down_v;
225  trace_t trace;
226 // float down_dist, up_dist;
227 // vec3_t delta, delta2;
228  vec3_t up, down;
229  float stepSize;
230 
231  VectorCopy (pm->ps->origin, start_o);
232  VectorCopy (pm->ps->velocity, start_v);
233 
234  if ( PM_SlideMove( gravity ) == 0 ) {
235  return; // we got exactly where we wanted to go first try
236  }
237 
238  VectorCopy(start_o, down);
239  down[2] -= STEPSIZE;
240  pm->trace (&trace, start_o, pm->mins, pm->maxs, down, pm->ps->clientNum, pm->tracemask);
241  VectorSet(up, 0, 0, 1);
242  // never step up when you still have up velocity
243  if(pm->ps->velocity[2] > 0 && (trace.fraction == 1.0 || DotProduct(trace.plane.normal, up) < 0.7)) {return;}
244  VectorCopy (pm->ps->origin, down_o);
245  VectorCopy (pm->ps->velocity, down_v);
246  VectorCopy (start_o, up);
247  up[2] += STEPSIZE;
248  pm->trace (&trace, start_o, pm->mins, pm->maxs, up, pm->ps->clientNum, pm->tracemask);
249  if(trace.allsolid){
250  return; // can't step up
251  }
252 
253  stepSize = trace.endpos[2] - start_o[2];
254  // try slidemove from this position
255  VectorCopy (trace.endpos, pm->ps->origin);
256  VectorCopy (start_v, pm->ps->velocity);
257 
258  PM_SlideMove( gravity );
259 
260  // push down the final amount
261  VectorCopy (pm->ps->origin, down);
262  down[2] -= stepSize;
263  pm->trace (&trace, pm->ps->origin, pm->mins, pm->maxs, down, pm->ps->clientNum, pm->tracemask);
264  if ( !trace.allsolid ) {
265  VectorCopy (trace.endpos, pm->ps->origin);
266  }
267  if ( trace.fraction < 1.0 ) {
268  PM_ClipVelocity( pm->ps->velocity, trace.plane.normal, pm->ps->velocity, OVERCLIP );
269  }
270 
271  {
272  float delta;
273  delta = pm->ps->origin[2] - start_o[2];
274  if ( delta > 2 ) {
275  if ( delta < 7 ) {
276  PM_AddEvent( EV_STEP_4 );
277  } else if ( delta < 11 ) {
278  PM_AddEvent( EV_STEP_8 );
279  } else if ( delta < 15 ) {
280  PM_AddEvent( EV_STEP_12 );
281  } else {
282  PM_AddEvent( EV_STEP_16 );
283  }
284  }
285  if ( pm->debugLevel ) {
286  Com_Printf("%i:stepped\n", c_pmove);
287  }
288  }
289 }
