16 /*
17 ========================
18 CG_Aura_BuildTailPoint
19 ========================
20  Builds position of the aura's tail point, dependant on velocity vector of player entity
21 */
22 static void CG_Aura_BuildTailPoint(centity_t *player, auraState_t *state, auraConfig_t *config){
23  vec3_t tailDir;
24 
25  // Find the direction the tail should point. Low speeds will have it pointing up.
26  if(VectorNormalize2( player->currentState.pos.trDelta, tailDir) < 180)
27  VectorSet( tailDir, 0, 0, 1);
28  else
29  VectorInverse( tailDir);
30 
31  // Using the established direction, set up the tail's potential hull point.
32  VectorMA( state->origin, config->auraScale * config->tailLength, tailDir, state->tailPos);
33 }

36 /*
37 =======================
38 CG_Aura_MarkRootPoint
39 =======================
40  Marks a point as the aura's root point. The root point is the point that is placed
41  furthest away from the tail. It is where the aura 'opens up'.
42 */
43 void CG_Aura_MarkRootPoint( auraState_t *state){
44  vec3_t prjPoint;
45  float maxDist, newDist;
46  int i;
47 
48  // NOTE:
49  // The root point is the point that is placed furthest down the aura.
50  // This means the distance between the tail point and the root point's projection
51  // onto the tail axis is maximal.
52 
53  maxDist = 0;
54  for(i = 0;i < state->convexHullCount;i++){
55  // Project the point onto the tail axis
56  ProjectPointOnLine( state->convexHull[i].pos_world, state->origin, state->tailPos, prjPoint);
57 
58  // Measure the distance and copy over the projection of the point
59  // as the new root if it's further away.
60  newDist = Distance( prjPoint, state->tailPos);
61  if(newDist > maxDist){
62  maxDist = newDist;
63  VectorCopy( prjPoint, state->rootPos);
64  }
65  }
66 
67  // HACK:
68  // Nudge the root away a bit to account for the expansion
69  // of the aura through the hull normals.
70  {
71  vec3_t tailDir;
72 
73  VectorSubtract( state->tailPos, state->origin, tailDir);
74  VectorNormalize( tailDir);
75  VectorMA( state->rootPos, -4.8f, tailDir, state->rootPos);
76  }
77 }

157 /*
158 ====================
159 CG_Aura_QSortAngle
160 ====================
161  Quicksort by ascending angles
162 */
163 static void CG_Aura_QSortAngle( auraTag_t *points, float* angles, int lowbound, int highbound){
164  int low, high;
165  float mid;
166  auraTag_t tempPoint;
167  float tempAngle;
168 
169  low = lowbound;
170  high = highbound;
171  mid = angles[(low + high) / 2];
172 
173  do{
174  while(angles[low] < mid) low++;
175  while(angles[high] > mid) high--;
176 
177  if(low <= high){
178  // swap points
179  tempPoint = points[low];
180  points[low] = points[high];
181  points[high] = tempPoint;
182 
183  // swap angles
184  tempAngle = angles[low];
185  angles[low] = angles[high];
186  angles[high] = tempAngle;
187 
188  low++;
189  high--;
190  }
191 
192  } while(!(low > high));
193 
194  if(high > lowbound) CG_Aura_QSortAngle( points, angles, lowbound, high);
195  if(low < highbound) CG_Aura_QSortAngle( points, angles, low, highbound);
196 }

199 /*
200 ===========================
201 CG_Aura_ArrangeConvexHull
202 ===========================
203  Rearranges *points to contain its convex hull in the first *nr_points points.
204 */
205 static qboolean CG_Aura_ArrangeConvexHull( auraTag_t *points, int *nr_points){
206  float angles[MAX_AURATAGS + 1];// +1 for tail
207  int amount, index, pivotIndex;
208  auraTag_t pivot;
209  auraTag_t behind, infront;
210  qboolean rightTurn;
211  auraTag_t vecPoint;
212  auraTag_t buffer[MAX_AURATAGS + 1];
213 
214 
215  amount = *nr_points;
216 
217  if(amount == 3) return qtrue;// Already a convex hull
218  if(amount < 3){
219  return qfalse;
220  }
221 
222  pivotIndex = 0;
223  // Find pivot point, which is known to be on the hull.
224  // Point with lowest y - if there are multiple, point with highest x.
225  for(index = 1;index < amount;index++){
226 
227  if(points[index].pos_screen[1] < points[pivotIndex].pos_screen[1]){
228  pivotIndex = index;
229 
230  } else if(points[index].pos_screen[1] == points[pivotIndex].pos_screen[1]){
231 
232  if (points[index].pos_screen[0] > points[pivotIndex].pos_screen[0]){
233  pivotIndex = index;
234  }
235  }
236  }
237 
238  // Put pivot into seperate variable and remove from array.
239  pivot = points[pivotIndex];
240  points[pivotIndex] = points[amount - 1];
241  amount--;
242 
243  // Calculate angle to pivot for each point in the array.
244  for(index = 0;index < amount;index++){
245 
246  // point vector
247  vecPoint.pos_screen[0] = pivot.pos_screen[0] - points[index].pos_screen[0];
248  vecPoint.pos_screen[1] = pivot.pos_screen[1] - points[index].pos_screen[1];
249 
250  // reduce to a unit-vector - length 1
251  angles[index] = vecPoint.pos_screen[0] / Q_hypot(vecPoint.pos_screen[0], vecPoint.pos_screen[1]);
252  }
253 
254  // Sort the points by angle.
255  CG_Aura_QSortAngle(points, angles, 0, amount - 1);
256 
257  // Step through array to remove points that are not p of the convex hull.
258  index = 1;
259 
260  do{
261  // Assign points behind and in front of current point.
262  if(index == 0){
263  rightTurn = qtrue;
264  } else{
265  behind = points[index - 1];
266 
267  if(index == (amount - 1)){
268  infront = pivot;
269  } else{
270  infront = points[index + 1];
271  }
272 
273  // Work out if we are making a right or left turn using vector product.
274  if(( (behind.pos_screen[0] - points[index].pos_screen[0]) * (infront.pos_screen[1] - points[index].pos_screen[1]) -
275  (infront.pos_screen[0] - points[index].pos_screen[0]) * (behind.pos_screen[1] - points[index].pos_screen[1])) < 0){
276  rightTurn = qtrue;
277  }else{
278  rightTurn = qfalse;
279  }
280  }
281 
282  if(rightTurn){
283  // point is currently considered part of the hull
284  index++;
285  } else{
286  // point is not part of the hull
287 
288  // remove point from convex hull
289  if(index == (amount - 1)){
290  amount--;
291  } else{
292  // move everything after the current value one step forward
293  memcpy( buffer, &points[index + 1], sizeof(auraTag_t) * (amount - index - 1));
294  memcpy( &points[index], buffer, sizeof(auraTag_t) * (amount - index - 1));
295  amount--;
296  }
297 
298  // backtrack to previous point
299  index--;
300  }
301 
302  } while(!(index == (amount - 1)));
303 
304  // add pivot back into points array
305  points[amount] = pivot;
306  amount++;
307 
308  *nr_points = amount;
309  return qtrue;
310 }

313 /*
314 ===========================
315 CG_Aura_SetHullAttributes
316 ===========================
317  Set segment lengths, normals, circumference, etc.
318 */
319 static void CG_Aura_SetHullAttributes( auraState_t *state){
320  int index, behind, infront;
321  vec3_t line_behind, line_infront, viewLine;
322  vec3_t temp_behind, temp_infront;
323  float circumference;
324 
325  auraTag_t *points;
326  int nr_points;
327 
328  points = state->convexHull;
329  nr_points = state->convexHullCount;
330 
331  circumference = 0.0f;
332 
333  for(index = 0;index < nr_points;index++){
334 
335  // Set successor and predeccessor
336  if(index == 0){
337  behind = nr_points - 1;
338  infront = index + 1;
339  } else if(index == nr_points - 1){
340  behind = index - 1;
341  infront = 0;
342  } else{
343  behind = index - 1;
344  infront = index + 1;
345  }
346 
347  VectorSubtract( points[index].pos_world, cg.refdef.vieworg, viewLine);
348 
349  // Calculate the normal
350  VectorSubtract( points[index].pos_world, points[behind].pos_world, line_behind );
351  VectorSubtract( points[infront].pos_world, points[index].pos_world, line_infront);
352 
353  VectorNormalize( line_behind );
354  circumference += (points[index].length = VectorNormalize( line_infront));
355 
356  CrossProduct( line_behind, viewLine, temp_behind);
357  CrossProduct( line_infront, viewLine, temp_infront);
358  if(!VectorNormalize(temp_behind)){
359  viewLine[2] += 0.1f;
360  CrossProduct( line_behind, viewLine, temp_behind);
361  VectorNormalize(temp_behind);
362  viewLine[2] -= 0.1f;
363  }
364 
365  if(!VectorNormalize(temp_infront)){
366  viewLine[2] += 0.1f;
367  CrossProduct( line_behind, viewLine, temp_behind);
368  VectorNormalize(temp_behind);
369  viewLine[2] -= 0.1f;
370  }
371 
372  VectorAdd( temp_behind, temp_infront, points[index].normal);
373  VectorNormalize( points[index].normal);
374  }
375 
376  state->convexHullCircumference = circumference;
377 }

382 CG_Aura_BuildConvexHull
383 =========================
384  Calls all relevant functions to build up the aura's convex hull.
385  Returns false if no hull can be made.
386 */
387 static qboolean CG_Aura_BuildConvexHull( centity_t *player, auraState_t *state, auraConfig_t *config){
388 
389  // Retrieve hull points
390  CG_Aura_GetHullPoints( player, state, config);
391

396

399 
400  // Hull building completed succesfully
401  return qtrue;
402 }

427  vec3_t line, offset, viewLine;

429  float len;

448  verts[2].st[0] = 0;
449  verts[2].st[1] = 1;
450  VectorMA (start, -width, offset, verts[3].xyz);

453

459

464 /*
465 ==========================
466 CG_Aura_LerpSpikeSegment
467 ==========================
468  Lerps the position the aura spike should have along a segment of the convex hull
469 */
470 static void CG_Aura_LerpSpikeSegment( auraState_t *state, int spikeNr, int *start, int *end, float *progress_pct){
471  float length_pos, length_sofar;
472  int i, j;
473 
474  // Map i onto the circumference of the convex hull.
475  length_pos = state->convexHullCircumference *((float)spikeNr / (float)(NR_AURASPIKES - 1));
476 
477  // Find the segment we are in right now.
478  length_sofar = 0;
479  for(i = 0;(( length_sofar + state->convexHull[i].length) < length_pos) &&(i < NR_AURASPIKES);i++){
480  length_sofar += state->convexHull[i].length;
481  }
482  j = i + 1;
483  if(j == state->convexHullCount){
484  j = 0;
485  }
486 
487  // Return found values.
488  *start = i;
489  *end = j;
490  *progress_pct = (length_pos - length_sofar) / state->convexHull[i].length;
491 }

494 /*
495 ==============
496 CG_LerpSpike
497 ==============
498  Lerps one spike in the aura
499 */
500 static void CG_LerpSpike( auraState_t *state, auraConfig_t *config, int spikeNr, float alphaModulate){
501  int start, end;
502  float progress_pct;
503  vec3_t viewLine;
504 
505 
506  vec3_t lerpPos, lerpNormal, lerpDir, endPos, basePos;
507  float lerpTime, lerpModulate, lerpBorder, baseBorder, lerpSize;
508  vec4_t lerpColor;
509 
510  vec3_t edge, tailDir, tempVec;
511 
512  // Decide on which type of spike to use.
513  if(!( spikeNr % 3)){
514  lerpTime =(cg.time % 400) / 400.0f;
515  lerpSize = 14.0f * (1 + lerpTime);
516  lerpBorder = 2.5f * (1.75f + lerpTime);
517  baseBorder = 2.5f * 1.75f;
518 
519  } else if(!( spikeNr % 2)){
520  lerpTime = (( cg.time + 200) % 400) / 400.0f;
521  lerpSize = 12.0f * (1 + lerpTime);
522  lerpBorder = 3.0f * (1.75f + lerpTime);
523  baseBorder = 3.0f * 1.75f;
524 
525  } else{
526  lerpTime =(cg.time % 500) / 500.0f;
527  lerpSize = 10.0f * (1 + lerpTime);
528  lerpBorder = 2.75f * (1.75f + lerpTime);
529  baseBorder = 2.75f * 1.75f;
530  }
531  lerpModulate = -4 * lerpTime * lerpTime + 4 * lerpTime;
532 
533  // NOTE: Prepared for a cvar switch between additive and
534  // blended aura.
535  if(0){
536  lerpColor[0] = config->auraColor[0] * lerpModulate * alphaModulate;
537  lerpColor[1] = config->auraColor[1] * lerpModulate * alphaModulate;
538  lerpColor[2] = config->auraColor[2] * lerpModulate * alphaModulate;
539  lerpColor[3] = 1.0f;
540  } else{
541  lerpColor[0] = config->auraColor[0];
542  lerpColor[1] = config->auraColor[1];
543  lerpColor[2] = config->auraColor[2];
544  lerpColor[3] = lerpModulate * alphaModulate;
545  }
546 
547  // Get our position in the hull
548  CG_Aura_LerpSpikeSegment( state, spikeNr, &start, &end, &progress_pct);
549 
550  // Lerp the position using the stored normal to expand the aura a bit
551  VectorSet( lerpNormal, 0.0f, 0.0f, 0.0f);
552  VectorMA( lerpNormal, 1.0f - progress_pct, state->convexHull[start].normal, lerpNormal);
553  VectorMA( lerpNormal, progress_pct, state->convexHull[end].normal, lerpNormal);
554  VectorNormalize(lerpNormal);
555 
556  VectorSubtract( state->convexHull[end].pos_world, state->convexHull[start].pos_world, edge);
557  VectorMA( state->convexHull[start].pos_world, progress_pct, edge, lerpPos);
558  VectorMA( lerpPos, baseBorder, lerpNormal, basePos);
559  VectorMA( lerpPos, lerpBorder, lerpNormal, lerpPos);
560 
561 
562  // Create the direction
563  VectorSubtract( lerpPos, state->rootPos, lerpDir);
564 
565  // Flatten the direction a bit so it doesn't point to the tip too drasticly.
566  VectorSubtract( state->tailPos, state->origin, tailDir);
567  VectorPllComponent( lerpDir, tailDir, edge);
568  VectorScale( edge, AURA_FLATTEN_NORMAL, edge);
569  VectorSubtract( lerpDir, edge, lerpDir);
570  VectorNormalize( lerpDir);
571 
572  // Set the viewing direction
573  VectorSubtract( lerpPos, cg.refdef.vieworg, viewLine);
574  VectorNormalize( viewLine);
575 
576  // Don't display this spike if it would be travelling into / out of
577  // the screen too much: It is part of the blank area surrounding the root.
578  CrossProduct( viewLine, lerpDir, tempVec);
579  if(VectorLength(tempVec) < AURA_ROOTCUTOFF_FRAQ){
580 
581  // Only disallow drawing if it's actually a segment originating
582  // from close enough to the root.
583  if(Distance( basePos, state->rootPos) < AURA_ROOTCUTOFF_DIST){
584  return;
585  }
586  }
587 
588  VectorMA( lerpPos, lerpBorder, lerpDir, lerpPos);
589  VectorMA( lerpPos, lerpSize, lerpDir, endPos);
590  CG_Aura_DrawSpike( lerpPos, endPos, lerpSize / 1.25f, config->auraShader, lerpColor);
591 }

659 
660  player->lastTrailTime = cg.time;
