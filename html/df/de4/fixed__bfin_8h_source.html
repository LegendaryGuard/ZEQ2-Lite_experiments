41 {
42  spx_word32_t res, bb;
43  bb = b;
44  a += b>>1;
45  __asm__ (
46  "P0 = 15;\n\t"
47  "R0 = %1;\n\t"
48  "R1 = %2;\n\t"
49  //"R0 = R0 + R1;\n\t"
