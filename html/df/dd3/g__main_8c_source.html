273 /*
274 ================
275 G_FindTeams
276 
277 Chain together all entities with a matching team field.
278 Entity teams are used for item groups and multi-entity mover groups.
279 
280 All but the first will have the FL_TEAMSLAVE flag set and teammaster field set
281 All but the last will have the teamchain field set to the next one
282 ================
283 */
284 void G_FindTeams( void ) {
285  gentity_t *e, *e2;
286  int i, j;
287  int c, c2;
288 
289  c = 0;
290  c2 = 0;
291  for ( i=1, e=g_entities+i ; i < level.num_entities ; i++,e++ ){
292  if (!e->inuse)
293  continue;
294  if (!e->team)
295  continue;
296  if (e->flags & FL_TEAMSLAVE)
297  continue;
298  e->teammaster = e;
299  c++;
300  c2++;
301  for (j=i+1, e2=e+1 ; j < level.num_entities ; j++,e2++)
302  {
303  if (!e2->inuse)
304  continue;
305  if (!e2->team)
306  continue;
307  if (e2->flags & FL_TEAMSLAVE)
308  continue;
309  if (!strcmp(e->team, e2->team))
310  {
311  c2++;
312  e2->teamchain = e->teamchain;
313  e->teamchain = e2;
314  e2->teammaster = e;
315  e2->flags |= FL_TEAMSLAVE;
316 
317  // make sure that targets only point at the master
318  if ( e2->targetname ) {
319  e->targetname = e2->targetname;
320  e2->targetname = NULL;
321  }
322  }
323  }
324  }
325 
326  G_Printf ("%i teams with %i entities\n", c, c2);
327 }

550 /*
551 =============
552 AddTournamentPlayer
553 
554 If there are less than two tournament players, put a
555 spectator in the game and restart
556 =============
557 */
558 void AddTournamentPlayer( void ) {
559  int i;
560  gclient_t *client;
561  gclient_t *nextInLine;
562 
563  if ( level.numPlayingClients >= 2 ) {
564  return;
565  }
566 
567  // never change during intermission
568  if ( level.intermissiontime ) {
569  return;
570  }
571 
572  nextInLine = NULL;
573 
574  for ( i = 0 ; i < level.maxclients ; i++ ) {
575  client = &level.clients[i];
576  if ( client->pers.connected != CON_CONNECTED ) {
577  continue;
578  }
579  if ( client->sess.sessionTeam != TEAM_SPECTATOR ) {
580  continue;
581  }
582  // never select the dedicated follow or scoreboard clients
583  if ( client->sess.spectatorState == SPECTATOR_SCOREBOARD ||
584  client->sess.spectatorClient < 0 ) {
585  continue;
586  }
587 
588  if(!nextInLine || client->sess.spectatorNum > nextInLine->sess.spectatorNum)
589  nextInLine = client;
590  }
591 
592  if ( !nextInLine ) {
593  return;
