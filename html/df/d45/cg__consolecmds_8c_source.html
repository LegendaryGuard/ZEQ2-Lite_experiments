82 static void CG_ScoresDown_f( void ) {
83  if ( cg.scoresRequestTime + 2000 < cg.time ) {
84  // the scores are more than two seconds out of data,
85  // so request new ones
86  cg.scoresRequestTime = cg.time;
87  trap_SendClientCommand( "score" );
88 
89  // leave the current scores up if they were already
90  // displayed, but if this is the first hit, clear them out
91  if ( !cg.showScores ) {
92  cg.showScores = qtrue;
93  cg.numScores = 0;
94  }
95  } else {
96  // show the cached contents even if they just pressed if it
97  // is within two seconds
98  cg.showScores = qtrue;
99  }
100 }

169 /*
170 ==================
171 CG_Draw2D_f
172 Example : draw2D <shader> <x> <y> <width> <height> <duration>
173 ==================
174 */
175 static void CG_Draw2D_f(void){
176  char shaderName[256];
177  overlay2D *current;
178  int index;
179  if(trap_Argc()<2){
180  CG_Printf("^3Usage : ^7draw2D <^5shaderName^7> <^5x^7> <^5y^7> <^5width^7> <^5height^7> <^5duration^7>\n");
181  return;
182  }
183  Q_strncpyz(shaderName,CG_Argv(1),256);
184  for(index=0;index<16;++index){
185  if(!cg.scripted2D[index].active){break;}
186  if(index == 15){
187  CG_Printf("^3Warning : ^7Cannot add overlay. Maximum overlay surfaces of [^3%i^7] active.\n",16);
188  return;
189  }
190  }
191  current = &cg.scripted2D[index];
192  current->shader = trap_R_RegisterShader(shaderName);
193  if(!current->shader){
194  CG_Printf("^1Error : ^7Shader by the name [^1%s^7] could not be found.\n",shaderName);
195  return;
196  }
197  current->active = qtrue;
198  current->x = atoi(CG_Argv(2));
199  current->y = atoi(CG_Argv(3));
200  current->width = atoi(CG_Argv(4)) ? atoi(CG_Argv(4)) : 640;
201  current->height = atoi(CG_Argv(5)) ? atoi(CG_Argv(5)) : 480;
202  current->endTime = atoi(CG_Argv(6)) ? atoi(CG_Argv(6)) : -1;
203  if(current->endTime != -1){
204  current->endTime += cg.time;
205  }
206 }

266 /*
267 =================
268 CG_ConsoleCommand
269 
270 The string has been tokenized and can be retrieved with
271 Cmd_Argc() / Cmd_Argv()
272 =================
273 */
274 qboolean CG_ConsoleCommand( void ) {
275  const char *cmd;
276  int i;
277 
278  cmd = CG_Argv(0);
279 
280  for ( i = 0 ; i < ARRAY_LEN( commands ) ; i++ ) {
281  if ( !Q_stricmp( cmd, commands[i].cmd ) ) {
282  commands[i].function();
283  return qtrue;
284  }
285  }
286 
287  return qfalse;
288 }
