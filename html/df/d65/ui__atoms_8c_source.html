104 
105  // default cursor position

107  menu->cursor_prev = 0;
108 
109  m_entersound = qtrue;
110 
111  trap_Key_SetCatcher( KEYCATCH_UI );
112 
113  // force first available item to have focus
114  for (i=0; i<menu->nitems; i++)
115  {
116  item = (menucommon_s *)menu->items[i];
117  if (!(item->flags & (QMF_GRAYED|QMF_MOUSEONLY|QMF_INACTIVE)))
118  {
119  menu->cursor_prev = -1;
