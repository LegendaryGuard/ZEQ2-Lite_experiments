66 void BG_EvaluateTrajectory( entityState_t *es, const trajectory_t *tr, int atTime, vec3_t result ) {
67  float deltaTime;
68  float phase;
69  float v0;
70  vec3_t dir;
71 
72  switch( tr->trType ) {
73  case TR_STATIONARY:
74  case TR_INTERPOLATE:
75  VectorCopy( tr->trBase, result );
76  break;
77 
78  case TR_LINEAR:
79  case TR_DRUNKEN:
80  deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
81  VectorMA( tr->trBase, deltaTime, tr->trDelta, result );
82 
83  if ( tr->trType == TR_DRUNKEN ) { // trDuration is how much the missile will wobble
84  vec3_t axis[3];
85  float ph1, ph2;
86 
87  if ( VectorNormalize2( tr->trDelta, axis[0] ) == 0.0f ) {
88  VectorSet( axis[0], 0, 0, 1 );
89  }
90  MakeNormalVectors( tr->trDelta, axis[1], axis[2] );
91  RotateAroundDirection( axis, deltaTime );
92  ph1 = sin( deltaTime * M_PI * 2 );
93  ph2 = (cos( deltaTime * M_PI * 2 ) - ph1) + cos( deltaTime * M_PI * 4 );
94  VectorMA( result, tr->trDuration * ph1, axis[1], result );
95  VectorMA( result, tr->trDuration * ph2, axis[2], result );
96  }
97  break;
98 
99  case TR_SINE:
100  deltaTime = ( atTime - tr->trTime ) / (float) tr->trDuration;
101  phase = sin( deltaTime * M_PI * 2 );
102  VectorMA( tr->trBase, phase, tr->trDelta, result );
103  break;
104 
105  case TR_LINEAR_STOP:
106  if ( atTime > tr->trTime + tr->trDuration ) {
107  atTime = tr->trTime + tr->trDuration;
108  }
109  deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
110  if ( deltaTime < 0 ) {
111  deltaTime = 0;
112  }
113  VectorMA( tr->trBase, deltaTime, tr->trDelta, result );
114  break;
115 
116  case TR_GRAVITY:
117  deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
118  VectorMA( tr->trBase, deltaTime, tr->trDelta, result );
119  result[2] -= 0.5 * DEFAULT_GRAVITY * deltaTime * deltaTime; // FIXME: local gravity...
120  break;
121 
122  case TR_ACCEL:
123  deltaTime = ( atTime - tr->trTime ) * 0.001f; // milliseconds to seconds
124 
125  /*
126  NOTE:
127  There's a problem with accelerated missiles being able to
128  get a negative acceleration and thus 'boomeranging' backwards.
129  We want to set a minimum velocity of 0 to prevent this.
130 
131  NOTE:
132  trDuration = acceleration,
133  */
134 
135  VectorCopy( tr->trDelta, dir );
136  v0 = VectorNormalize( dir );
137  phase = v0 + tr->trDuration * deltaTime;
138  if ( phase < 0 ) {
139 
140  // We have to find the point in time at which the velocity
141  // becomes zero. We solve 0 = v0 + a * t for t.
142 
143  deltaTime = -v0 / tr->trDuration;
144 
145  // Now we can use this time to get the missile's position
146  }
147 
148  // the 0.5*a*t^2 part.
149  phase = (tr->trDuration / 2.0f) * (deltaTime * deltaTime);
150  VectorMA (tr->trBase, phase, dir, result);
151 
152  // The u*t part.
153  VectorMA (result, deltaTime, tr->trDelta, result);
154  break;
155 
156  case TR_ARCH:
157  if (!es) {
158  Com_Error( ERR_DROP, "BG_EvaluateTrajectory: NULL entityState: %i", tr->trTime );
159  } else {
160  deltaTime = 1.0f - ( ( atTime - tr->trTime ) / (float) tr->trDuration );
161  BG_LerpQuadraticSpline( tr->trBase, es->angles2, tr->trDelta, deltaTime, result );
162  }
163  break;
164 
165  default:
166  Com_Error( ERR_DROP, "BG_EvaluateTrajectory: unknown trType: %i", tr->trTime );
167  break;
168  }
169 }

174

178 void BG_EvaluateTrajectoryDelta( entityState_t *es, const trajectory_t *tr, int atTime, vec3_t result ) {
179  float deltaTime;
180  float phase;
181  vec3_t dir;
182 
183  switch( tr->trType ) {
184  case TR_STATIONARY:
185  case TR_INTERPOLATE:
186  VectorClear( result );
187  break;
188  case TR_LINEAR:
189  case TR_DRUNKEN:
190  VectorCopy( tr->trDelta, result );
191  break;
192  case TR_SINE:
193  deltaTime = ( atTime - tr->trTime ) / (float) tr->trDuration;
194  phase = cos( deltaTime * M_PI * 2 ); // derivative of sin = cos
195  phase *= 0.5;
196  VectorScale( tr->trDelta, phase, result );
197  break;
198  case TR_LINEAR_STOP:
199  if ( atTime > tr->trTime + tr->trDuration ) {
200  VectorClear( result );
201  return;
202  }
203  VectorCopy( tr->trDelta, result );
204  break;
205  case TR_GRAVITY:
206  deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
207  VectorCopy( tr->trDelta, result );
208  result[2] -= DEFAULT_GRAVITY * deltaTime; // FIXME: local gravity...
209  break;
210  case TR_ACCEL:
211  deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
212 
213  // Turn magnitude of acceleration into a vector
214  VectorCopy(tr->trDelta, dir);
215  phase = VectorNormalize (dir);
216  phase = phase + tr->trDuration * deltaTime;
217  if ( phase < 0 ) { // prevent 'negative' velocity, clamp it at zero.
218  VectorSet( dir, 0, 0, 0);
219  } else {
220  VectorScale (dir, tr->trDuration, dir);
221  }
222 
223  // u + t * a = v
224  VectorMA (tr->trDelta, deltaTime, dir, result);
225  break;
226  case TR_ARCH:
227  if (!es) {
228  Com_Error( ERR_DROP, "BG_EvaluateTrajectory: NULL entityState: %i", tr->trTime );
229  } else {
230  deltaTime = 1.0f - ( ( atTime - tr->trTime ) / (float) tr->trDuration );
231  BG_LerpQuadraticSplineDelta( tr->trBase, es->angles2, tr->trDelta, deltaTime, result );
232  VectorNormalize( result );
233 
234  // Reverse the speed calculation we did in user_missile.c under HOM_ARCH
235  VectorScale( result, 1000.0f * Distance( tr->trBase, tr->trDelta ) / (float) tr->trDuration, result );
236  }
237  break;
238  default:
239  Com_Error( ERR_DROP, "BG_EvaluateTrajectoryDelta: unknown trType: %i", tr->trTime );
240  break;
241  }
242 }

345 void BG_TouchJumpPad( playerState_t *ps, entityState_t *jumppad ) {
346  vec3_t angles;
347  float p;
348  int effectNum;
349 
350  // spectators don't use jump pads
351  if ( ps->pm_type != PM_NORMAL ) {
352  return;
353  }
354 
355  // if we didn't hit this same jumppad the previous frame
356  // then don't play the event sound again if we are in a fat trigger
357  if ( ps->jumppad_ent != jumppad->number ) {
358 
359  vectoangles( jumppad->origin2, angles);
360  p = fabs( AngleNormalize180( angles[PITCH] ) );
361  if( p < 45 ) {
362  effectNum = 0;
363  } else {
364  effectNum = 1;
365  }
366  BG_AddPredictableEventToPlayerstate( EV_JUMP_PAD, effectNum, ps );
367  }
368  // remember hitting this jumppad this frame
369  ps->jumppad_ent = jumppad->number;
370  ps->jumppad_frame = ps->pmove_framecount;
371  // give the player the velocity from the jumppad
372  VectorCopy( jumppad->origin2, ps->velocity );
373 }
