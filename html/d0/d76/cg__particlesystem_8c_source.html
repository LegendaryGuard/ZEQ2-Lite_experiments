132 static void PSys_FreeParticle( PSys_Particle_t *particle ) {
133 
134  if ( !particle->prev ) {
135  CG_Error( "PSys_FreeParticle: not active" );
136  }
137 
138  // remove from the doubly linked global active list
139  particle->prev->next = particle->next;
140  particle->next->prev = particle->prev;
141 
142  // remove from the doubly linked local active list
143  particle->prev_local->next_local = particle->next_local;
144  particle->next_local->prev_local = particle->prev_local;
145 
146  // the free list is only singly linked
147  particle->next = PSys_Particles_free;
148  PSys_Particles_free = particle;

347 static void PSys_FreeSystem( PSys_System_t *system ) {
348  PSys_Particle_t *particle, *next_p;
349  PSys_Emitter_t *emitter, *next_e;
350  PSys_Force_t *force, *next_f;
351  PSys_Constraint_t *constraint,*next_c;
352 
353  if ( !system->prev ) {
354  CG_Error( "PSys_FreeSystem: not active" );
355  }
356 
357  // free the local list of particles
358  particle = system->particles.prev_local;
359  for ( ; particle != &(system->particles) ; particle = next_p ) {
360  // Grab next now, so if the entity is freed we still have the next one.
361  next_p = particle->prev_local;
362 
363  PSys_FreeParticle( particle );
364  }
365 
366  // free the local list of emitters
367  emitter = system->emitters.prev_local;
368  for ( ; emitter != &(system->emitters) ; emitter = next_e ) {
369  // Grab next now, so if the entity is freed we still have the next one.
370  next_e = emitter->prev_local;
371 
372  PSys_FreeEmitter( emitter );
373  }
374 
375  // free the local list of forces
376  force = system->forces.prev_local;
377  for ( ; force != &(system->forces) ; force = next_f ) {
378  // Grab next now, so if the entity is freed we still have the next one.
379  next_f = force->prev_local;
380 
381  PSys_FreeForce( force );
382  }
383 
384  // free the local list of constraints
385  constraint = system->constraints.prev_local;
386  for ( ; constraint != &(system->constraints) ; constraint = next_c ) {
387  // Grab next now, so if the entity is freed we still have the next one.
388  next_c = constraint->prev_local;
389 
390  PSys_FreeConstraint( constraint );
391  }
392 
393  // remove from the doubly linked active list
394  system->prev->next = system->next;
395  system->next->prev = system->prev;
396 
397  // the free list is only singly linked
398  system->next = PSys_Systems_free;
399  PSys_Systems_free = system;
400 }

445 static qboolean PSys_UpdateOrientation( PSys_System_t *system, PSys_Orientation_t *orient ) {
446 
447  if ( orient->entity ) {
448 
449  // Check if the linked entity is not in the PVS
450  if (!CG_FrameHist_IsInPVS( orient->entity->currentState.number )) {
451  return qfalse;
452  }
453 
454  // Check if we should link against the aura and if it is still up and going
455  if ( orient->linkAuraState && !CG_FrameHist_HasAura(orient->entity->currentState.number)) {
456  return qfalse;
457  }
458 
459  // Check if we should link against the weapon
460  if ( orient->linkWeapon ) {
461  // If we switched weapons
462  if ( CG_FrameHist_IsWeaponNr( orient->entity->currentState.number ) != orient->weaponNr ) {
463  return qfalse;
464  }
465  // If we switched weaponstate (aka stopped charging or stopped firing)
466  if ( CG_FrameHist_IsWeaponState( orient->entity->currentState.number ) != orient->weaponState ) {
467  return qfalse;
468  }
469  }
470 
471  if ( !CG_GetTagOrientationFromPlayerEntity( orient->entity, orient->tagName, &(orient->geometry)) ) {
472  VectorCopy( orient->entity->lerpOrigin, orient->geometry.origin );
473  AnglesToAxis( orient->entity->lerpAngles, orient->geometry.axis );
474  }
475  } else {
476  VectorCopy( system->rootPos, orient->geometry.origin );
477  VectorCopy( system->rootAxis[0], orient->geometry.axis[0] );
478  VectorCopy( system->rootAxis[1], orient->geometry.axis[1] );
479  VectorCopy( system->rootAxis[2], orient->geometry.axis[2] );
480  }
481 
482  // Local offset
483  VectorMA( orient->geometry.origin, orient->offset[0], orient->geometry.axis[0], orient->geometry.origin );
484  VectorMA( orient->geometry.origin, orient->offset[1], orient->geometry.axis[1], orient->geometry.origin );
485  VectorMA( orient->geometry.origin, orient->offset[2], orient->geometry.axis[2], orient->geometry.origin );
486 
487  // Possibly override the resulting direction with a new global direction
488  if ( VectorLength( orient->dir ) > 0.0f ) {
489  VectorNormalize2( orient->dir, orient->geometry.axis[0] );
490  MakeNormalVectors( orient->geometry.axis[0], orient->geometry.axis[1], orient->geometry.axis[2] );
491  }
492 
493  return qtrue;
494 }

511 static void PSys_UpdateEmitters( PSys_System_t *system ) {
512  PSys_Emitter_t *emitter, *next;
513 
514  emitter = system->emitters.prev_local;
515  for ( ; emitter != &(system->emitters) ; emitter = next ) {
516  orientation_t root;
517  vec3_t groundPoint;
518  trace_t trace;
519 
520  next = emitter->prev_local;
521 
522  // If there are no particle templates, just trash the emitter
523  if ( emitter->nrTemplates == 0 ) {
524  PSys_FreeEmitter( emitter );
525  continue;
526  }
527 
528  // Check if the emitter had a maximum life time, and if that lifetime has been exceeded proceed to destroy
529  // the emitter.
530  if ( emitter->lifeTime > 0 ) {
531  if ((emitter->startTime + emitter->lifeTime) < cg.time ) {
532  PSys_FreeEmitter( emitter );
533  continue;
534  }
535  }
536 
537  // Updates emitter's position and direction
538  if ( !PSys_UpdateOrientation( system, &(emitter->orientation))) {
539  // PSys_UpdateOrientation returned false, indicating a broken link,
540  // meaning the emitter should be dropped.
541  PSys_FreeEmitter( emitter );
542  continue;
543  }
544 
545  // Check whether the emitter is still within its initialization period or not.
546  if ( emitter->initTime > cg.time ) {
547  // Make sure we don't build up more particles while waiting for the
548  // initialization to end.
549  emitter->lastTime = cg.time - emitter->waitTime;
550  continue;
551  }
552 
553  // Handle regular and surface emitters slightly differently
554  if ( emitter->type < ETYPE_POINT_SURFACE ) {
555  // Regular emitters
556  VectorCopy( emitter->orientation.geometry.origin, root.origin );
557  AxisCopy( emitter->orientation.geometry.axis, root.axis );
558  trace.fraction = 0;
559  } else {
560  // Ground emitters
561  VectorCopy( emitter->orientation.geometry.origin, groundPoint );
562  groundPoint[2] -= emitter->grndDist;
563 
564  CG_Trace( &trace, emitter->orientation.geometry.origin, NULL, NULL, groundPoint, -1, CONTENTS_SOLID );
565  if ( trace.allsolid || trace.startsolid ) {
566  trace.fraction = 1.0f;
567  }
568 
569  if ( trace.fraction < 1.0f ) {
570  // Don't do these calculations when they're not necessary
571  VectorMA( trace.endpos, 1, trace.plane.normal, root.origin ); // <-- Make sure we don't start exactly on the surface!
572  VectorCopy( trace.plane.normal, root.axis[0] );
573  MakeNormalVectors( root.axis[0], root.axis[1], root.axis[2] );
574  } else {
575  // Make sure we don't build up more and more particles while not connected to
576  // the ground.
577  emitter->lastTime = cg.time - emitter->waitTime;
578 
579  // If no waitTime is set, then we should kill the emitter if it can't spawn its set of particles
580  // immediately.
581  if ( emitter->waitTime == 0 ) {
582  PSys_FreeEmitter( emitter );
583  continue;
584  }
585  }
586  }
587 
588  while ((( emitter->lastTime + emitter->waitTime ) <= cg.time ) && ( trace.fraction < 1.0f )) {
589  PSys_Particle_t *particle;
590  vec3_t jitVec, sphereVec;
591  vec3_t tempAxis[3];
592  int i, templateIndex;
593 
594  for ( i = 0; i < emitter->amount; i++ ) {
595  particle = PSys_SpawnParticle( system );
596 
597  // Set starting point based on emitter type
598  VectorSet( jitVec,
599  (crandom() - crandom()) * emitter->posJit,
600  (crandom() - crandom()) * emitter->posJit,
601  (crandom() - crandom()) * emitter->posJit );
602 
603  switch ( emitter->type ) {
604  case ETYPE_POINT:
605  case ETYPE_POINT_SURFACE:
606  VectorAdd( root.origin, jitVec, particle->position );
607  break;
608 
609  case ETYPE_RADIUS:
610  case ETYPE_RADIUS_SURFACE:
611  AxisCopy( root.axis, tempAxis );
612 
613  // NOTE: This function takes deg, not rad
614  RotateAroundDirection( tempAxis, crandom() * 360 );
615 
616  VectorMA( root.origin, emitter->radius, tempAxis[1], particle->position );
617  VectorMA( particle->position, emitter->offset, root.axis[0], particle->position );
618  VectorAdd( particle->position, jitVec, particle->position );
619  break;
620 
621  case ETYPE_SPHERE:
622  VectorSet( sphereVec, crandom() - crandom(), crandom() - crandom(), crandom() - crandom() );
623  VectorNormalize( sphereVec );
624  VectorMA( root.origin, emitter->radius, sphereVec, particle->position );
625  VectorAdd( particle->position, jitVec, particle->position );
626  break;
627 
628  default:
629  VectorCopy( root.origin, particle->position );
630  break;
631  }
632 
633  templateIndex = rand() % emitter->nrTemplates;
634 
635  // Set initial speed
636  VectorMA( particle->position, -emitter->particleTemplates[templateIndex].speed, root.axis[0], particle->oldPosition );
637 
638  // Set other initial particle physics
639  particle->lifeTime = emitter->particleTemplates[templateIndex].lifeTime;
640  particle->spawnTime = cg.time;
641  particle->mass = emitter->particleTemplates[templateIndex].mass;
642 
643  // If the particle has infinite aka zero mass, then the inverse mass must be zero.
644  // Avoid division by zero error.
645  if ( particle->mass ) {
646  particle->invMass = 1 / particle->mass;
647  } else {
648  particle->invMass = 0;
649  }
650 
651  // Assign a look to the particle

710 static float PSys_ApplyFalloff( PSys_Force_t *force, vec3_t particlePos, float sourceVal ) {
711  vec3_t distance;
712  float lerp;
713 
714  switch ( force->AOItype ) {
715  case AOI_INFINITE:
716  return sourceVal;
717  break;
718 
719  case AOI_SPHERE:
720  distance[0] = Distance( force->orientation.geometry.origin, particlePos );
721  if ( force->falloff ) {
722  lerp = 1 - ((distance[0] - force->AOIrange[0]) / force->falloff);
723  if ( lerp < 0 ) lerp = 0;
724  if ( lerp > 1 ) lerp = 1;
725  } else {
726  if ( distance[0] > force->AOIrange[0] ) lerp = 0; else lerp = 1;
727  }
728  return ( lerp * sourceVal );
729  break;
730 
731  case AOI_CYLINDER:
732  // Distance towards either of the horizontal axes is the vertical distance component
733  distance[0] = DistancePointToLine( particlePos, force->orientation.geometry.origin, force->orientation.geometry.axis[1]);
734  distance[0] = distance[0] - force->AOIrange[0];
735  if ( distance[0] < 0 ) distance[0] = 0;
736  // Distance towards the vertical axis is the horizontal distance component (in the circle)
737  distance[1] = DistancePointToLine( particlePos, force->orientation.geometry.origin, force->orientation.geometry.axis[0]);
738  distance[1] = distance[1] - force->AOIrange[1];
739  if ( distance[1] < 0 ) distance[1] = 0;
740  if ( force->falloff ) {
741  distance[2] = sqrt( distance[0] * distance[0] + distance[1] * distance[1] );
742  lerp = 1 - (distance[2] / force->falloff);
743  if ( lerp < 0 ) lerp = 0;
744  if ( lerp > 1 ) lerp = 1;
745  } else {
746  if ( distance[0] > force->AOIrange[0] || distance[1] > force->AOIrange[1] ) lerp = 0; else lerp = 1;
747  }
748  return ( lerp * sourceVal );
749  break;
750 
751  default:
752  break;
753  }
754 
755  return sourceVal;
756 }

758 static void PSys_AccumulateParticle( PSys_System_t *system, PSys_Particle_t *particle ) {
759  PSys_Force_t *force, *next;
760  vec3_t sphereForce;
761  vec3_t dragForce;
762  vec3_t swirlForce, swirlOut;
763  vec3_t v;
764 
765  VectorAdd( particle->accelAccum, system->gravity, particle->accelAccum );
766 
767  force = system->forces.prev_local;
768  for ( ; force != &(system->forces) ; force = next ) {
769  // Grab next now, so if the entity is freed we still have the next one.
770  next = force->prev_local;
771 
772  switch ( force->type ) {
773  case FTYPE_DIRECTIONAL:
774  VectorMA( particle->forceAccum, PSys_ApplyFalloff( force, particle->position, force->value ), force->orientation.geometry.axis[0], particle->forceAccum );
775  break;
776 
777  case FTYPE_SPHERICAL:
778  VectorSubtract( particle->position, force->orientation.geometry.origin, sphereForce );
779  if ( VectorNormalize( sphereForce ) == 0.0f ) {
780  // If the point is placed exactly on the point of force, shoot it upwards instead
781  VectorSet( sphereForce, 0, 0, 1 );
782  }
783 
784  VectorMA( particle->forceAccum, PSys_ApplyFalloff( force, particle->position, force->value ), sphereForce, particle->forceAccum );
785  break;
786 
787  case FTYPE_DRAG:
788  PSys_GetParticleVelocity( particle, v );
789  VectorScale( v, PSys_ApplyFalloff( force, particle->position, force->value ) * -1, dragForce );
790  VectorAdd( particle->forceAccum, dragForce, particle->forceAccum );
791  break;
792 
793  case FTYPE_SWIRL:
794  VectorSubtract( particle->position, force->orientation.geometry.origin, swirlOut );
795  if ( VectorNormalize( swirlOut ) == 0.0f ) {
796  // zero radius means we're at the 'center of the storm'
797  break;
798  }
799  CrossProduct( force->orientation.geometry.axis[0], swirlOut, swirlForce );
800  VectorNormalize( swirlForce );
801  VectorMA( particle->forceAccum, PSys_ApplyFalloff( force, particle->position, force->value ), swirlForce, particle->forceAccum );
802  VectorMA( particle->forceAccum, -1 * PSys_ApplyFalloff( force, particle->position, force->pullIn ), swirlOut, particle->forceAccum );
803  break;
804 
805  default:
806  // should never happen
807  break;
808  }
809  }

825 static void PSys_IntegrateParticle( PSys_Particle_t *particle, float timeStepSquare, float timeStepCorrected ) {
826  vec3_t vel;
827 
828  // Handle (infinite) mass
829  VectorScale( particle->forceAccum, particle->invMass, particle->forceAccum );
830  if ( particle->mass != 0 ) {
831  VectorAdd( particle->forceAccum, particle->accelAccum, particle->forceAccum );
832  }
833 
834  // Save the old position
835  VectorCopy( particle->position, particle->accelAccum );
836 
837 
838  // Timestep corrected Verlet integration:
839  // xi+1 = xi + (xi - xi-1) * (dti / dti-1) + a * dti * dti
840 
841  VectorScale( particle->forceAccum, timeStepSquare, particle->forceAccum );
842  VectorSubtract( particle->position, particle->oldPosition, vel );
843  VectorScale( vel, timeStepCorrected, vel );
844  VectorAdd( particle->position, vel, particle->position );
845  VectorAdd( particle->position, particle->forceAccum, particle->position );
846 
847  // Restore the old position
848  VectorCopy( particle->accelAccum, particle->oldPosition );
849 
850  // Reset the accumulators for the next frame
851  VectorSet( particle->forceAccum, 0, 0, 0 );
852  VectorSet( particle->accelAccum, 0, 0, 0 );
853 
854 
855  // Update ray information if necessary
856  if ( particle->rayParent ) {
857  VectorAdd( particle->rayParent->orientation.geometry.origin, particle->rayOffset, particle->rayOrigin );
858  }
859 }

907  float dist, tempDist;
908  vec3_t dir;
909  qboolean retval;
910 
911  retval = qtrue;
912  pt1 = system->particles.prev_local;
913  for ( ; pt1 != &(system->particles) ; pt1 = next1 ) {
914  next1 = pt1->prev_local;
915 
916  // Determine shortest distance to another particle that has
917  // not yet been affected by the constraint.
918  // NOTE: This last bit is important! Otherwise the particles
919  // will form seperate clusters instead of one cluster!

932  }

943  // ... slide both half the distance overshoot closer together and ...

950  }

1046 static qboolean PSys_ApplyPlaneConstraint( PSys_System_t *system, float value ) {

1048  qboolean retval;
1049  trace_t trace;
1050 
1051  retval = qtrue;

1053  for ( ; particle != &(system->particles) ; particle = next ) {
1054  // Grab next now, so if the entity is freed we still have the next one.
1055  next = particle->prev_local;

1058  // experienced this frame.
1059  CG_Trace( &trace, particle->oldPosition, NULL, NULL, particle->position, -1, CONTENTS_SOLID );
1060  if ( trace.startsolid || trace.allsolid ) {
1061  // make sure the entityNum is set to the one we're stuck in
1062  CG_Trace( &trace, particle->position, NULL, NULL, particle->position, -1, CONTENTS_SOLID );
1063  trace.fraction = 0.0f;
1064  }
1065 
1066  if ( trace.surfaceFlags & SURF_NOIMPACT ) {
1067  PSys_FreeParticle( particle );
1068  continue;
1069  }
1070 
1071  // If the particle moved the whole frame without encountering a solid, then
1072  // everything is okay, else we have to do some math to bounce the particle.
1073  if ( trace.fraction < 1.0f ) {
1074  float dp;

1076 
1077  // If we don't bounce at all, just remove the particle instead of having
1078  // it 'slide' along the ground indefinately.

1102  continue;
1103  }
1104  }

1107  // position to prevent a trace.startsolid when re-evaluating constraints.

1109  VectorAdd( particle->position, v, particle->position );
