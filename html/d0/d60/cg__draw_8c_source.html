44 static void CG_DrawField (int x, int y, int width, int value) {
45  char num[16], *ptr;
46  int l;
47  int frame;
48 
49  if ( width < 1 ) {
50  return;
51  }
52 
53  // draw number string
54  if ( width > 5 ) {
55  width = 5;
56  }
57 
58  switch ( width ) {
59  case 1:
60  value = value > 9 ? 9 : value;
61  value = value < 0 ? 0 : value;
62  break;
63  case 2:
64  value = value > 99 ? 99 : value;
65  value = value < -9 ? -9 : value;
66  break;
67  case 3:
68  value = value > 999 ? 999 : value;
69  value = value < -99 ? -99 : value;
70  break;
71  case 4:
72  value = value > 9999 ? 9999 : value;
73  value = value < -999 ? -999 : value;
74  break;
75  }
76 
77  Com_sprintf (num, sizeof(num), "%i", value);
78  l = strlen(num);
79  if (l > width)
80  l = width;
81  x += 2 + CHAR_WIDTH*(width - l);
82 
83  ptr = num;
84  while (*ptr && l)
85  {
86  if (*ptr == '-')
87  frame = STAT_MINUS;
88  else
89  frame = *ptr -'0';
90 
91  CG_DrawPic(qfalse, x,y, CHAR_WIDTH, CHAR_HEIGHT, cgs.media.numberShaders[frame] );
92  x += CHAR_WIDTH;
93  ptr++;
94  l--;
95  }
96 }

285 static void CG_DrawStatusBarHead( float x ) {
286  vec3_t angles;
287  float size, stretch;
288  float frac;
289 
290  VectorClear( angles );
291 
292  if ( cg.damageTime && cg.time - cg.damageTime < DAMAGE_TIME ) {
293  frac = (float)(cg.time - cg.damageTime ) / DAMAGE_TIME;
294  size = ICON_SIZE * 1.25 * ( 1.5 - frac * 0.5 );
295 
296  stretch = size - ICON_SIZE * 1.25;
297  // kick in the direction of damage
298  x -= stretch * 0.5 + cg.damageX * stretch * 0.5;
299 
300  cg.headStartYaw = 180 + cg.damageX * 45;
301 
302  cg.headEndYaw = 180 + 20 * cos( crandom()*M_PI );
303  cg.headEndPitch = 5 * cos( crandom()*M_PI );
304 
305  cg.headStartTime = cg.time;
306  cg.headEndTime = cg.time + 100 + random() * 2000;
307  } else {
308  if ( cg.time >= cg.headEndTime ) {
309  // select a new head angle
310  cg.headStartYaw = cg.headEndYaw;
311  cg.headStartPitch = cg.headEndPitch;
312  cg.headStartTime = cg.headEndTime;
313  cg.headEndTime = cg.time + 100 + random() * 2000;
314 
315  cg.headEndYaw = 180 + 20 * cos( crandom()*M_PI );
316  cg.headEndPitch = 5 * cos( crandom()*M_PI );
317  }
318 
319  size = ICON_SIZE * 1.25;
320  }
321 
322  // if the server was frozen for a while we may have a bad head start time
323  if ( cg.headStartTime > cg.time ) {
324  cg.headStartTime = cg.time;
325  }
326 
327  frac = ( cg.time - cg.headStartTime ) / (float)( cg.headEndTime - cg.headStartTime );
328  frac = frac * frac * ( 3 - 2 * frac );
329  angles[YAW] = cg.headStartYaw + ( cg.headEndYaw - cg.headStartYaw ) * frac;
330  angles[PITCH] = cg.headStartPitch + ( cg.headEndPitch - cg.headStartPitch ) * frac;
331 
332  CG_DrawHead( x, 480 - size, size, size,
333  cg.snap->ps.clientNum, angles );
334 }
