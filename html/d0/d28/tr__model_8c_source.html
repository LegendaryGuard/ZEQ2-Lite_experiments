31 /*
32 ====================
33 R_RegisterMD3
34 ====================
35 */
36 qhandle_t R_RegisterMD3(const char *name, model_t *mod)
37 {
38  union {
39  unsigned *u;
40  void *v;
41  } buf;
42  int lod;
43  int ident;
44  qboolean loaded = qfalse;
45  int numLoaded;
46  char filename[MAX_QPATH], namebuf[MAX_QPATH+20];
47  char *fext, defex[] = "md3";
48 
49  numLoaded = 0;
50 
51  strcpy(filename, name);
52 
53  fext = strchr(filename, '.');
54  if(!fext)
55  fext = defex;
56  else
57  {
58  *fext = '\0';
59  fext++;
60  }
61 
62  for (lod = MD3_MAX_LODS - 1 ; lod >= 0 ; lod--)
63  {
64  if(lod)
65  Com_sprintf(namebuf, sizeof(namebuf), "%s_%d.%s", filename, lod, fext);
66  else
67  Com_sprintf(namebuf, sizeof(namebuf), "%s.%s", filename, fext);
68 
69  ri.FS_ReadFile( namebuf, &buf.v );
70  if(!buf.u)
71  continue;
72 
73  ident = LittleLong(* (unsigned *) buf.u);
74  if (ident == MD3_IDENT)
75  loaded = R_LoadMD3(mod, lod, buf.u, name);
76  else
77  ri.Printf(PRINT_WARNING,"R_RegisterMD3: unknown fileid for %s\n", name);
78 
79  ri.FS_FreeFile(buf.v);
80 
81  if(loaded)
82  {
83  mod->numLods++;
84  numLoaded++;
85  }
86  else
87  break;
88  }
89 
90  if(numLoaded)
91  {
92  // duplicate into higher lod spots that weren't
93  // loaded, in case the user changes r_lodbias on the fly
94  for(lod--; lod >= 0; lod--)
95  {
96  mod->numLods++;
97  mod->md3[lod] = mod->md3[lod + 1];
98  }
99 
100  return mod->index;
101  }
102 
103 #ifdef _DEBUG
104  ri.Printf(PRINT_WARNING,"R_RegisterMD3: couldn't load %s\n", name);
105 #endif
106 
107  mod->type = MOD_BAD;
108  return 0;
109 }

201 /*
202 ====================
203 RE_RegisterModel
204 
205 Loads in a model for the given name
206 
207 Zero will be returned if the model fails to load.
208 An entry will be retained for failed models as an
209 optimization to prevent disk rescanning if they are
210 asked for again.
211 ====================
212 */
213 qhandle_t RE_RegisterModel( const char *name ) {
214  model_t *mod;
215  qhandle_t hModel;
216  qboolean orgNameFailed = qfalse;
217  int orgLoader = -1;
218  int i;
219  char localName[ MAX_QPATH ];
220  const char *ext;
221  char altName[ MAX_QPATH ];
222 
223  if ( !name || !name[0] ) {
224  ri.Printf( PRINT_ALL, "RE_RegisterModel: NULL name\n" );
225  return 0;
226  }
227 
228  if ( strlen( name ) >= MAX_QPATH ) {
229  ri.Printf( PRINT_ALL, "Model name exceeds MAX_QPATH\n" );
230  return 0;
231  }
232 
233  //
234  // search the currently loaded models
235  //
236  for ( hModel = 1 ; hModel < tr.numModels; hModel++ ) {
237  mod = tr.models[hModel];
238  if ( !strcmp( mod->name, name ) ) {
239  if( mod->type == MOD_BAD ) {
240  return 0;
241  }
242  return hModel;
243  }
244  }
245 
246  // allocate a new model_t
247 
248  if ( ( mod = R_AllocModel() ) == NULL ) {
249  ri.Printf( PRINT_WARNING, "RE_RegisterModel: R_AllocModel() failed for '%s'\n", name);
250  return 0;
251  }
252 
253  // only set the name after the model has been successfully loaded
254  Q_strncpyz( mod->name, name, sizeof( mod->name ) );
255 
256 
257  // make sure the render thread is stopped
258  R_SyncRenderThread();
259 
260  mod->type = MOD_BAD;
261  mod->numLods = 0;
262 
263  //
264  // load the files
265  //
266  Q_strncpyz( localName, name, MAX_QPATH );
267 
268  ext = COM_GetExtension( localName );
269 
270  if( *ext )
271  {
272  // Look for the correct loader and use it
273  for( i = 0; i < numModelLoaders; i++ )
274  {
275  if( !Q_stricmp( ext, modelLoaders[ i ].ext ) )
276  {
277  // Load
278  hModel = modelLoaders[ i ].ModelLoader( localName, mod );
279  break;
280  }
281  }
282 
283  // A loader was found
284  if( i < numModelLoaders )
285  {
286  if( !hModel )
287  {
288  // Loader failed, most likely because the file isn't there;
289  // try again without the extension
290  orgNameFailed = qtrue;
291  orgLoader = i;
292  COM_StripExtension( name, localName, MAX_QPATH );
293  }
294  else
295  {
296  // Something loaded
297  return mod->index;
298  }
299  }
300  }
301 
302  // Try and find a suitable match using all
303  // the model formats supported
304  for( i = 0; i < numModelLoaders; i++ )
305  {
306  if (i == orgLoader)
307  continue;
308 
309  Com_sprintf( altName, sizeof (altName), "%s.%s", localName, modelLoaders[ i ].ext );
310 
311  // Load
312  hModel = modelLoaders[ i ].ModelLoader( altName, mod );
313 
314  if( hModel )
315  {
316  if( orgNameFailed )
317  {
318  ri.Printf( PRINT_DEVELOPER, "WARNING: %s not present, using %s instead\n",
319  name, altName );
320  }
321 
322  break;
323  }
324  }
325 
326  return hModel;
327 }

334 static qboolean R_LoadMD3 (model_t *mod, int lod, void *buffer, const char *mod_name ) {
335  int i, j;
336  md3Header_t *pinmodel;
337  md3Frame_t *frame;
338  md3Surface_t *surf;
339  md3Shader_t *shader;
340  md3Triangle_t *tri;
341  md3St_t *st;
342  md3XyzNormal_t *xyz;
343  md3Tag_t *tag;
344  int version;
345  int size;
346 
347  pinmodel = (md3Header_t *)buffer;
348 
349  version = LittleLong (pinmodel->version);
350  if (version != MD3_VERSION) {
351  ri.Printf( PRINT_WARNING, "R_LoadMD3: %s has wrong version (%i should be %i)\n",
352  mod_name, version, MD3_VERSION);
353  return qfalse;
354  }
355 
356  mod->type = MOD_MESH;
357  size = LittleLong(pinmodel->ofsEnd);
358  mod->dataSize += size;
359  mod->md3[lod] = ri.Hunk_Alloc( size, h_low );
360 
361  Com_Memcpy (mod->md3[lod], buffer, LittleLong(pinmodel->ofsEnd) );
362 
363  LL(mod->md3[lod]->ident);
364  LL(mod->md3[lod]->version);
365  LL(mod->md3[lod]->numFrames);
366  LL(mod->md3[lod]->numTags);
367  LL(mod->md3[lod]->numSurfaces);
368  LL(mod->md3[lod]->ofsFrames);
369  LL(mod->md3[lod]->ofsTags);
370  LL(mod->md3[lod]->ofsSurfaces);
371  LL(mod->md3[lod]->ofsEnd);
372 
373  if ( mod->md3[lod]->numFrames < 1 ) {
374  ri.Printf( PRINT_WARNING, "R_LoadMD3: %s has no frames\n", mod_name );
375  return qfalse;
376  }
377 
378  // swap all the frames
379  frame = (md3Frame_t *) ( (byte *)mod->md3[lod] + mod->md3[lod]->ofsFrames );
380  for ( i = 0 ; i < mod->md3[lod]->numFrames ; i++, frame++) {
381  frame->radius = LittleFloat( frame->radius );
382  for ( j = 0 ; j < 3 ; j++ ) {
383  frame->bounds[0][j] = LittleFloat( frame->bounds[0][j] );
384  frame->bounds[1][j] = LittleFloat( frame->bounds[1][j] );
385  frame->localOrigin[j] = LittleFloat( frame->localOrigin[j] );
386  }
387  }
388 
389  // swap all the tags
390  tag = (md3Tag_t *) ( (byte *)mod->md3[lod] + mod->md3[lod]->ofsTags );
391  for ( i = 0 ; i < mod->md3[lod]->numTags * mod->md3[lod]->numFrames ; i++, tag++) {
392  for ( j = 0 ; j < 3 ; j++ ) {
393  tag->origin[j] = LittleFloat( tag->origin[j] );
394  tag->axis[0][j] = LittleFloat( tag->axis[0][j] );
395  tag->axis[1][j] = LittleFloat( tag->axis[1][j] );
396  tag->axis[2][j] = LittleFloat( tag->axis[2][j] );
397  }
398  }
399 
400  // swap all the surfaces
401  surf = (md3Surface_t *) ( (byte *)mod->md3[lod] + mod->md3[lod]->ofsSurfaces );
402  for ( i = 0 ; i < mod->md3[lod]->numSurfaces ; i++) {
403 
404  LL(surf->ident);
405  LL(surf->flags);
406  LL(surf->numFrames);
407  LL(surf->numShaders);
408  LL(surf->numTriangles);
409  LL(surf->ofsTriangles);
410  LL(surf->numVerts);
411  LL(surf->ofsShaders);
412  LL(surf->ofsSt);
413  LL(surf->ofsXyzNormals);
414  LL(surf->ofsEnd);
415 
416  if ( surf->numVerts > SHADER_MAX_VERTEXES ) {
417  ri.Printf(PRINT_WARNING, "R_LoadMD3: %s has more than %i verts on a surface (%i).\n",
418  mod_name, SHADER_MAX_VERTEXES, surf->numVerts );
419  return qfalse;
420  }
421  if ( surf->numTriangles*3 > SHADER_MAX_INDEXES ) {
422  ri.Printf(PRINT_WARNING, "R_LoadMD3: %s has more than %i triangles on a surface (%i).\n",
423  mod_name, SHADER_MAX_INDEXES / 3, surf->numTriangles );
424  return qfalse;
425  }
426 
427  // change to surface identifier
428  surf->ident = SF_MD3;
429 
430  // lowercase the surface name so skin compares are faster
431  Q_strlwr( surf->name );
432 
433  // strip off a trailing _1 or _2
434  // this is a crutch for q3data being a mess
435  j = strlen( surf->name );
436  if ( j > 2 && surf->name[j-2] == '_' ) {
437  surf->name[j-2] = 0;
438  }
439 
440  // register the shaders
441  shader = (md3Shader_t *) ( (byte *)surf + surf->ofsShaders );
442  for ( j = 0 ; j < surf->numShaders ; j++, shader++ ) {
443  shader_t *sh;
444 
445  sh = R_FindShader( shader->name, LIGHTMAP_NONE, qtrue );
446  if ( sh->defaultShader ) {
447  shader->shaderIndex = 0;
448  } else {
449  shader->shaderIndex = sh->index;
450  }
451  }
452 
453  // swap all the triangles
454  tri = (md3Triangle_t *) ( (byte *)surf + surf->ofsTriangles );
455  for ( j = 0 ; j < surf->numTriangles ; j++, tri++ ) {
456  LL(tri->indexes[0]);
457  LL(tri->indexes[1]);
458  LL(tri->indexes[2]);
459  }
460 
461  // swap all the ST
462  st = (md3St_t *) ( (byte *)surf + surf->ofsSt );
463  for ( j = 0 ; j < surf->numVerts ; j++, st++ ) {
464  st->st[0] = LittleFloat( st->st[0] );
465  st->st[1] = LittleFloat( st->st[1] );
466  }
467 
468  // swap all the XyzNormals
469  xyz = (md3XyzNormal_t *) ( (byte *)surf + surf->ofsXyzNormals );
470  for ( j = 0 ; j < surf->numVerts * surf->numFrames ; j++, xyz++ )
471  {
472  xyz->xyz[0] = LittleShort( xyz->xyz[0] );
473  xyz->xyz[1] = LittleShort( xyz->xyz[1] );
474  xyz->xyz[2] = LittleShort( xyz->xyz[2] );
475 
476  xyz->normal = LittleShort( xyz->normal );
477  }
478 
479 
480  // find the next surface
481  surf = (md3Surface_t *)( (byte *)surf + surf->ofsEnd );
482  }
483 
484  return qtrue;
485 }

591 ================
592 R_LerpTag
593 ================
594 */
595 int R_LerpTag( orientation_t *tag, qhandle_t handle, int startFrame, int endFrame,
596  float frac, const char *tagName ) {
597  md3Tag_t *start, *end;
598  int i;
599  float frontLerp, backLerp;
600  model_t *model;
601 
602  model = R_GetModelByHandle( handle );
603  if ( !model->md3[0] )
604  {
605 
606  if( model->type == MOD_IQM ) {
607  return R_IQMLerpTag( tag, model->modelData,
608  startFrame, endFrame,
609  frac, tagName );
610  } else {
611 
612  AxisClear( tag->axis );
613  VectorClear( tag->origin );
614  return qfalse;
615 
616  }
617  }
618  else
619  {
620  start = R_GetTag( model->md3[0], startFrame, tagName );
621  end = R_GetTag( model->md3[0], endFrame, tagName );
622  if ( !start || !end ) {
623  AxisClear( tag->axis );
624  VectorClear( tag->origin );
625  return qfalse;
626  }
627  }
628 
629  frontLerp = frac;
630  backLerp = 1.0f - frac;
631 
632  for ( i = 0 ; i < 3 ; i++ ) {
633  tag->origin[i] = start->origin[i] * backLerp + end->origin[i] * frontLerp;
634  tag->axis[0][i] = start->axis[0][i] * backLerp + end->axis[0][i] * frontLerp;
635  tag->axis[1][i] = start->axis[1][i] * backLerp + end->axis[1][i] * frontLerp;
636  tag->axis[2][i] = start->axis[2][i] * backLerp + end->axis[2][i] * frontLerp;
637  }
638  VectorNormalize( tag->axis[0] );
639  VectorNormalize( tag->axis[1] );
640  VectorNormalize( tag->axis[2] );
641  return qtrue;
642 }
