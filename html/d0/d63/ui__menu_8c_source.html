71 int aspectRatioCorrectedY(int ycoord){
72  char heightCvar[8];
73  char widthCvar[8];
74  char modeCvar[8];
75  float height;
76  float width;
77  float ratio;
78  float percentage;
79  int ret = ycoord;
80  trap_Cvar_VariableStringBuffer("r_customheight", heightCvar, sizeof(heightCvar));
81  trap_Cvar_VariableStringBuffer("r_customwidth", widthCvar, sizeof(widthCvar));
82  trap_Cvar_VariableStringBuffer("r_mode", modeCvar, sizeof(modeCvar));
83  height = atof(heightCvar);
84  width = atof(widthCvar);
85  if(height > 0 && width > 0 && atoi(modeCvar) == -1){
86  ratio = height / width;
87  if(ratio < 3.0f / 4.0f){
88  ratio *= 0.95f;
89  percentage = 3.0f / (4.0f * ratio);
90  ret = percentage * ycoord;
91  }
92  }
93 
94  return ret;
95 }
