137 PB_DS_CLASS_C_DEC::
138 get_nearest_larger_size(size_type n) const
139 {
140  const std::size_t* const p_upper = std::upper_bound(detail::g_a_sizes,
141  detail::g_a_sizes + detail::num_distinct_sizes, n);
142 
143  if (p_upper == detail::g_a_sizes + detail::num_distinct_sizes)
144  __throw_resize_error();
145  return *p_upper;
146 }

151 get_nearest_smaller_size(size_type n) const
152 {
153  const std::size_t* p_lower = std::lower_bound(detail::g_a_sizes,
154  detail::g_a_sizes + detail::num_distinct_sizes, n);
155 
156  if (*p_lower >= n && p_lower != detail::g_a_sizes)
157  --p_lower;
158  if (*p_lower < m_start_size)
159  return m_start_size;
160  return *p_lower;
