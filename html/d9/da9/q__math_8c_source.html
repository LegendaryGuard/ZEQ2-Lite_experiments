319  m[2][2] = vf[2];
320 
321  memcpy( im, m, sizeof( im ) );
322 
323  im[0][1] = m[1][0];
324  im[0][2] = m[2][0];
325  im[1][0] = m[0][1];
326  im[1][2] = m[2][1];
327  im[2][0] = m[0][2];
328  im[2][1] = m[1][2];

330  memset( zrot, 0, sizeof( zrot ) );

335  zrot[0][1] = sin( rad );

337  zrot[1][1] = cos( rad );
