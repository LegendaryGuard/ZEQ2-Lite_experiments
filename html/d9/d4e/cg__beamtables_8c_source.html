42 ==================
43 CG_WipeBeamTable
44 ==================
45  Wipes the given beamtable clean, preparing it for a new set of waypoints.
46 */
47 static void CG_WipeBeamTable( beamTable_t *beamTable ) {
48  int i;
49 
50  if (!beamTable->alreadyWiped) {
51 
52  memset( beamTable, 0, sizeof(beamTable_t) );
53 
54  beamTable->table_activeList.next = &(beamTable->table_activeList);
55  beamTable->table_activeList.prev = &(beamTable->table_activeList);
56  beamTable->table_freeList = beamTable->table;
57 
58  // singly link the free list
59  for ( i = 0 ; i < MAX_BEAMTABLE_SIZE - 1 ; i++ ) {
60  beamTable->table[i].next = &(beamTable->table[i+1]);
61  }
62 
63  beamTable->alreadyWiped = qtrue;
64  }
65 
66  // Ensure we can always do an update after a shorter time
67  // for the first segment. This keeps the beam from 'weirding out',
68  // and doing things like cutting through the ground
69  beamTable->updateTime = cg.time + BEAMTABLE_UPDATE / 3.0f;
70 }

94 /*
95 ========================
96 CG_AllocBeamTableElem
97 ========================
98  Will always succeed, even if it requires returning the previous tableElem.
99 */
100 static beamTableElem_t *CG_AllocBeamTableElem( beamTable_t *beamTable ) {
101  beamTableElem_t *tableElem;
102 
103  if ( !(tableElem = beamTable->table_freeList) ) {
104  // No free entities, so use the one at the end of the chain.
105  tableElem = beamTable->table_activeList.next;
106  return tableElem;
107  }
108 
109  beamTable->table_freeList = beamTable->table_freeList->next;
110 
111  // link into the active list
112  tableElem->next = beamTable->table_activeList.next;
113  tableElem->prev = &(beamTable->table_activeList);
114  beamTable->table_activeList.next->prev = tableElem;
115  beamTable->table_activeList.next = tableElem;
116 
117  return tableElem;
118 }

122 ====================
123 CG_BeamTableUpdate
124 ====================
125  Updates the beamtable associated with the client entity.
126 */
127 void CG_BeamTableUpdate( centity_t *cent, float width, qhandle_t shader, char *tagName ) {
128  beamTable_t *currentTable;
129 
130  // Failsafe against arrays running out of bounds.
131  if ( ( cent->currentState.clientNum < 0 ) || ( cent->currentState.clientNum >= MAX_CLIENTS ) ) {
132  CG_Error( "Bad clientNum on beamhead entity" );
133  return; // Don't know if return is necessary, but do it anyway.
134  }
135 
136  // Do we work on the alternate fire tables, or the primary fire ones?
137  if ( cent->currentState.weapon > ALTWEAPON_OFFSET) {
138  currentTable = &(beamTableAlternate[cent->currentState.clientNum]);
139  } else {
140  currentTable = &(beamTablePrimary[cent->currentState.clientNum]);
141  }
142 
143  // set the width, shader and tag to lock on to.
144  currentTable->width = width;
145  currentTable->shader = shader;
146  Q_strncpyz(currentTable->tagName, tagName, sizeof(currentTable->tagName));
147 
148  // always update the terminator symbol
149  VectorCopy(cent->lerpOrigin, currentTable->table_activeList.pos);
150  BG_EvaluateTrajectoryDelta( &(cent->currentState), &(cent->currentState.pos), cg.time, currentTable->table_activeList.tangent );
151  if ( !VectorNormalize( currentTable->table_activeList.tangent ) ) {
152  VectorSet( currentTable->table_activeList.tangent, 0, 0, 1);
153  }
154 
155  // If the entity has reached or exceeded the spawntime of a new waypoint,
156  // place a new one.
157  if ( currentTable->updateTime < cg.time ) {
158  beamTableElem_t *tableElem;
159 
160  // allocate a waypoint cell
161  tableElem = CG_AllocBeamTableElem( currentTable );
162 
163  // get the position and tangent from the terminator, and store them
164  VectorCopy( currentTable->table_activeList.pos, tableElem->pos );
165  VectorCopy( currentTable->table_activeList.tangent, tableElem->tangent );
166 
167  // set the next update time
168  currentTable->updateTime = cg.time + BEAMTABLE_UPDATE;
169  }
170 
171  // mark the table as having been active
172  currentTable->activeThisFrame = qtrue;
173  currentTable->alreadyWiped = qfalse;
174 }

181 /*
182 ====================
183 CG_BezierMidPoints
184 ====================
185  Calculates the mid points to form a set of four control points.
186  Uses the tangent (aka beamhead direction) to do so.
187 */
188 static void CG_BezierMidPoints( const vec3_t startPos, const vec3_t endPos,
189  const vec3_t startTan, const vec3_t endTan,
190  vec3_t midPos1, vec3_t midPos2 ) {
191  float tangentScale;
192 
193  // Set the scale of the tangents to reach halfway. This will give fluent curves
194  // through the calculated midpoints we're going to calculate.
195  tangentScale = 0.5 * Distance(startPos, endPos);
196 
197  // Scale and add startTan to the startPos to get midPos1
198  VectorCopy( startTan, midPos1 );
199  VectorNormalize( midPos1 );
200  VectorScale( midPos1, tangentScale, midPos1 );
201  VectorAdd( startPos, midPos1, midPos1 );
202 
203  // Invert, scale and add endTan to the endPos to get midPos2
204  VectorCopy( endTan, midPos2 );
205  VectorNormalize( midPos2 );
206  VectorScale( midPos2, -tangentScale, midPos2 );
207  VectorAdd( endPos, midPos2, midPos2 );
208 }

210 /*
211 ================
212 CG_BezierPoint
213 ================
214  Calculates the position and tangent within a bezier curve at a set
215  parameter value between 0 and 1.
216 */
217 static void CG_BezierPoint( const vec3_t point0, const vec3_t point1,
218  const vec3_t point2, const vec3_t point3,
219  float t, vec3_t outPoint, vec3_t outTangent ) {
220  float T[4];
221  float H[4];
222  vec3_t tmp[4];
223 
224  // Initialize the T-matrix for the point;
225  T[0] = t * t * t;
226  T[1] = t * t;
227  T[2] = t;
228  T[3] = 1;
229 
230  // Initialize the H-matrix
231  H[0] = -1 * T[0] + 3 * T[1] - 3 * T[2] + 1 * T[3];
232  H[1] = 3 * T[0] - 6 * T[1] + 3 * T[2];
233  H[2] = -3 * T[0] + 3 * T[1];
234  H[3] = 1 * T[0];
235 
236  // Calculate the new point
237  VectorScale(point0, H[0], tmp[0]);
238  VectorScale(point1, H[1], tmp[1]);
239  VectorScale(point2, H[2], tmp[2]);
240  VectorScale(point3, H[3], tmp[3]);
241  VectorAdd( tmp[0], tmp[1], outPoint);
242  VectorAdd( tmp[2], outPoint, outPoint);
243  VectorAdd( tmp[3], outPoint, outPoint);
244 
245 
246  // Initialize the T-matrix for the tangent;
247  T[0] = 3 * t * t;
248  T[1] = 2 * t;
249  T[2] = 1;
250  T[3] = 0;
251 
252  // Re-initialize the H-matrix
253  H[0] = -1 * T[0] + 3 * T[1] - 3 * T[2] + 1 * T[3];
254  H[1] = 3 * T[0] - 6 * T[1] + 3 * T[2];
255  H[2] = -3 * T[0] + 3 * T[1];
256  H[3] = 1 * T[0];
257 
258  // Calculate the new point's tangent
259  VectorScale(point0, H[0], tmp[0]);
260  VectorScale(point1, H[1], tmp[1]);
261  VectorScale(point2, H[2], tmp[2]);
262  VectorScale(point3, H[3], tmp[3]);
263  VectorAdd( tmp[0], tmp[1], outTangent);
264  VectorAdd( tmp[2], outTangent, outTangent);
265  VectorAdd( tmp[3], outTangent, outTangent);
266 }

311 // NOTE: This old way of doing the tesscount is bad for beams that
312 // have wide curves. It uses more tesselations than is
313 // affordable.
314 /*
315 static int CG_TessCount (const vec3_t startPos, const vec3_t endPos) {
316  float dist;
317  int tess;
318 
319 
320  dist = Distance(startPos, endPos);
321  tess = dist / TESS_DISTANCE;
322  if ( tess < 2 ) {
323  tess = 2;
324  }
325 
326  return tess;
327 }
328 */
329 static int CG_TessCount( const beamTableElem_t *startElem, const beamTableElem_t *endElem ) {
330  float directTangentRatio;
331  float indirectTangentRatio;
332  float tangentRatio;
333  float dist;
334  vec3_t dir;
335  int detail;
336 
337  VectorSubtract( endElem->pos, startElem->pos, dir );
338  if ( (dist = VectorNormalize( dir )) == 0.0f ) {
339  //If the distance is 0, then we don't make ANY tesselations at all.
340  return 0;
341  }
342 
343  directTangentRatio = 1 - DotProduct( startElem->tangent, endElem->tangent );
344  indirectTangentRatio = 1 - DotProduct( dir, endElem->tangent );
345  // NOTE: These should produce numbers between 0 and 2. 2 When the tangents
346  // differ 180 degrees, 0 when they do not differ.
347 
348  // Pick the biggest one
349  tangentRatio = indirectTangentRatio > directTangentRatio ? indirectTangentRatio : directTangentRatio;
350 
351  // Clamp the detail value
352  if ( r_beamDetail.value <= 0 ) {
353  r_beamDetail.value = 0;
354  detail = 0;
355  }
356 
357  if ( r_beamDetail.value >= 4 ) {
358  detail = 40;
359  if ( tangentRatio < 1 ) {
360  tangentRatio = 1;
361  }
362  } else if ( r_beamDetail.value == 3 ) {
363  detail = 30;
364  if ( tangentRatio < 0.5 ) {
365  tangentRatio = 0.5;
366  }
367  } else if ( r_beamDetail.value == 2 ) {
368  detail = 20;
369  if ( tangentRatio < 0.25 ) {
370  tangentRatio = 0.25;
371  }
372  } else if ( r_beamDetail.value == 1 ) {
373  detail = 10;
374  if ( tangentRatio < 0.1 ) {
375  tangentRatio = 0.1;
376  }
377  }
378 
379  // On small distances always use the maximum tangent ratio
380  if ( dist < 100 ) {
381  tangentRatio = 2;
382  }
383 
384  return 1 + floor( tangentRatio * detail );
385 }

469  // If the tag can not be found, wipe the table and don't display the beam

520  }
521

524  currentElem = currentElem->prev;
525  }
526

528  currentTable->activeThisFrame = qfalse;
