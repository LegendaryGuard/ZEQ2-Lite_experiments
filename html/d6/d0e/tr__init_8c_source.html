379 /*
380 ==================
381 RB_ReadPixels
382 
383 Reads an image but takes care of alignment issues for reading RGB images.
384 
385 Reads a minimum offset for where the RGB data starts in the image from
386 integer stored at pointer offset. When the function has returned the actual
387 offset was written back to address offset. This address will always have an
388 alignment of packAlign to ensure efficient copying.
389 
390 Stores the length of padding after a line of pixels to address padlen
391 
392 Return value must be freed with ri.Hunk_FreeTempMemory()
393 ==================
394 */
395 
396 byte *RB_ReadPixels(int x, int y, int width, int height, size_t *offset, int *padlen)
397 {
398  byte *buffer, *bufstart;
399  int padwidth, linelen;
400  GLint packAlign;
401 
402  qglGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
403 
404  linelen = width * 3;
405  padwidth = PAD(linelen, packAlign);
406 
407  // Allocate a few more bytes so that we can choose an alignment we like
408  buffer = ri.Hunk_AllocateTempMemory(padwidth * height + *offset + packAlign - 1);
409 
410  bufstart = PADP((intptr_t) buffer + *offset, packAlign);
411  qglReadPixels(x, y, width, height, GL_RGB, GL_UNSIGNED_BYTE, bufstart);
412 
413  *offset = bufstart - buffer;
414  *padlen = padwidth - linelen;
415 
416  return buffer;
417 }

419 /*
420 ==================
421 RB_TakeScreenshot
422 ==================
423 */
424 void RB_TakeScreenshot(int x, int y, int width, int height, char *fileName)
425 {
426  byte *allbuf, *buffer;
427  byte *srcptr, *destptr;
428  byte *endline, *endmem;
429  byte temp;
430 
431  int linelen, padlen;
432  size_t offset = 18, memcount;
433 
434  allbuf = RB_ReadPixels(x, y, width, height, &offset, &padlen);
435  buffer = allbuf + offset - 18;
436 
437  Com_Memset (buffer, 0, 18);
438  buffer[2] = 2; // uncompressed type
439  buffer[12] = width & 255;
440  buffer[13] = width >> 8;
441  buffer[14] = height & 255;
442  buffer[15] = height >> 8;
443  buffer[16] = 24; // pixel size
444 
445  // swap rgb to bgr and remove padding from line endings
446  linelen = width * 3;
447 
448  srcptr = destptr = allbuf + offset;
449  endmem = srcptr + (linelen + padlen) * height;
450 
451  while(srcptr < endmem)
452  {
453  endline = srcptr + linelen;
454 
455  while(srcptr < endline)
456  {
457  temp = srcptr[0];
458  *destptr++ = srcptr[2];
459  *destptr++ = srcptr[1];
460  *destptr++ = temp;
461 
462  srcptr += 3;
463  }
464 
465  // Skip the pad
466  srcptr += padlen;
467  }
468 
469  memcount = linelen * height;
470 
471  // gamma correct
472  if(glConfig.deviceSupportsGamma)
473  R_GammaCorrect(allbuf + offset, memcount);
474 
475  ri.FS_WriteFile(fileName, buffer, memcount + 18);
476 
477  ri.Hunk_FreeTempMemory(allbuf);
478 }

595 /*
596 ====================
597 R_LevelShot
598 
599 levelshots are specialized 128*128 thumbnails for
600 the menu system, sampled down from full screen distorted images
601 ====================
602 */
603 void R_LevelShot( void ) {
604  char checkname[MAX_OSPATH];
605  byte *buffer;
606  byte *source, *allsource;
607  byte *src, *dst;
608  size_t offset = 0;
609  int padlen;
610  int x, y;
611  int r, g, b;
612  float xScale, yScale;
613  int xx, yy;
614 
615  Com_sprintf(checkname, sizeof(checkname), "levelshots/%s.tga", tr.world->baseName);
616 
617  allsource = RB_ReadPixels(0, 0, glConfig.vidWidth, glConfig.vidHeight, &offset, &padlen);
618  source = allsource + offset;
619 
620  buffer = ri.Hunk_AllocateTempMemory(128 * 128*3 + 18);
621  Com_Memset (buffer, 0, 18);
622  buffer[2] = 2; // uncompressed type
623  buffer[12] = 128;
624  buffer[14] = 128;
625  buffer[16] = 24; // pixel size
626 
627  // resample from source
628  xScale = glConfig.vidWidth / 512.0f;
629  yScale = glConfig.vidHeight / 384.0f;
630  for ( y = 0 ; y < 128 ; y++ ) {
631  for ( x = 0 ; x < 128 ; x++ ) {
632  r = g = b = 0;
633  for ( yy = 0 ; yy < 3 ; yy++ ) {
634  for ( xx = 0 ; xx < 4 ; xx++ ) {
635  src = source + (3 * glConfig.vidWidth + padlen) * (int)((y*3 + yy) * yScale) +
636  3 * (int) ((x*4 + xx) * xScale);
637  r += src[0];
638  g += src[1];
639  b += src[2];
640  }
641  }
642  dst = buffer + 18 + 3 * ( y * 128 + x );
643  dst[0] = b / 12;
644  dst[1] = g / 12;
645  dst[2] = r / 12;
646  }
647  }
648 
649  // gamma correct
650  if ( glConfig.deviceSupportsGamma ) {
651  R_GammaCorrect( buffer + 18, 128 * 128 * 3 );
652  }
653 
654  ri.FS_WriteFile( checkname, buffer, 128 * 128*3 + 18 );
655 
656  ri.Hunk_FreeTempMemory(buffer);
657  ri.Hunk_FreeTempMemory(allsource);
658 
659  ri.Printf( PRINT_ALL, "Wrote %s\n", checkname );
660 }

783 ==================
784 RB_TakeVideoFrameCmd
785 ==================
786 */
787 const void *RB_TakeVideoFrameCmd( const void *data )
788 {
789  const videoFrameCommand_t *cmd;
790  byte *cBuf;
791  size_t memcount, linelen;
792  int padwidth, avipadwidth, padlen, avipadlen;
793  GLint packAlign;
794 
795  cmd = (const videoFrameCommand_t *)data;
796 
797  qglGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
798 
799  linelen = cmd->width * 3;
800 
801  // Alignment stuff for glReadPixels
802  padwidth = PAD(linelen, packAlign);
803  padlen = padwidth - linelen;
804  // AVI line padding
805  avipadwidth = PAD(linelen, AVI_LINE_PADDING);
806  avipadlen = avipadwidth - linelen;
807 
808  cBuf = PADP(cmd->captureBuffer, packAlign);
809 
810  qglReadPixels(0, 0, cmd->width, cmd->height, GL_RGB,
811  GL_UNSIGNED_BYTE, cBuf);
812 
813  memcount = padwidth * cmd->height;
814 
815  // gamma correct
816  if(glConfig.deviceSupportsGamma)
817  R_GammaCorrect(cBuf, memcount);
818 
819  if(cmd->motionJpeg)
820  {
821  memcount = RE_SaveJPGToBuffer(cmd->encodeBuffer, linelen * cmd->height,
822  r_aviMotionJpegQuality->integer,
823  cmd->width, cmd->height, cBuf, padlen);
824  ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, memcount);
825  }
826  else
827  {
828  byte *lineend, *memend;
829  byte *srcptr, *destptr;
830 
831  srcptr = cBuf;
832  destptr = cmd->encodeBuffer;
833  memend = srcptr + memcount;
834 
835  // swap R and B and remove line paddings
836  while(srcptr < memend)
837  {
838  lineend = srcptr + linelen;
839  while(srcptr < lineend)
840  {
841  *destptr++ = srcptr[2];
842  *destptr++ = srcptr[1];
843  *destptr++ = srcptr[0];
844  srcptr += 3;
845  }
846 
847  Com_Memset(destptr, '\0', avipadlen);
848  destptr += avipadlen;
849 
850  srcptr += padlen;
851  }
852 
853  ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, avipadwidth * cmd->height);
854  }
855 
856  return (const void *)(cmd + 1);
857 }
