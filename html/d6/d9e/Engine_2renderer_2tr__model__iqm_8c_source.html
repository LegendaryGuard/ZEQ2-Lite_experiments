28 static qboolean IQM_CheckRange( iqmHeader_t *header, int offset,
29  int count,int size ) {
30  // return true if the range specified by offset, count and size
31  // doesn't fit into the file
32  return ( count <= 0 ||
33  offset < 0 ||
34  offset > header->filesize ||
35  offset + count * size < 0 ||
36  offset + count * size > header->filesize );
37 }
38 // "multiply" 3x4 matrices, these are assumed to be the top 3 rows
39 // of a 4x4 matrix with the last row = (0 0 0 1)
40 static void Matrix34Multiply( float *a, float *b, float *out ) {
41  out[ 0] = a[0] * b[0] + a[1] * b[4] + a[ 2] * b[ 8];
42  out[ 1] = a[0] * b[1] + a[1] * b[5] + a[ 2] * b[ 9];
43  out[ 2] = a[0] * b[2] + a[1] * b[6] + a[ 2] * b[10];
44  out[ 3] = a[0] * b[3] + a[1] * b[7] + a[ 2] * b[11] + a[ 3];
45  out[ 4] = a[4] * b[0] + a[5] * b[4] + a[ 6] * b[ 8];
46  out[ 5] = a[4] * b[1] + a[5] * b[5] + a[ 6] * b[ 9];
47  out[ 6] = a[4] * b[2] + a[5] * b[6] + a[ 6] * b[10];
48  out[ 7] = a[4] * b[3] + a[5] * b[7] + a[ 6] * b[11] + a[ 7];
49  out[ 8] = a[8] * b[0] + a[9] * b[4] + a[10] * b[ 8];
50  out[ 9] = a[8] * b[1] + a[9] * b[5] + a[10] * b[ 9];
51  out[10] = a[8] * b[2] + a[9] * b[6] + a[10] * b[10];
52  out[11] = a[8] * b[3] + a[9] * b[7] + a[10] * b[11] + a[11];
53 }
54 static void InterpolateMatrix( float *a, float *b, float lerp, float *mat ) {
55  float unLerp = 1.0f - lerp;
56 
57  mat[ 0] = a[ 0] * unLerp + b[ 0] * lerp;
58  mat[ 1] = a[ 1] * unLerp + b[ 1] * lerp;
59  mat[ 2] = a[ 2] * unLerp + b[ 2] * lerp;
60  mat[ 3] = a[ 3] * unLerp + b[ 3] * lerp;
61  mat[ 4] = a[ 4] * unLerp + b[ 4] * lerp;
62  mat[ 5] = a[ 5] * unLerp + b[ 5] * lerp;
63  mat[ 6] = a[ 6] * unLerp + b[ 6] * lerp;
64  mat[ 7] = a[ 7] * unLerp + b[ 7] * lerp;
65  mat[ 8] = a[ 8] * unLerp + b[ 8] * lerp;
66  mat[ 9] = a[ 9] * unLerp + b[ 9] * lerp;
67  mat[10] = a[10] * unLerp + b[10] * lerp;
68  mat[11] = a[11] * unLerp + b[11] * lerp;
69 }
70 static void JointToMatrix( vec4_t rot, vec3_t scale, vec3_t trans,
71  float *mat ) {
72  float xx = 2.0f * rot[0] * rot[0];
73  float yy = 2.0f * rot[1] * rot[1];
74  float zz = 2.0f * rot[2] * rot[2];
75  float xy = 2.0f * rot[0] * rot[1];
76  float xz = 2.0f * rot[0] * rot[2];
77  float yz = 2.0f * rot[1] * rot[2];
78  float wx = 2.0f * rot[3] * rot[0];
79  float wy = 2.0f * rot[3] * rot[1];
80  float wz = 2.0f * rot[3] * rot[2];
81 
82  mat[ 0] = scale[0] * (1.0f - (yy + zz));
83  mat[ 1] = scale[0] * (xy - wz);
84  mat[ 2] = scale[0] * (xz + wy);
85  mat[ 3] = trans[0];
86  mat[ 4] = scale[1] * (xy + wz);
87  mat[ 5] = scale[1] * (1.0f - (xx + zz));
88  mat[ 6] = scale[1] * (yz - wx);
89  mat[ 7] = trans[1];
90  mat[ 8] = scale[2] * (xz - wy);
91  mat[ 9] = scale[2] * (yz + wx);
92  mat[10] = scale[2] * (1.0f - (xx + yy));
93  mat[11] = trans[2];
94 }
95 static void Matrix34Invert( float *inMat, float *outMat )
96 {
97  vec3_t trans;
98  float invSqrLen, *v;
99 
100  outMat[ 0] = inMat[ 0]; outMat[ 1] = inMat[ 4]; outMat[ 2] = inMat[ 8];
101  outMat[ 4] = inMat[ 1]; outMat[ 5] = inMat[ 5]; outMat[ 6] = inMat[ 9];
102  outMat[ 8] = inMat[ 2]; outMat[ 9] = inMat[ 6]; outMat[10] = inMat[10];
103 
104  v = outMat + 0; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
105  v = outMat + 4; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
106  v = outMat + 8; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
107 
108  trans[0] = inMat[ 3];
109  trans[1] = inMat[ 7];
110  trans[2] = inMat[11];
111 
112  outMat[ 3] = -DotProduct(outMat + 0, trans);
113  outMat[ 7] = -DotProduct(outMat + 4, trans);
114  outMat[11] = -DotProduct(outMat + 8, trans);
115 }

124 qboolean R_LoadIQM( model_t *mod, void *buffer, int filesize, const char *mod_name ) {
125  iqmHeader_t *header;
126  iqmVertexArray_t *vertexarray;
127  iqmTriangle_t *triangle;
128  iqmMesh_t *mesh;
129  iqmJoint_t *joint;
130  iqmPose_t *pose;
131  iqmBounds_t *bounds;
132  unsigned short *framedata;
133  char *str;
134  int i, j;
135  float jointMats[IQM_MAX_JOINTS * 2 * 12];
136  float *mat;
137  size_t size, joint_names;
138  iqmData_t *iqmData;
139  srfIQModel_t *surface;
140 
141  if( filesize < sizeof(iqmHeader_t) ) {
142  return qfalse;
143  }
144 
145  header = (iqmHeader_t *)buffer;
146  if( Q_strncmp( header->magic, IQM_MAGIC, sizeof(header->magic) ) ) {
147  return qfalse;
148  }
149 
150  LL( header->version );
151  if( header->version != IQM_VERSION ) {
152  ri.Printf(PRINT_WARNING, "R_LoadIQM: %s is a unsupported IQM version (%d), only version %d is supported.\n",
153  mod_name, header->version, IQM_VERSION);
154  return qfalse;
155  }
156 
157  LL( header->filesize );
158  if( header->filesize > filesize || header->filesize > 16<<20 ) {
159  return qfalse;
160  }
161 
162  LL( header->flags );
163  LL( header->num_text );
164  LL( header->ofs_text );
165  LL( header->num_meshes );
166  LL( header->ofs_meshes );
167  LL( header->num_vertexarrays );
168  LL( header->num_vertexes );
169  LL( header->ofs_vertexarrays );
170  LL( header->num_triangles );
171  LL( header->ofs_triangles );
172  LL( header->ofs_adjacency );
173  LL( header->num_joints );
174  LL( header->ofs_joints );
175  LL( header->num_poses );
176  LL( header->ofs_poses );
177  LL( header->num_anims );
178  LL( header->ofs_anims );
179  LL( header->num_frames );
180  LL( header->num_framechannels );
181  LL( header->ofs_frames );
182  LL( header->ofs_bounds );
183  LL( header->num_comment );
184  LL( header->ofs_comment );
185  LL( header->num_extensions );
186  LL( header->ofs_extensions );
187 
188  // check ioq3 joint limit
189  if ( header->num_joints > IQM_MAX_JOINTS ) {
190  ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has more than %d joints (%d).\n",
191  mod_name, IQM_MAX_JOINTS, header->num_joints);
192  return qfalse;
193  }
194 
195  // check and swap vertex arrays
196  if( IQM_CheckRange( header, header->ofs_vertexarrays,
197  header->num_vertexarrays,
198  sizeof(iqmVertexArray_t) ) ) {
199  return qfalse;
200  }
201  vertexarray = (iqmVertexArray_t *)((byte *)header + header->ofs_vertexarrays);
202  for( i = 0; i < header->num_vertexarrays; i++, vertexarray++ ) {
203  int j, n, *intPtr;
204 
205  if( vertexarray->size <= 0 || vertexarray->size > 4 ) {
206  return qfalse;
207  }
208 
209  // total number of values
210  n = header->num_vertexes * vertexarray->size;
211 
212  switch( vertexarray->format ) {
213  case IQM_BYTE:
214  case IQM_UBYTE:
215  // 1 byte, no swapping necessary
216  if( IQM_CheckRange( header, vertexarray->offset,
217  n, sizeof(byte) ) ) {
218  return qfalse;
219  }
220  break;
221  case IQM_INT:
222  case IQM_UINT:
223  case IQM_FLOAT:
224  // 4-byte swap
225  if( IQM_CheckRange( header, vertexarray->offset,
226  n, sizeof(float) ) ) {
227  return qfalse;
228  }
229  intPtr = (int *)((byte *)header + vertexarray->offset);
230  for( j = 0; j < n; j++, intPtr++ ) {
231  LL( *intPtr );
232  }
233  break;
234  default:
235  // not supported
236  return qfalse;
237  break;
238  }
239 
240  switch( vertexarray->type ) {
241  case IQM_POSITION:
242  case IQM_NORMAL:
243  if( vertexarray->format != IQM_FLOAT ||
244  vertexarray->size != 3 ) {
245  return qfalse;
246  }
247  break;
248  case IQM_TANGENT:
249  if( vertexarray->format != IQM_FLOAT ||
250  vertexarray->size != 4 ) {
251  return qfalse;
252  }
253  break;
254  case IQM_TEXCOORD:
255  if( vertexarray->format != IQM_FLOAT ||
256  vertexarray->size != 2 ) {
257  return qfalse;
258  }
259  break;
260  case IQM_BLENDINDEXES:
261  case IQM_BLENDWEIGHTS:
262  if( vertexarray->format != IQM_UBYTE ||
263  vertexarray->size != 4 ) {
264  return qfalse;
265  }
266  break;
267  case IQM_COLOR:
268  if( vertexarray->format != IQM_UBYTE ||
269  vertexarray->size != 4 ) {
270  return qfalse;
271  }
272  break;
273  }
274  }
275 
276  // check and swap triangles
277  if( IQM_CheckRange( header, header->ofs_triangles,
278  header->num_triangles, sizeof(iqmTriangle_t) ) ) {
279  return qfalse;
280  }
281  triangle = (iqmTriangle_t *)((byte *)header + header->ofs_triangles);
282  for( i = 0; i < header->num_triangles; i++, triangle++ ) {
283  LL( triangle->vertex[0] );
284  LL( triangle->vertex[1] );
285  LL( triangle->vertex[2] );
286 
287  if( triangle->vertex[0] > header->num_vertexes ||
288  triangle->vertex[1] > header->num_vertexes ||
289  triangle->vertex[2] > header->num_vertexes ) {
290  return qfalse;
291  }
292  }
293 
294  // check and swap meshes
295  if( IQM_CheckRange( header, header->ofs_meshes,
296  header->num_meshes, sizeof(iqmMesh_t) ) ) {
297  return qfalse;
298  }
299  mesh = (iqmMesh_t *)((byte *)header + header->ofs_meshes);
300  for( i = 0; i < header->num_meshes; i++, mesh++) {
301  LL( mesh->name );
302  LL( mesh->material );
303  LL( mesh->first_vertex );
304  LL( mesh->num_vertexes );
305  LL( mesh->first_triangle );
306  LL( mesh->num_triangles );
307 
308  // check ioq3 limits
309  if ( mesh->num_vertexes > SHADER_MAX_VERTEXES )
310  {
311  ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has more than %i verts on a surface (%i).\n",
312  mod_name, SHADER_MAX_VERTEXES, mesh->num_vertexes );
313  return qfalse;
314  }
315  if ( mesh->num_triangles*3 > SHADER_MAX_INDEXES )
316  {
317  ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has more than %i triangles on a surface (%i).\n",
318  mod_name, SHADER_MAX_INDEXES / 3, mesh->num_triangles );
319  return qfalse;
320  }
321 
322  if( mesh->first_vertex >= header->num_vertexes ||
323  mesh->first_vertex + mesh->num_vertexes > header->num_vertexes ||
324  mesh->first_triangle >= header->num_triangles ||
325  mesh->first_triangle + mesh->num_triangles > header->num_triangles ||
326  mesh->name >= header->num_text ||
327  mesh->material >= header->num_text ) {
328  return qfalse;
329  }
330  }
331 
332  // check and swap joints
333  if( IQM_CheckRange( header, header->ofs_joints,
334  header->num_joints, sizeof(iqmJoint_t) ) ) {
335  return qfalse;
336  }
337  joint = (iqmJoint_t *)((byte *)header + header->ofs_joints);
338  joint_names = 0;
339  for( i = 0; i < header->num_joints; i++, joint++ ) {
340  LL( joint->name );
341  LL( joint->parent );
342  LL( joint->translate[0] );
343  LL( joint->translate[1] );
344  LL( joint->translate[2] );
345  LL( joint->rotate[0] );
346  LL( joint->rotate[1] );
347  LL( joint->rotate[2] );
348  LL( joint->rotate[3] );
349  LL( joint->scale[0] );
350  LL( joint->scale[1] );
351  LL( joint->scale[2] );
352 
353  if( joint->parent < -1 ||
354  joint->parent >= (int)header->num_joints ||
355  joint->name >= (int)header->num_text ) {
356  return qfalse;
357  }
358  joint_names += strlen( (char *)header + header->ofs_text +
359  joint->name ) + 1;
360  }
361 
362  // check and swap poses
363  if( header->num_poses != header->num_joints ) {
364  return qfalse;
365  }
366  if( IQM_CheckRange( header, header->ofs_poses,
367  header->num_poses, sizeof(iqmPose_t) ) ) {
368  return qfalse;
369  }
370  pose = (iqmPose_t *)((byte *)header + header->ofs_poses);
371  for( i = 0; i < header->num_poses; i++, pose++ ) {
372  LL( pose->parent );
373  LL( pose->mask );
374  LL( pose->channeloffset[0] );
375  LL( pose->channeloffset[1] );
376  LL( pose->channeloffset[2] );
377  LL( pose->channeloffset[3] );
378  LL( pose->channeloffset[4] );
379  LL( pose->channeloffset[5] );
380  LL( pose->channeloffset[6] );
381  LL( pose->channeloffset[7] );
382  LL( pose->channeloffset[8] );
383  LL( pose->channeloffset[9] );
384  LL( pose->channelscale[0] );
385  LL( pose->channelscale[1] );
386  LL( pose->channelscale[2] );
387  LL( pose->channelscale[3] );
388  LL( pose->channelscale[4] );
389  LL( pose->channelscale[5] );
390  LL( pose->channelscale[6] );
391  LL( pose->channelscale[7] );
392  LL( pose->channelscale[8] );
393  LL( pose->channelscale[9] );
394  }
395 
396  if (header->ofs_bounds)
397  {
398  // check and swap model bounds
399  if(IQM_CheckRange(header, header->ofs_bounds,
400  header->num_frames, sizeof(*bounds)))
401  {
402  return qfalse;
403  }
404  bounds = (iqmBounds_t *) ((byte *) header + header->ofs_bounds);
405  for(i = 0; i < header->num_frames; i++)
406  {
407  LL(bounds->bbmin[0]);
408  LL(bounds->bbmin[1]);
409  LL(bounds->bbmin[2]);
410  LL(bounds->bbmax[0]);
411  LL(bounds->bbmax[1]);
412  LL(bounds->bbmax[2]);
413 
414  bounds++;
415  }
416  }
417 
418  // allocate the model and copy the data
419  size = sizeof(iqmData_t);
420  size += header->num_meshes * sizeof( srfIQModel_t );
421  size += header->num_joints * header->num_frames * 12 * sizeof( float );
422  if(header->ofs_bounds)
423  size += header->num_frames * 6 * sizeof(float); // model bounds
424  size += header->num_vertexes * 3 * sizeof(float); // positions
425  size += header->num_vertexes * 2 * sizeof(float); // texcoords
426  size += header->num_vertexes * 3 * sizeof(float); // normals
427  size += header->num_vertexes * 4 * sizeof(float); // tangents
428  size += header->num_vertexes * 4 * sizeof(byte); // blendIndexes
429  size += header->num_vertexes * 4 * sizeof(byte); // blendWeights
430  size += header->num_vertexes * 4 * sizeof(byte); // colors
431  size += header->num_joints * sizeof(int); // parents
432  size += header->num_triangles * 3 * sizeof(int); // triangles
433  size += joint_names; // joint names
434 
435  mod->type = MOD_IQM;
436  iqmData = (iqmData_t *)ri.Hunk_Alloc( size, h_low );
437  mod->modelData = iqmData;
438 
439  // fill header
440  iqmData->num_vertexes = header->num_vertexes;
441  iqmData->num_triangles = header->num_triangles;
442  iqmData->num_frames = header->num_frames;
443  iqmData->num_surfaces = header->num_meshes;
444  iqmData->num_joints = header->num_joints;
445  iqmData->surfaces = (srfIQModel_t *)(iqmData + 1);
446  iqmData->poseMats = (float *) (iqmData->surfaces + iqmData->num_surfaces);
447  if(header->ofs_bounds)
448  {
449  iqmData->bounds = iqmData->poseMats + 12 * header->num_joints * header->num_frames;
450  iqmData->positions = iqmData->bounds + 6 * header->num_frames;
451  }
452  else
453  iqmData->positions = iqmData->poseMats + 12 * header->num_joints * header->num_frames;
454  iqmData->texcoords = iqmData->positions + 3 * header->num_vertexes;
455  iqmData->normals = iqmData->texcoords + 2 * header->num_vertexes;
456  iqmData->tangents = iqmData->normals + 3 * header->num_vertexes;
457  iqmData->blendIndexes = (byte *)(iqmData->tangents + 4 * header->num_vertexes);
458  iqmData->blendWeights = iqmData->blendIndexes + 4 * header->num_vertexes;
459  iqmData->colors = iqmData->blendWeights + 4 * header->num_vertexes;
460  iqmData->jointParents = (int *)(iqmData->colors + 4 * header->num_vertexes);
461  iqmData->triangles = iqmData->jointParents + header->num_joints;
462  iqmData->names = (char *)(iqmData->triangles + 3 * header->num_triangles);
463 
464  // calculate joint matrices and their inverses
465  // they are needed only until the pose matrices are calculated
466  mat = jointMats;
467  joint = (iqmJoint_t *)((byte *)header + header->ofs_joints);
468  for( i = 0; i < header->num_joints; i++, joint++ ) {
469  float baseFrame[12], invBaseFrame[12];
470 
471  JointToMatrix( joint->rotate, joint->scale, joint->translate, baseFrame );
472  Matrix34Invert( baseFrame, invBaseFrame );
473 
474  if ( joint->parent >= 0 )
475  {
476  Matrix34Multiply( jointMats + 2 * 12 * joint->parent, baseFrame, mat );
477  mat += 12;
478  Matrix34Multiply( invBaseFrame, jointMats + 2 * 12 * joint->parent + 12, mat );
479  mat += 12;
480  }
481  else
482  {
483  Com_Memcpy( mat, baseFrame, sizeof(baseFrame) );
484  mat += 12;
485  Com_Memcpy( mat, invBaseFrame, sizeof(invBaseFrame) );
486  mat += 12;
487  }
488  }
489 
490  // calculate pose matrices
491  framedata = (unsigned short *)((byte *)header + header->ofs_frames);
492  mat = iqmData->poseMats;
493  for( i = 0; i < header->num_frames; i++ ) {
494  pose = (iqmPose_t *)((byte *)header + header->ofs_poses);
495  for( j = 0; j < header->num_poses; j++, pose++ ) {
496  vec3_t translate;
497  vec4_t rotate;
498  vec3_t scale;
499  float mat1[12], mat2[12];
500 
501  translate[0] = pose->channeloffset[0];
502  if( pose->mask & 0x001)
503  translate[0] += *framedata++ * pose->channelscale[0];
504  translate[1] = pose->channeloffset[1];
505  if( pose->mask & 0x002)
506  translate[1] += *framedata++ * pose->channelscale[1];
507  translate[2] = pose->channeloffset[2];
508  if( pose->mask & 0x004)
509  translate[2] += *framedata++ * pose->channelscale[2];
510 
511  rotate[0] = pose->channeloffset[3];
512  if( pose->mask & 0x008)
513  rotate[0] += *framedata++ * pose->channelscale[3];
514  rotate[1] = pose->channeloffset[4];
515  if( pose->mask & 0x010)
516  rotate[1] += *framedata++ * pose->channelscale[4];
517  rotate[2] = pose->channeloffset[5];
518  if( pose->mask & 0x020)
519  rotate[2] += *framedata++ * pose->channelscale[5];
520  rotate[3] = pose->channeloffset[6];
521  if( pose->mask & 0x040)
522  rotate[3] += *framedata++ * pose->channelscale[6];
523 
524  scale[0] = pose->channeloffset[7];
525  if( pose->mask & 0x080)
526  scale[0] += *framedata++ * pose->channelscale[7];
527  scale[1] = pose->channeloffset[8];
528  if( pose->mask & 0x100)
529  scale[1] += *framedata++ * pose->channelscale[8];
530  scale[2] = pose->channeloffset[9];
531  if( pose->mask & 0x200)
532  scale[2] += *framedata++ * pose->channelscale[9];
533 
534  // construct transformation matrix
535  JointToMatrix( rotate, scale, translate, mat1 );
536 
537  if( pose->parent >= 0 ) {
538  Matrix34Multiply( jointMats + 12 * 2 * pose->parent,
539  mat1, mat2 );
540  } else {
541  Com_Memcpy( mat2, mat1, sizeof(mat1) );
542  }
543 
544  Matrix34Multiply( mat2, jointMats + 12 * (2 * j + 1), mat );
545  mat += 12;
546  }
547  }
548 
549  // register shaders
550  // overwrite the material offset with the shader index
551  mesh = (iqmMesh_t *)((byte *)header + header->ofs_meshes);
552  surface = iqmData->surfaces;

667 static int R_CullIQM( iqmData_t *data, trRefEntity_t *ent ) {
668  vec3_t bounds[2];

670  int i;
671 
672  if (!data->bounds) {
673  tr.pc.c_box_cull_md3_clip++;

675  }

678  oldBounds = data->bounds + 6*ent->e.oldframe;
679  newBounds = data->bounds + 6*ent->e.frame;

682  for (i = 0 ; i < 3 ; i++) {
683  bounds[0][i] = oldBounds[i] < newBounds[i] ? oldBounds[i] : newBounds[i];
684  bounds[1][i] = oldBounds[i+3] > newBounds[i+3] ? oldBounds[i+3] : newBounds[i+3];

693  tr.pc.c_box_cull_md3_clip++;
694  return CULL_CLIP;
695  case CULL_OUT:
696  default:
697  tr.pc.c_box_cull_md3_out++;

709  int i, j;

712  const vec_t defaultBounds[6] = { -8, -8, -8, 8, 8, 8 };

716

723  bounds = data->bounds + 6*ent->e.frame;
724  } else {
725  bounds = defaultBounds;
726  }
727  VectorSubtract( bounds+3, bounds, diag );
728  VectorMA( bounds, 0.5f, diag, center );

731

735  if ( localOrigin[j] - radius >= fog->bounds[1][j] ) {
736  break;
737  }
738  if ( localOrigin[j] + radius <= fog->bounds[0][j] ) {
739  break;
740  }
741  }
742  if ( j == 3 ) {
743  return i;
744  }
745  }
746 
747  return 0;

864 static void ComputeJointMats( iqmData_t *data, int frame, int oldframe,
865  float backlerp, float *mat ) {

872  for( i = 0; i < data->num_joints; i++, joint++ ) {
873  if( *joint >= 0 ) {
874  Matrix34Multiply( mat + 12 * *joint,
875  mat1 + 12*i, mat + 12*i );
876  } else {
877  Com_Memcpy( mat + 12*i, mat1 + 12*i, 12 * sizeof(float) );
878  }
879  }
880  } else {
881  mat1 = data->poseMats + 12 * data->num_joints * frame;
882  mat2 = data->poseMats + 12 * data->num_joints * oldframe;
883

885  if( *joint >= 0 ) {
886  float tmpMat[12];
887  InterpolateMatrix( mat1 + 12*i, mat2 + 12*i,
888  backlerp, tmpMat );
889  Matrix34Multiply( mat + 12 * *joint,
890  tmpMat, mat + 12*i );
891 
892  } else {
893  InterpolateMatrix( mat1 + 12*i, mat2 + 12*i,
894  backlerp, mat );
895  }
896  }
897  }

908 void RB_IQMSurfaceAnim( surfaceType_t *surface ) {
909  srfIQModel_t *surf = (srfIQModel_t *)surface;
910  iqmData_t *data = surf->data;
911  float jointMats[IQM_MAX_JOINTS * 12];
912  int i;
913 
914  vec4_t *outXYZ = &tess.xyz[tess.numVertexes];
915  vec4_t *outNormal = &tess.normal[tess.numVertexes];
916  vec2_t (*outTexCoord)[2] = &tess.texCoords[tess.numVertexes];
917  color4ub_t *outColor = &tess.vertexColors[tess.numVertexes];
918 
919  int frame = backEnd.currentEntity->e.frame % data->num_frames;
920  int oldframe = backEnd.currentEntity->e.oldframe % data->num_frames;
921  float backlerp = backEnd.currentEntity->e.backlerp;
922 
923  int *tri;
924  glIndex_t *ptr;
925  glIndex_t base;
926 
927  RB_CHECKOVERFLOW( surf->num_vertexes, surf->num_triangles * 3 );
928 
929  // compute interpolated joint matrices
930  ComputeJointMats( data, frame, oldframe, backlerp, jointMats );
931 
932  // transform vertexes and fill other data
933  for( i = 0; i < surf->num_vertexes;
934  i++, outXYZ++, outNormal++, outTexCoord++, outColor++ ) {
935  int j, k;
936  float vtxMat[12];
937  float nrmMat[9];
938  int vtx = i + surf->first_vertex;
939 
940  // compute the vertex matrix by blending the up to
941  // four blend weights
942  for( k = 0; k < 12; k++ )
943  vtxMat[k] = data->blendWeights[4*vtx]
944  * jointMats[12*data->blendIndexes[4*vtx] + k];
945  for( j = 1; j < 4; j++ ) {
946  if( data->blendWeights[4*vtx + j] <= 0 )
947  break;
948  for( k = 0; k < 12; k++ )
949  vtxMat[k] += data->blendWeights[4*vtx + j]
950  * jointMats[12*data->blendIndexes[4*vtx + j] + k];
951  }
952  for( k = 0; k < 12; k++ )
953  vtxMat[k] *= 1.0f / 255.0f;
954 
955  // compute the normal matrix as transpose of the adjoint
956  // of the vertex matrix
957  nrmMat[ 0] = vtxMat[ 5]*vtxMat[10] - vtxMat[ 6]*vtxMat[ 9];
958  nrmMat[ 1] = vtxMat[ 6]*vtxMat[ 8] - vtxMat[ 4]*vtxMat[10];
959  nrmMat[ 2] = vtxMat[ 4]*vtxMat[ 9] - vtxMat[ 5]*vtxMat[ 8];
960  nrmMat[ 3] = vtxMat[ 2]*vtxMat[ 9] - vtxMat[ 1]*vtxMat[10];
961  nrmMat[ 4] = vtxMat[ 0]*vtxMat[10] - vtxMat[ 2]*vtxMat[ 8];
962  nrmMat[ 5] = vtxMat[ 1]*vtxMat[ 8] - vtxMat[ 0]*vtxMat[ 9];
963  nrmMat[ 6] = vtxMat[ 1]*vtxMat[ 6] - vtxMat[ 2]*vtxMat[ 5];
964  nrmMat[ 7] = vtxMat[ 2]*vtxMat[ 4] - vtxMat[ 0]*vtxMat[ 6];
965  nrmMat[ 8] = vtxMat[ 0]*vtxMat[ 5] - vtxMat[ 1]*vtxMat[ 4];
966 
967  (*outTexCoord)[0][0] = data->texcoords[2*vtx + 0];
968  (*outTexCoord)[0][1] = data->texcoords[2*vtx + 1];
969  (*outTexCoord)[1][0] = (*outTexCoord)[0][0];
970  (*outTexCoord)[1][1] = (*outTexCoord)[0][1];
971 
972  (*outXYZ)[0] =
973  vtxMat[ 0] * data->positions[3*vtx+0] +
974  vtxMat[ 1] * data->positions[3*vtx+1] +
975  vtxMat[ 2] * data->positions[3*vtx+2] +
976  vtxMat[ 3];
977  (*outXYZ)[1] =
978  vtxMat[ 4] * data->positions[3*vtx+0] +
979  vtxMat[ 5] * data->positions[3*vtx+1] +
980  vtxMat[ 6] * data->positions[3*vtx+2] +
981  vtxMat[ 7];
982  (*outXYZ)[2] =
983  vtxMat[ 8] * data->positions[3*vtx+0] +
984  vtxMat[ 9] * data->positions[3*vtx+1] +
985  vtxMat[10] * data->positions[3*vtx+2] +
986  vtxMat[11];
987  (*outXYZ)[3] = 1.0f;
988 
989  (*outNormal)[0] =
990  nrmMat[ 0] * data->normals[3*vtx+0] +
991  nrmMat[ 1] * data->normals[3*vtx+1] +
992  nrmMat[ 2] * data->normals[3*vtx+2];
993  (*outNormal)[1] =
994  nrmMat[ 3] * data->normals[3*vtx+0] +
995  nrmMat[ 4] * data->normals[3*vtx+1] +
996  nrmMat[ 5] * data->normals[3*vtx+2];
997  (*outNormal)[2] =
998  nrmMat[ 6] * data->normals[3*vtx+0] +
999  nrmMat[ 7] * data->normals[3*vtx+1] +
1000  nrmMat[ 8] * data->normals[3*vtx+2];
1001  (*outNormal)[3] = 0.0f;
1002 
1003  (*outColor)[0] = data->colors[4*vtx+0];
1004  (*outColor)[1] = data->colors[4*vtx+1];
1005  (*outColor)[2] = data->colors[4*vtx+2];
1006  (*outColor)[3] = data->colors[4*vtx+3];
1007  }
1008 
1009  tri = data->triangles + 3 * surf->first_triangle;
1010  ptr = &tess.indexes[tess.numIndexes];
1011  base = tess.numVertexes;
1012 
1013  for( i = 0; i < surf->num_triangles; i++ ) {

1020  tess.numVertexes += surf->num_vertexes;
