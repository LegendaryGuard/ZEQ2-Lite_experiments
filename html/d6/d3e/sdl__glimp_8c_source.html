217 /*
218 ===============
219 GLimp_CompareModes
220 ===============
221 */
222 static int GLimp_CompareModes( const void *a, const void *b )
223 {
224  const float ASPECT_EPSILON = 0.001f;
225  SDL_Rect *modeA = *(SDL_Rect **)a;
226  SDL_Rect *modeB = *(SDL_Rect **)b;
227  float aspectA = (float)modeA->w / (float)modeA->h;
228  float aspectB = (float)modeB->w / (float)modeB->h;
229  int areaA = modeA->w * modeA->h;
230  int areaB = modeB->w * modeB->h;
231  float aspectDiffA = fabs( aspectA - displayAspect );
232  float aspectDiffB = fabs( aspectB - displayAspect );
233  float aspectDiffsDiff = aspectDiffA - aspectDiffB;
234 
235  if( aspectDiffsDiff > ASPECT_EPSILON )
236  return 1;
237  else if( aspectDiffsDiff < -ASPECT_EPSILON )
238  return -1;
239  else
240  return areaA - areaB;
241 }

244 /*
245 ===============
246 GLimp_DetectAvailableModes
247 ===============
248 */
249 static void GLimp_DetectAvailableModes(void)
250 {
251  char buf[ MAX_STRING_CHARS ] = { 0 };
252  SDL_Rect **modes;
253  int numModes;
254  int i;
255 
256  modes = SDL_ListModes( videoInfo->vfmt, SDL_OPENGL | SDL_FULLSCREEN );
257 
258  if( !modes )
259  {
260  ri.Printf( PRINT_WARNING, "Can't get list of available modes\n" );
261  return;
262  }
263 
264  if( modes == (SDL_Rect **)-1 )
265  {
266  ri.Printf( PRINT_ALL, "Display supports any resolution\n" );
267  return; // can set any resolution
268  }
269 
270  for( numModes = 0; modes[ numModes ]; numModes++ );
271 
272  if( numModes > 1 )
273  qsort( modes, numModes, sizeof( SDL_Rect* ), GLimp_CompareModes );
274 
275  for( i = 0; i < numModes; i++ )
276  {
277  const char *newModeString = va( "%ux%u ", modes[ i ]->w, modes[ i ]->h );
278 
279  if( strlen( newModeString ) < (int)sizeof( buf ) - strlen( buf ) )
280  Q_strcat( buf, sizeof( buf ), newModeString );
281  else
282  ri.Printf( PRINT_WARNING, "Skipping mode %ux%x, buffer too small\n", modes[i]->w, modes[i]->h );
283  }
284 
285  if( *buf )
286  {
287  buf[ strlen( buf ) - 1 ] = 0;
288  ri.Printf( PRINT_ALL, "Available modes: '%s'\n", buf );
289  ri.Cvar_Set( "r_availableModes", buf );
290  }
291 }

295 GLimp_SetMode

298 static int GLimp_SetMode(int mode, qboolean fullscreen, qboolean noborder)
299 {
300  const char* glstring;
301  int sdlcolorbits;
302  int colorbits, depthbits, stencilbits;
303  int tcolorbits, tdepthbits, tstencilbits;
304  int samples;
305  int i = 0;
306  SDL_Surface *vidscreen = NULL;
307  Uint32 flags = SDL_OPENGL;
308 
309  ri.Printf( PRINT_ALL, "Initializing OpenGL display\n");
310 
311  if ( r_allowResize->integer )
312  flags |= SDL_RESIZABLE;
313 
314  if( videoInfo == NULL )
315  {
316  static SDL_VideoInfo sVideoInfo;
317  static SDL_PixelFormat sPixelFormat;
318 
319  videoInfo = SDL_GetVideoInfo( );
320 
321  // Take a copy of the videoInfo
322  Com_Memcpy( &sPixelFormat, videoInfo->vfmt, sizeof( SDL_PixelFormat ) );
323  sPixelFormat.palette = NULL; // Should already be the case
324  Com_Memcpy( &sVideoInfo, videoInfo, sizeof( SDL_VideoInfo ) );
325  sVideoInfo.vfmt = &sPixelFormat;
326  videoInfo = &sVideoInfo;
327 
328  if( videoInfo->current_h > 0 )
329  {
330  // Guess the display aspect ratio through the desktop resolution
331  // by assuming (relatively safely) that it is set at or close to
332  // the display's native aspect ratio
333  displayAspect = (float)videoInfo->current_w / (float)videoInfo->current_h;
334 
335  ri.Printf( PRINT_ALL, "Estimated display aspect: %.3f\n", displayAspect );
336  }
337  else
338  {
339  ri.Printf( PRINT_ALL,
340  "Cannot estimate display aspect, assuming 1.333\n" );
341  }
342  }
343 
344  ri.Printf (PRINT_ALL, "...setting mode %d:", mode );
345 
346  if ( !R_GetModeInfo( &glConfig.vidWidth, &glConfig.vidHeight, &glConfig.windowAspect, mode ) )
347  {
348  ri.Printf( PRINT_ALL, " invalid mode\n" );
349  return RSERR_INVALID_MODE;
350  }
351  ri.Printf( PRINT_ALL, " %d %d\n", glConfig.vidWidth, glConfig.vidHeight);
352 
353  if (fullscreen)
354  {
355  flags |= SDL_FULLSCREEN;
356  glConfig.isFullscreen = qtrue;
357  }
358  else
359  {
360  if (noborder)
361  flags |= SDL_NOFRAME;
362 
363  glConfig.isFullscreen = qfalse;
364  }
365 
366  colorbits = r_colorbits->value;
367  if ((!colorbits) || (colorbits >= 32))
368  colorbits = 24;
369 
370  if (!r_depthbits->value)
371  depthbits = 24;
372  else
373  depthbits = r_depthbits->value;
374  stencilbits = r_stencilbits->value;
375  samples = r_ext_multisample->value;
376 
377  for (i = 0; i < 16; i++)
378  {
379  // 0 - default
380  // 1 - minus colorbits
381  // 2 - minus depthbits
382  // 3 - minus stencil
383  if ((i % 4) == 0 && i)
384  {
385  // one pass, reduce
386  switch (i / 4)
387  {
388  case 2 :
389  if (colorbits == 24)
390  colorbits = 16;
391  break;
392  case 1 :
393  if (depthbits == 24)
394  depthbits = 16;
395  else if (depthbits == 16)
396  depthbits = 8;
397  case 3 :
398  if (stencilbits == 24)
399  stencilbits = 16;
400  else if (stencilbits == 16)
401  stencilbits = 8;
402  }
403  }
404 
405  tcolorbits = colorbits;
406  tdepthbits = depthbits;
407  tstencilbits = stencilbits;
408 
409  if ((i % 4) == 3)
410  { // reduce colorbits
411  if (tcolorbits == 24)
412  tcolorbits = 16;
413  }
414 
415  if ((i % 4) == 2)
416  { // reduce depthbits
417  if (tdepthbits == 24)
418  tdepthbits = 16;
419  else if (tdepthbits == 16)
420  tdepthbits = 8;
421  }
422 
423  if ((i % 4) == 1)
424  { // reduce stencilbits
425  if (tstencilbits == 24)
426  tstencilbits = 16;
427  else if (tstencilbits == 16)
428  tstencilbits = 8;
429  else
430  tstencilbits = 0;
431  }
432 
433  sdlcolorbits = 4;
434  if (tcolorbits == 24)
435  sdlcolorbits = 8;
436 
437 #ifdef __sgi /* Fix for SGIs grabbing too many bits of color */
438  if (sdlcolorbits == 4)
439  sdlcolorbits = 0; /* Use minimum size for 16-bit color */
440 
441  /* Need alpha or else SGIs choose 36+ bit RGB mode */
442  SDL_GL_SetAttribute( SDL_GL_ALPHA_SIZE, 1);
443 #endif
444 
445  SDL_GL_SetAttribute( SDL_GL_RED_SIZE, sdlcolorbits );
446  SDL_GL_SetAttribute( SDL_GL_GREEN_SIZE, sdlcolorbits );
447  SDL_GL_SetAttribute( SDL_GL_BLUE_SIZE, sdlcolorbits );
448  SDL_GL_SetAttribute( SDL_GL_DEPTH_SIZE, tdepthbits );
449  SDL_GL_SetAttribute( SDL_GL_STENCIL_SIZE, tstencilbits );
450 
451  SDL_GL_SetAttribute( SDL_GL_MULTISAMPLEBUFFERS, samples ? 1 : 0 );
452  SDL_GL_SetAttribute( SDL_GL_MULTISAMPLESAMPLES, samples );
453 
454  if(r_stereoEnabled->integer)
455  {
456  glConfig.stereoEnabled = qtrue;
457  SDL_GL_SetAttribute(SDL_GL_STEREO, 1);
458  }
459  else
460  {
461  glConfig.stereoEnabled = qfalse;
462  SDL_GL_SetAttribute(SDL_GL_STEREO, 0);
463  }
464 
465  SDL_GL_SetAttribute( SDL_GL_DOUBLEBUFFER, 1 );
466 
467 #if 0 // See http://bugzilla.icculus.org/show_bug.cgi?id=3526
468  // If not allowing software GL, demand accelerated
469  if( !r_allowSoftwareGL->integer )
470  {
471  if( SDL_GL_SetAttribute( SDL_GL_ACCELERATED_VISUAL, 1 ) < 0 )
472  {
473  ri.Printf( PRINT_ALL, "Unable to guarantee accelerated "
474  "visual with libSDL < 1.2.10\n" );
475  }
476  }
477 #endif
478 
479  if( SDL_GL_SetAttribute( SDL_GL_SWAP_CONTROL, r_swapInterval->integer ) < 0 )
480  ri.Printf( PRINT_ALL, "r_swapInterval requires libSDL >= 1.2.10\n" );
481 
482 //#ifdef USE_ICON
483  {
484  /* SDL_Surface *icon = SDL_CreateRGBSurfaceFrom(
485  (void *)CLIENT_WINDOW_ICON.pixel_data,
486  CLIENT_WINDOW_ICON.width,
487  CLIENT_WINDOW_ICON.height,
488  CLIENT_WINDOW_ICON.bytes_per_pixel * 8,
489  CLIENT_WINDOW_ICON.bytes_per_pixel * CLIENT_WINDOW_ICON.width,
490 #ifdef Q3_LITTLE_ENDIAN
491  0x000000FF, 0x0000FF00, 0x00FF0000, 0xFF000000
492 #else
493  0xFF000000, 0x00FF0000, 0x0000FF00, 0x000000FF
494 #endif
495  );*/
496 
497  SDL_WM_SetIcon( SDL_LoadBMP("ZEQ2/zeq2.bmp"), NULL );
498  //SDL_FreeSurface( icon );
499  }
500 //#endif
501 
502  SDL_WM_SetCaption(CLIENT_WINDOW_TITLE, CLIENT_WINDOW_MIN_TITLE);
503  SDL_ShowCursor(0);
504 
505  if (!(vidscreen = SDL_SetVideoMode(glConfig.vidWidth, glConfig.vidHeight, colorbits, flags)))
506  {
507  ri.Printf( PRINT_DEVELOPER, "SDL_SetVideoMode failed: %s\n", SDL_GetError( ) );
508  continue;
509  }
510 
511  opengl_context = GLimp_GetCurrentContext();
512 
513  ri.Printf( PRINT_ALL, "Using %d/%d/%d Color bits, %d depth, %d stencil display.\n",
514  sdlcolorbits, sdlcolorbits, sdlcolorbits, tdepthbits, tstencilbits);
515 
516  glConfig.colorBits = tcolorbits;
517  glConfig.depthBits = tdepthbits;
518  glConfig.stencilBits = tstencilbits;
519  break;
520  }
521 
522  GLimp_DetectAvailableModes();
523 
524  if (!vidscreen)
525  {
526  ri.Printf( PRINT_ALL, "Couldn't get a visual\n" );
527  return RSERR_INVALID_MODE;
528  }
529 
530  screen = vidscreen;
531 
532  glstring = (char *) qglGetString (GL_RENDERER);
533  ri.Printf( PRINT_ALL, "GL_RENDERER: %s\n", glstring );
534 
535  return RSERR_OK;
536 }

903  if(GLimp_StartDriverAndSetMode(r_mode->integer, r_fullscreen->integer, r_noborder->integer))
904  goto success;

907  ri.Sys_GLimpSafeInit( );

910  goto success;

913  if( r_mode->integer != R_MODE_FALLBACK )

916  r_mode->integer, R_MODE_FALLBACK );

918

920  goto success;
921  }

927  // This values force the UI to disable driver selection
928  glConfig.driverType = GLDRV_ICD;
929  glConfig.hardwareType = GLHW_GENERIC;
930  glConfig.deviceSupportsGamma = SDL_SetGamma( 1.0f, 1.0f, 1.0f ) >= 0;
