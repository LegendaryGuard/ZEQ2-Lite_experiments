25  animation_t* anim;
26  lf->animationNumber = newAnimation;
27  newAnimation &= ~ANIM_TOGGLEBIT;
28  if(newAnimation < 0 || newAnimation >= MAX_TOTALANIMATIONS){
29  CG_Error("Bad animation number: %i",newAnimation);

33  lf->animationTime = lf->frameTime + anim->initialLerp;

37 }

46 void CG_RunLerpFrame(clientInfo_t* ci,lerpFrame_t* lf,int newAnimation,float speedScale,qboolean isCamera){
47  int f;
48  int numFrames;
49  animation_t* anim;
50  // debugging tool to get no animations
51  if(!cg_animSpeed.integer){
52  lf->oldFrame = lf->frame = lf->backlerp = 0;
53  return;
54  }
55  // see if the animation sequence is switching
56  if(!lf->animation){
57  CG_SetLerpFrameAnimation(ci,lf,newAnimation,isCamera);
58  }
59  if(newAnimation != lf->animationNumber){
60  // If the only difference is the togglebit, and the animation is supposed to be
61  // continuous.
62  if(newAnimation & ~ANIM_TOGGLEBIT == lf->animationNumber & ~ANIM_TOGGLEBIT && lf->animation->continuous){
63  // do nothing, animation should continue to loop
64  }
65  else{
66  CG_SetLerpFrameAnimation( ci, lf, newAnimation, isCamera );
67  }
68  }
69  // if we have passed the current frame, move it to
70  // oldFrame and calculate a new frame
71  if(cg.time >= lf->frameTime){
72  lf->oldFrame = lf->frame;
73  lf->oldFrameTime = lf->frameTime;
74  // get the next frame based on the animation
75  anim = lf->animation;
76  // shouldn't happen
77  if(!anim->frameLerp){return;}
78  if(cg.time < lf->animationTime){
79  // initial lerp
80  lf->frameTime = lf->animationTime;
81  }
82  else{
83  lf->frameTime = lf->oldFrameTime + anim->frameLerp;
84  }
85  f = (lf->frameTime - lf->animationTime) / anim->frameLerp;
86  // adjust for haste, etc
87  f *= speedScale;
88  numFrames = anim->numFrames;
89  if(anim->flipflop){
90  numFrames *= 2;
91  }
92  if(f >= numFrames){
93  f -= numFrames;
94  if(anim->loopFrames){
95  f %= anim->loopFrames;
96  f += anim->numFrames - anim->loopFrames;
97  }
98  else{
99  f = numFrames - 1;
100  // the animation is stuck at the end, so it
101  // can immediately transition to another sequence
102  lf->frameTime = cg.time;
103  }
104  }
105  if(anim->reversed){
106  lf->frame = anim->firstFrame + anim->numFrames - 1 - f;
107  }
108  else if(anim->flipflop && f>=anim->numFrames){
109  lf->frame = anim->firstFrame + anim->numFrames - 1 - (f%anim->numFrames);
110  }
111  else{
112  lf->frame = anim->firstFrame + f;
113  }
114  if(cg.time > lf->frameTime){
115  lf->frameTime = cg.time;
116  if(cg_debugAnim.integer){
117  CG_Printf("Clamp lf->frameTime\n");
118  }
119  }
120  }
121  if(lf->frameTime > cg.time + 200){
122  lf->frameTime = cg.time;
123  }
124  if(lf->oldFrameTime > cg.time){
125  lf->oldFrameTime = cg.time;
126  }
127  // calculate current lerp value
128  lf->backlerp = lf->frameTime == lf->oldFrameTime ? 0 : 1 - (float)(cg.time - lf->oldFrameTime) / (lf->frameTime - lf->oldFrameTime);
129 }

134 }
