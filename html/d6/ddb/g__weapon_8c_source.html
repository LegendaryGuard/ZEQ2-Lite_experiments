53 ================
54 */
55 void G_BounceProjectile( vec3_t start, vec3_t impact, vec3_t dir, vec3_t endout ) {
56  vec3_t v, newv;
57  float dot;
58 
59  VectorSubtract( impact, start, v );
60  dot = DotProduct( v, dir );
61  VectorMA( v, -2*dot, dir, newv );
62 
63  VectorNormalize(newv);
64  VectorMA(impact, 8192, newv, endout);
65 }

74 /*======================
75 SnapVectorTowards
76 
77 Round a vector to integers for more efficient network
78 transmission, but make sure that it rounds towards a given point
79 rather than blindly truncating. This prevents it from truncating
80 into a wall.
81 ======================*/
82 void SnapVectorTowards( vec3_t v, vec3_t to ) {
83  int i;
84 
85  for ( i = 0 ; i < 3 ; i++ ) {
86  if ( to[i] <= v[i] ) {
87  v[i] = (int)v[i];
88  } else {
89  v[i] = (int)v[i] + 1;
90  }
91  }
92 }
