27 P_DamageFeedback
28 ===============*/
29 void P_DamageFeedback( gentity_t *player ) {
30  gclient_t *client;
31  float count;
32  vec3_t angles;
33 
34  client = player->client;
35  // total points of damage shot at the player this frame
36  count = client->damage_blood; // + client->damage_armor;
37  if ( count == 0 ) { // didn't take any damage
38  // ADDING FOR ZEQ2
39  // but we DO need to reset this to 0 for the deductHP to work!
40  client->ps.damageCount = 0;
41  return;
42  }
43 
44  if ( count > 255 ) {
45  count = 255;
46  }
47 
48  // send the information to the client
49 
50  // world damage (falling, slime, etc) uses a special code
51  // to make the blend blob centered instead of positional
52  if ( client->damage_fromWorld ) {
53  client->ps.damagePitch = 255;
54  client->ps.damageYaw = 255;
55 
56  client->damage_fromWorld = qfalse;
57  } else {
58  vectoangles( client->damage_from, angles );
59  client->ps.damagePitch = angles[PITCH]/360.0 * 256;
60  client->ps.damageYaw = angles[YAW]/360.0 * 256;
61  }
62 
63  // play an apropriate pain sound
64  if ( (level.time > player->pain_debounce_time) ) {
65  player->pain_debounce_time = level.time + 700;
66  G_AddEvent( player, EV_PAIN, player->powerLevelTotal );
67  client->ps.damageEvent++;
68  }
69 
70 
71  client->ps.damageCount = count;
72 
73  //
74  // clear totals
75  //
76  client->damage_blood = 0;
77  //client->damage_armor = 0;
78  client->damage_knockback = 0;
79 }

109 ClientImpacts

112 void ClientImpacts( gentity_t *ent, pmove_t *pm ) {
113  int i, j;
114  trace_t trace;
115  gentity_t *other;
116 
117  memset( &trace, 0, sizeof( trace ) );
118  for (i=0 ; i<pm->numtouch ; i++) {
119  for (j=0 ; j<i ; j++) {
120  if (pm->touchents[j] == pm->touchents[i] ) {
121  break;
122  }
123  }
124  if (j != i) {
125  continue; // duplicated
126  }
127  other = &g_entities[ pm->touchents[i] ];
128 
129  if ( !other->touch ) {
130  continue;
131  }
132 
133  other->touch( other, ent, &trace );
134  }
135 
136 }

138 /*
139 ============
140 G_TouchTriggers
141 
142 Find all trigger entities that ent's current position touches.
143 Spectators will only interact with teleporters.
144 ============
145 */
146 void G_TouchTriggers( gentity_t *ent ) {
147  int i, num;
148  int touch[MAX_GENTITIES];
149  gentity_t *hit;
150  trace_t trace;
151  vec3_t mins, maxs;
152  static vec3_t range = { 40, 40, 52 };
153 
154  if ( !ent->client ) {
155  return;
156  }
157  VectorSubtract( ent->client->ps.origin, range, mins );
158  VectorAdd( ent->client->ps.origin, range, maxs );
159 
160  num = trap_EntitiesInBox( mins, maxs, touch, MAX_GENTITIES );
161 
162  // can't use ent->absmin, because that has a one unit pad
163  VectorAdd( ent->client->ps.origin, ent->r.mins, mins );
164  VectorAdd( ent->client->ps.origin, ent->r.maxs, maxs );
165 
166  for ( i=0 ; i<num ; i++ ) {
167  hit = &g_entities[touch[i]];
168 
169  if ( !hit->touch && !ent->touch ) {
170  continue;
171  }
172  if ( !( hit->r.contents & CONTENTS_TRIGGER ) ) {
173  continue;
174  }
175 
176  // ignore most entities if a spectator
177  if ( ent->client->sess.sessionTeam == TEAM_SPECTATOR ) {
178  if ( hit->s.eType != ET_TELEPORT_TRIGGER &&
179  // this is ugly but adding a new ET_? type will
180  // most likely cause network incompatibilities
181  hit->touch != Touch_DoorTrigger) {
182  continue;
183  }
184  }
185 
186  if ( !trap_EntityContact( mins, maxs, hit ) ) {
187  continue;
188  }
189  memset( &trace, 0, sizeof(trace) );
190 
191  if ( hit->touch ) {
192  hit->touch (hit, ent, &trace);
193  }
194  }
195 
196  // if we didn't touch a jump pad this pmove frame
197  if ( ent->client->ps.jumppad_frame != ent->client->ps.pmove_framecount ) {
198  ent->client->ps.jumppad_frame = 0;
199  ent->client->ps.jumppad_ent = 0;
200  }
201 }

253 qboolean ClientInactivityTimer( gclient_t *client ) {
254  if ( ! g_inactivity.integer ) {
255  // give everyone some time, so if the operator sets g_inactivity during
256  // gameplay, everyone isn't kicked
257  client->inactivityTime = level.time + 60 * 1000;
258  client->inactivityWarning = qfalse;
259  } else if ( client->pers.cmd.forwardmove ||
260  client->pers.cmd.rightmove ||
261  client->pers.cmd.upmove ||
262  (client->pers.cmd.buttons & BUTTON_ATTACK) ) {
263  client->inactivityTime = level.time + g_inactivity.integer * 1000;
264  client->inactivityWarning = qfalse;
265  } else if ( !client->pers.localClient ) {
266  if ( level.time > client->inactivityTime ) {
267  trap_DropClient( client - level.clients, "Dropped due to inactivity" );
268  return qfalse;
269  }
270  if ( level.time > client->inactivityTime - 10000 && !client->inactivityWarning ) {
271  client->inactivityWarning = qtrue;
272  trap_SendServerCommand( client - level.clients, "cp \"Ten seconds until inactivity drop!\n\"" );
273  }
274  }
275  return qtrue;
276 }

409 void SendPendingPredictableEvents( playerState_t *ps ) {
410  gentity_t *t;
411  int event, seq;
412  int extEvent, number;
413 
414  // if there are still events pending
415  if ( ps->entityEventSequence < ps->eventSequence ) {
416  // create a temporary entity for this event which is sent to everyone
417  // except the client who generated the event
418  seq = ps->entityEventSequence & (MAX_PS_EVENTS-1);
419  event = ps->events[ seq ] | ( ( ps->entityEventSequence & 3 ) << 8 );
420  // set external event to zero before calling BG_PlayerStateToEntityState
421  extEvent = ps->externalEvent;
422  ps->externalEvent = 0;
423  // create temporary entity for event
424  t = G_TempEntity( ps->origin, event );
425  number = t->s.number;
426  BG_PlayerStateToEntityState( ps, &t->s,qtrue );
427  t->s.number = number;
428  t->s.eType = ET_EVENTS + event;
429  t->s.eFlags |= EF_PLAYER_EVENT;
430  t->s.otherEntityNum = ps->clientNum;
431  // send to everyone except the client who generated the event
432  t->r.svFlags |= SVF_NOTSINGLECLIENT;
433  t->r.singleClient = ps->clientNum;
434  // set back external event
435  ps->externalEvent = extEvent;
436  }

438 void LockonCheck(gclient_t *client){
439  int entityNum = -1;
440  playerState_t *ps;
441  ps = &client->ps;
442  if(ps->lockedTarget>0){
443  if(!&g_entities[ps->lockedTarget-1].client || &g_entities[ps->lockedTarget-1].client->pers.connected == CON_DISCONNECTED){
444  ps->lockonData[lkLastLockedPlayer] = -1;
445  ps->lockedPosition = 0;
446  ps->lockedPlayer = 0;
447  ps->lockedTarget = 0;
448  return;
449  }
450  if(ps->lockedPlayer){
451  ps->timers[tmLockon] += pml.msec;
452  if(ps->lockedPlayer->bitFlags & isTransforming || ps->lockedPlayer->bitFlags & isStruggling || ps->lockedPlayer->bitFlags & isDead ||
453  ps->lockedPlayer->bitFlags & isUnconcious || ps->lockedPlayer->bitFlags & isCrashed){
454  ps->lockedPosition = 0;
455  ps->lockedPlayer = 0;
456  ps->lockedTarget = 0;
457  return;
458  }
459  entityNum = PM_VerifyTrace(250);
460  if(!(ps->bitFlags & usingZanzoken) && ps->lockedPlayer && ((entityNum == -1) || ps->lockedPlayer->bitFlags & usingZanzoken)){
461  ps->lockonData[lkLastLockedPlayer] = ps->lockedPlayer->clientNum;
462  ps->lockedPosition = 0;
463  ps->lockedTarget = 0;
464  ps->timers[tmLockon] = 0;
465  return;
466  }
467  }
468  ps->lockedPosition = &g_entities[ps->lockedTarget-1].r.currentOrigin;
469  ps->lockedPlayer = &g_entities[ps->lockedTarget-1].client->ps;
470  }
471 }
472 /*
473 ==============
474 ClientThink
475 
476 This will be called once for each client frame, which will
477 usually be a couple times for each server frame on fast clients.
478 
479 If "g_synchronousClients 1" is set, this will be called exactly
480 once for each server frame, which makes for smooth demo recording.
481 ==============
482 */
483 void ClientThink_real( gentity_t *ent ) {
484  gclient_t *client;
485  pmove_t pm;
486  int oldEventSequence;
487  int msec;
488  usercmd_t *ucmd;
489  int i;
490  client = ent->client;
491  if (client->pers.connected != CON_CONNECTED){return;}
492  ucmd = &ent->client->pers.cmd;
493  if ( ucmd->serverTime > level.time + 200 ) {ucmd->serverTime = level.time + 200;}
494  if ( ucmd->serverTime < level.time - 1000 ) {ucmd->serverTime = level.time - 1000;}
495  msec = ucmd->serverTime - client->ps.commandTime;
496  if(msec < 1 && client->sess.spectatorState != SPECTATOR_FOLLOW){return;}
497  if(msec > 200){msec = 200;}
498  if(pmove_msec.integer < 8 ){trap_Cvar_Set("pmove_msec", "8");}
499  else if(pmove_msec.integer > 33){trap_Cvar_Set("pmove_msec", "33");}
500  if(pmove_fixed.integer || client->pers.pmoveFixed){
501  ucmd->serverTime = ((ucmd->serverTime + pmove_msec.integer-1) / pmove_msec.integer) * pmove_msec.integer;
502  }
503  if(level.intermissiontime){
504  ClientIntermissionThink( client );
505  return;
506  }
507  if(client->sess.sessionTeam == TEAM_SPECTATOR){
508  if ( client->sess.spectatorState == SPECTATOR_SCOREBOARD ) {
509  return;
510  }
511  SpectatorThink( ent, ucmd );
512  return;
513  }
514  if(client->noclip){
515  client->ps.pm_type = PM_NOCLIP;
516  }
517  else{
518  client->ps.pm_type = PM_NORMAL;
519  }
520  ent->s.playerBitFlags = client->ps.bitFlags;
521  ent->s.attackPowerCurrent = client->ps.powerLevel[plHealth];
522  ent->s.attackPowerTotal = client->ps.powerLevel[plMaximum];
523  G_LinkUserWeaponData( &(client->ps) );
524  G_CheckSkills(&(client->ps));
525  // set up for pmove
526  oldEventSequence = client->ps.eventSequence;
527  memset(&pm,0,sizeof(pm));
528  pm.ps = &client->ps;
529  pm.cmd = *ucmd;
530  pm.tracemask = MASK_PLAYERSOLID;
531  pm.trace = trap_Trace;
532  pm.pointcontents = trap_PointContents;
533  pm.debugLevel = g_debugMove.integer;
534  pm.noFootsteps = (g_dmflags.integer & DF_NO_FOOTSTEPS) > 0;
535  pm.pmove_fixed = pmove_fixed.integer | client->pers.pmoveFixed;
536  pm.pmove_msec = pmove_msec.integer;
537  VectorCopy(client->ps.origin,client->oldOrigin);
538  Pmove(&pm);
539  checkTier(client);
540  if(pm.ps->powerLevel[plTierChanged] == 1)
541  {
542  if(pm.ps->powerLevel[plTierChanged] == 1){
543  if(!PM_WeaponSelectable(pm.ps->weapon)){
544  for(i = 6; i > 0; i--){
545  if(PM_WeaponSelectable(i)){
546  pm.ps->weapon = i;
547  break;
548  }
549  }
550  }
551  }
552  }
553  LockonCheck(client);
554  if ( ent->client->ps.eventSequence != oldEventSequence ) {
555  ent->eventTime = level.time;
556  }
557  BG_PlayerStateToEntityState( &ent->client->ps, &ent->s, qtrue );
558  SendPendingPredictableEvents(&ent->client->ps);
559  if ( !( ent->client->ps.eFlags & EF_FIRING ) ) {
560  client->fireHeld = qfalse;
561  }
562  VectorCopy( ent->s.pos.trBase, ent->r.currentOrigin );
563  VectorCopy (pm.mins, ent->r.mins);
564  VectorCopy (pm.maxs, ent->r.maxs);
565  ent->waterlevel = pm.waterlevel;
566  ent->watertype = pm.watertype;
567  ClientEvents(ent,oldEventSequence);
568  trap_LinkEntity (ent);
569  if(!ent->client->noclip){
570  G_TouchTriggers(ent);
571  }
572  VectorCopy( ent->client->ps.origin, ent->r.currentOrigin );
573  ClientImpacts(ent,&pm);
574  if(ent->client->ps.eventSequence != oldEventSequence) {
575  ent->eventTime = level.time;
576  }
577  client->oldbuttons = client->buttons;
578  client->buttons = ucmd->buttons;
579  client->latched_buttons |= client->buttons & ~client->oldbuttons;
580  if (client->ps.bitFlags & isDead){
581  if(level.time > client->respawnTime){ClientRespawn(ent);}
582  return;
583  }
584 }

621 void SpectatorClientEndFrame( gentity_t *ent ) {
622  gclient_t *cl;
623 
624  // if we are doing a chase cam or a remote view, grab the latest info
625  if ( ent->client->sess.spectatorState == SPECTATOR_FOLLOW ) {
626  int clientNum, flags;
627 
628  clientNum = ent->client->sess.spectatorClient;
629 
630  // team follow1 and team follow2 go to whatever clients are playing
631  if ( clientNum == -1 ) {
632  clientNum = level.follow1;
633  } else if ( clientNum == -2 ) {
634  clientNum = level.follow2;
635  }
636  if ( clientNum >= 0 ) {
637  cl = &level.clients[ clientNum ];
638  if ( cl->pers.connected == CON_CONNECTED && cl->sess.sessionTeam != TEAM_SPECTATOR ) {
639  flags = (cl->ps.eFlags & ~(EF_VOTED | EF_TEAMVOTED)) | (ent->client->ps.eFlags & (EF_VOTED | EF_TEAMVOTED));
640  ent->client->ps = cl->ps;
641  ent->client->ps.pm_flags |= PMF_FOLLOW;
642  ent->client->ps.eFlags = flags;
643  return;
644  } else {
645  // drop them to free spectators unless they are dedicated camera followers
646  if ( ent->client->sess.spectatorClient >= 0 ) {
647  ent->client->sess.spectatorState = SPECTATOR_FREE;
648  ClientBegin( ent->client - level.clients );
649  }
650  }
651  }
652  }
653 
654  if ( ent->client->sess.spectatorState == SPECTATOR_SCOREBOARD ) {
655  ent->client->ps.pm_flags |= PMF_SCOREBOARD;
656  } else {
657  ent->client->ps.pm_flags &= ~PMF_SCOREBOARD;
658  }
659 }

661 /*
662 ==============
663 ClientEndFrame
664 
665 Called at the end of each server frame for each connected client
666 A fast client will have multiple ClientThink for each ClientEdFrame,
667 while a slow client may have multiple ClientEndFrame between ClientThink.
668 ==============
669 */
670 void ClientEndFrame( gentity_t *ent ) {
671 // int i;
672  if ( ent->client->sess.sessionTeam == TEAM_SPECTATOR ) {
673  SpectatorClientEndFrame( ent );
674  return;
675  }
676 
677  //
678  // If the end of unit layout is displayed, don't give
679  // the player any normal movement attributes
680  //
681  if ( level.intermissiontime ) {
682  return;
683  }
684 
685  // burn from lava, etc
686  P_WorldEffects( ent );
687 
688  // apply all the damage taken this frame
689  P_DamageFeedback( ent );
690 
691  // add the EF_CONNECTION flag if we haven't gotten commands recently
692  if ( level.time - ent->client->lastCmdTime > 1000 ) {
