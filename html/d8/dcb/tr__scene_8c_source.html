113 =====================
114 RE_AddPolyToScene
115 
116 =====================
117 */
118 void RE_AddPolyToScene( qhandle_t hShader, int numVerts, const polyVert_t *verts, int numPolys ) {
119  srfPoly_t *poly;
120  int i, j;
121  int fogIndex;
122  fog_t *fog;
123  vec3_t bounds[2];
124 
125  if ( !tr.registered ) {
126  return;
127  }
128 
129  if ( !hShader ) {
130  ri.Printf( PRINT_WARNING, "WARNING: RE_AddPolyToScene: NULL poly shader\n");
131  return;
132  }
133 
134  for ( j = 0; j < numPolys; j++ ) {
135  if ( r_numpolyverts + numVerts > max_polyverts || r_numpolys >= max_polys ) {
136  /*
137  NOTE TTimo this was initially a PRINT_WARNING
138  but it happens a lot with high fighting scenes and particles
139  since we don't plan on changing the const and making for room for those effects
140  simply cut this message to developer only
141  */
142  ri.Printf( PRINT_DEVELOPER, "WARNING: RE_AddPolyToScene: r_max_polys or r_max_polyverts reached\n");
143  return;
144  }
145 
146  poly = &backEndData[tr.smpFrame]->polys[r_numpolys];
147  poly->surfaceType = SF_POLY;
148  poly->hShader = hShader;
149  poly->numVerts = numVerts;
150  poly->verts = &backEndData[tr.smpFrame]->polyVerts[r_numpolyverts];
151 
152  Com_Memcpy( poly->verts, &verts[numVerts*j], numVerts * sizeof( *verts ) );
153 
154  if ( glConfig.hardwareType == GLHW_RAGEPRO ) {
155  poly->verts->modulate[0] = 255;
156  poly->verts->modulate[1] = 255;
157  poly->verts->modulate[2] = 255;
158  poly->verts->modulate[3] = 255;
159  }
160  // done.
161  r_numpolys++;
162  r_numpolyverts += numVerts;
163 
164  // if no world is loaded
165  if ( tr.world == NULL ) {
166  fogIndex = 0;
167  }
168  // see if it is in a fog volume
169  else if ( tr.world->numfogs == 1 ) {
170  fogIndex = 0;
171  } else {
172  // find which fog volume the poly is in
173  VectorCopy( poly->verts[0].xyz, bounds[0] );
174  VectorCopy( poly->verts[0].xyz, bounds[1] );
175  for ( i = 1 ; i < poly->numVerts ; i++ ) {
176  AddPointToBounds( poly->verts[i].xyz, bounds[0], bounds[1] );
177  }
178  for ( fogIndex = 1 ; fogIndex < tr.world->numfogs ; fogIndex++ ) {
179  fog = &tr.world->fogs[fogIndex];
180  if ( bounds[1][0] >= fog->bounds[0][0]
181  && bounds[1][1] >= fog->bounds[0][1]
182  && bounds[1][2] >= fog->bounds[0][2]
183  && bounds[0][0] <= fog->bounds[1][0]
184  && bounds[0][1] <= fog->bounds[1][1]
185  && bounds[0][2] <= fog->bounds[1][2] ) {
186  break;
187  }
188  }
189  if ( fogIndex == tr.world->numfogs ) {
190  fogIndex = 0;
191  }
192  }
193  poly->fogIndex = fogIndex;
194  }
195 }

207 void RE_AddRefEntityToScene( const refEntity_t *ent ) {
208  if ( !tr.registered ) {
209  return;
210  }
211  if ( r_numentities >= MAX_ENTITIES ) {
212  ri.Printf(PRINT_DEVELOPER, "RE_AddRefEntityToScene: Dropping refEntity, reached MAX_ENTITIES\n");
213  return;
214  }
215  if ( Q_isnan(ent->origin[0]) || Q_isnan(ent->origin[1]) || Q_isnan(ent->origin[2]) ) {
216  static qboolean firstTime = qtrue;
217  if (firstTime) {
218  firstTime = qfalse;
219  ri.Printf( PRINT_WARNING, "RE_AddRefEntityToScene passed a refEntity which has an origin with a NaN component\n");
220  }
221  return;
222  }
223  if ( (int)ent->reType < 0 || ent->reType >= RT_MAX_REF_ENTITY_TYPE ) {
224  ri.Error( ERR_DROP, "RE_AddRefEntityToScene: bad reType %i", ent->reType );
225  }
226 
227  backEndData[tr.smpFrame]->entities[r_numentities].e = *ent;
228  backEndData[tr.smpFrame]->entities[r_numentities].lightingCalculated = qfalse;
229 
230  r_numentities++;
231 }

240 void RE_AddDynamicLightToScene( const vec3_t org, float intensity, float r, float g, float b, int additive ) {
241  dlight_t *dl;
242 
243  if ( !tr.registered ) {
244  return;
245  }
246  if ( r_numdlights >= MAX_DLIGHTS ) {
247  return;
248  }
249  if ( intensity <= 0 ) {
250  return;
251  }
252  // these cards don't have the correct blend mode
253  if ( glConfig.hardwareType == GLHW_RIVA128 || glConfig.hardwareType == GLHW_PERMEDIA2 ) {
254  return;
255  }
256  dl = &backEndData[tr.smpFrame]->dlights[r_numdlights++];
257  VectorCopy (org, dl->origin);
258  dl->radius = intensity;
259  dl->color[0] = r;
260  dl->color[1] = g;
261  dl->color[2] = b;
262  dl->additive = additive;
263 }

266 =====================
267 RE_AddFogToScene
268 
269 =====================
270 */
271 void RE_AddFogToScene( float start, float end, float r, float g, float b, float opacity, float mode, float hint ) {
272  int fogMode[] = { GL_EXP, GL_EXP2, GL_LINEAR }; // Storage For Three Types Of Fog
273  int fogModefilter = mode; // Which Fog To Use
274  int fogHint[] = { GL_DONT_CARE, GL_FASTEST, GL_NICEST };// Storage For Three Types Of Fog
275  int fogHintfilter = hint; // Which Fog To Use
276  float fogColor[4] = {r, g, b, 1.0f}; // Fog Color
277 
278  if ( ( start == 0 && end == 0 ) || opacity == 0 ){
279  qglDisable(GL_FOG); // Disables GL_FOG
280  return;
281  }
282  else
283  {
284  qglEnable(GL_FOG); // Enables GL_FOG
285  }
286 
287  qglFogi(GL_FOG_MODE, fogMode[fogModefilter]); // Fog Mode
288  qglFogfv(GL_FOG_COLOR, fogColor); // Set Fog Color
289  qglFogf(GL_FOG_DENSITY, opacity); // How Dense Will The Fog Be
290  qglHint(GL_FOG_HINT, fogHint[fogHintfilter]); // Fog Hint Value
291  qglFogf(GL_FOG_START, start); // Fog Start Depth
292  qglFogf(GL_FOG_END, end); // Fog End Depth
293 }

315 /*
316 @@@@@@@@@@@@@@@@@@@@@
317 RE_RenderScene
318 
319 Draw a 3D view into a part of the window, then return
320 to 2D drawing.
321 
322 Rendering a scene may require multiple views to be rendered
323 to handle mirrors,
324 @@@@@@@@@@@@@@@@@@@@@
325 */
326 void RE_RenderScene( const refdef_t *fd ) {
327  viewParms_t parms;
328  int startTime;
329 
330  if ( !tr.registered ) {
331  return;
332  }
333  GLimp_LogComment( "====== RE_RenderScene =====\n" );
334 
335  if ( r_norefresh->integer ) {
336  return;
337  }
338 
339  startTime = ri.Milliseconds();
340 
341  if (!tr.world && !( fd->rdflags & RDF_NOWORLDMODEL ) ) {
342  ri.Error (ERR_DROP, "R_RenderScene: NULL worldmodel");
343  }
344 
345  Com_Memcpy( tr.refdef.text, fd->text, sizeof( tr.refdef.text ) );
346 
347  tr.refdef.x = fd->x;
348  tr.refdef.y = fd->y;
349  tr.refdef.width = fd->width;
350  tr.refdef.height = fd->height;
351  tr.refdef.fov_x = fd->fov_x;
352  tr.refdef.fov_y = fd->fov_y;
353 
354  VectorCopy( fd->vieworg, tr.refdef.vieworg );
355  VectorCopy( fd->viewaxis[0], tr.refdef.viewaxis[0] );
356  VectorCopy( fd->viewaxis[1], tr.refdef.viewaxis[1] );
357  VectorCopy( fd->viewaxis[2], tr.refdef.viewaxis[2] );
358 
359  tr.refdef.time = fd->time;
360  tr.refdef.rdflags = fd->rdflags;
361 
362  // copy the areamask data over and note if it has changed, which
363  // will force a reset of the visible leafs even if the view hasn't moved
364  tr.refdef.areamaskModified = qfalse;
365  if ( ! (tr.refdef.rdflags & RDF_NOWORLDMODEL) ) {
366  int areaDiff;
367  int i;
368 
369  // compare the area bits
370  areaDiff = 0;
371  for (i = 0 ; i < MAX_MAP_AREA_BYTES/4 ; i++) {
372  areaDiff |= ((int *)tr.refdef.areamask)[i] ^ ((int *)fd->areamask)[i];
373  ((int *)tr.refdef.areamask)[i] = ((int *)fd->areamask)[i];
374  }
375 
376  if ( areaDiff ) {
377  // a door just opened or something
378  tr.refdef.areamaskModified = qtrue;
379  }
380  }
381 
382 
383  // derived info
384 
385  tr.refdef.floatTime = tr.refdef.time * 0.001f;
386 
387  tr.refdef.numDrawSurfs = r_firstSceneDrawSurf;
388  tr.refdef.drawSurfs = backEndData[tr.smpFrame]->drawSurfs;
389 
390  tr.refdef.num_entities = r_numentities - r_firstSceneEntity;
391  tr.refdef.entities = &backEndData[tr.smpFrame]->entities[r_firstSceneEntity];
392 
393  tr.refdef.num_dlights = r_numdlights - r_firstSceneDlight;
394  tr.refdef.dlights = &backEndData[tr.smpFrame]->dlights[r_firstSceneDlight];
395 
396  tr.refdef.numPolys = r_numpolys - r_firstScenePoly;
397  tr.refdef.polys = &backEndData[tr.smpFrame]->polys[r_firstScenePoly];
398 
399  // turn off dynamic lighting globally by clearing all the
400  // dlights if it needs to be disabled or if vertex lighting is enabled
401  if ( r_dynamiclight->integer == 0 ||
402  r_vertexLight->integer == 1 ||
403  glConfig.hardwareType == GLHW_PERMEDIA2 ) {
404  tr.refdef.num_dlights = 0;
405  }
406 
407  // a single frame may have multiple scenes draw inside it --
408  // a 3D game view, 3D status bar renderings, 3D menus, etc.
409  // They need to be distinguished by the light flare code, because
410  // the visibility state for a given surface may be different in
411  // each scene / view.
412  tr.frameSceneNum++;
413  tr.sceneCount++;
414 
415  // setup view parms for the initial view
416  //
417  // set up viewport
418  // The refdef takes 0-at-the-top y coordinates, so
419  // convert to GL's 0-at-the-bottom space
420  //
421  Com_Memset( &parms, 0, sizeof( parms ) );
422  parms.viewportX = tr.refdef.x;
423  parms.viewportY = glConfig.vidHeight - ( tr.refdef.y + tr.refdef.height );
424  parms.viewportWidth = tr.refdef.width;
425  parms.viewportHeight = tr.refdef.height;
426  parms.isPortal = qfalse;
427 
428  parms.fovX = tr.refdef.fov_x;
429  parms.fovY = tr.refdef.fov_y;
430 
431  parms.stereoFrame = tr.refdef.stereoFrame;
432 
433  VectorCopy( fd->vieworg, parms.or.origin );
434  VectorCopy( fd->viewaxis[0], parms.or.axis[0] );
435  VectorCopy( fd->viewaxis[1], parms.or.axis[1] );
436  VectorCopy( fd->viewaxis[2], parms.or.axis[2] );
437 
438  VectorCopy( fd->vieworg, parms.pvsOrigin );
439 
440  R_RenderView( &parms );
441 
442  // the next scene rendered in this frame will tack on after this one
443  r_firstSceneDrawSurf = tr.refdef.numDrawSurfs;
444  r_firstSceneEntity = r_numentities;
445  r_firstSceneDlight = r_numdlights;
446  r_firstScenePoly = r_numpolys;
447 
448  tr.frontEndMsec += ri.Milliseconds() - startTime;
449 }
