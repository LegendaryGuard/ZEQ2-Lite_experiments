93 static void CG_LerpTrails( void ) {

97  trail_t *trail;
98 
99  for ( j = 0; j < MAX_GENTITIES; j++ ) {
100 
101  trail = &cg_trails[j];
102  // Don't bother updating if the very end and very start are already
103  // the same. We'd either be on the start frame or the last frame of
104  // the trails existence. It's not going to be drawn either way.
105 
106  // FIXED: Just incase we'd be able to get a full loop back of the first
107  // to the last point in with guided missiles, ALWAYS check those.
108  if ( !Distance(trail->pos[0], trail->pos[TRAIL_SEGMENTS - 1] ) &&
109  !( cg_entities[j].currentState.eFlags & EF_GUIDED ) ) {
110  continue;
111  }
112 
113  for ( i = 1; i < TRAIL_SEGMENTS; i++ ) {
114  // Set up the distance and direction
115  VectorSubtract( trail->pos[i-1], trail->pos[i], dir );
116  dist = VectorNormalize( dir );
117 
118  distDelta = trail->baseSpeed * ((28 - i) / 40.0f) * cg.frametime * 0.001f;
119 
120  // If we'd overshoot, halt on the previous point
121  if ( distDelta > dist ) {
122  VectorCopy( trail->pos[i-1], trail->pos[i] );
123  VectorCopy( trail->tangent[i-1], trail->tangent[i] );
124  continue;
125  }
126 
127 
128  // If the trail section gets too long, snap it forward like a rubberband.
129  if ( dist - distDelta > TRAIL_MAX_LENGTH ) {
130  VectorMA( trail->pos[i-1], -TRAIL_MAX_LENGTH, dir, trail->pos[i] );
131  VectorCopy( dir, trail->tangent[i] );
132  continue;
133  }
134 
135  VectorMA( trail->pos[i], distDelta, dir, trail->pos[i] );
136  VectorCopy( dir, trail->tangent[i] );
137  }

167 Calculates the vertices for drawing the trail's polygons out of the position of a node in a trail.
168 Is a utility function used by CG_AddTrailsToScene.
169 point: The position of the node
170 tangent: The direction from the current node to the next node

172 verts: An array of 4 polyVert_t structures to operate on,
173  though only the 0th and 1st element are of importance.
174 */
175 static void CG_GetTrailVerts( vec3_t point, vec3_t tangent, float width, polyVert_t *verts ) {
176  vec3_t offset, viewLine, dummyPoint;
177  float len;
178 
179  VectorSubtract( point, cg.refdef.vieworg, viewLine );
180  CrossProduct( viewLine, tangent, offset );
181  len = VectorNormalize( offset );
182 
183  // HACK: if len would be 0, then we wouldn't get a proper width, so
184  // in this case nudge the point upwards by 0.1 on its z-coordinate.
185  if ( !len ) {
186  VectorCopy( point, dummyPoint );
187  dummyPoint[2] += 0.1f;
188  VectorSubtract( dummyPoint, cg.refdef.vieworg, viewLine );
189  CrossProduct( viewLine, tangent, offset );
190  VectorNormalize( offset );
191  }
192 
193  VectorMA( point, width, offset, verts[0].xyz );
194  VectorMA( point, -width, offset, verts[1].xyz );
195 }
