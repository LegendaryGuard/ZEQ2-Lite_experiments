65 {
66  int e=0;

69  {
70  sign = 1;
71  x = -x;
72  }
73  if (x==0)
74  {
75  spx_float_t r = {0,0};

86  }

98 {

101  return b;
102  else if (b.m==0)
103  return a;
104  if ((a).e > (b).e)
105  {
106  r.m = ((a).m>>1) + ((b).m>>MIN(15,(a).e-(b).e+1));

115  {

306  }
307  e += e0;

314 static inline spx_float_t FLOAT_DIVU(spx_float_t a, spx_float_t b)
315 {
316  int e=0;
317  spx_int32_t num;
318  spx_float_t r;
319  if (b.m<=0)
320  {
321  speex_warning_int("Attempted to divide by", b.m);
322  return FLOAT_ONE;
323  }
324  num = a.m;
325  a.m = ABS16(a.m);
326  while (a.m >= b.m)
327  {
328  e++;
329  a.m >>= 1;
330  }
331  num = num << (15-e);
332  r.m = DIV32_16(num,b.m);
333  r.e = a.e-b.e-15+e;
334  return r;
335 }
