30 ==================
31 CG_BubbleTrail
32 
33 Bullets shot underwater
34 ==================
35 */
36 void CG_BubbleTrail( vec3_t start, vec3_t end, float spacing ) {
37  vec3_t move;
38  vec3_t vec;
39  float len;
40  int i;
41 
42 
43  VectorCopy (start, move);
44  VectorSubtract (end, start, vec);
45  len = VectorNormalize (vec);
46 
47  // advance a random amount first
48  i = rand() % (int)spacing;
49  VectorMA( move, i, vec, move );
50 
51  VectorScale (vec, spacing, vec);
52 
53  for ( ; i < len; i += spacing ) {
54  localEntity_t *le;
55  refEntity_t *re;
56 
57  le = CG_AllocLocalEntity();
58  le->leFlags = LEF_PUFF_DONT_SCALE;
59  le->leType = LE_MOVE_SCALE_FADE;
60  le->startTime = cg.time;
61  le->endTime = cg.time + 1000 + random() * 250;
62  le->lifeRate = 1.0 / ( le->endTime - le->startTime );
63 
64  re = &le->refEntity;
65  re->shaderTime = cg.time / 1000.0f;
66 
67  re->reType = RT_SPRITE;
68  re->rotation = 0;
69  re->radius = 3;
70  re->customShader = cgs.media.waterBubbleMediumShader;
71  re->shaderRGBA[0] = 0xff;
72  re->shaderRGBA[1] = 0xff;
73  re->shaderRGBA[2] = 0xff;
74  re->shaderRGBA[3] = 0xff;
75 
76  le->color[3] = 1.0;
77 
78  le->pos.trType = TR_LINEAR;
79  le->pos.trTime = cg.time;
80  VectorCopy( move, le->pos.trBase );
81  le->pos.trDelta[0] = crandom()*5;
82  le->pos.trDelta[1] = crandom()*5;
83  le->pos.trDelta[2] = crandom()*5 + 6;
84 
85  VectorAdd (move, vec, move);
86  }
87 }
