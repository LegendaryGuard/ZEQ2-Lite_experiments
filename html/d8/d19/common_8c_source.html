150 =============
151 Com_Printf
152 
153 Both client and server can use this, and it will output
154 to the apropriate place.
155 
156 A raw string should NEVER be passed as fmt, because of "%f" type crashers.
157 =============
158 */
159 void QDECL Com_Printf( const char *fmt, ... ) {
160  va_list argptr;
161  char msg[MAXPRINTMSG];
162  static qboolean opening_qconsole = qfalse;
163 
164 
165  va_start (argptr,fmt);
166  Q_vsnprintf (msg, sizeof(msg), fmt, argptr);
167  va_end (argptr);
168 
169  if ( rd_buffer ) {
170  if ((strlen (msg) + strlen(rd_buffer)) > (rd_buffersize - 1)) {
171  rd_flush(rd_buffer);
172  *rd_buffer = 0;
173  }
174  Q_strcat(rd_buffer, rd_buffersize, msg);
175  // TTimo nooo .. that would defeat the purpose
176  //rd_flush(rd_buffer);
177  //*rd_buffer = 0;
178  return;
179  }
180 
181 #ifndef DEDICATED
182  CL_ConsolePrint( msg );
183 #endif
184 
185  // echo to dedicated console and early console
186  Sys_Print( msg );
187 
188  // logfile
189  if ( com_logfile && com_logfile->integer ) {
190  // TTimo: only open the qconsole.log if the filesystem is in an initialized state
191  // also, avoid recursing in the qconsole.log opening (i.e. if fs_debug is on)
192  if ( !logfile && FS_Initialized() && !opening_qconsole) {
193  struct tm *newtime;
194  time_t aclock;
195 
196  opening_qconsole = qtrue;
197 
198  time( &aclock );
199  newtime = localtime( &aclock );
200 
201  logfile = FS_FOpenFileWrite( "console.log" );
202 
203  if(logfile)
204  {
205  Com_Printf( "logfile opened on %s\n", asctime( newtime ) );
206 
207  if ( com_logfile->integer > 1 )
208  {
209  // force it to not buffer so we get valid
210  // data even if we are crashing
211  FS_ForceFlush(logfile);
212  }
213  }
214  else
215  {
216  Com_Printf("Opening console.log failed!\n");
217  Cvar_SetValue("logfile", 0);
218  }
219 
220  opening_qconsole = qfalse;
221  }
222  if ( logfile && FS_Initialized()) {
223  FS_Write(msg, strlen(msg), logfile);
224  }
225  }
226 }

251 /*
252 =============
253 Com_Error
254 
255 Both client and server can use this, and it will
256 do the appropriate thing.
257 =============
258 */
259 void QDECL Com_Error( int code, const char *fmt, ... ) {
260  va_list argptr;
261  static int lastErrorTime;
262  static int errorCount;
263  int currentTime;
264 
265  if(com_errorEntered)
266  Sys_Error("recursive error after: %s", com_errorMessage);
267 
268  com_errorEntered = qtrue;
269 
270  Cvar_Set("com_errorCode", va("%i", code));
271 
272  // when we are running automated scripts, make sure we
273  // know if anything failed
274  if ( com_buildScript && com_buildScript->integer ) {
275  code = ERR_FATAL;
276  }
277 
278  // if we are getting a solid stream of ERR_DROP, do an ERR_FATAL
279  currentTime = Sys_Milliseconds();
280  if ( currentTime - lastErrorTime < 100 ) {
281  if ( ++errorCount > 3 ) {
282  code = ERR_FATAL;
283  }
284  } else {
285  errorCount = 0;
286  }
287  lastErrorTime = currentTime;
288 
289  va_start (argptr,fmt);
290  Q_vsnprintf (com_errorMessage, sizeof(com_errorMessage),fmt,argptr);
291  va_end (argptr);
292 
293  if (code != ERR_DISCONNECT)
294  Cvar_Set("com_errorMessage", com_errorMessage);
295 
296  if (code == ERR_DISCONNECT || code == ERR_SERVERDISCONNECT) {
297  VM_Forced_Unload_Start();
298  SV_Shutdown( "Server disconnected" );
299  CL_Disconnect( qtrue );
300  CL_FlushMemory( );
301  VM_Forced_Unload_Done();
302  // make sure we can get at our local stuff
303  FS_PureServerSetLoadedPaks("", "");
304  com_errorEntered = qfalse;
305  longjmp (abortframe, -1);
306  } else if (code == ERR_DROP) {
307  Com_Printf ("********************\nERROR: %s\n********************\n", com_errorMessage);
308  VM_Forced_Unload_Start();
309  SV_Shutdown (va("Server crashed: %s", com_errorMessage));
310  CL_Disconnect( qtrue );
311  CL_FlushMemory( );
312  VM_Forced_Unload_Done();
313  FS_PureServerSetLoadedPaks("", "");
314  com_errorEntered = qfalse;
315  longjmp (abortframe, -1);
316  } else {
317  VM_Forced_Unload_Start();
318  CL_Shutdown(va("Client fatal crashed: %s", com_errorMessage), qtrue, qtrue);
319  SV_Shutdown(va("Server fatal crashed: %s", com_errorMessage));
320  VM_Forced_Unload_Done();
321  }
322 
323  Com_Shutdown ();
324 
325  Sys_Error ("%s", com_errorMessage);
326 }

333 Both client and server can use this, and it will

337 void Com_Quit_f( void ) {
338  // don't try to shutdown if we are in a recursive error
339  char *p = Cmd_Args( );

350  FS_Shutdown(qtrue);
351  }
352  Sys_Quit ();
353 }

381 
382 Break it up into multiple console lines

385 void Com_ParseCommandLine( char *commandLine ) {
386  int inq = 0;
387  com_consoleLines[0] = commandLine;
388  com_numConsoleLines = 1;
389 
390  while ( *commandLine ) {
391  if (*commandLine == '"') {
392  inq = !inq;
393  }
394  // look for a + seperating character
395  // if commandLine came from a file, we might have real line seperators
396  if ( (*commandLine == '+' && !inq) || *commandLine == '\n' || *commandLine == '\r' ) {
397  if ( com_numConsoleLines == MAX_CONSOLE_LINES ) {
398  return;
399  }
400  com_consoleLines[com_numConsoleLines] = commandLine + 1;
401  com_numConsoleLines++;
402  *commandLine = 0;
403  }
404  commandLine++;
405  }
406 }

575 /*
576 ============
577 Com_Filter
578 ============
579 */
580 int Com_Filter(char *filter, char *name, int casesensitive)
581 {
582  char buf[MAX_TOKEN_CHARS];
583  char *ptr;
584  int i, found;
585 
586  while(*filter) {
587  if (*filter == '*') {
588  filter++;
589  for (i = 0; *filter; i++) {
590  if (*filter == '*' || *filter == '?') break;
591  buf[i] = *filter;
592  filter++;
593  }
594  buf[i] = '\0';
595  if (strlen(buf)) {
596  ptr = Com_StringContains(name, buf, casesensitive);
597  if (!ptr) return qfalse;
598  name = ptr + strlen(buf);
599  }
600  }
601  else if (*filter == '?') {
602  filter++;
603  name++;
604  }
605  else if (*filter == '[' && *(filter+1) == '[') {
606  filter++;
607  }
608  else if (*filter == '[') {
609  filter++;
610  found = qfalse;
611  while(*filter && !found) {
612  if (*filter == ']' && *(filter+1) != ']') break;
613  if (*(filter+1) == '-' && *(filter+2) && (*(filter+2) != ']' || *(filter+3) == ']')) {
614  if (casesensitive) {
615  if (*name >= *filter && *name <= *(filter+2)) found = qtrue;
616  }
617  else {
618  if (toupper(*name) >= toupper(*filter) &&
619  toupper(*name) <= toupper(*(filter+2))) found = qtrue;
620  }
621  filter += 3;
622  }
623  else {
624  if (casesensitive) {
625  if (*filter == *name) found = qtrue;
626  }
627  else {
628  if (toupper(*filter) == toupper(*name)) found = qtrue;
629  }
630  filter++;
631  }
632  }
633  if (!found) return qfalse;
634  while(*filter) {
635  if (*filter == ']' && *(filter+1) != ']') break;
636  filter++;
637  }
638  filter++;
639  name++;
640  }
641  else {
642  if (casesensitive) {
643  if (*filter != *name) return qfalse;
644  }
645  else {
646  if (toupper(*filter) != toupper(*name)) return qfalse;
647  }
648  filter++;
649  name++;
650  }
651  }
652  return qtrue;
653 }

831  memzone_t *zone;
832 
833  if (!ptr) {
834  Com_Error( ERR_DROP, "Z_Free: NULL pointer" );
835  }
836 
837  block = (memblock_t *) ( (byte *)ptr - sizeof(memblock_t));
838  if (block->id != ZONEID) {
839  Com_Error( ERR_FATAL, "Z_Free: freed a pointer without ZONEID" );
840  }
841  if (block->tag == 0) {
842  Com_Error( ERR_FATAL, "Z_Free: freed a freed pointer" );
843  }
844  // if static memory
845  if (block->tag == TAG_STATIC) {
846  return;
847  }
848 
849  // check the memory trash tester
850  if ( *(int *)((byte *)block + block->size - 4 ) != ZONEID ) {
851  Com_Error( ERR_FATAL, "Z_Free: memory block wrote past end" );
852  }
853 
854  if (block->tag == TAG_SMALL) {
855  zone = smallzone;
856  }
857  else {
858  zone = mainzone;
859  }
860 
861  zone->used -= block->size;
862  // set the block to something that should cause problems
863  // if it is referenced...
864  Com_Memset( ptr, 0xaa, block->size - sizeof( *block ) );
865 
866  block->tag = 0; // mark as free
867 
868  other = block->prev;
869  if (!other->tag) {
870  // merge with previous free block
871  other->size += block->size;
872  other->next = block->next;
873  other->next->prev = other;
874  if (block == zone->rover) {
875  zone->rover = other;
876  }
877  block = other;
878  }
879 
880  zone->rover = block;
881 
882  other = block->next;
883  if ( !other->tag ) {
884  // merge the next free block onto the end
885  block->size += other->size;
886  block->next = other->next;
887  block->next->prev = block;
888  if (other == zone->rover) {
889  zone->rover = block;
890  }
891  }

1195  reproducable without history effects -- no out of memory errors on weird map to map changes

1197  minimize total pages in use at run time

1213  could occur.

1216  that side.

1218  If not, we can choose to make either side the new temp side and push future
